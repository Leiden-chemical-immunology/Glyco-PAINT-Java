038package createexperiment;
039
040import paint.shared.io.ExperimentInfoTableIO;
041import paint.shared.objects.ExperimentInfo;
042import tech.tablesaw.api.Table;
043
044import java.io.File;
045import java.io.IOException;
046import java.util.ArrayList;
047import java.util.List;
048import java.util.regex.Matcher;
049import java.util.regex.Pattern;
050
051import static paint.shared.constants.PaintConstants.EXPERIMENT_INFO_CSV;
052
053/**
054 * Provides functionality to generate and write an experiment information CSV file
055 * based on a list of ND2 recording files. Automatically parses and extracts condition
056 * and replicate numbers from filenames, and ensures unique file names to avoid overwriting.
057 */
058public class ExperimentInfoWriter {
059
060    /**
061     * Regular expression pattern for extracting metadata from filenames of experiment recordings.
062     * <p>
063     * Expected filename format:
064     * <pre>
065     * [digits]-Exp-[digits]-[uppercase letter][digits]-[digits].nd2
066     * </pre>
067     * Groups:
068     * <ul>
069     *   <li>Group 1 – Condition number</li>
070     *   <li>Group 2 – Replicate number</li>
071     * </ul>
072     */
073    private static final Pattern FILENAME_PATTERN =
074            Pattern.compile("^\\d+-Exp-(\\d+)-[A-Z]\\d+-(\\d+)\\.nd2$");
075
076    /**
077     * Writes experiment recording metadata into a CSV file located in the specified experiment directory.
078     * <p>
079     * Each ND2 recording file is parsed into an {@link ExperimentInfo} instance containing structured
080     * metadata such as condition and replicate numbers. The resulting table is serialized as
081     * {@code Experiment Info.csv}, with numeric suffixing to avoid overwriting existing files.
082     *
083     * @param experimentDir the directory where the experiment CSV will be created
084     * @param recordings    list of ND2 recording files to process
085     * @return the created {@code File} object pointing to the resulting CSV file
086     * @throws IOException if directory creation or file writing fails
087     */
088    public static File writeExperimentInfo(File experimentDir, List<File> recordings) throws IOException {
089        if (!experimentDir.exists() && !experimentDir.mkdirs()) {
090            throw new IOException("Failed to create experiment directory: " + experimentDir);
091        }
092
093        // Build ExperimentInfo objects from filenames
094        List<ExperimentInfo> infos = new ArrayList<>();
095
096        for (File rec : recordings) {
097            String name = rec.getName();
098            ExperimentInfo info = new ExperimentInfo();
099            info.setRecordingName(name);
100
101            // Try to parse condition and replicate from filename
102            Matcher m = FILENAME_PATTERN.matcher(name);
103            if (m.matches()) {
104                info.setConditionNumber(Integer.parseInt(m.group(1)));
105                info.setReplicateNumber(Integer.parseInt(m.group(2)));
106            } else {
107                info.setConditionNumber(0);
108                info.setReplicateNumber(0);
109            }
110
111            // Default values
112            info.setProcessFlag(true);
113            info.setProbeName("");
114            info.setProbeType("");
115            info.setCellType("");
116            info.setAdjuvant("");
117            info.setConcentration(0.0);
118            info.setThreshold(0.0);
119
120            infos.add(info);
121        }
122
123        // --- Write table via ExperimentInfoTableIO ---
124        ExperimentInfoTableIO io = new ExperimentInfoTableIO();
125        Table table = io.toTable(infos);
126
127        File csvFile = uniqueFile(experimentDir, EXPERIMENT_INFO_CSV);
128        io.writeCsv(table, csvFile.toPath());
129
130        return csvFile;
131    }
132
133    /**
134     * Ensures the output filename is unique by appending an incremented suffix if needed.
135     *
136     * @param dir      directory for the file
137     * @param fileName desired file name
138     * @return a unique file reference that does not overwrite existing files
139     */
140    private static File uniqueFile(File dir, String fileName) {
141        int dot = fileName.lastIndexOf('.');
142        String stem = (dot >= 0) ? fileName.substring(0, dot) : fileName;
143        String ext  = (dot >= 0) ? fileName.substring(dot) : "";
144
145        File candidate = new File(dir, fileName);
146        int n = 1;
147        while (candidate.exists()) {
148            candidate = new File(dir, stem + "-" + n + ext);
149            n++;
150        }
151        return candidate;
152    }
153}
