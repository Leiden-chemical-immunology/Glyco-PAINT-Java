049/**
050 * The CreateExperimentUI class provides a graphical user interface for generating experiment files.
051 * It enables users to select a directory, apply regex filters for file selection, and execute a process
052 * to create an {@code Experiment Info.csv} file. The class also handles saving regex histories and
053 * refreshing filtered file lists.
054 * <p>
055 * This class serves as the entry point for launching the Swing-based GUI application.
056 */
057public class CreateExperimentUI {
058
059    // === Preferences keys ===
060    private static final String PREF_NODE   = "paint/create-experiment";
061    private static final String KEY_IMAGES  = "lastImagesDir";
062    private static final String KEY_PROJECT = "lastProjectDir";
063
064    // Default baseline regexes (always included)
065    private static final String[] DEFAULT_REGEXES = {
066            "",  // blank
067            "^\\d-Exp-\\d+-[A-Z]\\d+-\\d+\\.nd2$",
068            "^(?!.*BF).*\\.nd2$"
069    };
070
071    /**
072     * The main method serves as the entry point for the application, initializing
073     * and displaying the Create Experiment GUI.
074     *
075     * @param args the command-line arguments passed to the application
076     */
077    public static void main(String[] args) {
078        SwingUtilities.invokeLater(CreateExperimentUI::createAndShowGUI);
079    }

081    /**
082     * Configures and displays a graphical user interface (GUI) for managing regex filters,
083     * selecting input/output directories, and processing files. The GUI consists of multiple
084     * sections, including regex controls, file list display, input/output directory configurations,
085     * and action buttons. This method also uses preferences to remember user choices
086     * (e.g., regex history and last-used directories) and dynamically updates its components
087     * based on user actions.
088     * <p>
089     * Key Features:
090     * - Regex management: Users can add, select, and delete regex patterns.
091     * - Input/output directory selection: Users can specify directories for source files
092     * and output storage.
093     * - File display: Files in the selected input directory are listed and filtered based
094     * on the regex.
095     * - Persisted preferences: User choices (like regex patterns or directories) are saved
096     * and reloaded on subsequent application launches.
097     * - Modular panel layout: Organized interface using labeled sections for regex filtering,
098     * file selection, and processing actions.
099     * <p>
100     * Components:
101     * - Regex Controls: Dropdown allowing users to input regex patterns, supported by history and deletion menus.
102     * - File List: A dynamically updated list of files matching the selected regex in the input directory.
103     * - Input/Output Controls: Buttons and labels for selecting directories and displaying their current paths.
104     * - Action Buttons: Buttons for initiating the file processing operation and closing the application.
105     * <p>
106     * Dialogs:
107     * - Uses JFileChooser dialogs for selecting directories.
108     * <p>
109     * Event Handling:
110     * - Handles user interactions via buttons and combo box events (e.g., regex filtering, file refreshing).
111     * <p>
112     * Constraints:
113     * - Limits regex entry length to a maximum of 100 characters.
114     * - Ensures only valid directories are selected and persistently stores their paths.
115     * <p>
116     * Dependencies:
117     * - javax.swing (for GUI elements like JFrame, JPanel, JComboBox, JList, JButton).
118     * - java.util.prefs.Preferences (for storing user-specific persistent settings).
119     * - java.io.File (for directory and file handling).
120     * <p>
121     * Usage:
122     * Typically called during application initialization to present the user interface.
123     * This method does not return, as it focuses on creating and displaying the application
124     * window with active event handling for user interaction.
125     */
126    private static void createAndShowGUI() {
127        Preferences prefs = Preferences.userRoot().node(PREF_NODE);
128
129        JFrame frame = new JFrame("Create Experiment");
130        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
131        frame.setLayout(new BorderLayout());
132
133        // Use a small clean font (not bold)
134        Font smallFont = new JLabel().getFont().deriveFont(Font.PLAIN, 12f);
135
136        // === Regex filter controls ===
137        JPanel topPanel = new JPanel(new BorderLayout(5, 5));
138        topPanel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
139
140        JComboBox<String> regexCombo = new JComboBox<>();
141        regexCombo.setEditable(true);
142        regexCombo.setFont(smallFont);
143
144        // Always insert defaults first
145        for (String def : DEFAULT_REGEXES) {
146            regexCombo.addItem(def);
147        }
148
149        // Load regex history from preferences
150        boolean hasAnySaved = false;
151        String lastRegex = "";
152        for (int i = 0; ; i++) {
153            String rx = prefs.get("regex." + i, null);
154            if (rx == null) {
155                break;
156            }
157            if (!rx.trim().isEmpty()) {
158                hasAnySaved = true;
159                if (((DefaultComboBoxModel<String>) regexCombo.getModel()).getIndexOf(rx.trim()) == -1) {
160                    regexCombo.addItem(rx.trim());
161                }
162                lastRegex = rx.trim();
163            }
164        }
165
166        // Select last used regex if any
167        if (!lastRegex.isEmpty()) {
168            regexCombo.setSelectedItem(lastRegex);
169        }
170
171        // If first installation (no saved regex), persist defaults
172        if (!hasAnySaved) {
173            saveRegexHistory(regexCombo, prefs);
174        }
175
176        // === Right-click delete regex ===
177        JTextField regexEditor = (JTextField) regexCombo.getEditor().getEditorComponent();
178        JPopupMenu regexMenu = new JPopupMenu();
179        JMenuItem deleteItem = new JMenuItem("Delete this regex");
180        regexMenu.add(deleteItem);
181        regexEditor.setComponentPopupMenu(regexMenu);
182        deleteItem.addActionListener(ev -> {
183            String selected = regexEditor.getText().trim();
184            if (!selected.isEmpty()) {
185                regexCombo.removeItem(selected);
186                saveRegexHistory(regexCombo, prefs);
187            }
188        });
189
190        JButton filterButton = new JButton("Filter");
191        filterButton.setFont(smallFont);
192        filterButton.setPreferredSize(new Dimension(70, 22));
193
194        topPanel.add(new JLabel("Regex:"), BorderLayout.WEST);
195        topPanel.add(regexCombo, BorderLayout.CENTER);
196        topPanel.add(filterButton, BorderLayout.EAST);
197
198        // === File list ===
199        DefaultListModel<File> listModel = new DefaultListModel<>();
200        JList<File> fileList = new JList<>(listModel);
201        fileList.setSelectionMode(ListSelectionModel.MULTIPLE_INTERVAL_SELECTION);
202        fileList.setFont(smallFont);
203        fileList.setCellRenderer(new DefaultListCellRenderer() {
204            @Override
205            public Component getListCellRendererComponent(
206                    JList<?> list, Object value, int index,
207                    boolean isSelected, boolean cellHasFocus) {
208                Component c = super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
209                if (value instanceof File) {
210                    setText(((File) value).getName());
211                }
212                setFont(smallFont);
213                return c;
214            }
215        });
216        JScrollPane scrollPane = new JScrollPane(fileList);
217        scrollPane.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
218
219        // === Input/output controls ===
220        JPanel ioPanel = new JPanel();
221        ioPanel.setLayout(new BoxLayout(ioPanel, BoxLayout.Y_AXIS));
222        ioPanel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
223
224        Dimension buttonSize = new Dimension(90, 22);
225
226        final File[] inputDir = {new File(prefs.get(KEY_IMAGES, System.getProperty("user.home")))};
227        final File[] outputDir = {new File(prefs.get(KEY_PROJECT, System.getProperty("user.home")))};
228
229        // Images row
230        JPanel imagesRow = new JPanel(new FlowLayout(FlowLayout.LEFT, 5, 2));
231        JButton imagesButton = new JButton("Images");
232        imagesButton.setPreferredSize(buttonSize);
233        imagesButton.setFont(smallFont);
234        JLabel inputDirLabel = new JLabel(inputDir[0].getAbsolutePath());
235        inputDirLabel.setFont(smallFont);
236        imagesRow.add(imagesButton);
237        imagesRow.add(inputDirLabel);
238
239        // Project row
240        JPanel projectRow = new JPanel(new FlowLayout(FlowLayout.LEFT, 5, 2));
241        JButton projectButton = new JButton("Project");
242        projectButton.setPreferredSize(buttonSize);
243        projectButton.setFont(smallFont);
244        JLabel outputDirLabel = new JLabel(outputDir[0].getAbsolutePath());
245        outputDirLabel.setFont(smallFont);
246        projectRow.add(projectButton);
247        projectRow.add(outputDirLabel);
248
249        ioPanel.add(imagesRow);
250        ioPanel.add(projectRow);
251
252        // === Action buttons ===
253        JPanel actionPanel = new JPanel(new FlowLayout(FlowLayout.RIGHT));
254        JButton processButton = new JButton("Process");
255        JButton closeButton = new JButton("Close");
256        processButton.setPreferredSize(new Dimension(80, 22));
257        processButton.setFont(smallFont);
258        closeButton.setPreferredSize(new Dimension(80, 22));
259        closeButton.setFont(smallFont);
260        actionPanel.add(processButton);
261        actionPanel.add(closeButton);
262
263        // Bottom container
264        JPanel bottomPanel = new JPanel(new BorderLayout(0, 8));
265        bottomPanel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
266        bottomPanel.add(ioPanel, BorderLayout.CENTER);
267        bottomPanel.add(actionPanel, BorderLayout.SOUTH);
268
269        // === Layout ===
270        frame.add(topPanel, BorderLayout.NORTH);
271        frame.add(scrollPane, BorderLayout.CENTER);
272        frame.add(bottomPanel, BorderLayout.SOUTH);
273
274        // Refresh helper (reloads visible files)
275        Runnable refresh = () -> {
276            String regex = ((String) regexCombo.getEditor().getItem()).trim();
277            if (!regex.isEmpty() && regex.length() <= 100 &&
278                    ((DefaultComboBoxModel<String>) regexCombo.getModel()).getIndexOf(regex) == -1) {
279                regexCombo.addItem(regex);
280                saveRegexHistory(regexCombo, prefs);
281            }
282            refreshList(listModel, inputDir[0], regex, frame);
283        };
284
285        // === Button actions ===
286
287        // Image directory chooser
288        imagesButton.addActionListener((ActionEvent e) -> {
289            JFileChooser chooser = new JFileChooser(inputDir[0]);
290            chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
291            chooser.setDialogTitle("Select Images Directory");
292            chooser.setPreferredSize(new Dimension(600, 350));
293            int result = chooser.showOpenDialog(frame);
294            if (result == JFileChooser.APPROVE_OPTION) {
295                File chosen = chooser.getSelectedFile();
296                if (chosen != null && chosen.isDirectory() && !chosen.isHidden()) {
297                    inputDir[0] = chosen;
298                    inputDirLabel.setText(inputDir[0].getAbsolutePath());
299                    prefs.put(KEY_IMAGES, inputDir[0].getAbsolutePath());
300                    refresh.run();
301                }
302            }
303        });
304
305        // Project directory chooser
306        projectButton.addActionListener((ActionEvent e) -> {
307            JFileChooser chooser = new JFileChooser(outputDir[0]);
308            chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
309            chooser.setDialogTitle("Select Project Directory");
310            int result = chooser.showOpenDialog(frame);
311            if (result == JFileChooser.APPROVE_OPTION) {
312                File chosen = chooser.getSelectedFile();
313                if (chosen != null && chosen.isDirectory() && !chosen.isHidden()) {
314                    outputDir[0] = chosen;
315                    outputDirLabel.setText(outputDir[0].getAbsolutePath());
316                    prefs.put(KEY_PROJECT, outputDir[0].getAbsolutePath());
317                }
318            }
319        });
320
321        // Regex combo and filter refresh
322        regexCombo.addActionListener(e -> refresh.run());
323        filterButton.addActionListener(e -> refresh.run());
324
325        // Process button — generate Experiment Info CSV
326        processButton.addActionListener((ActionEvent e) -> {
327            if (outputDir[0] == null) {
328                JOptionPane.showMessageDialog(frame, "No project directory chosen.");
329                return;
330            }
331            if (inputDir[0] == null) {
332                JOptionPane.showMessageDialog(frame, "No images directory chosen.");
333                return;
334            }
335            List<File> selectedFiles = fileList.getSelectedValuesList();
336            if (selectedFiles.isEmpty()) {
337                JOptionPane.showMessageDialog(frame, "No files selected to process.");
338                return;
339            }
340
341            try {
342                String experimentName = inputDir[0].getName();
343                File experimentDir = new File(outputDir[0], experimentName);
344                if (!experimentDir.exists() && !experimentDir.mkdirs()) {
345                    throw new IOException("Failed to create experiment directory: " + experimentDir);
346                }
347
348                // Delegate to ExperimentInfoWriter
349                File createdFile = ExperimentInfoWriter.writeExperimentInfo(experimentDir, selectedFiles);
350
351                fileList.clearSelection();
352                JOptionPane.showMessageDialog(
353                        frame,
354                        "Experiment info written to:\n" + createdFile.getAbsolutePath(),
355                        "Success",
356                        JOptionPane.INFORMATION_MESSAGE
357                );
358            } catch (IOException ex) {
359                JOptionPane.showMessageDialog(frame, "Error writing ExperimentInfo.csv: " + ex.getMessage());
360            }
361        });
362
363        // Close button
364        closeButton.addActionListener(e -> frame.dispose());
365
366        // Final setup
367        frame.setSize(400, 500);
368        frame.setLocationRelativeTo(null);
369        frame.setVisible(true);
370
371        if (inputDir[0] != null && inputDir[0].isDirectory()) {
372            refresh.run();
373        }
374    }

376    /**
377     * Saves the history of regex patterns from a given JComboBox to the specified Preferences object.
378     * The method first clears any existing regex entries in the Preferences storage, and then adds
379     * non-empty and valid regex patterns (trimmed and with a maximum length of 100 characters) from
380     * the JComboBox to the Preferences with sequential keys.
381     *
382     * @param combo the JComboBox containing the regex patterns to be saved.
383     * @param prefs the Preferences object used to persist the regex history.
384     */
385    private static void saveRegexHistory(JComboBox<String> combo, Preferences prefs) {
386        try {
387            for (int i = 0; ; i++) {
388                String key = "regex." + i;
389                if (prefs.get(key, null) == null) {
390                    break;
391                }
392                prefs.remove(key);
393            }
394        } catch (Exception ignored) {
395        }
396
397        int idx = 0;
398        for (int i = 0; i < combo.getItemCount(); i++) {
399            String item = combo.getItemAt(i).trim();
400            if (!item.isEmpty() && item.length() <= 100) {
401                prefs.put("regex." + idx, item);
402                idx++;
403            }
404        }
405    }

407    /**
408     * Refreshes the given list model with files from the specified directory that match the given regular expression.
409     * If the directory is null or does not contain any files, the list model will be cleared. If a regex is provided,
410     * only files matching the pattern will be added to the list model. If the regex is invalid, an error message will
411     * be displayed in a dialog.
412     *
413     * @param model  the DefaultListModel used to display the filtered list of files
414     * @param dir    the directory whose files will be scanned
415     * @param regex  the regular expression used to filter file names; if null or empty, all files are included
416     * @param parent the parent component for displaying error dialogs in case of an invalid regex
417     */
418    private static void refreshList(DefaultListModel<File> model, File dir, String regex, Component parent) {
419        model.clear();
420        if (dir == null) {
421            return;
422        }
423
424        Pattern pattern = null;
425        String rx = (regex == null) ? "" : regex.trim();
426        if (!rx.isEmpty()) {
427            try {
428                pattern = Pattern.compile(rx);
429            } catch (PatternSyntaxException ex) {
430                JOptionPane.showMessageDialog(parent, "Invalid regex: " + ex.getDescription());
431                return;
432            }
433        }
434
435        File[] children = dir.listFiles(f -> !f.isHidden());
436        if (children == null) {
437            return;
438        }
439
440        for (File f : children) {
441            if (f.isFile() && (pattern == null || pattern.matcher(f.getName()).matches())) {
442                model.addElement(f);
443            }
444        }
445    }
