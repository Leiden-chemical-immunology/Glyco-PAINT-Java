045package paint.fiji.trackmate;
046
047import com.google.gson.JsonObject;
048import org.scijava.command.Command;
049import paint.shared.config.PaintConfig;
050import paint.shared.utils.PaintConsoleWindow;
051import paint.shared.utils.PaintLogger;
052import paint.shared.utils.PaintPrefs;
053
054import java.nio.file.Path;
055import java.nio.file.Paths;
056import java.util.ArrayList;
057import java.util.List;
058
059import static paint.shared.constants.PaintConstants.PAINT_CONFIGURATION_JSON;
060import static paint.shared.utils.ValidProjectPath.getValidProjectPath;
061
062/**
063 * Executes the Paint TrackMate analysis in a headless (non-interactive)
064 * Fiji or ImageJ session. This class implements {@link Command} and can
065 * be invoked either manually or via plugin registration.
066 *
067 * <p>Responsibilities include:</p>
068 * <ul>
069 *   <li>Validating the project root path.</li>
070 *   <li>Initializing PaintConfig and PaintLogger.</li>
071 *   <li>Identifying active experiments.</li>
072 *   <li>Executing the TrackMate pipeline on selected datasets.</li>
073 * </ul>
074 */
075public class TrackMateHeadless implements Command {
076
077    /**
078     * Main command entry point — executes the TrackMate pipeline for all
079     * experiments marked as active in the Paint configuration file.
080     *
081     * <p>Execution steps:</p>
082     * <ol>
083     *   <li>Retrieve and validate project root path.</li>
084     *   <li>Initialize configuration and logging systems.</li>
085     *   <li>Identify all active experiments.</li>
086     *   <li>Run the TrackMate workflow using {@link RunTrackMateOnProject}.</li>
087     * </ol>
088     *
089     * <p>Logs all progress and exceptions via {@link PaintLogger}.</p>
090     */
091    @Override
092    public void run() {
093        try {
094            // -----------------------------------------------------------------
095            // Step 1 — Validate project path
096            // -----------------------------------------------------------------
097            Path projectPath = getValidProjectPath();
098            if (projectPath == null) {
099                PaintLogger.errorf("No valid Project Root found.");
100                return;
101            }
102
103            // -----------------------------------------------------------------
104            // Step 2 — Initialize configuration
105            // -----------------------------------------------------------------
106            Path jsonPath = projectPath.resolve(PAINT_CONFIGURATION_JSON);
107            if (!jsonPath.toFile().exists()) {
108                PaintLogger.errorf("Invalid or missing configuration file: %s", jsonPath);
109                return;
110            }
111            PaintConfig.initialise(projectPath);
112            PaintConfig.reinitialise(projectPath);  // ensures consistent reload
113
114            Path imagesPath = Paths.get(PaintPrefs.getString("Path", "Images Root", ""));
115
116            // -----------------------------------------------------------------
117            // Step 3 — Initialize console and logging
118            // -----------------------------------------------------------------
119            PaintConsoleWindow.createConsoleFor("TrackMate Headless");
120            PaintLogger.initialise(projectPath, "TrackMateHeadless");
121
122            // -----------------------------------------------------------------
123            // Step 4 — Identify active experiments
124            // -----------------------------------------------------------------
125            List<String> experiments = getSelectedExperiments();
126            if (experiments.isEmpty()) {
127                PaintLogger.warnf("No experiments marked as true in configuration. Nothing to process.");
128                return;
129            }
130
131            PaintLogger.infof("Running TrackMate in headless mode...");
132            PaintLogger.infof("Project path: %s", projectPath);
133            PaintLogger.infof("Images root:  %s", imagesPath);
134            PaintLogger.infof("Experiments:  %s", experiments);
135
136            // -----------------------------------------------------------------
137            // Step 5 — Execute TrackMate on project
138            // -----------------------------------------------------------------
139            boolean success = RunTrackMateOnProject.runProject(projectPath, imagesPath, experiments, null, null);
140            if (success) {
141                PaintLogger.infof("TrackMate completed successfully.");
142            } else {
143                PaintLogger.errorf("TrackMate encountered errors.");
144            }
145
146        } catch (Exception e) {
147            PaintLogger.errorf("TrackMate headless execution failed: %s", e.getMessage());
148        }
149    }
150
151    /**
152     * Retrieves all experiment names from the Paint configuration that are
153     * explicitly marked as <code>true</code> in the "Experiments" section.
154     *
155     * <p>If no valid section is found, or if all values are false, a warning is logged.</p>
156     *
157     * @return a list of selected experiment names; empty if none found
158     */
159    private static List<String> getSelectedExperiments() {
160        List<String> selected = new ArrayList<>();
161        JsonObject experiments = PaintConfig.instance().getSection("Experiments");
162
163        if (experiments == null) {
164            PaintLogger.warnf("No 'Experiments' section found in configuration.");
165            return selected;
166        }
167
168        for (String key : experiments.keySet()) {
169            try {
170                if (experiments.get(key).getAsBoolean()) {
171                    selected.add(key);
172                }
173            } catch (Exception ignored) {
174                // Ignore malformed entries (non-boolean)
175            }
176        }
177        return selected;
178    }
179}
