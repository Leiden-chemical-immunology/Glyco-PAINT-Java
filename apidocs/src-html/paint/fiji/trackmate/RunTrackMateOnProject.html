048package paint.fiji.trackmate;
049
050import paint.shared.dialogs.ProjectDialog;
051import paint.shared.utils.PaintLogger;
052import paint.shared.validate.ValidationResult;
053
054import java.nio.file.Files;
055import java.nio.file.Path;
056import java.time.Duration;
057import java.time.LocalDateTime;
058import java.util.List;
059
060import static paint.shared.constants.PaintConstants.EXPERIMENT_INFO_CSV;
061import static paint.shared.utils.Miscellaneous.formatDuration;
062import static paint.shared.validate.ImageRootValidator.validateImageRoot;
063import static paint.shared.validate.ValidationHandler.validateExperiments;
064
065/**
066 * The {@code RunTrackMateOnProject} class orchestrates the TrackMate analysis
067 * across multiple experiments contained within a project directory.
068 * <p>
069 * It validates image directories and configuration files before processing,
070 * handles per-experiment execution, supports user-initiated cancellation, and
071 * logs all key events and outcomes.
072 * </p>
073 * <p>
074 * Each experiment is validated and processed independently, ensuring that a
075 * failure in one experiment does not halt others. At completion, a total
076 * runtime summary is logged.
077 * </p>
078 */
079public class RunTrackMateOnProject extends RunTrackMateOnExperiment {
080
081    /**
082     * Executes the TrackMate analysis pipeline for all specified experiments
083     * within the given project directory. Performs validation on both image
084     * roots and CSV metadata, then runs the core processing for each experiment.
085     *
086     * @param projectPath     the root directory of the Paint project containing experiment subfolders
087     * @param imagesPath      the directory containing image files for the experiments
088     * @param experimentNames list of experiment folder names to be processed
089     * @param dialog          optional {@link ProjectDialog} used for user-driven cancellation
090     * @param sweepDir        optional override directory for writing analysis results
091     *                        (defaults to {@code projectPath} if {@code null})
092     * @return {@code true} if all experiments processed successfully;
093     * {@code false} if any validation or runtime errors occurred
094     */
095    public static boolean runProject(Path projectPath,
096                                     Path imagesPath,
097                                     List<String> experimentNames,
098                                     ProjectDialog dialog,
099                                     Path sweepDir) {
100
101        boolean status = true;                               // Overall success flag
102        LocalDateTime start = LocalDateTime.now();           // Timestamp for runtime measurement
103
104        // ---------------------------------------------------------------------
105        // Phase 1 – Validate that required images exist
106        // ---------------------------------------------------------------------
107        ValidationResult validateResult = validateImageRoot(projectPath, imagesPath, experimentNames);
108        if (!validateResult.isValid()) {
109            for (String err : validateResult.getErrors()) {
110                PaintLogger.errorf(err);
111            }
112            return false;                                     // Abort on image validation failure
113        }
114
115        // ---------------------------------------------------------------------
116        // Phase 2 – Validate experiment configuration (Experiment Info.csv)
117        // ---------------------------------------------------------------------
118        validateResult = validateExperiments(projectPath, experimentNames, EXPERIMENT_INFO_CSV);
119        if (!validateResult.isValid()) {
120            for (String err : validateResult.getErrors()) {
121                PaintLogger.errorf(err);
122            }
123            return false;                                   // Abort on configuration validation failure
124        }
125
126        // ---------------------------------------------------------------------
127        // Phase 3 – Execute TrackMate per experiment
128        // ---------------------------------------------------------------------
129        for (String experimentName : experimentNames) {
130            // Allow user to cancel processing mid-run
131            if (dialog != null && dialog.isCancelled()) {
132                PaintLogger.warnf("Processing aborted by user.");
133                break;
134            }
135
136            // Determine target experiment path (sweepDir override if given)
137            Path experimentPath = (sweepDir != null)
138                    ? sweepDir.resolve(experimentName)
139                    : projectPath.resolve(experimentName);
140
141            // Verify the directory actually exists
142            if (!Files.isDirectory(experimentPath)) {
143                PaintLogger.errorf("Experiment directory '%s' does not exist.", experimentPath);
144                continue;
145            }
146
147            // Run the TrackMate workflow for this experiment
148            try {
149                boolean ok = runTrackMateOnExperiment(
150                        experimentPath,
151                        imagesPath.resolve(experimentName),
152                        dialog);
153
154                if (!ok) {
155                    status = false;                         // Mark failure if sub-run failed
156                }
157
158            } catch (Exception e) {
159                PaintLogger.errorf("Error during TrackMate run for '%s': %s",
160                                   experimentName, e.getMessage());
161                status = false;
162            }
163        }
164
165        // ---------------------------------------------------------------------
166        // Phase 4 – Report summary and duration
167        // ---------------------------------------------------------------------
168        Duration totalDuration = Duration.between(start, LocalDateTime.now());
169        PaintLogger.infof("Processed %d experiments in %s.",
170                          experimentNames.size(),
171                          formatDuration((int) (totalDuration.toMillis() / 1000)));
172
173        return status;
174    }
175}
