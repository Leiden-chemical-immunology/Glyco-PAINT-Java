001/******************************************************************************
002 *  Class:        RunTrackMateOnProjectSweep.java
003 *  Package:      paint.fiji.trackmate
004 *
005 *  PURPOSE:
006 *    Executes the Paint TrackMate analysis pipeline across multiple experiments
007 *    under varying parameter configurations ("sweep mode"). This allows
008 *    systematic exploration of parameter sensitivity by iteratively running
009 *    TrackMate under different settings.
010 *
011 *  DESCRIPTION:
012 *    • Loads the Sweep configuration from Paint's JSON configuration file.
013 *    • Iterates over all active TrackMate parameters marked for sweeping.
014 *    • Creates separate subdirectories for each parameter–value combination.
015 *    • Rewrites Paint configuration files dynamically for each sweep run.
016 *    • Executes the TrackMate workflow for all experiments under the new setup.
017 *    • Optionally flattens the sweep results for summary analysis.
018 *
019 *  RESPONSIBILITIES:
020 *    • Manage configuration cloning and parameter substitution per sweep run.
021 *    • Invoke {@link RunTrackMateOnProject} for each parameter combination.
022 *    • Maintain logging and recovery of the original Paint configuration.
023 *
024 *  USAGE EXAMPLE:
025 *    boolean ok = RunTrackMateOnProjectSweep.runWithSweep(
026 *                     Paths.get("/Paint Project"),
027 *                     Paths.get("/Volumes/Images"),
028 *                     Arrays.asList("221108", "221122"));
029 *
030 *  DEPENDENCIES:
031 *    – paint.fiji.trackmate.RunTrackMateOnProject
032 *    – paint.shared.config.PaintConfig
033 *    – paint.shared.config.SweepConfig
034 *    – paint.shared.utils.PaintLogger
035 *    – paint.fiji.utils.SweepFlattener

067/**
068 * Coordinates a "parameter sweep" execution for TrackMate across multiple
069 * experiments. For each active sweep parameter, the Paint configuration
070 * is duplicated and modified, and the entire project is reprocessed.
071 * <p>
072 * This class enables systematic parameter-space exploration for TrackMate
073 * within Paint’s experiment framework.
074 * </p>
075 */
076public class RunTrackMateOnProjectSweep {
077
078    /**
079     * Executes the TrackMate analysis workflow across multiple parameter
080     * configurations defined in the sweep configuration file.
081     * <p>
082     * For each parameter–value combination, this method:
083     * <ul>
084     *   <li>Creates a clean subdirectory under the project’s "Sweep" folder.</li>
085     *   <li>Copies and modifies the Paint configuration file.</li>
086     *   <li>Runs all selected experiments under that parameter setting.</li>
087     *   <li>Logs results and restores the original configuration afterward.</li>
088     * </ul>
089     *
090     * @param projectPath     the root directory of the Paint project
091     * @param imagesPath      the directory containing experiment image data
092     * @param experimentNames list of experiments to include in the sweep
093     * @return {@code true} if all sweep combinations complete successfully;
094     *         {@code false} otherwise
095     * @throws IOException if any file or directory operation fails
096     */
097    public static boolean runWithSweep(Path projectPath,
098                                       Path imagesPath,
099                                       List<String> experimentNames) throws IOException {
100
101        // ---------------------------------------------------------------------
102        // Phase 1 – Load sweep configuration
103        // ---------------------------------------------------------------------
104        Path sweepFile = projectPath.resolve(PAINT_SWEEP_CONFIGURATION_JSON);
105        if (!Files.exists(sweepFile)) {
106            PaintLogger.infof("No sweep configuration found at %s", sweepFile);
107            return false;
108        }
109
110        SweepConfig sweepConfig = new SweepConfig(sweepFile.toString());
111        Map<String, List<Number>> sweeps = sweepConfig.getActiveSweepValues("TrackMate Sweep");
112
113        if (sweeps.isEmpty()) {
114            PaintLogger.infof("Sweep enabled, but no active sweep parameters defined.");
115            return false;
116        }
117
118        boolean overallStatus = true;
119        List<String[]> summaryRows = new ArrayList<>();
120
121        // ---------------------------------------------------------------------
122        // Phase 2 – Generate sweep summary for logging
123        // ---------------------------------------------------------------------
124        List<String> sweepSummary = new ArrayList<>();
125        sweepSummary.add("");
126        try {
127            for (Map.Entry<String, List<Number>> entry : sweeps.entrySet()) {
128                String parameter = entry.getKey();       // parameter name
129                List<Number> values = entry.getValue();  // list of values
130
131                // Convert values to comma-separated string
132                String valueList = values.stream()
133                        .map(Object::toString)
134                        .reduce((a, b) -> a + ", " + b)
135                        .orElse("(none)");
136
137                // Add a header line for the parameter
138                sweepSummary.add(String.format("Parameter: %s", parameter));
139                sweepSummary.add(String.format("Values:    %s", valueList));
140                sweepSummary.add("");
141            }
142
143            // Add the experiments section at the end
144            if (experimentNames != null && !experimentNames.isEmpty()) {
145                String exps = String.join(", ", experimentNames);
146                sweepSummary.add(String.format("Experiments: %s", exps));
147            }
148        } catch (Exception e) {
149            PaintLogger.errorf("Error building sweep summary: %s", e.getMessage());
150        }
151        PaintLogger.doc("Sweep analysis to be performed", sweepSummary);
152
153        // ---------------------------------------------------------------------
154        // Phase 3 – Perform parameter sweep
155        // ---------------------------------------------------------------------
156        try {
157            for (Map.Entry<String, List<Number>> entry : sweeps.entrySet()) {
158                // Cycle through the parameters
159                String parameter    = entry.getKey();       // Holds a parameter that we sweep
160                List<Number> values = entry.getValue();     // Values holds the values
161
162                // Store original parameter value for restoration
163                String originalValue = PaintConfig.getString("TrackMate", parameter, "undefined");
164
165                for (Number val : values) {
166                    PaintLogger.infof("Running sweep for %s = %s", parameter, val);
167
168                    // Create clean sweep directory
169                    Path sweepPath = projectPath.resolve("Sweep").resolve("[" + parameter + "]-[" + val + "]");
170                    FileUtils.deleteDirectory(sweepPath.toFile());
171                    Files.createDirectories(sweepPath);
172
173                    // Copy baseline PaintConfig.json
174                    Path baselineConfig = projectPath.resolve(PAINT_CONFIGURATION_JSON);
175                    Path configCopy     = sweepPath.resolve(PAINT_CONFIGURATION_JSON);
176                    try {
177                        Files.copy(baselineConfig, configCopy, StandardCopyOption.REPLACE_EXISTING);
178                        PaintLogger.infof("Copied baseline PaintConfig.json to %s", configCopy);
179                    } catch (IOException e) {
180                        PaintLogger.errorf("Failed to copy PaintConfig.json to %s: %s", configCopy, e.getMessage());
181                    }
182
183                    // Reinitialize PaintConfig for this sweep
184                    PaintConfig.reinitialise(sweepPath);
185
186                    // Apply updated parameter value
187                    if (val.doubleValue() == val.intValue()) {
188                        PaintConfig.setInt("TrackMate", parameter, val.intValue());
189                    } else {
190                        PaintConfig.setDouble("TrackMate", parameter, val.doubleValue());
191                    }
192
193                    // Persist modified configuration
194                    PaintConfig.instance().save();
195
196                    // Copy Experiment Info.csv files into sweep directory
197                    for (String expName : experimentNames) {
198                        Path expSrc    = projectPath.resolve(expName).resolve(EXPERIMENT_INFO_CSV);
199                        Path expDstDir = sweepPath.resolve(expName);
200                        Path expDst    = expDstDir.resolve(EXPERIMENT_INFO_CSV);
201                        try {
202                            if (Files.exists(expSrc)) {
203                                Files.createDirectories(expDstDir);
204                                Files.copy(expSrc, expDst, StandardCopyOption.REPLACE_EXISTING);
205                            } else {
206                                PaintLogger.warnf("Experiment Info.csv not found for %s at %s", expName, expSrc);
207                            }
208                        } catch (IOException e) {
209                            PaintLogger.errorf("Failed to copy Experiment Info.csv for %s: %s", expName, e.getMessage());
210                        }
211                    }
212
213                    // Run TrackMate workflow for all experiments under current sweep
214                    boolean status = RunTrackMateOnProject.runProject(projectPath, imagesPath, experimentNames, null, sweepPath);
215
216                    summaryRows.add(new String[]{
217                            parameter, val.toString(), sweepPath.toString(),
218                            status ? "SUCCESS" : "FAILED"
219                    });
220
221                    if (!status) {
222                        overallStatus = false;
223                    }
224                }
225
226                // Optionally restore original value after each parameter sweep
227                PaintConfig.setString("TrackMate", parameter, originalValue);
228            }
229
230        } finally {
231            // Always restore PaintConfig context to the project root
232            PaintConfig.reinitialise(projectPath);
233            PaintLogger.infof("Restored PaintConfig to project root: %s", projectPath);
234        }
235
236        // ---------------------------------------------------------------------
237        // Phase 4 – Flatten results (optional)
238        // ---------------------------------------------------------------------
239        if (overallStatus) {
240            flattenSweep(projectPath.resolve("Sweep"), experimentNames, true);
241        }
242
243        return overallStatus;
244    }
245
246    /**
247     * Command-line entry point for testing the sweep execution.
248     */
249    public static void main(String[] args) throws IOException {
250        runWithSweep(
251                Paths.get("/Users/hans/Paint Test Project"),
252                Paths.get("/Volumes/Extreme Pro/Omero"),
253                Arrays.asList("221012", "AnyName"));
254    }
