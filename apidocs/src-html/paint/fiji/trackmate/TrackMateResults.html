001/******************************************************************************
002 *  Class:        TrackMateResults.java
003 *  Package:      paint.fiji.trackmate
004 *
005 *  PURPOSE:
006 *    Represents the outcome of a TrackMate analysis run, encapsulating
007 *    success flags, detection statistics, frame counts, and runtime duration.
008 *
009 *  DESCRIPTION:
010 *    • Captures and stores per-recording results after TrackMate processing.
011 *    • Tracks number of detected spots, total and filtered tracks, and
012 *      the number of frames analyzed.
013 *    • Provides duration metrics for performance reporting.
014 *    • Used by {@link RunTrackMateOnRecording} and higher-level classes
015 *      to summarize execution results.
016 *
017 *  RESPONSIBILITIES:
018 *    • Store immutable TrackMate result data for one recording.
019 *    • Provide structured accessors for logging and reporting.
020 *    • Support consistent summary formatting via {@link #toString()}.
021 *
022 *  USAGE EXAMPLE:
023 *    TrackMateResults result = new TrackMateResults(
024 *        true, true, 452, 87, 73, 300, Duration.ofSeconds(28), 6124);
025 *
026 *  DEPENDENCIES:
027 *    – java.time.Duration
028 *    – paint.fiji.trackmate.RunTrackMateOnRecording

040package paint.fiji.trackmate;
041
042import java.time.Duration;
043
044/**
045 * Immutable container for the results of a TrackMate analysis.
046 * <p>
047 * Encapsulates statistics such as the number of detected spots,
048 * total and filtered tracks, number of analyzed frames, and
049 * runtime duration.
050 * </p>
051 */
052public class TrackMateResults {
053    private final boolean  success;
054    private final boolean  calculationPerformed;
055    private final int      numberOfSpots;
056    private final int      numberOfTracks;
057    private final int      numberOfFilteredTracks;
058    private final int      numberOfFrames;
059    private final Duration duration;
060    private final int      numberOfSpotsInAllTracks;
061
062
063    /**
064     * Default constructor.
065     * <p>
066     * Creates an immutable {@code TrackMateResults} instance with all
067     * values initialized to default (false, zero, or null).
068     * </p>
069     */
070    public TrackMateResults() {
071        this(false, false, 0, 0, 0, 0, null, 0);
072    }
073
074    /**
075     * Constructs a {@code TrackMateResults} instance with a success flag and
076     * calculation state only, setting all numeric values to zero.
077     *
078     * @param success              whether the analysis succeeded
079     * @param calculationPerformed whether the TrackMate pipeline was executed
080     */
081    public TrackMateResults(boolean success, boolean calculationPerformed) {
082        this(success, calculationPerformed, 0, 0, 0, 0, null, 0);
083    }
084
085    /**
086     * Constructs a {@code TrackMateResults} instance with only a success flag.
087     * Assumes no calculations were performed and sets all numeric fields to zero.
088     *
089     * @param success whether the analysis succeeded
090     */
091    public TrackMateResults(boolean success) {
092        this(success, false, 0, 0, 0, 0, null, 0);
093    }
094
095    /**
096     * Constructs a fully detailed {@code TrackMateResults} instance.
097     *
098     * @param success                true if the analysis succeeded
099     * @param calculationPerformed   true if processing was executed
100     * @param numberOfSpots          number of detected spots
101     * @param numberOfTracks         total number of identified tracks
102     * @param numberOfFilteredTracks number of tracks after filtering
103     * @param numberOfFrames         total number of frames analyzed
104     * @param duration               runtime duration of analysis
105     * @param numberOfSpotsInAllTracks total number of spots in all tracks
106     */
107    public TrackMateResults(boolean success,
108                            boolean calculationPerformed,
109                            int numberOfSpots,
110                            int numberOfTracks,
111                            int numberOfFilteredTracks,
112                            int numberOfFrames,
113                            Duration duration,
114                            int numberOfSpotsInAllTracks) {
115        this.success                  = success;
116        this.calculationPerformed     = calculationPerformed;
117        this.numberOfSpots            = numberOfSpots;
118        this.numberOfTracks           = numberOfTracks;
119        this.numberOfFilteredTracks   = numberOfFilteredTracks;
120        this.numberOfFrames           = numberOfFrames;
121        this.duration                 = duration;
122        this.numberOfSpotsInAllTracks = numberOfSpotsInAllTracks;
123    }
124
125    // -------------------------------------------------------------------------
126    // Getters
127    // -------------------------------------------------------------------------
128
129    /**
130     * @return true if the analysis completed successfully
131     */
132    public boolean isSuccess() {
133        return success;
134    }
135
136    /**
137     * @return true if the calculation was performed
138     */
139    public boolean isCalculationPerformed() {
140        return calculationPerformed;
141    }
142
143    /**
144     * @return total number of detected spots
145     */
146    public int getNumberOfSpots() {
147        return numberOfSpots;
148    }
149
150    /**
151     * @return total number of generated tracks
152     */
153    public int getNumberOfTracks() {
154        return numberOfTracks;
155    }
156
157    /**
158     * @return number of tracks that passed filtering
159     */
160    public int getNumberOfFilteredTracks() {
161        return numberOfFilteredTracks;
162    }
163
164    /**
165     * @return number of frames analyzed
166     */
167    public int getNumberOfFrames() {
168        return numberOfFrames;
169    }
170
171    /**
172     * @return duration of the analysis, or {@code null} if unavailable
173     */
174    public Duration getDuration() {
175        return duration;
176    }
177
178    /**
179     * @return number of spots across all tracks
180     */
181    public int getNumberOfSpotsInAllTracks() {
182        return numberOfSpotsInAllTracks;
183    }
184
185    // -------------------------------------------------------------------------
186    // Representation
187    // -------------------------------------------------------------------------
188
189    /**
190     * Returns a formatted summary string containing all key metrics.
191     *
192     * @return a string summarizing TrackMate run results
193     */
194    @Override
195    public String toString() {
196        return String.format(
197                "Success: %b, Spots: %d, Tracks: %d, Filtered: %d, Frames: %d, Duration(ms): %d",
198                success,
199                numberOfSpots,
200                numberOfTracks,
201                numberOfFilteredTracks,
202                numberOfFrames,
203                duration != null ? duration.toMillis() : 0
204        );
205    }
206}
