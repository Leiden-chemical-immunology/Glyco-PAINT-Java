077/**
078 * Provides functionality to execute the TrackMate analysis workflow for a single
079 * experiment. Each experiment typically contains one or more recordings that
080 * are processed sequentially.
081 * <p>
082 * The workflow includes:
083 * <ul>
084 *   <li>Initializing configuration and runtime settings.</li>
085 *   <li>Running TrackMate per recording in a monitored thread.</li>
086 *   <li>Recording runtime statistics and aggregating results.</li>
087 *   <li>Supporting user cancellation and timeouts via watchdog control.</li>
088 * </ul>
089 */
090public class RunTrackMateOnExperiment extends RunTrackMateOnRecording {
091
092    /**
093     * Verbosity flag derived from {@link PaintRuntime}.
094     */
095    static final boolean verbose = PaintRuntime.isVerbose();
096
097    /**
098     * Executes a given task within a monitored thread using a watchdog.
099     * The watchdog ensures that the task completes within a specified time limit
100     * or terminates early if the user cancels processing.
101     *
102     * @param task                   the {@link Runnable} task to execute
103     * @param maxSecondsPerRecording time limit for execution in seconds
104     * @param dialog                 optional {@link ProjectDialog} that can signal cancellation
105     * @return {@code true} if the task completed successfully;
106     * {@code false} if cancelled or timed out
107     */
108    private static boolean runWithWatchdog(Runnable task,
109                                           int maxSecondsPerRecording,
110                                           ProjectDialog dialog) {
111
112        Thread thread = new Thread(task, "TrackMateThread");
113        thread.start();
114
115        int numberOfInterrupts = 0;
116        int numberOfDotsOnline = 0;
117
118        for (int i = 0; i < maxSecondsPerRecording; i++) {
119            try {
120                thread.join(1000); // check every second
121            } catch (InterruptedException e) {
122                Thread.currentThread().interrupt();
123                PaintLogger.errorf("Watchdog thread was interrupted.");
124                return false;
125            }
126
127            // ✅ Finished normally
128            if (!thread.isAlive()) {
129                return true;
130            }
131
132            // ✅ User requested cancellation
133            if (dialog != null && dialog.isCancelled()) {
134                PaintLogger.warnf("User requested cancellation — stopping TrackMate gracefully...");
135                return false;
136            }
137
138            // Print progress dots
139            numberOfInterrupts++;
140            if (numberOfInterrupts >= 1) {
141                PaintLogger.raw(".");
142                numberOfDotsOnline++;
143                numberOfInterrupts = 0;
144            }
145            if (numberOfDotsOnline >= 80) {
146                PaintLogger.raw("\n                                                    ");
147                numberOfDotsOnline = 0;
148            }
149        }
150
151        // ⏱ Timeout reached
152        PaintLogger.errorf("   TrackMate - exceeded time limit of %d seconds.", maxSecondsPerRecording);
153        return false;
154    }
155
156    /**
157     * Runs the TrackMate workflow for a single experiment.
158     * Reads experiment configuration, processes each recording in sequence,
159     * and aggregates results into consolidated output files.
160     *
161     * @param experimentPath the directory of the experiment (contains CSV files)
162     * @param imagesPath     the directory containing corresponding image data
163     * @param dialog         optional {@link ProjectDialog} for user cancellation
164     * @return {@code true} if all recordings processed successfully;
165     *         {@code false} if errors or cancellations occurred
166     */
167    public static boolean runTrackMateOnExperiment(Path experimentPath,
168                                                   Path imagesPath,
169                                                   ProjectDialog dialog) {
170
171        // ---------------------------------------------------------------------
172        // Initial setup
173        // ---------------------------------------------------------------------
174        Duration totalDuration           = Duration.ZERO;
175        int numberRecordings             = 0;
176        boolean status                   = true;
177        List<Path> processedTrackFiles   = new ArrayList<>();
178
179        // Initialize configuration
180        Path configPath = experimentPath.getParent().resolve(PAINT_CONFIGURATION_JSON);
181        if (Files.exists(configPath)) {
182            PaintConfig.initialise(experimentPath.getParent());
183        } else {
184            PaintLogger.warnf("No PaintConfig.json found in %s, defaults will be used.",
185                              experimentPath.getParent());
186        }
187
188        TrackMateConfig trackMateConfig = new TrackMateConfig();
189        int maxSecondsPerRecording      = trackMateConfig.getMaxNumberOfSecondsPerImage();
190
191        PaintLogger.debugf(trackMateConfig.toString());
192        if (verbose) {
193            PaintLogger.infof(trackMateConfig.toString());
194        }
195
196        // Save configuration snapshot
197        try {
198            Path filePath = experimentPath.resolve("Output").resolve("ParametersUsed.txt");
199            Files.createDirectories(filePath.getParent());
200            trackMateConfigToFile(trackMateConfig, filePath);
201        } catch (Exception ex) {
202            PaintLogger.errorf("Could not write file '%s'", "ParametersUsed.txt");
203        }
204
205        // ---------------------------------------------------------------------
206        // Validate input files
207        // ---------------------------------------------------------------------
208        Path experimentFilePath   = experimentPath.resolve(EXPERIMENT_INFO_CSV);
209        Path allRecordingFilePath = experimentPath.resolve(RECORDINGS_CSV);
210
211        if (!Files.exists(experimentFilePath)) {
212            PaintLogger.errorf("Experiment info file does not exist: %s", experimentFilePath);
213            return false;
214        }
215
216        int numberRecordingsToProcess = countProcessed(experimentFilePath);
217        String experimentName         = experimentPath.getFileName().toString();
218        String projectName            = experimentPath.getParent().getFileName().toString();
219
220        PaintLogger.blankline();
221        PaintLogger.infof("Processing %d %s in experiment '%s' (project '%s').",
222                          numberRecordingsToProcess,
223                          numberRecordingsToProcess == 1 ? "recording" : "recordings",
224                          experimentName,
225                          projectName);
226        PaintLogger.blankline();
227
228        // ---------------------------------------------------------------------
229        // Process Experiment Info and Recordings
230        // ---------------------------------------------------------------------
231        try (Reader experimentInfoReader = Files.newBufferedReader(experimentFilePath);
232             CSVParser experimentInfoParser = new CSVParser(experimentInfoReader,
233                                                            CSVFormat.DEFAULT.builder()
234                                                                    .setHeader()
235                                                                    .setSkipHeaderRecord(true)
236                                                                    .build());
237             BufferedWriter allRecordingsWriter = Files.newBufferedWriter(allRecordingFilePath);
238             CSVPrinter allRecordingsPrinter = new CSVPrinter(allRecordingsWriter, CSVFormat.DEFAULT.builder().build())) {
239
240            // Extend the Experiment Info header for RECORDINGS_CSV
241            List<String> header = new ArrayList<>(experimentInfoParser.getHeaderMap().keySet());
242            header.addAll(Arrays.asList(
243                    "Number of Spots",
244                    "Number of Tracks",
245                    "Number of Tracks in Background",
246                    "Number of Squares in Background",
247                    "Average Tracks in Background",
248                    "Number of Spots in All Tracks",
249                    "Number of Frames",
250                    "Run Time",
251                    "Time Stamp",
252                    "Exclude",
253                    "Tau",
254                    "R Squared",
255                    "Density"
256            ));
257            allRecordingsPrinter.printRecord(header);
258
259            // Iterate through experiment records
260            for (CSVRecord experientInfoRecord : experimentInfoParser) {
261
262                if (dialog != null && dialog.isCancelled()) {
263                    PaintLogger.warnf("User requested cancellation. Stopping.");
264                    break;
265                }
266
267                try {
268                    Map<String, String> row = new LinkedHashMap<>();
269                    for (String key : experimentInfoParser.getHeaderMap().keySet()) {
270                        row.put(key, experientInfoRecord.get(key));
271                    }
272
273                    ExperimentInfo experimentInfo           = new ExperimentInfo(row);
274                    String         recordingName            = experimentInfo.getRecordingName();
275                    int            numberOfSpots            = 0;
276                    int            numberOfFilteredTracks   = 0;
277                    int            numberOfSpotsInAllTracks = 0;
278                    int            numberOfFrames           = 0;
279                    int            runTime                  = 0;
280                    String         timeStamp                = "";
281
282                    if (experimentInfo.isProcessFlag()) {
283                        double threshold = experimentInfo.getThreshold();
284
285                        Files.createDirectories(experimentPath.resolve(DIR_BRIGHTFIELD_IMAGES));
286                        Files.createDirectories(experimentPath.resolve(DIR_TRACKMATE_IMAGES));
287
288                        PaintLogger.infof("   Recording '%s' started TrackMate processing.", recordingName);
289
290                        // The following is necessary because of how Java handles variable capture inside lambdas or inner classes.
291                        // We are not changing the variable trackMateResults itself (the reference to the array never changes),
292                        // We pass the address of the array that does not change, but the contents can change.
293                        final TrackMateResults[] trackMateResults = new TrackMateResults[1];
294
295                        // Run TrackMate in a monitored thread
296                        boolean finished = runWithWatchdog(() -> {
297                            try {
298                                trackMateResults[0] = RunTrackMateOnRecording.runTrackMateOnRecording(
299                                        experimentPath, imagesPath, trackMateConfig, threshold, experimentInfo, dialog);
300                            } catch (Exception e) {
301                            }
302                        }, maxSecondsPerRecording, dialog);
303
304                        // Handle failures and cancellations
305                        if (!finished) {
306                            if (dialog != null && dialog.isCancelled()) {
307                                PaintLogger.infof("Recording '%s' cancelled cleanly.", recordingName);
308                                break;
309                            } else {
310                                PaintLogger.errorf("   TrackMate failed or timed out for '%s'.", recordingName);
311                                PaintLogger.blankline();
312                                status = false;
313                                continue;
314                            }
315                        }
316
317                        // Validate processing results
318                        if (trackMateResults[0] == null || !trackMateResults[0].isSuccess()) {
319                            PaintLogger.errorf("   TrackMate failed for '%s'.", recordingName);
320                            status = false;
321                            continue;
322                        }
323                        if (!trackMateResults[0].isCalculationPerformed()) {
324                            numberRecordings++;
325                            continue;
326                        }
327
328                        int durationInSeconds = (int) (trackMateResults[0].getDuration().toMillis() / 1000);
329                        PaintLogger.infof("   Recording '%s' processed in %s.", recordingName, formatDuration(durationInSeconds));
330                        PaintLogger.blankline();
331
332                        Path trackFilePath = experimentPath.resolve(recordingName + "-tracks.csv");
333                        processedTrackFiles.add(trackFilePath);
334                        totalDuration = totalDuration.plus(trackMateResults[0].getDuration());
335                        numberRecordings++;
336
337                        // Extract output values
338                        numberOfSpots            = trackMateResults[0].getNumberOfSpots();
339                        numberOfFilteredTracks   = trackMateResults[0].getNumberOfFilteredTracks();
340                        numberOfFrames           = trackMateResults[0].getNumberOfFrames();
341                        numberOfSpotsInAllTracks = trackMateResults[0].getNumberOfSpotsInAllTracks();
342                        runTime                  = durationInSeconds;
343                        timeStamp                = LocalDateTime.now().format(DateTimeFormatter.ISO_LOCAL_DATE_TIME);
344
345                    } else if (verbose) {
346                        PaintLogger.infof("   Recording '%s' skipped.", recordingName);
347                    }
348
349                    // Write summary output for this recording
350                    List<String> output = new ArrayList<>();
351                    for (String key : experimentInfoParser.getHeaderMap().keySet()) {
352                        output.add(row.get(key));
353                    }
354                    output.addAll(Arrays.asList(
355                            String.valueOf(numberOfSpots),
356                            String.valueOf(numberOfFilteredTracks),
357                            "",
358                            "",
359                            "",
360                            String.valueOf(numberOfSpotsInAllTracks),
361                            String.valueOf(numberOfFrames),
362                            String.valueOf(runTime),
363                            timeStamp,
364                            "False",
365                            "",
366                            "",
367                            ""
368                    ));
369                    allRecordingsPrinter.printRecord(output);
370
371                } catch (Exception e) {
372                    if (dialog == null || !dialog.isCancelled()) {
373                        PaintLogger.errorf("Error processing recording: %s", e.getMessage());
374                    }
375                }
376            }
377
378        } catch (IOException e) {
379            if (dialog == null || !dialog.isCancelled()) {
380                PaintLogger.errorf("Error reading Experiment Info: %s", e.getMessage());
381            }
382            status = false;
383        }
384
385        // ---------------------------------------------------------------------
386        // Finalize outputs and summaries
387        // ---------------------------------------------------------------------
388        if (dialog != null && dialog.isCancelled()) {
389            PaintLogger.warnf("Cancellation processed.");
390            PaintLogger.blankline();
391            return false;
392        }
393
394        // Merge all per-recording tracks into a single CSV
395        Path tracksFilePath = experimentPath.resolve(TRACKS_CSV);
396        try {
397            concatenateCsvFiles(processedTrackFiles, tracksFilePath, true);
398        } catch (IOException e) {
399            PaintLogger.errorf("Error concatenating tracks: %s", e.getMessage());
400            status = false;
401        }
402
403        // Log final summary
404        PaintLogger.infof("Processed %d recordings in %s.",
405                          numberRecordings, formatDuration((int) (totalDuration.toMillis() / 1000)));
406        PaintLogger.blankline();
407        return status;
408    }
