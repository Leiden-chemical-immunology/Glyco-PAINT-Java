042package paint.fiji.trackmate;
043
044import org.scijava.command.Command;
045import org.scijava.plugin.Plugin;
046import paint.generatesquares.GenerateSquaresHeadless;
047import paint.shared.config.PaintConfig;
048import paint.shared.dialogs.ProjectDialog;
049import paint.shared.utils.PaintPrefs;
050import paint.shared.utils.JarInfoLogger;
051import paint.shared.utils.PaintConsoleWindow;
052import paint.shared.utils.PaintLogger;
053import paint.shared.utils.PaintRuntime;
054
055import javax.swing.*;
056import java.nio.file.Files;
057import java.nio.file.Path;
058import java.time.LocalDateTime;
059import java.time.format.DateTimeFormatter;
060
061import static paint.shared.constants.PaintConstants.PAINT_SWEEP_CONFIGURATION_JSON;
062import static paint.shared.utils.ValidProjectPath.getValidProjectPath;

064/**
065 * Main user interface class for running TrackMate interactively within the
066 * PAINT environment. Handles initialization, configuration, dialog management,
067 * and execution of TrackMate processes.
068 */
069@Plugin(type = Command.class, menuPath = "Plugins>Glyco-PAINT>Run")
070public class TrackMateUI extends RunTrackMateOnProjectSweep implements Command {
071
072    /**
073     * Prevents concurrent execution of multiple TrackMate runs.
074     */
075    private static volatile boolean running = false;
076
077    /**
078     * Executes the TrackMate workflow through an interactive GUI dialog.
079     * <p>
080     * The method:
081     * <ul>
082     *   <li>Ensures single-instance execution.</li>
083     *   <li>Initializes logging and configuration state.</li>
084     *   <li>Displays a project dialog for experiment selection.</li>
085     *   <li>Runs the appropriate TrackMate pipeline (sweep or standard).</li>
086     *   <li>Optionally triggers "Generate Squares" after completion.</li>
087     * </ul>
088     */
089    @Override
090    public void run() {
091
092        // ---------------------------------------------------------------------
093        // Step 1 – Setup exception handler and concurrency lock
094        // ---------------------------------------------------------------------
095        Thread.setDefaultUncaughtExceptionHandler((thread, throwable) ->
096                                                          PaintLogger.debugf("AWT complained: %s", throwable.getMessage())
097        );
098
099        if (running) {
100            showWarning("TrackMate processing is already running.\nPlease wait until it finishes.");
101            return;
102        }
103
104        // ---------------------------------------------------------------------
105        // Step 2 – Retrieve project root
106        // ---------------------------------------------------------------------
107        Path projectPath = getValidProjectPath();
108        if (projectPath == null) {
109            return;
110        }
111
112        // ---------------------------------------------------------------------
113        // Step 3 – Initialize logging, configuration, and runtime settings
114        // ---------------------------------------------------------------------
115        PaintConsoleWindow.createConsoleFor("TrackMate");
116        PaintConfig.initialise(projectPath);
117
118        String debugLevel = PaintPrefs.getString("Runtime", "Log Level", "INFO");
119        PaintLogger.setLevel(debugLevel);
120        PaintLogger.initialise(projectPath, "TrackMateOnProject.log");
121        PaintLogger.debugf("TrackMate plugin started (Interactive).");
122
123        PaintRuntime.initialiseFromPrefs();
124
125        // Log version and timestamp
126        JarInfoLogger.JarInfo info = JarInfoLogger.getJarInfo(TrackMateUI.class);
127        if (info != null) {
128            PaintLogger.infof("Compilation date: %s", info.implementationDate);
129            PaintLogger.infof("Version: %s", info.implementationVersion);
130        }
131
132        LocalDateTime now = LocalDateTime.now();
133        String formattedTime = now.format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss"));
134        PaintLogger.infof("Current time: %s", formattedTime);
135        PaintLogger.blankline();
136
137        // ---------------------------------------------------------------------
138        // Step 4 – Show experiment dialog
139        // ---------------------------------------------------------------------
140        ProjectDialog dialog = new ProjectDialog(null, projectPath, ProjectDialog.DialogMode.TRACKMATE);
141        PaintConsoleWindow.closeOnDialogDispose(dialog.getDialog());
142
143        // ---------------------------------------------------------------------
144        // Step 5 – Handle OK action callback
145        // ---------------------------------------------------------------------
146        dialog.setCalculationCallback(project -> {
147
148            if (running) {
149                showWarning("TrackMate processing is already running.\nPlease wait until it finishes.");
150                return false;
151            }
152
153            running = true;
154            dialog.setOkEnabled(false);
155
156            try {
157                boolean debug = PaintConfig.getBoolean("Debug", "Debug RunTrackMateOnProject", false);
158                Path imagesPath = project.getImagesRootPath();
159                Path currentProjectRoot = project.getProjectRootPath();
160
161                if (debug) {
162                    PaintLogger.debugf("TrackMate processing started.");
163                    PaintLogger.debugf("Experiments: %s", project.getExperimentNames());
164                }
165
166                boolean success;
167                boolean sweepEnabled = dialog.isSweepSelected();
168                Path sweepFile = currentProjectRoot.resolve(PAINT_SWEEP_CONFIGURATION_JSON);
169
170                if (sweepEnabled) {
171                    if (Files.exists(sweepFile)) {
172                        success = RunTrackMateOnProjectSweep.runWithSweep(
173                                currentProjectRoot, imagesPath, project.getExperimentNames());
174                    } else {
175                        PaintLogger.infof("No Sweep configuration detected at %s", sweepFile);
176                        return false;
177                    }
178                } else {
179                    success = RunTrackMateOnProject.runProject(
180                            projectPath, imagesPath, project.getExperimentNames(), dialog, null);
181
182                    if (success && PaintConfig.getBoolean("TrackMate", "Run Generate Squares After", true)) {
183                        PaintLogger.infof("TrackMate finished successfully. Starting Generate Squares...");
184                        PaintLogger.blankline();
185                        GenerateSquaresHeadless.run(currentProjectRoot, project.getExperimentNames());
186                        PaintLogger.infof("Generate Squares completed successfully.");
187                    }
188                }
189
190                return success;
191
192            } catch (Exception e) {
193                PaintLogger.errorf("Error during TrackMate execution: %s", e.getMessage());
194                return false;
195            } finally {
196                running = false;
197            }
198        });
199
200        dialog.showDialog();
201    }
202
203    // -------------------------------------------------------------------------
204    // Utility Methods
205    // -------------------------------------------------------------------------
206
207    /**
208     * Displays a warning dialog with the specified message.
209     *
210     * @param message warning message text to display
211     */
212    private void showWarning(String message) {
213        JOptionPane optionPane = new JOptionPane(message, JOptionPane.WARNING_MESSAGE);
214        JDialog warnDialog = optionPane.createDialog(null, "Warning");
215        warnDialog.setAlwaysOnTop(true);
216        warnDialog.setVisible(true);
217    }
