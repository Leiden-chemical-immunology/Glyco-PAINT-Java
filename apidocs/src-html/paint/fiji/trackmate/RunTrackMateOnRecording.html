018 *    • Manage end-to-end TrackMate processing for a single recording.

021 *    • Collect and summarize analysis results in {@link TrackMateResults}.
022 *
023 *  USAGE EXAMPLE:
024 *    TrackMateResults result = RunTrackMateOnRecording.runTrackMateOnRecording(
025 *        experimentPath,

081/**
082 * Provides functionality to run TrackMate analysis for a single microscopy
083 * recording. Handles setup, detection, tracking, visualization, and export.
084 * <p>
085 * This class is typically invoked by {@link RunTrackMateOnExperiment} as part
086 * of batch processing within a Paint project.
087 * </p>
088 */
089public class RunTrackMateOnRecording extends TrackMateHeadless {
090
091    /**
092     * Executes the TrackMate pipeline on a given recording.
093     * Loads the ND2 image, applies analysis parameters, and writes results.
094     *
095     * @param experimentPath       directory where results will be written
096     * @param imagesPath           directory containing the ND2 image
097     * @param trackMateConfig      configuration object with TrackMate parameters
098     * @param threshold            threshold for spot detection
099     * @param experimentInfoRecord metadata describing this recording
100     * @param dialog               optional dialog for user cancellation
101     * @return {@link TrackMateResults} containing analysis statistics,
102     *         or a cancellation result if aborted
103     */
104    public static TrackMateResults runTrackMateOnRecording(Path experimentPath,
105                                                           Path imagesPath,
106                                                           TrackMateConfig trackMateConfig,
107                                                           double threshold,
108                                                           ExperimentInfo experimentInfoRecord,
109                                                           ProjectDialog dialog) {
110
111        final boolean debugFlag = getBoolean("Debug", "Debug RunTrackMateOnRecording", false);
112        LocalDateTime start = LocalDateTime.now();
113        DebugTools.setRootLevel("OFF");
114
115        ImagePlus imp            = null;
116        ImagePlus impBrightfield = null;
117        ImagePlus capture        = null;
118
119        try {
120            // -----------------------------------------------------------------
121            // Step 1 – Load ND2 image
122            // -----------------------------------------------------------------
123            PaintLogger.raw("                       TrackMate - Image Loading:   ");
124            if (isCancelled(Thread.currentThread(), dialog)) {
125                return cancelEarly(imp, impBrightfield);
126            }
127
128            File nd2File = new File(imagesPath.toFile(), experimentInfoRecord.getRecordingName() + ".nd2");
129            if (!nd2File.exists()) {
130                if (!isCancelled(Thread.currentThread(), dialog)) {
131                    PaintLogger.errorf("Could not open image file: %s", nd2File.getAbsolutePath());
132                }
133                return cancelEarly(imp, impBrightfield);
134            }
135
136            try {
137                imp = IJ.openImage(nd2File.getAbsolutePath());
138            } catch (Exception e) {
139                if (!isCancelled(Thread.currentThread(), dialog)) {
140                    PaintLogger.errorf("Could not load image file: %s", nd2File.getAbsolutePath());
141                }
142                return cancelEarly(imp, impBrightfield);
143            }
144
145            if (isCancelled(Thread.currentThread(), dialog)) {
146                return cancelEarly(imp, impBrightfield);
147            }
148
149            if (imp == null) {
150                if (!isCancelled(Thread.currentThread(), dialog)) {
151                    PaintLogger.errorf("Unsupported format or file not found: %s", nd2File);
152                }
153                return cancelEarly(imp, impBrightfield);
154            }
155
156            imp.show();
157            IJ.wait(100);
158
159            IJ.run(imp, "Enhance Contrast", "saturated=0.35");
160            IJ.run("Grays");
161
162            // -----------------------------------------------------------------
163            // Step 2 – Save Brightfield snapshot
164            // -----------------------------------------------------------------
165            Path jpgPath = experimentPath.resolve("Brightfield Images")
166                    .resolve(experimentInfoRecord.getRecordingName() + ".jpg");
167            if (Files.notExists(jpgPath.getParent())) {
168                Files.createDirectories(jpgPath.getParent());
169            }
170            if (isCancelled(Thread.currentThread(), dialog)) {
171                return cancelEarly(imp, impBrightfield);
172            }
173
174            Path brightFieldPath = null;
175            String baseName = experimentInfoRecord.getRecordingName();
176
177            List<String> candidates = Arrays.asList(
178                    baseName + "-BF.nd2",
179                    baseName + "-BF1.nd2",
180                    baseName + "-BF2.nd2");
181            try {
182                brightFieldPath = candidates.stream()
183                        .map(imagesPath::resolve)
184                        .filter(Files::exists)
185                        .findFirst()
186                        .orElse(null);
187            } catch (Exception e) {
188                PaintLogger.errorf("Could not locate brightfield file: %s", candidates);
189            }
190
191            if (brightFieldPath == null || Files.notExists(brightFieldPath)) {
192                PaintLogger.warnf("      Could not open brightfield file: %s",
193                                  brightFieldPath == null ? "none found" : brightFieldPath.toString());
194            } else {
195                try {
196                    impBrightfield = IJ.openImage(brightFieldPath.toString());
197                    IJ.run(impBrightfield, "Enhance Contrast", "saturated=0.35");
198                    IJ.saveAs(impBrightfield, "Jpeg", jpgPath.toString());
199                } catch (Exception e) {
200                    PaintLogger.errorf("Error handling brightfield file: %s", e.getMessage());
201                }
202            }
203
204            // -----------------------------------------------------------------
205            // Step 3 – Configure TrackMate
206            // -----------------------------------------------------------------
207            Model model = new Model();
208            model.setLogger(Logger.VOID_LOGGER);
209
210            Settings settings = new Settings(imp);
211            // Detector configuration
212            settings.detectorFactory   = new LogDetectorFactory();
213            settings.detectorSettings  = settings.detectorFactory.getDefaultSettings();
214            settings.detectorSettings.put("TARGET_CHANNEL",           trackMateConfig.getTargetChannel());
215            settings.detectorSettings.put("RADIUS",                   trackMateConfig.getRadius());
216            settings.detectorSettings.put("DO_SUBPIXEL_LOCALIZATION", trackMateConfig.isDoSubpixelLocalization());
217            settings.detectorSettings.put("THRESHOLD",                threshold);
218            settings.detectorSettings.put("DO_MEDIAN_FILTERING",      trackMateConfig.isMedianFiltering());
219
220            // Tracker configuration
221            settings.trackerFactory    = new SparseLAPTrackerFactory();
222            settings.trackerSettings   = settings.trackerFactory.getDefaultSettings();
223            settings.trackerSettings.put("LINKING_MAX_DISTANCE",            trackMateConfig.getLinkingMaxDistance());
224            settings.trackerSettings.put("ALTERNATIVE_LINKING_COST_FACTOR", trackMateConfig.getAlternativeLinkingCostFactor());
225            settings.trackerSettings.put("ALLOW_GAP_CLOSING",               trackMateConfig.isAllowGapClosing());
226            settings.trackerSettings.put("GAP_CLOSING_MAX_DISTANCE",        trackMateConfig.getGapClosingMaxDistance());
227            settings.trackerSettings.put("MAX_FRAME_GAP",                   trackMateConfig.getMaxFrameGap());
228            settings.trackerSettings.put("ALLOW_TRACK_SPLITTING",           trackMateConfig.isAllowTrackSplitting());
229            settings.trackerSettings.put("SPLITTING_MAX_DISTANCE",          trackMateConfig.getSplittingMaxDistance());
230            settings.trackerSettings.put("ALLOW_TRACK_MERGING",             trackMateConfig.isAllowTrackMerging());
231            settings.trackerSettings.put("MERGING_MAX_DISTANCE",            trackMateConfig.getMergingMaxDistance());
232
233            // Deterministic execution
234            Locale.setDefault(Locale.US);
235            System.setProperty("user.language", "en");
236            System.setProperty("user.country", "US");
237            System.setProperty("java.util.concurrent.ForkJoinPool.common.parallelism", "1");  // Disable JVM-level parallelism (limits common pool threads to 1)
238            System.setProperty("trackmate.deterministic", "true");                            // Request deterministic behavior from TrackMate if supportedSo where I
239
240            settings.addSpotFilter(new FeatureFilter("QUALITY", 0, true));
241            settings.addAllAnalyzers();
242            settings.addTrackFilter(new FeatureFilter("NUMBER_SPOTS", trackMateConfig.getMinNumberOfSpotsInTrack(), true));
243
244            // -----------------------------------------------------------------
245            // Step 4 – Execute TrackMate
246            // -----------------------------------------------------------------
247
248            if (isCancelled(Thread.currentThread(), dialog)) {
249                return cancelEarly(imp, impBrightfield);
250            }
251
252            TrackMate trackmate = new TrackMate(model, settings);
253            if (!trackmate.checkInput()) {
254                PaintLogger.errorf("TrackMate input check failed: %s", trackmate.getErrorMessage());
255                return cancelEarly(imp, impBrightfield);
256            }
257
258            PaintLogger.raw("\n                       TrackMate - spot detection:  ");
259            try {
260                if (!trackmate.execDetection()) {
261                    PaintLogger.errorf("TrackMate - execDetection failed: %s", trackmate.getErrorMessage());
262                    return cancelEarly(imp, impBrightfield);
263                }
264            } catch (Exception e) {
265                PaintLogger.errorf("Unexpected error during detection: %s", e.getMessage());
266                return cancelEarly(imp, impBrightfield);
267            }
268
269            int numberOfSpots = model.getSpots().getNSpots(false);
270            if (numberOfSpots > trackMateConfig.getMaxNumberOfSpotsInImage()) {
271                PaintLogger.warnf("   TrackMate - Too many spots detected (%d). Limit is %d.",
272                                  numberOfSpots, trackMateConfig.getMaxNumberOfSpotsInImage());
273                return cancelEarly(imp, impBrightfield);
274            } else {
275                String numberOfSpotsString = " (" + numberOfSpots + " spots detected).";
276                PaintLogger.raw(numberOfSpotsString);
277            }
278
279            if (isCancelled(Thread.currentThread(), dialog)) {
280                return cancelEarly(imp, impBrightfield);
281            }
282
283            PaintLogger.raw("\n                       TrackMate - track detection: ");
284            try {
285                if (!trackmate.process()) {
286                    PaintLogger.errorf("TrackMate process failed: %s", trackmate.getErrorMessage());
287                    return cancelEarly(imp, impBrightfield);
288                }
289            } catch (Exception e) {
290                PaintLogger.errorf("Unexpected error during TrackMate process: %s", e.getMessage());
291                return cancelEarly(imp, impBrightfield);
292            }
293
294            // -----------------------------------------------------------------
295            // Step 5 – Visualization
296            // -----------------------------------------------------------------
297            if (isCancelled(Thread.currentThread(), dialog)) {
298                return cancelEarly(imp, impBrightfield);
299            }
300
301            final SelectionModel  selectionModel = new SelectionModel(model);
302            final DisplaySettings ds             = DisplaySettingsIO.readUserDefault();
303            ds.setSpotVisible(false);
304            ds.setTrackColorBy(DisplaySettings.TrackMateObject.TRACKS,
305                               trackMateConfig.getTrackColoring());
306            final HyperStackDisplayer displayer = new HyperStackDisplayer(model, selectionModel, imp, ds);
307            displayer.render();
308            displayer.refresh();
309
310            capture = CaptureOverlayAction.capture(imp, -1, 1, null);
311            Path imagePath = experimentPath.resolve("TrackMate Images")
312                    .resolve(experimentInfoRecord.getRecordingName() + ".jpg");
313            if (capture != null) {
314                if (!new FileSaver(capture).saveAsTiff(String.valueOf(imagePath))) {
315                    PaintLogger.errorf("Failed to save TIFF to: %s", imagePath);
316                }
317            }
318            IJ.wait(2000);
319
320            // -----------------------------------------------------------------
321            // Step 6 – Write tracks CSV
322            // -----------------------------------------------------------------
323            String tracksName = experimentInfoRecord.getRecordingName() + "-tracks.csv";
324            Path tracksPath = experimentPath.resolve(tracksName);
325            int totalSpotsInAllTracks = 0;
326
327            try {
328                totalSpotsInAllTracks = TrackCsvWriter.writeTracksCsv(
329                        trackmate,
330                        experimentInfoRecord.getExperimentName(),
331                        experimentInfoRecord.getRecordingName(),
332                        tracksPath.toFile(),
333                        true);
334            } catch (Exception e) {
335                PaintLogger.errorf("Failed to write tracks to '%s'", tracksPath);
336            }
337
338            // -----------------------------------------------------------------
339            // Step 7 – Summarize and return results
340            // -----------------------------------------------------------------
341            int numberOfSpotsTotal     = model.getSpots().getNSpots(true);
342            int numberOfTracks         = model.getTrackModel().nTracks(false);
343            int numberOfFilteredTracks = model.getTrackModel().nTracks(true);
344            int numberOfFrames         = imp.getNFrames();
345
346            Duration duration = Duration.between(start, LocalDateTime.now());
347
348            closeImages(imp, impBrightfield, capture);
349
350            return new TrackMateResults(true,
351                                        true,
352                                        numberOfSpotsTotal,
353                                        numberOfTracks,
354                                        numberOfFilteredTracks,
355                                        numberOfFrames,
356                                        duration,
357                                        totalSpotsInAllTracks);
358
359        } catch (Exception e) {
360            PaintLogger.errorf("Exception during TrackMate processing: %s", e.getMessage());
361            return cancelEarly(imp, impBrightfield);
362        } finally {
363            closeImages(imp, impBrightfield, capture);
364        }
365    }

370
371    /**
372     * Checks for thread interruption or user cancellation.
373     */
374    private static boolean isCancelled(Thread t, ProjectDialog dialog) {
375        return t.isInterrupted() || (dialog != null && dialog.isCancelled());
376    }
377
378    /**
379     * Safely closes visible images.
380     */
381    private static void closeImages(ImagePlus... images) {
382        for (ImagePlus img : images) {
383            try {
384                if (img != null && img.isVisible()) {
385                    img.close();
386                }
387            } catch (Exception e) {
388                PaintLogger.warnf("Error closing image: %s", e.getMessage());
389            }
390        }
391    }
392
393    /**
394     * Cancels processing early and closes images.
395     */
396    private static TrackMateResults cancelEarly(ImagePlus... images) {
397        closeImages(images);
398        PaintLogger.warnf("   Recording cancelled.");
399        return new TrackMateResults(false);
400    }
