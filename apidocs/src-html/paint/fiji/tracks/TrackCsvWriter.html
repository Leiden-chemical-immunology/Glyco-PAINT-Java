066/**
067 * Provides functionality to export TrackMate tracks to a CSV file.
068 * <p>
069 * Combines native TrackMate features with custom PAINT-calculated attributes,
070 * producing a structured dataset suitable for downstream statistical analysis.
071 * </p>
072 */
073public final class TrackCsvWriter {
074
075    private TrackCsvWriter() {
076        // Utility class; prevent instantiation
077    }
078
079    /**
080     * Writes track information from a TrackMate analysis into a CSV file.
081     * <p>
082     * Includes both TrackMate-calculated and PAINT-derived metrics, such as
083     * total distance, diffusion coefficients, and confinement ratio.
084     * </p>
085     *
086     * @param trackmate      TrackMate instance containing the model and features
087     * @param experimentName experiment name for metadata tagging
088     * @param recordingName  recording name associated with the tracks
089     * @param csvFile        destination CSV file
090     * @param visibleOnly    if true, exports only visible tracks
091     * @return total number of spots across all exported tracks
092     */
093    public static int writeTracksCsv(final TrackMate trackmate,
094                                     final String experimentName,
095                                     final String recordingName,
096                                     final File csvFile,
097                                     final boolean visibleOnly) {
098
099        // ---------------------------------------------------------------------
100        // Step 1 – Extract TrackMate components
101        // ---------------------------------------------------------------------
102        final Model model               = trackmate.getModel();
103        final TrackModel trackModel     = model.getTrackModel();
104        final FeatureModel featureModel = model.getFeatureModel();
105
106        // Collect and sort track IDs for deterministic output order
107        final Set<Integer> trackIDsSet = trackModel.trackIDs(visibleOnly);
108        final List<Integer> trackIDs = new ArrayList<>(trackIDsSet);
109        Collections.sort(trackIDs);
110
111        final List<Track> tracks = new ArrayList<>();
112        int totalSpots = 0;
113
114        // ---------------------------------------------------------------------
115        // Step 2 – Process each track
116        // ---------------------------------------------------------------------
117        for (Integer trackId : trackIDs) {
118            TrackAttributes trackAttributes = calculateTrackAttributes(trackModel, trackId, TIME_INTERVAL);
119
120            Track track = new Track();
121            track.setExperimentName(experimentName);
122            track.setRecordingName(recordingName);
123
124            // Native TrackMate features
125            track.setNumberOfSpots(     asInt(    featureModel.getTrackFeature(trackId, "NUMBER_SPOTS")));
126            track.setNumberOfGaps(      asInt(   featureModel.getTrackFeature(trackId, "NUMBER_GAPS")));
127            track.setLongestGap(        asInt(   featureModel.getTrackFeature(trackId, "LONGEST_GAP")));
128            track.setTrackDuration(     roundOr( featureModel.getTrackFeature(trackId, "TRACK_DURATION"),     3, -1));
129            track.setTrackXLocation(    roundOr( featureModel.getTrackFeature(trackId, "TRACK_X_LOCATION"),   2, -1));
130            track.setTrackYLocation(    roundOr( featureModel.getTrackFeature(trackId, "TRACK_Y_LOCATION"),   2, -1));
131            track.setTrackDisplacement( roundOr( featureModel.getTrackFeature(trackId, "TRACK_DISPLACEMENT"), 2, -1));
132            track.setTrackMaxSpeed(     roundOr( featureModel.getTrackFeature(trackId, "TRACK_MAX_SPEED"),    2, -1));
133            track.setTrackMedianSpeed(  roundOr( featureModel.getTrackFeature(trackId, "TRACK_MEDIAN_SPEED"), 2, -1));
134
135            // Custom PAINT-calculated attributes
136            track.setDiffusionCoefficient(    roundOr(trackAttributes.diffusionCoeff,    2, -1));
137            track.setDiffusionCoefficientExt( roundOr(trackAttributes.diffusionCoeffExt, 2, -1));
138            track.setTotalDistance(           roundOr(trackAttributes.totalDistance,     2, -1));
139            track.setConfinementRatio(        roundOr(trackAttributes.confinementRatio,  2, -1));
140            track.setSquareNumber(-1);
141            track.setLabelNumber(-1);
142
143            totalSpots += trackAttributes.numberOfSpotsInTracks;
144            tracks.add(track);
145        }
146
147        // ---------------------------------------------------------------------
148        // Step 3 – Build and export table
149        // ---------------------------------------------------------------------
150        try {
151            TrackTableIO trackTableIO = new TrackTableIO();
152            Table tracksTable = trackTableIO.toTable(tracks);
153
154            tracksTable = tracksTable.sortOn(
155                    "Recording Name",
156                    "Number of Spots",
157                    "Number of Gaps",
158                    "Longest Gap",
159                    "Track Duration",
160                    "Track X Location",
161                    "Track Y Location",
162                    "Track Displacement",
163                    "Track Max Speed",
164                    "Track Median Speed",
165                    "Diffusion Coefficient",
166                    "Diffusion Coefficient Ext",
167                    "Total Distance",
168                    "Confinement Ratio"
169            );
170
171            // Replace Track Ids and generate unique keys
172            IntColumn newIds       = IntColumn.create("Track Id");
173            StringColumn newUniqueKey = StringColumn.create("Unique Key");
174
175            for (int i = 0; i < tracksTable.rowCount(); i++) {
176                newIds.append(i);
177                newUniqueKey.append(recordingName + "-" + i);
178            }
179            tracksTable.replaceColumn("Track Id", newIds);
180            tracksTable.replaceColumn("Unique Key", newUniqueKey);
181
182            trackTableIO.writeCsv(tracksTable, csvFile.toPath());
183        } catch (Exception e) {
184            PaintLogger.errorf("Failed writing track CSV: %s", e.getMessage());
185            e.printStackTrace();
186        }
187
188        return totalSpots;
189    }

191    // -------------------------------------------------------------------------
192    // Helper Methods
193    // -------------------------------------------------------------------------
194
195    /**
196     * Rounds a double to the specified number of decimal places.
197     *
198     * @param v      input value (nullable)
199     * @param places number of decimal places
200     * @return rounded value, or {@code Double.NaN} if null
201     */
202    private static double roundTo(Double v, int places) {
203        if (v == null) {
204            return Double.NaN;
205        }
206        double scale = Math.pow(10, places);
207        return Math.round(v * scale) / scale;
208    }
209
210    /**
211     * Rounds a double or returns a fallback if {@code v} is null.
212     *
213     * @param v       input value (nullable)
214     * @param places  decimal precision
215     * @param ifNull  fallback value when input is null
216     * @return rounded value or fallback
217     */
218    private static Double roundOr(Double v, int places, double ifNull) {
219        return v == null ? ifNull : roundTo(v, places);
220    }
221
222    /**
223     * Converts a Double to an int, returning -1 for null or NaN.
224     *
225     * @param v Double value
226     * @return integer value or -1 for invalid input
227     */
228    private static int asInt(Double v) {
229        if (v == null || v.isNaN()) {
230            return -1;
231        }
232        return (int) Math.round(v);
233    }
