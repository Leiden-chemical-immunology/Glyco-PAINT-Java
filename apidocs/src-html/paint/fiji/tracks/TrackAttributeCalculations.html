039package paint.fiji.tracks;
040
041import fiji.plugin.trackmate.Spot;
042import fiji.plugin.trackmate.TrackModel;
043
044import java.util.ArrayList;
045import java.util.Comparator;
046import java.util.List;
047
048import static paint.shared.utils.Miscellaneous.round;
049
050/**
051 * Provides static methods to compute motion-related attributes for a given
052 * TrackMate track, including total distance, diffusion coefficients,
053 * confinement ratio, and displacement.
054 * <p>
055 * Designed as a utility class — not instantiable.
056 * </p>
057 */
058public final class TrackAttributeCalculations {
059
060    /**
061     * Private constructor to prevent instantiation.
062     */
063    private TrackAttributeCalculations() {
064        // Utility class; prevent instantiation
065    }
066
067    /**
068     * Computes motion and diffusion-related attributes for a single track.
069     * <p>
070     * Calculated attributes include:
071     * <ul>
072     *   <li>Number of spots in track</li>
073     *   <li>Total travel distance</li>
074     *   <li>Diffusion coefficients (standard and extended)</li>
075     *   <li>Confinement ratio</li>
076     *   <li>Net displacement</li>
077     * </ul>
078     *
079     * @param trackModel the {@link TrackModel} containing track and spot data
080     * @param trackId    the unique identifier of the track to analyze
081     * @param dtSeconds  the time interval between consecutive frames, in seconds
082     * @return a {@link TrackAttributes} object containing all computed metrics
083     */
084    public static TrackAttributes calculateTrackAttributes(TrackModel trackModel, int trackId, double dtSeconds) {
085
086        // ---------------------------------------------------------------------
087        // Step 1 – Collect and sort track spots
088        // ---------------------------------------------------------------------
089        final List<Spot> spots = new ArrayList<>(trackModel.trackSpots(trackId));
090        if (spots.size() < 2) {
091            return new TrackAttributes();
092        }
093        spots.sort(Comparator.comparingInt(s -> (int) Math.round(s.getFeature(Spot.FRAME))));
094
095        // ---------------------------------------------------------------------
096        // Step 2 – Initialize variables
097        // ---------------------------------------------------------------------
098        double totalDistance       = 0.0;
099        double cumMsd              = 0.0;
100        double cumMsdExt           = 0.0;
101        double diffusionCoeff      = Double.NaN;
102        double diffusionCoeffExt   = Double.NaN;
103        double confinementRatio    = Double.NaN;
104        int    numberSpotsInTrack  = spots.size();
105
106        // ---------------------------------------------------------------------
107        // Step 3 – Reference coordinates
108        // ---------------------------------------------------------------------
109        final double x0     = get(spots.get(0), Spot.POSITION_X);
110        final double y0     = get(spots.get(0), Spot.POSITION_Y);
111        final double xLast  = get(spots.get(spots.size() - 1), Spot.POSITION_X);
112        final double yLast  = get(spots.get(spots.size() - 1), Spot.POSITION_Y);
113
114        // ---------------------------------------------------------------------
115        // Step 4 – Compute distances and MSD values
116        // ---------------------------------------------------------------------
117        for (int i = 1; i < spots.size(); i++) {
118
119            // The previous point
120            final double x1 = get(spots.get(i - 1), Spot.POSITION_X);
121            final double y1 = get(spots.get(i - 1), Spot.POSITION_Y);
122
123            // The current point
124            final double x2 = get(spots.get(i), Spot.POSITION_X);
125            final double y2 = get(spots.get(i), Spot.POSITION_Y);
126
127            // For MSD Ext, take the distance to the previous point
128            final double dx = x2 - x1;
129            final double dy = y2 - y1;
130
131            // For MSD, take the distance to the first point
132            final double dx0   = x2 - x0;
133            final double dy0   = y2 - y0;
134
135            // Calculate the cumulative values
136            cumMsd            += dx0 * dx0 + dy0 * dy0;
137            cumMsdExt         += dx * dx + dy * dy;
138
139            // Also calculate the increase in total distance
140            totalDistance += Math.hypot(dx, dy);
141        }
142
143        // Diffusion coefficients (remain NaN if insufficient data or dtSeconds <= 0)
144        final int nSteps = spots.size() - 1;
145
146        if (nSteps > 0 && dtSeconds > 0.0) {
147            final double msd = cumMsd / nSteps;
148            final double msdExt = cumMsdExt / nSteps;
149
150            diffusionCoeff    = round(msd / (4.0 * dtSeconds), 2);
151            diffusionCoeffExt = round(msdExt / (4.0 * dtSeconds), 2);
152        }
153
154        // Calculate the displacement between the first and last spot
155        double displacement = Math.hypot(xLast - x0, yLast - y0);
156
157        // Calculate the Confinement ratio = displacement / total distance
158        if (totalDistance > 0.0) {
159            confinementRatio = round(displacement / totalDistance, 2);
160        }
161
162        // If totalDistance is zero, keep it as NaN for downstream consistency
163        if (totalDistance == 0.0) {
164            totalDistance = Double.NaN;
165        }
166
167        // ---------------------------------------------------------------------
168        // Step 6 – Return immutable result
169        // ---------------------------------------------------------------------
170        return new TrackAttributes(
171                numberSpotsInTrack,
172                totalDistance,
173                diffusionCoeff,
174                diffusionCoeffExt,
175                confinementRatio,
176                displacement
177        );
178    }
179
180    // -------------------------------------------------------------------------
181    // Helper methods
182    // -------------------------------------------------------------------------
183
184    /**
185     * Retrieves a numerical feature value from a {@link Spot}.
186     * Returns 0.0 if the feature is missing or {@code null}.
187     *
188     * @param s          the spot to read from
189     * @param featureKey the feature name (e.g. {@code Spot.POSITION_X})
190     * @return the feature value or 0.0 if unavailable
191     */
192    private static double get(Spot s, String featureKey) {
193        final Double v = s.getFeature(featureKey);
194        return v == null ? 0.0 : v;
195    }
196}
