016 *    • Optionally removes processed experiment subdirectories.
017 *
018 *  RESPONSIBILITIES:
019 *    • Automate post-sweep data consolidation.
020 *    • Maintain deterministic output structure for all sweeps.
021 *    • Integrate “Generate Squares” into the flattening pipeline.
022 *
023 *  USAGE EXAMPLE:
024 *    Path sweepPath = Paths.get("/Users/hans/Paint Test Project/Sweep");

043package paint.fiji.utils;
044
045import org.apache.commons.io.FileUtils;
046import paint.generatesquares.GenerateSquaresHeadless;
047import paint.shared.utils.PaintLogger;
048
049import java.io.IOException;
050import java.nio.file.DirectoryStream;
051import java.nio.file.Files;
052import java.nio.file.Path;
053import java.nio.file.Paths;
054import java.util.ArrayList;
055import java.util.Arrays;
056import java.util.List;
057
058import static paint.shared.constants.PaintConstants.*;
059import static paint.shared.utils.CsvUtils.addCase;
060import static paint.shared.utils.CsvUtils.concatenateNamedCsvFiles;

062/**
063 * Automates flattening and consolidation of multi-parameter sweep data.
064 * <p>
065 * Scans a sweep root directory containing subdirectories for parameter cases,
066 * processes each experiment, runs “Generate Squares” and concatenates CSV
067 * outputs into flattened tables suitable for downstream Paint analysis.
068 * </p>
069 */
070public final class SweepFlattener {
071
072    /**
073     * Private constructor to prevent instantiation.
074     */
075    private SweepFlattener() {
076        // Utility class; prevent instantiation
077    }
078
079    /**
080     * Flattens a sweep directory structure by processing each parameter subfolder.
081     * <p>
082     * Performs the following sequence:
083     * <ol>
084     *   <li>Identifies parameter directories under the given sweep root.</li>
085     *   <li>Runs the {@link GenerateSquaresHeadless} process for each case.</li>
086     *   <li>Adds “Case” columns to per-experiment CSVs.</li>
087     *   <li>Concatenates the four main CSVs (Squares, Tracks, Recordings, Experiment Info).</li>
088     *   <li>Optionally deletes subdirectories after successful processing.</li>
089     *   <li>Finally, merges all parameter-level CSVs into a global flattened output.</li>
090     * </ol>
091     *
092     * @param sweepPath       root directory containing the parameter subdirectories
093     * @param experimentNames list of experiment subdirectories to process per parameter
094     * @param deleteSubdirs   true to remove experiment folders after processing
095     * @throws IOException if I/O operations fail during flattening
096     */
097    public static void flattenSweep(Path sweepPath,
098                                    List<String> experimentNames,
099                                    boolean deleteSubdirs) throws IOException {
100
101        if (!Files.isDirectory(sweepPath)) {
102            throw new IOException("Not a directory: " + sweepPath);
103        }
104
105        List<String> paramDirsFound = new ArrayList<>();
106
107        // ---------------------------------------------------------------------
108        // Step 1 – Iterate over parameter directories (e.g., “[Param]-[Value]”)
109        // ---------------------------------------------------------------------
110        try (DirectoryStream<Path> paramPaths = Files.newDirectoryStream(sweepPath, "[[]*")) {
111            for (Path paramPath : paramPaths) {
112                if (!Files.isDirectory(paramPath)) {
113                    continue;
114                }
115
116                PaintLogger.infof("Flattening sweep case: %s", paramPath.getFileName());
117                paramDirsFound.add(paramPath.getFileName().toString());
118
119                if (experimentNames.isEmpty()) {
120                    PaintLogger.warnf("No experiment subdirectories specified under %s", paramPath);
121                    continue;
122                }
123
124                // -----------------------------------------------------------------
125                // Step 2 – Run Generate Squares
126                // -----------------------------------------------------------------
127                try {
128                    GenerateSquaresHeadless.run(paramPath, experimentNames);
129                } catch (Exception e) {
130                    PaintLogger.errorf("GenerateSquaresHeadless failed in %s: %s", paramPath, e.getMessage());
131                }
132
133                // -----------------------------------------------------------------
134                // Step 3 – Add “Case” field to experiment-level CSVs
135                // -----------------------------------------------------------------
136                try {
137                    String caseName = paramPath.getFileName().toString();
138                    addCase(paramPath, SQUARES_CSV,       experimentNames, caseName);
139                    addCase(paramPath, TRACKS_CSV,        experimentNames, caseName);
140                    addCase(paramPath, RECORDINGS_CSV,    experimentNames, caseName);
141                    addCase(paramPath, EXPERIMENT_INFO_CSV, experimentNames, caseName);
142                } catch (IOException e) {
143                    PaintLogger.errorf("  Error adding 'Case' field in %s: %s", paramPath, e.getMessage());
144                }
145
146                // -----------------------------------------------------------------
147                // Step 4 – Concatenate per-parameter experiment CSVs
148                // -----------------------------------------------------------------
149                try {
150                    concatenateNamedCsvFiles(paramPath, SQUARES_CSV,       experimentNames);
151                    concatenateNamedCsvFiles(paramPath, TRACKS_CSV,        experimentNames);
152                    concatenateNamedCsvFiles(paramPath, RECORDINGS_CSV,    experimentNames);
153                    concatenateNamedCsvFiles(paramPath, EXPERIMENT_INFO_CSV, experimentNames);
154                } catch (IOException e) {
155                    PaintLogger.errorf("  Error concatenating files in %s: %s", paramPath, e.getMessage());
156                }
157
158                // -----------------------------------------------------------------
159                // Step 5 – Optionally delete experiment subdirectories
160                // -----------------------------------------------------------------
161                if (deleteSubdirs) {
162                    for (String sub : experimentNames) {
163                        try {
164                            FileUtils.deleteDirectory(paramPath.resolve(sub).toFile());
165                        } catch (Exception e) {
166                            PaintLogger.warnf("Failed to delete subdirectory %s: %s", sub, e.getMessage());
167                        }
168                    }
169                }
170            }
171        }
172
173        // ---------------------------------------------------------------------
174        // Step 6 – Global concatenation across all parameter directories
175        // ---------------------------------------------------------------------
176        concatenateNamedCsvFiles(sweepPath, SQUARES_CSV,       paramDirsFound);
177        concatenateNamedCsvFiles(sweepPath, TRACKS_CSV,        paramDirsFound);
178        concatenateNamedCsvFiles(sweepPath, RECORDINGS_CSV,    paramDirsFound);
179        concatenateNamedCsvFiles(sweepPath, EXPERIMENT_INFO_CSV, paramDirsFound);
180
181        PaintLogger.blankline();
182        PaintLogger.infof("Completed Sweep flattening for %d parameter sets.", paramDirsFound.size());
183    }
184
185    /**
186     * Standalone entry point for debugging and testing the sweep flattening process.
187     *
188     * @param args command-line arguments (unused)
189     * @throws IOException if I/O operations fail during execution
190     */
191    public static void main(String[] args) throws IOException {
192        Path sweepPath = Paths.get("/Users/hans/Paint Test Project/Sweep");
193        flattenSweep(sweepPath, Arrays.asList("221012", "AnyName"), false);
194    }
