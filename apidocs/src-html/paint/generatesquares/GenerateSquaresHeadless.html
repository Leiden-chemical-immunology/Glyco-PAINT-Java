068/**
069 * Runs the Generate Squares pipeline headlessly:
070 * validates experiments, performs calculations, exports results,
071 * and builds project-level summary CSVs.
072 */
073public class GenerateSquaresHeadless {
074
075    public static void run(Path projectPath, List<String> experimentNames)  {
076
077        // --- Early abort check ---
078        if (Thread.currentThread().isInterrupted()) {
079            PaintLogger.infof("Generate Squares run aborted before start (user cancelled).");
080            return;
081        }
082
083        // --- Validate input data ---
084        PaintLogger.infof("Validating input data...");
085
086        ValidationResult validateResult = validateExperiments(
087                projectPath,
088                experimentNames,
089                Arrays.asList(EXPERIMENT_INFO_CSV, RECORDINGS_CSV, TRACKS_CSV)
090        );
091
092        if (!validateResult.isValid()) {
093            for (String line : validateResult.getReport().split("\n")) {
094                PaintLogger.errorf(line);
095            }
096            throw new IllegalStateException("Experiment validation failed.");
097        }
098
099        logContextAndConfiguration(projectPath, experimentNames);
100        LocalDateTime start = LocalDateTime.now();
101
102        // --- Prepare project ---
103        GenerateSquaresConfig generateSquaresConfig = new GenerateSquaresConfig();
104        Project project = new Project();
105        project.setProjectRootPath(projectPath);
106        project.setExperimentNames(experimentNames);
107        project.setGenerateSquaresConfig(generateSquaresConfig);
108
109        // --- Run each experiment ---
110        for (String experimentName : experimentNames) {
111
112            if (Thread.currentThread().isInterrupted()) {
113                PaintLogger.infof("Generate Squares run stopped early (user cancelled).");
114                return;
115            }
116
117            PaintLogger.infof("Running Generate Squares for experiment: %s", experimentName);
118
119            try {
120                generateSquaresForExperiment(project, experimentName);
121            } catch (Exception e) {
122                PaintLogger.errorf("Error processing experiment %s: %s", experimentName, e.getMessage());
123                continue;
124            }
125
126            if (Thread.currentThread().isInterrupted()) {
127                PaintLogger.infof("Cancelled before exporting histograms for %s", experimentName);
128                return;
129            }
130
131            try {
132                Experiment experiment = loadExperiment(projectPath, experimentName, true);
133                Path pdfOut = projectPath
134                        .resolve(experimentName)
135                        .resolve("Output")
136                        .resolve("Background.pdf");
137
138                Files.createDirectories(pdfOut.getParent());
139                exportExperimentHistogramsToPngs(experiment, projectPath.resolve(experimentName));
140
141            } catch (Exception e) {
142                PaintLogger.errorf("Failed to export histograms for %s: %s", experimentName, e.getMessage());
143            }
144        }
145
146        // --- Concatenate results ---
147        if (Thread.currentThread().isInterrupted()) {
148            PaintLogger.infof("Cancelled before concatenating project-level CSVs.");
149            return;
150        }
151
152        try {
153            PaintLogger.infof("Creating project-level summary files...");
154
155            PaintLogger.infof("   Creating %s", projectPath.resolve(SQUARES_CSV));
156            concatenateNamedCsvFiles(projectPath, SQUARES_CSV, experimentNames);
157            if (Thread.currentThread().isInterrupted()) {
158                return;
159            }
160
161            PaintLogger.infof("   Creating %s", projectPath.resolve(RECORDINGS_CSV));
162            concatenateNamedCsvFiles(projectPath, RECORDINGS_CSV, experimentNames);
163            if (Thread.currentThread().isInterrupted()) {
164                return;
165            }
166
167            PaintLogger.infof("   Creating %s", projectPath.resolve(EXPERIMENT_INFO_CSV));
168            concatenateNamedCsvFiles(projectPath, EXPERIMENT_INFO_CSV, experimentNames);
169            if (Thread.currentThread().isInterrupted()) {
170                return;
171            }
172
173            PaintLogger.infof("   Creating %s", projectPath.resolve(TRACKS_CSV));
174            concatenateNamedCsvFiles(projectPath, TRACKS_CSV, experimentNames);
175
176            PaintLogger.blankline();
177
178            Duration duration = Duration.between(start, LocalDateTime.now());
179            PaintLogger.infof("Finished Generate Squares for all experiments in %s", formatDuration(duration));
180
181        } catch (Exception e) {
182            PaintLogger.errorf("Failed to concatenate CSVs: %s", e.getMessage());
183        }
184    }
185
186    private static void logContextAndConfiguration(Path projectPath, List<String> experimentNames) {
187
188        int nSquares      = PaintConfig.getInt("Generate Squares", "Number of Squares in Recording", 400);
189        int side          = (int) Math.round(Math.sqrt(nSquares));
190        int minTracks     = PaintConfig.getInt("Generate Squares", "Min Tracks to Calculate Tau", 20);
191        double minRSq     = PaintConfig.getDouble("Generate Squares", "Min Required R Squared", 0.1);
192        double minDensity = PaintConfig.getDouble("Generate Squares", "Min Required Density Ratio", 2.0);
193        double maxVar     = PaintConfig.getDouble("Generate Squares", "Max Allowable Variability", 10.0);
194
195        // Neatly wrapped experiment list
196        String formattedExperiments;
197        if (experimentNames.isEmpty()) {
198            formattedExperiments = "                   (none selected — please verify selection)";
199        } else {
200            final int MAX_WIDTH = 100;
201            final String INDENT = "                   ";
202
203            StringBuilder sb = new StringBuilder("  "); // the first line has only 2 leading spaces
204            int currentLineLength = 2;
205            int effectiveMaxWidthFirstLine = MAX_WIDTH - (INDENT.length() - 2);
206
207            for (int i = 0; i < experimentNames.size(); i++) {
208                String exp = experimentNames.get(i);
209                int tokenLength = exp.length() + (i < experimentNames.size() - 1 ? 2 : 0);
210
211                // use narrower limit for first line, normal for later lines
212                int limit = (sb.indexOf("\n") == -1) ? effectiveMaxWidthFirstLine : MAX_WIDTH;
213
214                if (currentLineLength + tokenLength > limit) {
215                    sb.append("\n").append(INDENT);
216                    currentLineLength = INDENT.length();
217                }
218
219                sb.append(exp);
220                currentLineLength += exp.length();
221
222                if (i < experimentNames.size() - 1) {
223                    sb.append(", ");
224                    currentLineLength += 2;
225                }
226            }
227
228            formattedExperiments = sb.toString();
229        }
230
231        PaintLogger.doc("Generate Squares", Arrays.asList(
232                "Starting Generate Squares analysis for project: " + projectPath.getFileName(),
233                "",
234                "Selected experiments:",
235                formattedExperiments,
236                "",
237                "Using parameters:",
238                String.format(Locale.getDefault(), "  • Grid size:                 %dx%d (%d squares)", side, side, nSquares),
239                String.format(Locale.getDefault(), "  • Minimum tracks per square: %d", minTracks),
240                String.format(Locale.getDefault(), "  • Minimum R²:                %.2f", minRSq),
241                String.format(Locale.getDefault(), "  • Minimum density ratio:     %.1f", minDensity),
242                String.format(Locale.getDefault(), "  • Maximum variability:       %.1f", maxVar),
243                "",
244                "Each recording will be divided into spatial squares, and per-square track statistics will be calculated.",
245                "Results will be prepared per experiment and squares and tracks files updated",
246                "",
247                "The results will then be compiled into project level files:",
248                String.format("  • %s", SQUARES_CSV),
249                String.format("  • %s (with updated Square Number and Label Number fields)", TRACKS_CSV),
250                String.format("  • %s", RECORDINGS_CSV),
251                String.format("  • %s", EXPERIMENT_INFO_CSV)
252        ));
253    }
