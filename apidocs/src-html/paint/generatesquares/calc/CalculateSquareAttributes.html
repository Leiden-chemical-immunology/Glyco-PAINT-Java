067    /**
068     * Calculates attributes for each square in a recording, such as Tau, density, variability, and density ratio.
069     * It also applies visibility filtering based on given parameters and assigns label numbers to selected squares.
070     *
071     * @param experimentPath        the file path of the current experiment for saving generated files
072     * @param recording             the recording containing squares and associated track data
073     * @param generateSquaresConfig the configuration parameters used for generating square attributes and analysis
074     */
075    public static void calculateSquareAttributes(Path experimentPath,
076                                                 Recording recording,
077                                                 GenerateSquaresConfig generateSquaresConfig) {
078        double       minRequiredRSquared        = generateSquaresConfig.getMinRequiredRSquared();
079        int          minTracksForTau            = generateSquaresConfig.getMinTracksToCalculateTau();
080        double       maxAllowableVariability    = generateSquaresConfig.getMaxAllowableVariability();
081        double       minRequiredDensityRatio    = generateSquaresConfig.getMinRequiredDensityRatio();
082        String       neighbourMode              = generateSquaresConfig.getNeighbourMode();
083        int          numberOfSquaresInRecording = generateSquaresConfig.getNumberOfSquaresInRecording();
084        double       squareArea                 = calculateSquareArea(numberOfSquaresInRecording);    // Here we look at the single square
085        double       concentration              = recording.getConcentration();
086        List<Square> squaresOfRecording         = recording.getSquaresOfRecording();
087
088        SquareUtils.BackgroundEstimationResult result = calculateBackgroundDensity(squaresOfRecording);
089        double meanBackgroundTracks                   = result.getBackgroundMean();
090        double backgroundTracksOri                    = calcAverageTrackCountInBackgroundSquares(recording.getSquaresOfRecording(), (int) (0.1 * numberOfSquaresInRecording));
091
092        PaintLogger.debugf("Estimated Background track count = %.2f, n = %d%n",
093                           meanBackgroundTracks, result.getBackgroundSquares().size());
094
095        for (Square square : squaresOfRecording) {
096            List<Track> tracksInSquare  = square.getTracks();
097            Table       table           = square.getTracksTable();
098            int         squareNumber    = square.getSquareNumber();
099            Table tracksInSquareTable   = square.getTracksTable();
100
101            if (tracksInSquare == null || tracksInSquare.isEmpty()) {
102                continue;
103            }
104
105            if (tracksInSquare.size() >= minTracksForTau) {
106                CalculateTau.CalculateTauResult results = calculateTau(tracksInSquare, minRequiredRSquared);
107
108                if (paint.shared.config.PaintConfig.getBoolean("Generate Squares", "Plot Curve Fitting", false) &&
109                        tracksInSquare.size() >= minTracksForTau) {
110                    saveTauFitPlot(tracksInSquare, results, experimentPath, recording.getRecordingName(), squareNumber);
111                }
112
113                if (results.getStatus() == CalculateTau.CalculateTauResult.Status.TAU_SUCCESS) {
114                    square.setTau(round(results.getTau(), 0));
115                    square.setRSquared(round(results.getRSquared(), 3));
116                } else {
117                    square.setTau(Double.NaN);
118                    square.setRSquared(Double.NaN);
119                }
120            } else {
121                square.setTau(Double.NaN);
122                square.setRSquared(Double.NaN);
123            }
124
125            if (table.rowCount() == 0) {
126                continue;
127            }
128
129            // @format:off
130            square.setVariability(                   round(calculateVariability(table, squareNumber, numberOfSquaresInRecording, 10),    2));
131            square.setDensity(                       round(calculateDensity(tracksInSquare.size(), squareArea, RECORDING_DURATION, concentration), 3));
132            square.setDensityRatio(                  round(calculateDensityRatio(tracksInSquare.size(), meanBackgroundTracks),                     2));
133            square.setDensityRatioOri(               round(calculateDensityRatio(tracksInSquare.size(), backgroundTracksOri),                      2));
134
135            square.setMedianDiffusionCoefficient(    round(tracksInSquareTable.doubleColumn("Diffusion Coefficient").median(),         2));
136            square.setMedianDiffusionCoefficientExt( round(tracksInSquareTable.doubleColumn("Diffusion Coefficient Ext").median(),     2));
137
138            square.setMedianDisplacement(            round(tracksInSquareTable.doubleColumn("Track Displacement").median(),            1));
139            square.setMaxDisplacement(               round(tracksInSquareTable.doubleColumn("Track Displacement").max(),               1));
140            square.setTotalDisplacement(             round(tracksInSquareTable.doubleColumn("Track Displacement").sum(),               1));
141
142            square.setMedianMaxSpeed(                round(tracksInSquareTable.doubleColumn("Track Max Speed").median(),               1));
143            square.setMaxMaxSpeed(                   round(tracksInSquareTable.doubleColumn("Track Max Speed").max(),                  1));
144
145            square.setMedianMedianSpeed(             round(tracksInSquareTable.doubleColumn("Track Median Speed").median(),            1));
146            square.setMaxMedianSpeed(                round(tracksInSquareTable.doubleColumn("Track Median Speed").max(),               1));
147
148            square.setMaxTrackDuration(              round(tracksInSquareTable.doubleColumn("Track Duration").max(),                   1));
149            square.setTotalTrackDuration(            round(tracksInSquareTable.doubleColumn("Track Duration").sum(),                   1));
150            square.setMedianTrackDuration(           round(tracksInSquareTable.doubleColumn("Track Duration").median(),                1));
151            // @format:on
152
153        }
154
155        applyVisibilityFilter(squaresOfRecording,
156                              minRequiredDensityRatio,
157                              maxAllowableVariability,
158                              minRequiredRSquared,
159                              neighbourMode);
160
161        int labelNumber = 0;
162        for (Square sq : recording.getSquaresOfRecording()) {
163            if (sq.isSelected()) {
164                sq.setLabelNumber(labelNumber++);
165            }
166        }
167    }

169    /**
170     * Calculates various attributes for a recording, including Tau, density,
171     * background statistics, and R-squared values. This method processes detailed
172     * information about the recording and applies configuration settings to compute
173     * the necessary metrics.
174     *
175     * @param recording             the recording object containing square and track data to process
176     * @param generateSquaresConfig the configuration parameters for generating square attributes
177     */
178    public static void calculateRecordingAttributes(Recording recording,
179                                                    GenerateSquaresConfig generateSquaresConfig) {
180
181        double minRequiredRSquared = generateSquaresConfig.getMinRequiredRSquared();
182
183        BackgroundEstimationResult result = SquareUtils.calculateBackgroundDensity(recording.getSquaresOfRecording());
184        double       meanBackgroundTracks = result.getBackgroundMean();
185        int          backgroundTracks     = result.getBackgroundSquares().stream().mapToInt(Square::getNumberOfTracks).sum();
186
187        PaintLogger.debugf("Estimated Background track count = %.2f, n = %d%n", meanBackgroundTracks, result.getBackgroundSquares().size());
188
189        recording.setNumberOfSquaresInBackground(result.getBackgroundSquares().size());
190        recording.setNumberOfTracksInBackground(backgroundTracks);
191        recording.setAverageTracksInBackGround(round(meanBackgroundTracks, 3));
192
193        List<Track> tracksFromSelectedSquares = getTracksFromSelectedSquares(recording.getSquaresOfRecording());
194        CalculateTau.CalculateTauResult results = calculateTau(tracksFromSelectedSquares, minRequiredRSquared);
195        if (results.getStatus() == CalculateTau.CalculateTauResult.Status.TAU_SUCCESS) {
196            recording.setTau(round(results.getTau(), 0));
197            recording.setRSquared(round(results.getRSquared(), 3));
198        } else {
199            recording.setTau(Double.NaN);
200            recording.setRSquared(Double.NaN);
201        }
202
203        double density = calculateDensity(
204                tracksFromSelectedSquares.size(),
205                calculateSquareArea(getNumberOfSelectedSquares(recording)),
206                RECORDING_DURATION,
207                recording.getConcentration()
208        );
209        recording.setDensity(round(density, 2));
210    }

212    public static double calculateDensityRatio(int numberOfTracksInSquare, double numberOfTracksInBackgroundSquare) {
213        if (numberOfTracksInBackgroundSquare == 0) {
214            return 0;
215        } else {
216            return numberOfTracksInSquare / numberOfTracksInBackgroundSquare;
217        }
218    }

220    /**
221     * Computes the spatial variability of track positions within a selected square of a recording.
222     * The square is subdivided into a grid of size `granularity × granularity`, and the count
223     * of tracks falling into each grid cell is determined. Variability is then quantified
224     * as the coefficient of variation (standard deviation divided by mean) of the cell counts.
225     *
226     * @param tracks                     A table of track data, including x- and y-coordinate columns.
227     * @param squareNumber               The index of the square region within the recording which is being analysed.
228     * @param numberOfSquaresInRecording The total number of spatial squares defined for the recording.
229     * @param granularity                The number of subdivisions (cells) along each dimension of the square’s grid.
230     * @return                           The coefficient of variation (σ / μ) of the grid-cell counts;
231     *                                   returns 0.0 if the mean cell count is zero (no tracks).
232     */
233    public static double calculateVariability(Table tracks,
234                                              int   squareNumber,
235                                              int   numberOfSquaresInRecording,
236                                              int   granularity) {
237
238        // Matrix for variability analysis
239        int[][] matrix = new int[granularity][granularity];
240
241        // Width and height of a square
242        int dimension = (int) Math.sqrt(numberOfSquaresInRecording);
243        double width  = IMAGE_WIDTH / dimension;
244        double height = IMAGE_WIDTH / dimension;
245
246        // Access the columns once
247        DoubleColumn xCol = tracks.doubleColumn("Track X Location");
248        DoubleColumn yCol = tracks.doubleColumn("Track Y Location");
249
250        // Loop over the tracks and fill the matrix
251        for (int i = 0; i < tracks.rowCount(); i++) {
252            double x = xCol.get(i);  // The x-coordinate of the track
253            double y = yCol.get(i);  // The y-coordinate of the track
254
255            // Get grid indices
256            int[] indices = getIndices(x, y, width, height, squareNumber, dimension, granularity);
257            int xi = indices[0];
258            int yi = indices[1];
259
260            if (xi >= 0 && xi < granularity && yi >= 0 && yi < granularity) {
261                matrix[yi][xi]++;
262            }
263        }
264
265        // Flatten matrix into an 1D array for stats
266        int totalCells = granularity * granularity;
267        double[] values = new double[totalCells];
268        int idx = 0;
269        for (int r = 0; r < granularity; r++) {
270            for (int c = 0; c < granularity; c++) {
271                values[idx++] = matrix[r][c];
272            }
273        }
274
275        double mean = mean(values);
276        if (mean == 0) {
277            return 0.0;
278        }
279
280        double std = std(values, mean);
281        return std / mean; // coefficient of variation
282    }

284    /**
285     * Computes the mean of the given array of double values.
286     *
287     * @param values an array of double values for which the mean is to be calculated.
288     *               The array must not be empty and must contain at least one value.
289     * @return the mean (average) of the given array of double values.
290     */
291    // Utility: compute mean
292    private static double mean(double[] values) {
293        double sum = 0.0;
294        for (double v : values) {
295            sum += v;
296        }
297        return sum / values.length;
298    }
299
300    /**
301     * Calculates the population standard deviation of the given array of double values.
302     *
303     * @param values an array of double values for which the population standard deviation is to be calculated.
304     *               The array must not be empty and must contain at least one value.
305     * @param mean   the mean (average) of the values in the array, precomputed to optimize calculation.
306     * @return the population standard deviation of the given array of double values.
307     */
308    // Utility: compute std (population standard deviation)
309    private static double std(double[] values, double mean) {
310        double sumSq = 0.0;
311        for (double v : values) {
312            double diff = v - mean;
313            sumSq += diff * diff;
314        }
315        return Math.sqrt(sumSq / values.length);
316    }

318    /**
319     * Calculates the grid indices for a point within a specific square of a recording.
320     * The method determines the location of the point in a finer grid inside the square,
321     * based on the specified granularity.
322     *
323     * @param x1               the x-coordinate of the point in the global coordinate system
324     * @param y1               the y-coordinate of the point in the global coordinate system
325     * @param width            the width of each square in the grid
326     * @param height           the height of each square in the grid
327     * @param squareSeqNr      the sequence number of the square in the grid
328     * @param nrOfSquaresInRow the total number of squares in a single row of the grid
329     * @param granularity      the number of subdivisions along one axis within a square
330     * @return an array of two integers, where the first value is the x-index (column index)
331     * and the second value is the y-index (row index) of the point in the finer grid
332     */
333    private static int[] getIndices(double x1,
334                                    double y1,
335                                    double width,
336                                    double height,
337                                    int squareSeqNr,
338                                    int nrOfSquaresInRow,
339                                    int granularity) {
340        // Calculate the top-left corner (x0, y0) of the square
341        double x0 = (squareSeqNr % nrOfSquaresInRow) * width;
342        double y0 = (squareSeqNr / nrOfSquaresInRow) * height;     // Integer division is intended
343
344        // Calculate the grid indices (xi, yi) for the track
345        int xi = (int) (((x1 - x0) / width) * granularity);
346        int yi = (int) (((y1 - y0) / height) * granularity);
347
348        return new int[]{xi, yi};
349    }
