043import paint.shared.objects.Square;
044
045import java.util.ArrayList;
046import java.util.Collections;
047import java.util.Comparator;
048import java.util.List;
049import java.util.stream.Collectors;
050
051public class SquareUtils {
052
053    /**
054     * Private constructor to prevent instantiation.
055     */
056    private SquareUtils() {
057    }
058
059    /**
060     * Estimates the background density of track counts from a list of squares.
061     * The method iteratively filters squares with track counts exceeding a dynamically
062     * calculated threshold (mean + 2 * standard deviation), recalculates the mean,
063     * and repeats until the mean stabilizes or a maximum number of iterations is reached.
064     *
065     * @param squares List of Square objects containing track count information.
066     *                Must not be null or empty.
067     * @return A BackgroundEstimationResult object containing the estimated mean track count
068     * for the background and the list of squares identified as background.
069     */
070    public static BackgroundEstimationResult calculateBackgroundDensity(List<Square> squares) {
071        if (squares == null || squares.isEmpty()) {
072            return new BackgroundEstimationResult(Double.NaN, Collections.emptyList());
073        }
074
075        double mean = squares.stream()
076                .mapToDouble(Square::getNumberOfTracks)
077                .average().orElse(Double.NaN);
078
079        if (Double.isNaN(mean) || mean == 0) {
080            return new BackgroundEstimationResult(mean, Collections.emptyList());
081        }
082
083        final double EPSILON = 0.01;
084        final int MAX_ITER   = 10;
085        double prevMean;
086
087        List<Square> current = new ArrayList<>(squares);
088
089        for (int iter = 0; iter < MAX_ITER; iter++) {
090            prevMean = mean;
091            final double meanForLambda = mean;
092
093            double std = Math.sqrt(current.stream()
094                                           .mapToDouble(sq -> Math.pow(sq.getNumberOfTracks() - meanForLambda, 2))
095                                           .average().orElse(0));
096
097            final double threshold = meanForLambda + 2 * std;
098
099            List<Square> filtered = new ArrayList<>();
100            for (Square sq : current) {
101                if (sq.getNumberOfTracks() <= threshold) {
102                    filtered.add(sq);
103                }
104            }
105
106            if (filtered.isEmpty()) {
107                break;
108            }
109
110            mean = filtered.stream()
111                    .mapToDouble(Square::getNumberOfTracks)
112                    .average().orElse(mean);
113
114            current = filtered;
115
116            // Stop if the mean stabilizes
117            if (Math.abs(mean - prevMean) / prevMean < EPSILON) {
118                break;
119            }
120        }
121
122        return new BackgroundEstimationResult(mean, current);
123    }
124
125    /**
126     * Represents the result of a background estimation process for track counts.
127     * Contains the mean track count of the estimated background and the list of
128     * squares identified as background.
129     */
130    public static class BackgroundEstimationResult {
131        private final double backgroundMean;
132        private final List<Square> backgroundSquares;
133
134        /**
135         * Constructs a new BackgroundEstimationResult with the provided mean background value
136         * and the list of squares classified as background.
137         *
138         * @param backgroundMean    the mean value of tracks estimated as background
139         * @param backgroundSquares the list of squares identified as background
140         */
141        public BackgroundEstimationResult(double backgroundMean, List<Square> backgroundSquares) {
142            this.backgroundMean    = backgroundMean;
143            this.backgroundSquares = backgroundSquares;
144        }
145
146        /**
147         * @return Estimated mean background track count.
148         */
149        public double getBackgroundMean() {
150            return backgroundMean;
151        }
152
153        /**
154         * @return List of squares classified as background.
155         */
156        public List<Square> getBackgroundSquares() {
157            return backgroundSquares;
158        }
159    }
160
161    /**
162     * Calculates the average track count in a specified number of background squares
163     * by selecting a configurable number of the smallest non-zero track counts from
164     * the provided list of squares.
165     *
166     * @param squaresOfRecording      A list of Square objects, each containing track count data.
167     *                                Must not be null.
168     * @param nrOfAverageCountSquares The number of smallest non-zero track counts to include
169     *                                in the average calculation. Must be greater than zero.
170     * @return The average track count as a double, calculated from the smallest non-zero
171     * track counts in the specified number of squares. Returns 0.0 if none are found
172     * or if the number of valid squares is zero.
173     */
174    public static double calcAverageTrackCountInBackgroundSquares(List<Square> squaresOfRecording,
175                                                                  int nrOfAverageCountSquares) {
176
177        List<Integer> trackCounts = squaresOfRecording.stream()
178                .map(Square::getNumberOfTracks)
179                .collect(Collectors.toList());
180
181        // Sort descending
182        trackCounts.sort(Comparator.reverseOrder());
183
184        double total = 0.0;
185        int n = 0;
186
187        // Traverse backward (smallest to largest)
188        for (int i = trackCounts.size() - 1; i >= 0; i--) {
189            int value = trackCounts.get(i);
190            if (value > 0) {
191                total += value;
192                n++;
193                if (n >= nrOfAverageCountSquares) {
194                    break;
195                }
196            }
197        }
198
199        return n == 0 ? 0.0 : total / n;
200    }
