001/******************************************************************************
002 *  Class:        PlotUtils.java
003 *  Package:      paint.generatesquares.calc
004 *
005 *  PURPOSE:
006 *    Provides lightweight rendering utilities for generating Tau-fitting plots,
007 *    histograms and related graphical visualizations within the Paint experiment
008 *    workflow.
009 *
010 *  DESCRIPTION:
011 *    This utility class supports rendering of 2D visual representations such as:
012 *      • Frequency–duration scatter plots with exponential curve fits
013 *      • Histograms of track counts per square and background estimation
014 *      • Saving Tau-fit plots (PNG) under “Success” or “Failed” directories
015 *    Plots are rendered into BufferedImages, PDF documents or image files
016 *    and can be saved or embedded in reports and GUIs.
017 *
018 *  RESPONSIBILITIES:
019 *    • Draw frequency–duration scatter plots and fitted exponential curves
020 *    • Display Tau and R² annotations on plots
021 *    • Build histograms of track counts and background counts per square
022 *    • Save Tau-fit images, handling directory structure and failure/success classification
023 *    • Produce antialiased BufferedImage or PDF or PNG output for use in Paint analysis
024 *
025 *  USAGE EXAMPLE:
026 *    BufferedImage img = PlotUtils.renderTauPlot(x, y, result, false, 800,600);
027 *    PlotUtils.exportExperimentHistogramsToPdf(experiment, outputFile);
028 *    PlotUtils.saveTauFitPlot(tracks, tauResult, experimentPath, recordingName, squareIndex);
029 *
030 *  DEPENDENCIES:
031 *    – paint.shared.objects.{Experiment, Recording, Square, Track}
032 *    – paint.shared.utils.PaintLogger
033 *    – paint.generatesquares.calc.SquareUtils
034 *    – org.apache.pdfbox and de.rototor.pdfbox.graphics2d.PdfBoxGraphics2D

078    /**
079     * Renders a Tau plot with given data points, fit curve, axes, labels, and annotations.
080     *
081     * @param x         the array of x-coordinate values for the data points to be plotted
082     * @param y         the array of y-coordinate values for the data points to be plotted
083     * @param result    the calculated result containing the Tau value and R² value for the exponential fit
084     * @param fitFailed a flag indicating whether the exponential fit failed
085     * @param width     the width of the generated image in pixels
086     * @param height    the height of the generated image in pixels
087     * @return a BufferedImage representing the rendered Tau plot
088     */
089    public static BufferedImage renderTauPlot(double[] x, double[] y,
090                                              CalculateTau.CalculateTauResult result,
091                                              boolean fitFailed,
092                                              int width, int height) {
093
094        // --- Initialize blank image and 2D context ---
095        BufferedImage img = new BufferedImage(width, height, BufferedImage.TYPE_INT_ARGB);
096        Graphics2D g2 = img.createGraphics();
097        g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
098
099        // --- Draw background ---
100        g2.setColor(Color.WHITE);
101        g2.fillRect(0, 0, width, height);
102
103        // --- Define margins and plotting area ---
104
105        // @format:off
106        int marginLeft   = 70;
107        int marginRight  = 40;
108        int marginTop    = 40;
109        int marginBottom = 60;
110        int x0           = marginLeft;
111        int y0           = height - marginBottom;
112        // @format:on
113
114        // --- Determine axis ranges from data ---
115        double minX = Arrays.stream(x).min().orElse(0);
116        double maxX = Arrays.stream(x).max().orElse(1);
117        double minY = Arrays.stream(y).min().orElse(0);
118        double maxY = Arrays.stream(y).max().orElse(1);
119
120        // Add 5% padding on both axes so points stay within the frame
121        double padX = (maxX - minX) * 0.05;
122        double padY = (maxY - minY) * 0.05;
123        if (padX == 0) {
124            padX = 1.0; // avoid zero-division for flat x data
125        }
126        if (padY == 0) {
127            padY = 1.0; // avoid zero-division for flat y data
128        }
129
130        minX -= padX;
131        maxX += padX;
132        minY -= padY;
133        maxY += padY;
134
135        // --- Calculate scaling factors ---
136        double xScale = (width - marginLeft - marginRight) / (maxX - minX);
137        double yScale = (height - marginTop - marginBottom) / (maxY - minY);
138
139        // --- Draw X and Y axes ---
140        g2.setColor(Color.GRAY);
141        g2.drawLine(x0, y0, width - marginRight, y0); // X-axis
142        g2.drawLine(x0, y0, x0, marginTop);           // Y-axis
143
144        // --- Draw data points ---
145        g2.setColor(new Color(30, 100, 200)); // blue tone
146        for (int i = 0; i < x.length; i++) {
147            int px = (int)(marginLeft + (x[i] - minX) * xScale);
148            int py = (int)(y0 - (y[i] - minY) * yScale);
149            g2.fillOval(px - 3, py - 3, 6, 6);
150        }
151
152        // --- Draw fitted exponential curve (if available) ---
153        if (!fitFailed && result != null && Double.isFinite(result.getTau())) {
154            double tau = result.getTau();
155            // Convert to rate constant (1/ms) — arbitrary scaling for display
156            double t = (tau > 0) ? 1000.0 / tau : Double.NaN;
157
158            double m = Arrays.stream(y).max().orElse(1); // amplitude
159            double b = Arrays.stream(y).min().orElse(0); // baseline
160            g2.setColor(Color.RED);
161
162            int steps = 200;
163            double prevX = minX;
164            double prevY = m * Math.exp(-t * prevX) + b;
165            for (int i = 1; i <= steps; i++) {
166
167                // @format:off
168                double cx = minX + (maxX - minX) * i / steps;
169                double cy = m * Math.exp(-t * cx) + b;
170                int x1    = (int) (marginLeft + (prevX - minX) * xScale);
171                int y1    = (int) (y0 - (prevY - minY) * yScale);
172                int x2    = (int) (marginLeft + (cx - minX) * xScale);
173                int y2    = (int) (y0 - (cy - minY) * yScale);
174                // @format:on
175
176                g2.drawLine(x1, y1, x2, y2);
177                prevX = cx;
178                prevY = cy;
179            }
180        }
181
182        // --- Axis labels ---
183        g2.setColor(Color.BLACK);
184        g2.setFont(g2.getFont().deriveFont(Font.PLAIN, 12f));
185        g2.drawString("Duration", width / 2 - 30, height - 20);
186
187        g2.rotate(-Math.PI / 2);
188        g2.drawString("Frequency", -height / 2 - 30, 20);
189        g2.rotate(Math.PI / 2);
190
191        // --- Fit annotation (Tau/R² or "Fit failed") ---
192        g2.setColor(fitFailed ? Color.RED.darker() : new Color(0, 128, 0));
193        String msg = fitFailed
194                ? "Fit failed"
195                : String.format("Tau = %.1f ms, R² = %.3f",
196                                result.getTau(), result.getRSquared());
197        g2.setFont(g2.getFont().deriveFont(Font.BOLD, 14f));
198        g2.drawString(msg, marginLeft + 10, marginTop + 20);
199
200        // --- Cleanup and return image ---
201        g2.dispose();
202        return img;
203    }

205    /**
206     * Exports histograms of the experimental data to a PDF file. Each recording of the
207     * experiment is represented as a histogram in a separate page within the generated PDF.
208     *
209     * @param experiment The experiment containing recordings with track count data to be exported.
210     * @param outputFile The output {@code Path} where the generated PDF file will be saved.
211     * @throws IOException If an error occurs during the creation or saving of the PDF file.
212     */
213    public static void exportExperimentHistogramsToPdf(Experiment experiment, Path outputFile) throws IOException {
214        try (PDDocument doc = new PDDocument()) {
215            for (Recording recording : experiment.getRecordings()) {
216                List<Square> squares = recording.getSquaresOfRecording();
217                if (squares == null || squares.isEmpty()) {
218                    System.out.printf("Recording '%s': no squares%n", recording.getRecordingName());
219                    continue;
220                }
221
222                SquareUtils.BackgroundEstimationResult backgroundResult = calculateBackgroundDensity(squares);
223                Set<Square> backgroundSet = new HashSet<>(backgroundResult.getBackgroundSquares());
224                double backgroundTracksPerSquare = backgroundResult.getBackgroundMean();
225
226                int totalSquares = squares.size();
227                int nBackground  = backgroundSet.size();
228                int totalTracks  = squares.stream().mapToInt(Square::getNumberOfTracks).sum();
229                int backgroundTracksTotal = backgroundSet.stream().mapToInt(Square::getNumberOfTracks).sum();
230
231                int maxTracks = squares.stream().mapToInt(Square::getNumberOfTracks).max().orElse(0);
232                int binSize   = Math.max(1, maxTracks / 20);
233                int binCount  = (maxTracks / binSize) + 1;
234
235                int[] allBins = new int[binCount];
236                int[] bgBins  = new int[binCount];
237
238                for (Square sq : squares) {
239                    int n   = sq.getNumberOfTracks();
240                    int bin = Math.min(n / binSize, binCount - 1);
241                    allBins[bin]++;
242                    if (backgroundSet.contains(sq)) {
243                        bgBins[bin]++;
244                    }
245                }
246
247                PDPage page = new PDPage(PDRectangle.A4);
248                doc.addPage(page);
249
250                float pageWidth  = page.getMediaBox().getWidth();
251                float pageHeight = page.getMediaBox().getHeight();
252
253                int plotWidth  = 800;
254                int plotHeight = 600;
255
256                PdfBoxGraphics2D g2 = new PdfBoxGraphics2D(doc, plotWidth, plotHeight);
257
258                drawHistogram(g2,
259                              plotWidth,
260                              plotHeight,
261                              allBins,
262                              bgBins,
263                              binSize,
264                              recording.getRecordingName(),
265                              totalSquares,
266                              totalTracks,
267                              nBackground,
268                              backgroundTracksTotal,
269                              backgroundTracksPerSquare);
270
271                g2.dispose();
272                PDFormXObject formXObject = g2.getXFormObject();
273
274                float margin  = 36f;
275                float maxW    = pageWidth  - 2 * margin;
276                float maxH    = pageHeight - 2 * margin;
277                float scale   = Math.min(Math.min(maxW / plotWidth, maxH / plotHeight), 1.0f);
278                float scaledW = plotWidth  * scale;
279                float scaledH = plotHeight * scale;
280
281                float offsetX = (pageWidth  - scaledW) / 2f;
282                float offsetY = (pageHeight - scaledH) / 2f;
283
284                try (PDPageContentStream contentStream =
285                             new PDPageContentStream(doc, page, PDPageContentStream.AppendMode.APPEND, true, true)) {
286                    contentStream.saveGraphicsState();
287                    contentStream.transform(org.apache.pdfbox.util.Matrix.getTranslateInstance(offsetX, offsetY));
288                    contentStream.transform(org.apache.pdfbox.util.Matrix.getScaleInstance(scale, scale));
289                    contentStream.drawForm(formXObject);
290                    contentStream.restoreGraphicsState();
291                }
292                PaintLogger.debugf("Added histogram for recording '%s'%n", recording.getRecordingName());
293            }
294
295            PaintLogger.debugf("Added %d pages to %s%n", doc.getNumberOfPages(), outputFile);
296            doc.save(outputFile.toFile());
297        }
298    }

300    /**
301     * Exports histograms of the experimental data as individual PNG images.
302     * Each recording is rendered to a separate file under:
303     *    Output/Background Plots/<recordingName>.png
304     *
305     * @param experiment The experiment containing recordings with track count data.
306     * @param experimentPath The root directory of the experiment (used to resolve Output path).
307     * @throws IOException If any image cannot be written.
308     */
309    public static void exportExperimentHistogramsToPngs(Experiment experiment, Path experimentPath) throws IOException {
310        Path outputDir = experimentPath.resolve("Output").resolve("Background Plots");
311        Files.createDirectories(outputDir);
312
313        for (Recording recording : experiment.getRecordings()) {
314            List<Square> squares = recording.getSquaresOfRecording();
315            if (squares == null || squares.isEmpty()) {
316                PaintLogger.debugf("Recording '%s': no squares%n", recording.getRecordingName());
317                continue;
318            }
319
320            SquareUtils.BackgroundEstimationResult backgroundResult = calculateBackgroundDensity(squares);
321            Set<Square> backgroundSet = new HashSet<>(backgroundResult.getBackgroundSquares());
322            double backgroundTracksPerSquare = backgroundResult.getBackgroundMean();
323
324            int totalSquares = squares.size();
325            int nBackground  = backgroundSet.size();
326            int totalTracks  = squares.stream().mapToInt(Square::getNumberOfTracks).sum();
327            int backgroundTracksTotal = backgroundSet.stream().mapToInt(Square::getNumberOfTracks).sum();
328
329            int maxTracks = squares.stream()
330                    .mapToInt(Square::getNumberOfTracks).max().orElse(0);
331            int binSize   = Math.max(1, maxTracks / 20);
332            int binCount  = (maxTracks / binSize) + 1;
333
334            int[] allBins = new int[binCount];
335            int[] bgBins  = new int[binCount];
336
337            for (Square sq : squares) {
338                int n   = sq.getNumberOfTracks();
339                int bin = Math.min(n / binSize, binCount - 1);
340                allBins[bin]++;
341                if (backgroundSet.contains(sq)) {
342                    bgBins[bin]++;
343                }
344            }
345
346            int plotWidth  = 900;
347            int plotHeight = 600;
348            BufferedImage img = new BufferedImage(plotWidth, plotHeight, BufferedImage.TYPE_INT_ARGB);
349            Graphics2D g2 = img.createGraphics();
350            drawHistogram(g2,
351                          plotWidth,
352                          plotHeight,
353                          allBins,
354                          bgBins,
355                          binSize,
356                          recording.getRecordingName(),
357                          totalSquares,
358                          totalTracks,
359                          nBackground,
360                          backgroundTracksTotal,
361                          backgroundTracksPerSquare);
362            g2.dispose();
363
364            Path outFile = outputDir.resolve(recording.getRecordingName() + ".png");
365            ImageIO.write(img, "png", outFile.toFile());
366            PaintLogger.debugf("Wrote background histogram to %s%n", outFile);
367        }
368    }

372    /**
373     * Saves a plot visualizing the Tau-fit result for a square as a PNG file under
374     * “Success” or “Failed”, within the current experiment’s output directory.
375     *
376     * @param tracks         the list of tracks used to compute the frequency distribution
377     *                       (their durations will form the x-axis)
378     * @param tauResult      the result of the Tau calculation (may indicate failure)
379     * @param experimentPath the base path of the experiment directory
380     * @param recordingName  the name of the recording
381     * @param squareIndex    the index of the square region (used to name the file)
382     */
383    public static void saveTauFitPlot(List<Track> tracks,
384                                      CalculateTau.CalculateTauResult tauResult,
385                                      Path experimentPath,
386                                      String recordingName,
387                                      int squareIndex) {
388        boolean fitFailed = (tauResult == null ||
389                tauResult.getStatus() != CalculateTau.CalculateTauResult.Status.TAU_SUCCESS ||
390                !Double.isFinite(tauResult.getTau()) ||
391                !Double.isFinite(tauResult.getRSquared()));
392
393        Map<Double, Integer> freqMap = new TreeMap<>();
394        for (Track track : tracks) {
395            double d = track.getTrackDuration();
396            freqMap.put(d, freqMap.getOrDefault(d, 0) + 1);
397        }
398
399        double[] x = new double[freqMap.size()];
400        double[] y = new double[freqMap.size()];
401        int idx = 0;
402        for (Map.Entry<Double, Integer> e : freqMap.entrySet()) {
403            x[idx] = e.getKey();
404            y[idx] = e.getValue();
405            idx++;
406        }
407
408        BufferedImage img = renderTauPlot(x, y, tauResult, fitFailed, 900, 600);
409
410        Path rootDir   = experimentPath.resolve("Output").resolve("Tau Fitting Plots");
411        Path targetDir = rootDir.resolve(fitFailed ? "Failed" : "Success");
412
413        try {
414            Files.createDirectories(targetDir);
415            Path plotPath = targetDir.resolve(String.format("%s_square_%03d.png", recordingName, squareIndex));
416            ImageIO.write(img, "png", plotPath.toFile());
417        } catch (Exception e) {
418            // Log failure but do *not* throw
419            PaintLogger.errorf("Failed to save Tau-fit plot for '%s' square %03d: %s",
420                               recordingName, squareIndex, e.getMessage());
421        }
422    }

424    // ========== Helper method: drawHistogram ==========
425
426    private static void drawHistogram(Graphics2D g2,
427                                      int w,
428                                      int h,
429                                      int[] allBins,
430                                      int[] bgBins,
431                                      int binSize,
432                                      String title,
433                                      int totalSquares,
434                                      int totalTracks,
435                                      int nBackground,
436                                      int backgroundTracksTotal,
437                                      double avgTracksInBackground) {
438        g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
439                            RenderingHints.VALUE_ANTIALIAS_ON);
440        g2.setColor(Color.WHITE);
441        g2.fillRect(0, 0, w, h);
442
443        int marginLeft   = 70;
444        int marginBottom = 50;
445        int marginTop    = 50;
446        int marginRight  = 40;
447
448        int binCount     = allBins.length;
449        int maxBinCount = Arrays.stream(allBins).max().orElse(1);
450
451        double xStep  = (double)(w - marginLeft - marginRight) / binCount;
452        double yScale = (double)(h - marginTop - marginBottom) / maxBinCount;
453
454        g2.setColor(Color.GRAY);
455        int x0 = marginLeft;
456        int y0 = h - marginBottom;
457        g2.drawLine(x0, y0, w - marginRight, y0);
458        g2.drawLine(x0, y0, x0, marginTop);
459
460        g2.setColor(Color.BLACK);
461        int nTicks = 5;
462        for (int i = 0; i <= nTicks; i++) {
463            int y = y0 - (i * (h - marginTop - marginBottom) / nTicks);
464            int value = (int)Math.round(i * (double) maxBinCount / nTicks);
465            g2.drawLine(x0 - 5, y, x0, y);
466            g2.drawString(String.valueOf(value), x0 - 45, y + 5);
467        }
468
469        for (int i = 0; i < binCount; i++) {
470            int total      = allBins[i];
471            int background = bgBins[i];
472            int x          = (int)(x0 + i * xStep);
473            int barW       = (int)(xStep * 0.8);
474
475            if (background > 0) {
476                int bgHeight = (int)(background * yScale);
477                int bgY      = y0 - bgHeight;
478                g2.setColor(new Color(100,160,255,180));
479                g2.fillRect(x, bgY, barW, bgHeight);
480            }
481
482            int fgHeight = (int)((total - background) * yScale);
483            if (fgHeight > 0) {
484                int fgY = y0 - fgHeight - (int)(background * yScale);
485                g2.setColor(new Color(180,180,180));
486                g2.fillRect(x, fgY, barW, fgHeight);
487            }
488
489            g2.setColor(Color.BLACK);
490            g2.drawRect(x, y0 - (int)(total * yScale), barW, (int)(total * yScale));
491        }
492
493        g2.setFont(new Font("SansSerif", Font.PLAIN, 12));
494        g2.drawString("Track count bins (bin size ≈ " + binSize + ")", w / 2 - 90, h - 15);
495        g2.drawString("Number of squares", 10, marginTop - 10);
496
497        g2.setFont(new Font("SansSerif", Font.BOLD, 16));
498        g2.drawString("Track Count Histogram – " + title, w / 2 - 150, marginTop - 20);
499
500        g2.setFont(new Font("SansSerif", Font.BOLD, 14));
501        g2.setColor(new Color(0,70,180));
502        int textX = w - 370;
503        int textY = marginTop + 20;
504
505        g2.drawString(String.format("Number of squares in recording             : %d", totalSquares),      textX, textY);
506        g2.drawString(String.format("Number of tracks in recording              : %d", totalTracks),     textX, textY + 20);
507        g2.drawString(String.format("Number of background squares               : %d", nBackground),      textX, textY + 40);
508        g2.drawString(String.format("Number of tracks in background             : %d", backgroundTracksTotal), textX, textY + 60);
509        g2.drawString(String.format("Average number of tracks in the background : %.3f", avgTracksInBackground), textX, textY + 80);
510    }
