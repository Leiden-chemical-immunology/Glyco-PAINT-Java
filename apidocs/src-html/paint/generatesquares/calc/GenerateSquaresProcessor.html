071    /**
072     * Processes an experiment to generate square regions for each recording, compute attributes,
073     * and compile data tables for all squares and tracks. The method applies geometric segmentations,
074     * assigns tracks to the generated squares, and calculates additional square and recording-level attributes.
075     * Finally, it writes compiled results to the file system.
076     *
077     * @param project        the project containing configurations and experiment data
078     * @param experimentName the name of the experiment to process
079     */
080    public static void generateSquaresForExperiment(Project project, String experimentName) {
081        GenerateSquaresConfig generateSquaresConfig = project.getGenerateSquaresConfig();
082        Experiment            experiment            = null;
083        List<Recording>       recordings;
084
085        Path    projectPath       = project.getProjectRootPath();
086        boolean plotFittingCurves = getBoolean("Generate Squares", "Plot Curve Fitting", false);
087
088        LocalDateTime start = LocalDateTime.now();
089        PaintLogger.debugf("Loading Experiment '%s'", experimentName);
090
091        // Early Exit if the user cancelled before we start
092        if (Thread.currentThread().isInterrupted()) {
093            PaintLogger.infof("Cancelled before starting experiment %s", experimentName);
094            return;
095        }
096        try {
097            experiment = loadExperiment(project.getProjectRootPath(), experimentName, false);
098        } catch (Exception e) {
099            PaintLogger.errorf("Failed to load Experiment '%s'", experimentName);
100        }
101
102        if (experiment == null) {
103            PaintLogger.errorf("Failed to load experiment: %s", experimentName);
104            return;
105        }
106
107        PaintLogger.infof("Starting processing experiment '%s'", experimentName);
108
109        recordings = experiment.getRecordings();
110        for (Recording recording : recordings) {
111
112            // CHECK before starting each recording
113            if (Thread.currentThread().isInterrupted()) {
114                PaintLogger.infof("Cancelled before processing recording %s", recording.getRecordingName());
115                return;
116            }
117            PaintLogger.infof("   Processing: %s", recording.getRecordingName());
118            PaintLogger.debugf(recording.toString());
119
120            // Create the squares with basic geometric information
121            List<Square> squares = generateSquaresForRecording(recording, generateSquaresConfig);
122            recording.setSquaresOfRecording(squares);
123
124            // Assign the recording tracks to the squares
125            assignTracksToSquares(recording);
126
127            // CHECK mid-work before calculating attributes
128            if (Thread.currentThread().isInterrupted()) {
129                PaintLogger.infof("Cancelled before attribute calculation for %s", recording.getRecordingName());
130                return;
131            }
132
133            // Calculate square-level and recording-level attributes
134            Path experimentPath = project.getProjectRootPath().resolve(experiment.getExperimentName());
135            CalculateSquareAttributes.calculateSquareAttributes(experimentPath, recording, generateSquaresConfig);
136            CalculateSquareAttributes.calculateRecordingAttributes(recording, generateSquaresConfig);
137        }
138
139        Duration duration = Duration.between(start, LocalDateTime.now());
140        PaintLogger.infof("Finished processing experiment '%s' in %s", experimentName, formatDuration(duration));
141        PaintLogger.blankline();
142
143        // CHECK before writing output files
144        if (Thread.currentThread().isInterrupted()) {
145            PaintLogger.infof("Cancelled before writing output for %s", experimentName);
146            return;
147        }
148
149        // Compile all squares and write
150        Table allSquaresTable = compileAllSquares(experiment);
151        Path experimentPath = project.getProjectRootPath().resolve(experiment.getExperimentName());
152        writeAllSquares(experimentPath, allSquaresTable);
153
154        // Write recordings
155        writeAllRecordings(experimentPath, experiment.getRecordings());
156
157        // All Tracks
158        Table allTracksTable = compileAllTracks(experiment);
159        allTracksTable = allTracksTable.sortOn("Recording Name", "Track Id");
160        writeAllTracks(experimentPath, allTracksTable);
161
162    }

164    /**
165     * Generates a list of {@code Square} objects for the given recording. Each square corresponds to a
166     * segment of the recording area based on the configuration provided.
167     *
168     * @param recording             the recording for which squares are to be generated
169     * @param generateSquaresConfig the configuration specifying the number of squares and related parameters
170     * @return a list of {@code Square} objects representing the segmented areas of the recording
171     */
172    public static List<Square> generateSquaresForRecording(Recording recording, GenerateSquaresConfig generateSquaresConfig) {
173
174        // Total number of squares per recording.
175        int numberOfSquaresInRecording = generateSquaresConfig.getNumberOfSquaresInRecording();
176        numberOfSquaresInOneDimension = (int) Math.sqrt(numberOfSquaresInRecording);   // Number of squares in one dimension (e.g., 20 for 20x20).
177
178        
179        List<Square> squares = new ArrayList<>();
180        double squareWidth   = IMAGE_WIDTH / numberOfSquaresInOneDimension;
181        double squareHeight  = IMAGE_HEIGHT / numberOfSquaresInOneDimension;
182        
183
184        int squareNumber = 0;
185        for (int rowNumber = 0; rowNumber < numberOfSquaresInOneDimension; rowNumber++) {
186            for (int columnNumber = 0; columnNumber < numberOfSquaresInOneDimension; columnNumber++) {
187                double X0 = columnNumber * squareWidth;
188                double Y0 = rowNumber * squareHeight;
189                double X1 = (columnNumber + 1) * squareWidth;
190                double Y1 = (rowNumber + 1) * squareHeight;
191
192                squares.add(new Square(
193                        recording.getRecordingName() + '-' + squareNumber,
194                        recording.getExperimentName(),
195                        recording.getRecordingName(),
196                        squareNumber,
197                        rowNumber,
198                        columnNumber,
199                        X0,
200                        Y0,
201                        X1,
202                        Y1));
203
204                squareNumber++;
205            }
206        }
207        return squares;
208    }

210    /**
211     * Assigns tracks to the predefined square regions of a recording.
212     * It processes the tracks table of the recording, assigns each track to the relevant square,
213     * updates the square attributes, and compiles a complete tracks table for the recording.
214     *
215     * @param recording the {@code Recording} instance containing track and square data.
216     *                  The method modifies this object by assigning tracks to the corresponding squares
217     *                  and updating their track-related attributes.
218     */
219    public static void assignTracksToSquares(Recording recording) {
220        Table tracksOfRecording   = recording.getTracksTable();
221        TrackTableIO trackTableIO = new TrackTableIO();
222        Table recordingTrackTable = trackTableIO.emptyTable();
223
224        int lastRowCol            = numberOfSquaresInOneDimension - 1;
225        int labelNumber           = 0;
226        int incrementalTrackCount = 0;
227
228        PaintLogger.debugf("Assigning tracks to squares (%d total tracks)", tracksOfRecording.rowCount());
229        for (Square square : recording.getSquaresOfRecording()) {
230
231            Table squareTracksTable = filterTracksInSquare(tracksOfRecording, square, lastRowCol);
232            incrementalTrackCount += squareTracksTable.rowCount();
233
234            if (squareTracksTable.rowCount() == 0) {
235                square.setTracks(Collections.emptyList());
236                square.setTracksTable(squareTracksTable);
237                square.setNumberOfTracks(0);
238                continue;
239            }
240
241            // Convert rows to Track entities
242            List<Track> tracks = trackTableIO.toEntities(squareTracksTable);
243
244            // Update the fields on each Track
245            for (Track track : tracks) {
246                track.setSquareNumber(square.getSquareNumber());
247                track.setLabelNumber(labelNumber);
248            }
249
250            // Rebuild the table from the modified tracks (ensures the table reflects the updates)
251            Table updatedSquareTracks = trackTableIO.toTable(tracks);
252
253            // Append updated tracks into the global recording table
254            recordingTrackTable.append(updatedSquareTracks);
255
256            // Update the square
257            square.setTracks(tracks);
258            square.setTracksTable(updatedSquareTracks);
259            square.setNumberOfTracks(tracks.size());
260
261            // Log info
262            PaintLogger.debugf("Square %d: %d tracks assigned (label %d)",
263                               square.getSquareNumber(), tracks.size(), labelNumber);
264
265            labelNumber++;
266        }
267        PaintLogger.debugf("assignTracksToSquare - The numbers of tracks assigned is %d  the recording is %s", incrementalTrackCount, tracksOfRecording.rowCount());
268
269        // Update the recording table
270        recording.setTracksTable(recordingTrackTable);
271
272        PaintLogger.debugf("✅ Total %d tracks assigned to %d squares.",
273                           recordingTrackTable.rowCount(), recording.getSquaresOfRecording().size());
274    }

276    /**
277     * Compiles all square data from the recordings in the specified experiment into a single table.
278     * The method iterates through each recording in the experiment, retrieves its square data,
279     * and appends it to a cumulative table. If a recording does not have square data available,
280     * an error is logged.
281     *
282     * @param experiment the experiment containing recordings whose square data is to be combined
283     * @return a {@code Table} containing the aggregated square data from all recordings in the experiment,
284     *         or an empty table if no square data exists
285     */
286    private static Table compileAllSquares(Experiment experiment) {
287        SquareTableIO squaresTableIO = new SquareTableIO();
288        Table allSquaresTable = squaresTableIO.emptyTable();
289
290        for (Recording recording : experiment.getRecordings()) {
291            Table table = squaresTableIO.toTable(recording.getSquaresOfRecording());
292            if (table != null) {
293                squaresTableIO.appendInPlace(allSquaresTable, table);
294            } else {
295                PaintLogger.errorf("compileAllSquares - squares table does not exist for '%s'", recording.getRecordingName());
296            }
297        }
298        return allSquaresTable;
299    }
300
301    /**
302     * Compiles all track data from the recordings in the specified experiment into a single table.
303     * The method iterates through each recording in the experiment, retrieves its track data,
304     * and appends it to an aggregate table. If a recording does not have track data available,
305     * an error is logged.
306     *
307     * @param experiment the experiment containing recordings whose track data is to be combined
308     * @return a {@code Table} containing the aggregated track data from all recordings in the experiment,
309     *         or an empty table if no track data exists
310     */
311    private static Table compileAllTracks(Experiment experiment) {
312        TrackTableIO trackTableIO = new TrackTableIO();
313        Table allTracksTable = trackTableIO.emptyTable();
314
315        for (Recording recording : experiment.getRecordings()) {
316            PaintLogger.debugf("Processing squares for experiment '%s'  - recording '%s'", experiment.getExperimentName(), recording.getRecordingName());
317            Table table = recording.getTracksTable();
318            if (table != null) {
319                trackTableIO.appendInPlace(allTracksTable, table);
320            } else {
321                PaintLogger.errorf("compileAllSquares - squares table does not exist for '%s'", recording.getRecordingName());
322            }
323        }
324        return allTracksTable;
325    }
