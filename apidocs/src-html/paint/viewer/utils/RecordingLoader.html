007 *    project directory, constructing {@link RecordingEntry} instances that

009 *
010 *  DESCRIPTION:
011 *    The {@code RecordingLoader} iterates over all experiment folders within
012 *    a project, loads each experiment using {@link paint.shared.io.ExperimentDataLoader},
013 *    and constructs {@link RecordingEntry} objects for all recordings that meet
014 *    process and file-availability requirements.
015 *
016 *    Each recording entry combines metadata, images (TrackMate and Brightfield),
017 *    and configuration thresholds loaded from {@link paint.shared.config.PaintConfig}.
018 *    Invalid or incomplete recordings are skipped with diagnostic logging.
019 *
020 *  KEY FEATURES:
021 *    • Loads complete experiments with squares and track data.
022 *    • Validates existence of TrackMate and Brightfield images.
023 *    • Reads density, variability, and R² thresholds from configuration.
024 *    • Constructs structured {@link RecordingEntry} objects for UI or analysis use.
025 *    • Provides detailed logging via {@link paint.shared.utils.PaintLogger}.

040package paint.viewer.utils;
041
042import paint.shared.config.PaintConfig;
043import paint.shared.io.ExperimentDataLoader;
044import paint.shared.objects.Experiment;
045import paint.shared.objects.Project;
046import paint.shared.objects.Recording;
047import paint.shared.utils.PaintLogger;
048
049import java.nio.file.Files;
050import java.nio.file.Path;
051import java.util.ArrayList;
052import java.util.List;
053
054/**
055 * Provides functionality for loading and filtering recordings from a project directory.
056 * <p>
057 * The {@code RecordingLoader} ensures that required images and metadata exist for each
058 * recording, applying configuration-defined thresholds to build {@link RecordingEntry}
059 * instances ready for visualization or further processing.
060 * </p>
061 */
062public class RecordingLoader {
063
064    /**
065     * Loads all {@link RecordingEntry} instances from the specified {@link Project}.
066     * <p>
067     * This method iterates over all experiment directories, loads their full content
068     * (including squares and tracks), and constructs valid entries only for recordings
069     * meeting both process requirements and file availability conditions.
070     * </p>
071     *
072     * @param project the project context containing experiments, recordings, and file data
073     * @return list of valid {@link RecordingEntry} instances; empty if no valid recordings found
074     */
075    public static List<RecordingEntry> loadFromProject(Project project) {
076
077        List<RecordingEntry> recordingEntries = new ArrayList<>();
078
079        for (String experimentName : project.getExperimentNames()) {
080
081            Path experimentPath = project.getProjectRootPath().resolve(experimentName);
082
083            // ✅ Load full experiment (recordings + tracks + squares)
084            Experiment experiment = ExperimentDataLoader.loadExperiment(
085                    project.getProjectRootPath(),
086                    experimentName,
087                    true // matureProject: includes squares + tracks
088            );
089
090            if (experiment == null || experiment.getRecordings().isEmpty()) {
091                continue;
092            }
093
094            for (Recording recording : experiment.getRecordings()) {
095                String recordingName = recording.getRecordingName();
096                if (!recording.isProcessFlag()) {
097                    continue;
098                }
099
100                // --- Image paths ---
101                Path trackmateImagePath = experimentPath
102                        .resolve("TrackMate Images")
103                        .resolve(recordingName + ".jpg");
104
105                if (!Files.exists(trackmateImagePath)) {
106                    PaintLogger.errorf("Missing TrackMate image for '%s'", recordingName);
107                    continue;
108                }
109
110                Path brightfieldDirPath = experimentPath.resolve("BrightField Images");
111                Path brightfieldImagePath = null;
112
113                if (!Files.isDirectory(brightfieldDirPath)) {
114                    PaintLogger.errorf("Missing Brightfield directory '%s' image for recording '%s'",
115                                       brightfieldDirPath, recordingName);
116                    continue;
117                }
118
119                try {
120                    for (Path p : (Iterable<Path>) Files.list(brightfieldDirPath)::iterator) {
121                        String fileName = p.getFileName().toString();
122                        if ((fileName.startsWith(recordingName + "-BF") || fileName.startsWith(recordingName))
123                                && fileName.endsWith(".jpg")) {
124                            brightfieldImagePath = p;
125                            break;
126                        }
127                    }
128                } catch (Exception e) {
129                    e.printStackTrace();
130                }
131
132                if (brightfieldImagePath == null) {
133                    PaintLogger.errorf("Missing BrightField image for '%s'", recordingName);
134                    continue;
135                }
136
137                // --- Thresholds from config ---
138                double minDensityRatio = PaintConfig.getDouble("Generate Squares", "Min Required Density Ratio", 2.0);
139                double maxVariability  = PaintConfig.getDouble("Generate Squares", "Max Allowable Variability",  10.0);
140                double minRSquared     = PaintConfig.getDouble("Generate Squares", "Min Required R Squared",     0.1);
141                String neighbourMode   = PaintConfig.getString("Generate Squares", "Neighbour Mode",             "Free");
142
143                // --- Build final entry ---
144                RecordingEntry entry = new RecordingEntry(
145                        recording,
146                        trackmateImagePath,
147                        brightfieldImagePath,
148                        experimentName,
149                        minDensityRatio,
150                        maxVariability,
151                        minRSquared,
152                        neighbourMode
153                );
154
155                recordingEntries.add(entry);
156            }
157        }
158
159        return recordingEntries;
160    }
161}
