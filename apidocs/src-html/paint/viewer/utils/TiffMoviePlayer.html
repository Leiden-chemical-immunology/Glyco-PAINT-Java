037package paint.viewer.utils;
038
039import ij.IJ;
040import ij.ImagePlus;
041import ij.measure.Calibration;
042import ij.process.ImageProcessor;
043
044import javax.swing.*;
045import javax.swing.border.EmptyBorder;
046import java.awt.*;
047import java.awt.image.BufferedImage;
048import java.io.File;
049import java.io.OutputStream;
050import java.io.PrintStream;
051
052/**
053 * Provides functionality for playing multi-frame TIFF files as movie-like sequences.
054 * <p>
055 * The {@code TiffMoviePlayer} uses ImageJ for image handling and Swing for the GUI.
056 * It includes playback controls such as play/pause, speed adjustment, and frame navigation.
057 * </p>
058 */
059public class TiffMoviePlayer {
060
061    /**
062     * Plays a multi-frame TIFF file as a movie sequence.
063     *
064     * @param tiffPath the absolute or relative file path of the TIFF image stack
065     */
066    public void playMovie(String tiffPath) {
067        final String fileName = new File(tiffPath).getName();
068
069        // --- Simple static loading dialog (no progress bar) ---
070        final JDialog loadingDialog = new JDialog((Frame) null, "Loading Recording", false);
071        loadingDialog.setDefaultCloseOperation(JDialog.DO_NOTHING_ON_CLOSE);
072        loadingDialog.setLayout(new BorderLayout());
073
074        JLabel label = new JLabel("Loading " + fileName + "…", SwingConstants.CENTER);
075        label.setFont(new Font("SansSerif", Font.PLAIN, 13));
076        label.setBorder(new EmptyBorder(25, 20, 25, 20));
077
078        loadingDialog.add(label, BorderLayout.CENTER);
079        loadingDialog.setSize(300, 120);
080        loadingDialog.setResizable(false);
081        loadingDialog.setLocationRelativeTo(null);
082
083        SwingUtilities.invokeLater(() -> loadingDialog.setVisible(true));
084
085        // --- Load + show UI on background thread ---
086        new Thread(() -> {
087            System.setProperty("apple.awt.UIElement", "true");
088            IJ.redirectErrorMessages();
089            IJ.showStatus("");
090
091            // Silence ImageJ console
092            PrintStream originalOut = System.out;
093            System.setOut(new PrintStream(new OutputStream() {
094                @Override
095                public void write(int b) {
096                }
097            }));
098
099            final ImagePlus imp = IJ.openImage(tiffPath);
100
101            System.setOut(originalOut);
102            SwingUtilities.invokeLater(loadingDialog::dispose);
103
104            if (imp == null) {
105                SwingUtilities.invokeLater(() -> JOptionPane.showMessageDialog(
106                        null,
107                        "Failed to open image file:\n" + tiffPath,
108                        "Error",
109                        JOptionPane.ERROR_MESSAGE
110                ));
111                return;
112            }
113
114            IJ.run(imp, "Enhance Contrast", "saturated=0.35");
115
116            int delay = 50;
117            Calibration cal = imp.getCalibration();
118            if (cal != null && cal.frameInterval > 0) {
119                delay = (int) Math.round(cal.frameInterval * 1000);
120            }
121            final int baseDelayMs = delay;
122
123            SwingUtilities.invokeLater(() -> {
124                final JFrame frame = new JFrame("Movie Player - " + fileName);
125                frame.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
126                frame.setLayout(new BorderLayout());
127
128                final JLabel imageLabel = new JLabel("", SwingConstants.CENTER);
129                imageLabel.setOpaque(true);
130                imageLabel.setBackground(Color.DARK_GRAY);
131
132                JPanel imagePanel = new JPanel(new BorderLayout());
133                imagePanel.setBackground(Color.DARK_GRAY);
134                imagePanel.setBorder(new EmptyBorder(4, 4, 4, 4));
135                imagePanel.add(imageLabel, BorderLayout.CENTER);
136                frame.add(imagePanel, BorderLayout.CENTER);
137
138                final int totalFrames         = imp.getStackSize();
139                final JSlider frameSlider     = new JSlider(1, totalFrames, 1);
140                final JLabel frameLabel       = new JLabel("Frame: 1");
141                final JButton playPauseButton = new JButton("⏸ Pause");
142                final JSlider speedSlider     = new JSlider(50, 400, 100);
143                final JLabel speedLabel       = new JLabel("Speed: 1.0×");
144
145                JPanel controls = new JPanel(new GridLayout(2, 1, 0, 3));
146                controls.setBorder(new EmptyBorder(6, 6, 6, 6));
147                controls.setBackground(new Color(245, 245, 245));
148
149                JPanel topRow = new JPanel(new FlowLayout(FlowLayout.CENTER, 8, 0));
150                topRow.setBackground(controls.getBackground());
151                frameSlider.setPreferredSize(new Dimension(260, 25));
152                topRow.add(playPauseButton);
153                topRow.add(frameSlider);
154                topRow.add(frameLabel);
155
156                JPanel bottomRow = new JPanel(new FlowLayout(FlowLayout.CENTER, 8, 0));
157                bottomRow.setBackground(controls.getBackground());
158                speedSlider.setPreferredSize(new Dimension(100, 25));
159                speedSlider.setMajorTickSpacing(50);
160                speedSlider.setPaintTicks(true);
161                bottomRow.add(speedLabel);
162                bottomRow.add(speedSlider);
163
164                controls.add(topRow);
165                controls.add(bottomRow);
166
167                // --- Add Close button below controls ---
168                JButton closeButton = new JButton("Close");
169                closeButton.addActionListener(e -> {
170                    frame.dispose(); // closes the window
171                });
172                JPanel closePanel = new JPanel(new FlowLayout(FlowLayout.RIGHT, 8, 4));
173                closePanel.setBackground(controls.getBackground());
174                closePanel.add(closeButton);
175
176                JPanel bottomPanel = new JPanel(new BorderLayout());
177                bottomPanel.add(controls, BorderLayout.CENTER);
178                bottomPanel.add(closePanel, BorderLayout.SOUTH);
179
180                frame.add(bottomPanel, BorderLayout.SOUTH);
181
182                // --- Show frame ---
183                frame.setSize(Math.min(imp.getWidth() + 40, 1000),
184                              Math.min(imp.getHeight() + 180, 900));
185                frame.setLocationRelativeTo(null);
186                frame.setVisible(true);
187
188                final boolean[] playing = {true};
189                final int[] currentFrame = {1};
190
191                playPauseButton.addActionListener(e -> {
192                    playing[0] = !playing[0];
193                    playPauseButton.setText(playing[0] ? "⏸ Pause" : "▶️ Play");
194                });
195
196                speedSlider.addChangeListener(e -> {
197                    double raw = speedSlider.getValue() / 100.0;
198                    double speed = Math.round(raw * 2) / 2.0;
199                    speedLabel.setText(String.format("Speed: %.1fx", speed));
200                });
201
202                frameSlider.addChangeListener(e -> {
203                    if (!frameSlider.getValueIsAdjusting() && !playing[0]) {
204                        final int frameIndex = frameSlider.getValue();
205                        ImageProcessor ip2 = imp.getStack().getProcessor(frameIndex);
206                        final BufferedImage img2 = ip2.getBufferedImage();
207                        SwingUtilities.invokeLater(() -> {
208                            imageLabel.setIcon(new ImageIcon(img2));
209                            frameLabel.setText("Frame: " + frameIndex);
210                        });
211                        currentFrame[0] = frameIndex;
212                    }
213                });
214
215                ImageProcessor ip0 = imp.getStack().getProcessor(1);
216                imageLabel.setIcon(new ImageIcon(ip0.getBufferedImage()));
217
218                new Thread(() -> {
219                    while (frame.isVisible()) {
220                        if (playing[0]) {
221                            int frameIdx = currentFrame[0];
222                            ImageProcessor ip = imp.getStack().getProcessor(frameIdx);
223                            final BufferedImage img = ip.getBufferedImage();
224                            final int finalFrameIdx = frameIdx;
225
226                            SwingUtilities.invokeLater(() -> {
227                                imageLabel.setIcon(new ImageIcon(img));
228                                frameLabel.setText("Frame: " + finalFrameIdx);
229                                frameSlider.setValue(finalFrameIdx);
230                            });
231
232                            double raw = speedSlider.getValue() / 100.0;
233                            double speed = Math.round(raw * 2) / 2.0;
234                            if (speed <= 0.0) {
235                                speed = 0.5;
236                            }
237                            long sleepTime = (long) (baseDelayMs / speed);
238                            if (sleepTime < 5) {
239                                sleepTime = 5;
240                            }
241
242                            try {
243                                Thread.sleep(sleepTime);
244                            } catch (InterruptedException ignored) {
245                            }
246
247                            currentFrame[0]++;
248                            if (currentFrame[0] > totalFrames) {
249                                currentFrame[0] = 1;
250                            }
251                        } else {
252                            try {
253                                Thread.sleep(100);
254                            } catch (InterruptedException ignored) {
255                            }
256                        }
257                    }
258                }, "TiffMoviePlaybackThread").start();
259            });
260        }, "TiffLoaderThread").start();
261    }
262
263    /**
264     * Manual test entry point for running the TIFF player independently.
265     */
266    public static void main(String[] args) {
267        SwingUtilities.invokeLater(() -> new TiffMoviePlayer().playMovie("/Volumes/Extreme Pro/Omero/221012/221012-Exp-3-A4-3.tif"));
268    }
