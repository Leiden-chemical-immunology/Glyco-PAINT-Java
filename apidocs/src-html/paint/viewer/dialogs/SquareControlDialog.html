020 *    and triggers file updates through the associated listener.
021 *
022 *  KEY FEATURES:
023 *    • Live preview of slider and neighbour mode changes.
024 *    • Modeless operation allowing real-time visual feedback.
025 *    • Apply buttons for different persistence scopes.

042
043import paint.viewer.panels.RecordingControlsPanel;
044import paint.viewer.panels.SquareGridPanel;

047import javax.swing.*;
048import javax.swing.event.ChangeEvent;
049import javax.swing.event.ChangeListener;
050import java.awt.*;
051import java.text.DecimalFormat;

053/**
054 * A dialog for configuring square control parameters in the viewer.
055 * <p>
056 * Provides interactive sliders and radio buttons that allow users to adjust
057 * key square filtering parameters. Changes are previewed live in the main
058 * viewer; pressing an Apply button commits and saves the configuration.
059 */
060public class SquareControlDialog extends JDialog {
061    private static final DecimalFormat ONE_DEC = new DecimalFormat("0.0");
062    private final JSlider                         densityRatioSlider;
063    private final JSlider                         variabilitySlider;
064    private final JSlider                         rSquaredSlider;
065    private final JLabel                          densityRatioValue;
066    private final JLabel                          variabilityValue;
067    private final JLabel                          rSquaredValue;
068    private final JRadioButton                    neighbourFree;
069    private final JRadioButton                    neighbourRelaxed;
070    private final JRadioButton                    neighbourStrict;
071    private final RecordingControlsPanel.Listener listener;
072    private final double                          origDensityRatio;
073    private final double                          origVariability;
074    private final double                          origRSquared;
075    private final String                          origNeighbourMode;
076    /**
077     * Constructs a dialog that enables interactive adjustment of square grid
078     * parameters such as Density Ratio, Variability, and R² thresholds.
079     *
080     * @param owner      the parent frame that owns this dialog
081     * @param gridPanel  the grid panel to update visually during preview
082     * @param listener   listener receiving apply and preview callbacks
083     * @param initParams the initial square control parameters
084     */
085    public SquareControlDialog(JFrame owner,
086                               SquareGridPanel gridPanel,
087                               RecordingControlsPanel.Listener listener,
088                               SquareControlParams initParams) {
089        super(owner, "Square Controls", false);
090        this.listener = listener;
091
092        setLayout(new BorderLayout(10, 10));
093
094        JPanel content = new JPanel();
095        content.setLayout(new BoxLayout(content, BoxLayout.Y_AXIS));
096
097        // ─────────────────────────────────────────────────────────────────────
098        // Sliders
099        // ─────────────────────────────────────────────────────────────────────
100        densityRatioSlider = createSlider(0, 20000, (int) Math.round(initParams.minRequiredDensityRatio * 10));
101        variabilitySlider  = createSlider(0, 200,   (int) Math.round(initParams.maxAllowableVariability * 10));
102        rSquaredSlider     = createSlider(0, 100,   (int) Math.round(initParams.minRequiredRSquared * 100));
103
104        densityRatioValue = new JLabel();
105        variabilityValue  = new JLabel();
106        rSquaredValue     = new JLabel();
107
108        JPanel slidersPanel = new JPanel(new GridLayout(1, 3, 15, 0));
109        slidersPanel.setBorder(BorderFactory.createTitledBorder("Square Filters"));
110        slidersPanel.add(wrapSlider(densityRatioSlider, "Min Density Ratio", densityRatioValue));
111        slidersPanel.add(wrapSlider(variabilitySlider, "Max Variability", variabilityValue));
112        slidersPanel.add(wrapSlider(rSquaredSlider, "Min R²", rSquaredValue));
113
114        // ─────────────────────────────────────────────────────────────────────
115        // Neighbour mode radio buttons
116        // ─────────────────────────────────────────────────────────────────────
117
118        neighbourFree    = new JRadioButton("Free");
119        neighbourRelaxed = new JRadioButton("Relaxed");
120        neighbourStrict  = new JRadioButton("Strict");
121
122        ButtonGroup neighbourGroup = new ButtonGroup();
123        neighbourGroup.add(neighbourFree);
124        neighbourGroup.add(neighbourRelaxed);
125        neighbourGroup.add(neighbourStrict);
126
127        switch (initParams.neighbourMode) {
128            case "Relaxed":
129                neighbourRelaxed.setSelected(true);
130                break;
131            case "Strict":
132                neighbourStrict.setSelected(true);
133                break;
134            default:
135                neighbourFree.setSelected(true);
136        }
137
138        JPanel neighbourPanel = new JPanel(new FlowLayout(FlowLayout.LEFT));
139        neighbourPanel.setBorder(BorderFactory.createTitledBorder("Neighbour Mode"));
140        neighbourPanel.add(neighbourFree);
141        neighbourPanel.add(neighbourRelaxed);
142        neighbourPanel.add(neighbourStrict);
143        neighbourPanel.setAlignmentX(Component.CENTER_ALIGNMENT);
144
145        // ─────────────────────────────────────────────────────────────────────
146        // Apply / Cancel controls
147        // ─────────────────────────────────────────────────────────────────────
148
149        JPanel applyPanel       = new JPanel(new FlowLayout(FlowLayout.LEFT));
150        applyPanel.setBorder(BorderFactory.createTitledBorder("Apply Changes"));
151        JButton applyRecording  = new JButton("Apply to Recording");
152        JButton applyExperiment = new JButton("Apply to Experiment");
153        JButton applyProject    = new JButton("Apply to Project");
154        JButton cancelButton    = new JButton("Cancel");
155
156        for (JButton b : new JButton[]{applyRecording, applyExperiment, applyProject, cancelButton}) {
157            b.setPreferredSize(new Dimension(180, 28));
158        }
159
160        applyPanel.add(applyRecording);
161        applyPanel.add(applyExperiment);
162        applyPanel.add(applyProject);
163        applyPanel.add(cancelButton);
164        applyPanel.setAlignmentX(Component.CENTER_ALIGNMENT);
165
166        // ─────────────────────────────────────────────────────────────────────
167        // Layout assembly
168        // ─────────────────────────────────────────────────────────────────────
169        content.add(slidersPanel);
170        content.add(Box.createVerticalStrut(10));
171        content.add(neighbourPanel);
172        content.add(Box.createVerticalStrut(10));
173        content.add(applyPanel);
174        add(content, BorderLayout.CENTER);
175
176        // ─────────────────────────────────────────────────────────────────────
177        // Listeners for live preview
178        // ─────────────────────────────────────────────────────────────────────
179        ChangeListener sliderListener = new ChangeListener() {
180            @Override
181            public void stateChanged(ChangeEvent e) {
182                updateValueLabels();
183                propagatePreview();
184                SwingUtilities.invokeLater(gridPanel::repaint);
185            }
186        };
187        densityRatioSlider.addChangeListener(sliderListener);
188        variabilitySlider.addChangeListener(sliderListener);
189        rSquaredSlider.addChangeListener(sliderListener);
190
191        neighbourFree.addActionListener(e -> propagatePreview());
192        neighbourRelaxed.addActionListener(e -> propagatePreview());
193        neighbourStrict.addActionListener(e -> propagatePreview());
194
195        // Apply button actions (commit and write to file)
196        applyRecording.addActionListener(e -> {
197            listener.onApplySquareControl("Recording", collectParams());
198            dispose();
199        });
200        applyExperiment.addActionListener(e -> {
201            listener.onApplySquareControl("Experiment", collectParams());
202            dispose();
203        });
204        applyProject.addActionListener(e -> {
205            listener.onApplySquareControl("Project", collectParams());
206            dispose();
207        });
208        cancelButton.addActionListener(e -> {
209            restoreOriginals();
210            dispose();
211        });
212
213        // ─────────────────────────────────────────────────────────────────────
214        // Preserve original values for cancel/restore
215        // ─────────────────────────────────────────────────────────────────────
216
217        origDensityRatio  = initParams.minRequiredDensityRatio;
218        origVariability   = initParams.maxAllowableVariability;
219        origRSquared      = initParams.minRequiredRSquared;
220        origNeighbourMode = initParams.neighbourMode;
221
222        updateValueLabels();
223        propagatePreview();
224
225        pack();
226        setLocationRelativeTo(owner);
227    }
228
229    /**
230     * Creates a vertical slider with labeled ticks.
231     */
232    private JSlider createSlider(int min, int max, int value) {
233        JSlider slider = new JSlider(JSlider.VERTICAL, min, max, Math.min(max, Math.max(min, value)));
234        slider.setMajorTickSpacing(Math.max(1, (max - min) / 5));
235        slider.setPaintTicks(true);
236        slider.setPaintLabels(true);
237
238        boolean isRSquared = (max == 100);
239        double divisor = isRSquared ? 100.0 : 10.0;
240
241        java.util.Hashtable<Integer, JLabel> table = new java.util.Hashtable<>();
242        int major = Math.max(1, (max - min) / 5);
243        for (int v = min; v <= max; v += major) {
244            String text = isRSquared
245                    ? ONE_DEC.format(v / divisor)
246                    : String.valueOf((int) Math.round(v / divisor));
247            table.put(v, new JLabel(text));
248        }
249        slider.setLabelTable(table);
250        return slider;
251    }
252
253    /**
254     * Wraps a slider and value label inside a titled panel.
255     */
256    private JPanel wrapSlider(JSlider slider, String title, JLabel valueLabel) {
257        JPanel panel = new JPanel(new BorderLayout());
258        panel.setBorder(BorderFactory.createTitledBorder(title));
259        panel.add(slider, BorderLayout.CENTER);
260        valueLabel.setHorizontalAlignment(SwingConstants.CENTER);
261        panel.add(valueLabel, BorderLayout.SOUTH);
262        return panel;
263    }
264
265    /**
266     * Updates the numeric labels to reflect current slider values.
267     */
268    private void updateValueLabels() {
269        densityRatioValue.setText(ONE_DEC.format(densityRatioSlider.getValue() / 10.0));
270        variabilityValue.setText(ONE_DEC.format(variabilitySlider.getValue() / 10.0));
271        rSquaredValue.setText(ONE_DEC.format(rSquaredSlider.getValue() / 100.0));
272    }
273
274    /**
275     * Sends live parameter updates for real-time preview without persistence.
276     */
277    private void propagatePreview() {
278        SquareControlParams params = collectParams();
279        listener.onApplySquareControl("Preview", params);
280    }
281
282    /**
283     * Returns the currently selected neighbour mode string.
284     */
285    private String getNeighbourMode() {
286        if (neighbourFree.isSelected()) {
287            return "Free";
288        }
289        if (neighbourRelaxed.isSelected()) {
290            return "Relaxed";
291        }
292        return "Strict";
293    }
294
295    /**
296     * Restores original slider and neighbour mode values, updating the preview.
297     */
298    private void restoreOriginals() {
299        densityRatioSlider.setValue((int) Math.round(origDensityRatio * 10));
300        variabilitySlider.setValue((int) Math.round(origVariability * 10));
301        rSquaredSlider.setValue((int) Math.round(origRSquared * 100));
302
303        switch (origNeighbourMode) {
304            case "Relaxed":
305                neighbourRelaxed.setSelected(true);
306                break;
307            case "Strict":
308                neighbourStrict.setSelected(true);
309                break;
310            default:
311                neighbourFree.setSelected(true);
312        }
313
314        updateValueLabels();
315        propagatePreview();
316    }
317
318    /**
319     * Collects the current slider and neighbour mode state into a parameter object.
320     */
321    private SquareControlParams collectParams() {
322        return new SquareControlParams(
323                densityRatioSlider.getValue() / 10.0,
324                variabilitySlider.getValue() / 10.0,
325                rSquaredSlider.getValue() / 100.0,
326                getNeighbourMode()
327        );
328    }
