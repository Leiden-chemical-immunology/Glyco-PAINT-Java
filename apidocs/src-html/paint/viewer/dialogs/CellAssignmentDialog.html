003 *  Package:      paint.viewer.dialogs
004 *
005 *  PURPOSE:
006 *    Provides a dialog window for assigning, undoing, or canceling cell
007 *    assignments within the PAINT viewer’s square grid interface.
008 *
009 *  DESCRIPTION:
010 *    The dialog presents a scrollable list of available cell IDs, each
011 *    represented by a color-coded square and a radio button label.
012 *    The user can:
013 *      • Select a cell ID and assign it to the current selection.
014 *      • Undo the most recent cell assignment.
015 *      • Cancel the current selection.
016 *      • Close the dialog window.
017 *
018 *    The dialog interacts with {@link CellAssignmentDialog.Listener}, allowing
019 *    external components (e.g., the viewer frame) to handle user actions.
020 *    It operates modelessly, remaining accessible while the viewer remains active.
021 *
022 *  KEY FEATURES:
023 *    • Scrollable, color-coded list of cell options.
024 *    • Modeless behavior for fluid interaction with the main viewer.
025 *    • Integrated event callback through {@link Listener}.
026 *    • Intuitive visual feedback for selected cell options.

041package paint.viewer.dialogs;
042
043import paint.viewer.panels.SquareGridPanel;
044
045import javax.swing.*;
046import java.awt.*;
047import java.util.LinkedHashMap;
048import java.util.Map;
049
050/**
051 * A dialog window for assigning actions related to a specific cell within a
052 * grid-like interface. The dialog allows users to select a cell via radio
053 * buttons and perform actions such as assigning, undoing, canceling, or
054 * closing the dialog.
055 *
056 * <p>The selection is visually represented by a colored square alongside
057 * radio buttons, where the colors correspond to specific cells. The dialog
058 * interacts with a {@link CellAssignmentDialog.Listener} interface for
059 * handling user actions.</p>
060 *
061 * <p>Features of this dialog include:</p>
062 * <ul>
063 *   <li>Modeless behavior to remain accessible alongside the parent frame.</li>
064 *   <li>Dynamic creation of cell options based on the number of supported cells.</li>
065 *   <li>Visual UI components with color-coded squares and radio buttons.</li>
066 *   <li>Scrollable layout for handling a large number of cells.</li>
067 * </ul>
068 */
069public class CellAssignmentDialog extends JDialog {
070
071    /** Listener interface for handling cell assignment actions. */
072    public interface Listener {
073        void onAssign(int cellId);
074        void onUndo();
075        void onCancelSelection();
076    }
077
078    private final ButtonGroup group = new ButtonGroup();
079    private final Map<JRadioButton, JPanel> squareByRadio = new LinkedHashMap<>();
080
081    /**
082     * Constructs a modeless dialog for assigning cells to selected regions
083     * within the viewer.
084     *
085     * @param owner    the parent frame owning this dialog
086     * @param listener a {@link Listener} implementation to handle user actions
087     */
088    public CellAssignmentDialog(Frame owner, Listener listener) {
089        super(owner, "Assign Cells", false);
090        setDefaultCloseOperation(DISPOSE_ON_CLOSE);
091        setLayout(new BorderLayout());
092
093        JPanel list = new JPanel();
094        list.setLayout(new BoxLayout(list, BoxLayout.Y_AXIS));
095
096        // --- ID 0: No cell ---
097        list.add(createCellRow(0, "No cell", Color.GRAY, true));
098
099        // --- IDs 1..N vertically ---
100        int cellCount = SquareGridPanel.getSupportedCellCount();
101        for (int id = 1; id <= cellCount; id++) {
102            Color c = SquareGridPanel.getColorForCell(id);
103            list.add(createCellRow(id, "Assign to cell " + id, c, false));
104        }
105
106        JScrollPane scroll = new JScrollPane(list,
107                                             JScrollPane.VERTICAL_SCROLLBAR_AS_NEEDED,
108                                             JScrollPane.HORIZONTAL_SCROLLBAR_NEVER);
109        add(scroll, BorderLayout.CENTER);
110
111        // --- Controls ---
112        JPanel buttons    = new JPanel(new FlowLayout(FlowLayout.RIGHT));
113        JButton assignBtn = new JButton("Assign");
114        JButton undoBtn   = new JButton("Undo");
115        JButton cancelBtn = new JButton("Cancel");
116        JButton closeBtn  = new JButton("Close");
117
118        assignBtn.addActionListener(e -> listener.onAssign(getSelectedCellId()));
119        undoBtn.addActionListener(  e -> listener.onUndo());
120        cancelBtn.addActionListener(e -> listener.onCancelSelection());
121        closeBtn.addActionListener( e -> dispose());
122
123        buttons.add(assignBtn);
124        buttons.add(undoBtn);
125        buttons.add(cancelBtn);
126        buttons.add(closeBtn);
127        add(buttons, BorderLayout.SOUTH);
128
129        pack();
130        setLocationRelativeTo(owner);
131        updateVisualSelection();
132    }
133
134    /**
135     * Creates a panel row with a colored square and an associated radio button
136     * for selecting a particular cell.
137     *
138     * @param cellId   unique identifier for the cell
139     * @param label    descriptive label for the radio button
140     * @param baseColor the color to display for this cell
141     * @param selected whether this option should be preselected
142     * @return a configured JPanel containing the radio button and color square
143     */
144    private JPanel createCellRow(int cellId, String label, Color baseColor, boolean selected) {
145        JPanel row = new JPanel(new FlowLayout(FlowLayout.LEFT, 8, 6));
146
147        JRadioButton radio = new JRadioButton(label);
148        radio.setForeground(Color.BLACK);
149        radio.putClientProperty("cellId", cellId);
150        if (selected) {
151            radio.setSelected(true);
152        }
153        group.add(radio);
154
155        JPanel square = new JPanel() {
156            @Override
157            public Dimension getPreferredSize() {
158                return new Dimension(20, 20);
159            }
160
161            @Override
162            protected void paintComponent(Graphics g) {
163                super.paintComponent(g);
164                g.setColor(baseColor);
165                g.drawRect(1, 1, getWidth() - 3, getHeight() - 3);
166            }
167        };
168        square.setBackground(Color.WHITE);
169        square.setBorder(BorderFactory.createLineBorder(Color.DARK_GRAY, 1));
170        square.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
171
172        squareByRadio.put(radio, square);
173
174        radio.addActionListener(e -> updateVisualSelection());
175        square.addMouseListener(new java.awt.event.MouseAdapter() {
176            @Override
177            public void mouseClicked(java.awt.event.MouseEvent e) {
178                radio.setSelected(true);
179                updateVisualSelection();
180            }
181        });
182
183        row.add(square);
184        row.add(radio);
185        return row;
186    }
187
188    /**
189     * Retrieves the cell ID associated with the selected radio button.
190     *
191     * @return the selected cell ID, or 0 if no selection is active
192     */
193    private int getSelectedCellId() {
194        for (JRadioButton rb : squareByRadio.keySet()) {
195            if (rb.isSelected()) {
196                Object v = rb.getClientProperty("cellId");
197                if (v instanceof Integer) {
198                    return (Integer) v;
199                }
200            }
201        }
202        return 0;
203    }
204
205    /**
206     * Updates the visual appearance of the colored squares to reflect the
207     * currently selected radio button.
208     */
209    private void updateVisualSelection() {
210        for (Map.Entry<JRadioButton, JPanel> e : squareByRadio.entrySet()) {
211            JRadioButton rb = e.getKey();
212            JPanel square = e.getValue();
213            int id = (Integer) rb.getClientProperty("cellId");
214            Color base = (id == 0) ? Color.GRAY : SquareGridPanel.getColorForCell(id);
215
216            if (rb.isSelected()) {
217                square.setBackground(base);
218                square.setBorder(BorderFactory.createLineBorder(Color.BLACK, 2));
219            } else {
220                square.setBackground(Color.WHITE);
221                square.setBorder(BorderFactory.createLineBorder(Color.DARK_GRAY, 1));
222            }
223        }
224        repaint();
225    }
226}
