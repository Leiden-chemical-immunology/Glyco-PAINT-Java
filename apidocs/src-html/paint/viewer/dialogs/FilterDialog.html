021 *    • Multi-select filtering lists with per-column Filter/Reset buttons.

039package paint.viewer.dialogs;
040
041import paint.viewer.utils.RecordingEntry;
042
043import javax.swing.*;
044import java.awt.*;
045import java.util.ArrayList;
046import java.util.List;
047import java.util.Set;
048import java.util.TreeSet;
049import java.util.stream.Collectors;
050
051/**
052 * A modal dialog for filtering a list of {@link RecordingEntry} objects
053 * based on attributes such as cell type, probe name, probe type,
054 * adjuvant, and concentration.
055 *
056 * <p>Each filter category is displayed in a column with its own JList,
057 * “Filter” and “Reset” buttons. Users can apply individual filters,
058 * reset them, reset all filters, or cancel the operation.</p>
059 */
060public class FilterDialog extends JDialog {
061
062    private final List<RecordingEntry> originalRecordings;
063    private final JList<String>  cellTypeList;
064    private final JList<String>  probeNameList;
065    private final JList<String>  probeTypeList;
066    private final JList<String>  adjuvantList;
067    private final JList<String>  concentrationList;
068    private List<RecordingEntry> filteredRecordings;
069    private boolean              cancelled = true;
070
071    /**
072     * Constructs a modal {@code FilterDialog} for filtering a list of
073     * {@link RecordingEntry} objects. The dialog allows the user to select
074     * and apply filters on key attributes of the recordings.
075     *
076     * @param owner      the parent frame that owns this dialog
077     * @param recordings the list of {@code RecordingEntry} objects to be filtered
078     */
079    public FilterDialog(Frame owner, List<RecordingEntry> recordings) {
080        super(owner, "Filter Recordings", true);
081        this.originalRecordings = new ArrayList<>(recordings);
082        this.filteredRecordings = new ArrayList<>(recordings);
083
084        setLayout(new BorderLayout(10, 10));
085        setResizable(false);
086
087        // ─────────────────────────────────────────────────────────────────────
088        // Collect distinct values for each attribute
089        // ─────────────────────────────────────────────────────────────────────
090
091        Set<String> cellTypes      = new TreeSet<>();
092        Set<String> probeNames     = new TreeSet<>();
093        Set<String> probeTypes     = new TreeSet<>();
094        Set<String> adjuvants      = new TreeSet<>();
095        Set<String> concentrations = new TreeSet<>();
096
097        for (RecordingEntry entry : recordings) {
098            cellTypes.add(entry.getCellType());
099            probeNames.add(entry.getProbeName());
100            probeTypes.add(entry.getProbeType());
101            adjuvants.add(entry.getAdjuvant());
102            concentrations.add(String.valueOf(entry.getConcentration()));
103        }
104
105        cellTypeList      = createList(cellTypes);
106        probeNameList     = createList(probeNames);
107        probeTypeList     = createList(probeTypes);
108        adjuvantList      = createList(adjuvants);
109        concentrationList = createList(concentrations);
110
111        JPanel listPanel = new JPanel(new GridLayout(1, 6, 10, 0));
112        listPanel.add(createListBoxWithButtons("Cell Type", cellTypeList));
113        listPanel.add(createListBoxWithButtons("Probe Name", probeNameList));
114        listPanel.add(createListBoxWithButtons("Probe Type", probeTypeList));
115        listPanel.add(createListBoxWithButtons("Adjuvant", adjuvantList));
116        listPanel.add(createListBoxWithButtons("Concentration", concentrationList));
117
118        // ─────────────────────────────────────────────────────────────────────
119        // Global Apply / Reset All / Cancel controls
120        // ─────────────────────────────────────────────────────────────────────
121        JPanel rightButtonPanel = new JPanel();
122        rightButtonPanel.setLayout(new BoxLayout(rightButtonPanel, BoxLayout.Y_AXIS));
123
124        JButton applyButton    = new JButton("Apply");
125        JButton resetAllButton = new JButton("Reset All");
126        JButton cancelButton   = new JButton("Cancel");
127
128        Dimension btnSize = new Dimension(100, 30);
129        applyButton.setMaximumSize(btnSize);
130        resetAllButton.setMaximumSize(btnSize);
131        cancelButton.setMaximumSize(btnSize);
132
133        applyButton.addActionListener(e -> {
134            cancelled = false;
135            dispose();
136        });
137        resetAllButton.addActionListener(e -> resetAllFilters());
138        cancelButton.addActionListener(e -> {
139            cancelled = true;
140            dispose();
141        });
142
143        rightButtonPanel.add(applyButton);
144        rightButtonPanel.add(Box.createVerticalStrut(10));
145        rightButtonPanel.add(resetAllButton);
146        rightButtonPanel.add(Box.createVerticalStrut(10));
147        rightButtonPanel.add(cancelButton);
148        rightButtonPanel.add(Box.createVerticalGlue());
149
150        listPanel.add(rightButtonPanel);
151
152        add(listPanel, BorderLayout.CENTER);
153
154        pack();
155        setLocationRelativeTo(owner);
156    }
157
158    /**
159     * Creates a multi-select {@link JList} populated with a given set of values.
160     *
161     * @param values the distinct string values to populate the list
162     * @return a configured {@code JList} instance
163     */
164    private JList<String> createList(Set<String> values) {
165        JList<String> list = new JList<>(values.toArray(new String[0]));
166        list.setSelectionMode(ListSelectionModel.MULTIPLE_INTERVAL_SELECTION);
167        list.setVisibleRowCount(5);
168        return list;
169    }
170
171    /**
172     * Creates a bordered panel containing a labeled {@link JList} and
173     * two buttons (“Filter” and “Reset”) that act on that list.
174     *
175     * @param title the title for the panel border
176     * @param list  the {@code JList} instance to be displayed
177     * @return the assembled panel
178     */
179    private JPanel createListBoxWithButtons(String title, JList<String> list) {
180        JPanel panel = new JPanel(new BorderLayout());
181        panel.setBorder(BorderFactory.createTitledBorder(title));
182
183        JScrollPane scrollPane = new JScrollPane(list);
184        scrollPane.setPreferredSize(new Dimension(120, 160));
185        panel.add(scrollPane, BorderLayout.CENTER);
186
187        JPanel btnPanel   = new JPanel(new GridLayout(2, 1, 0, 5));
188        JButton filterBtn = new JButton("Filter");
189        JButton resetBtn  = new JButton("Reset");
190
191        filterBtn.addActionListener(e -> applySingleFilter(list));
192        resetBtn.addActionListener(e -> resetSingleFilter());
193
194        btnPanel.add(filterBtn);
195        btnPanel.add(resetBtn);
196        panel.add(btnPanel, BorderLayout.SOUTH);
197        return panel;
198    }
199
200    /**
201     * Applies filtering logic using the selected values from a specific list.
202     *
203     * @param list the {@code JList} containing selected filter criteria
204     */
205    private void applySingleFilter(JList<String> list) {
206        List<String> selected = list.getSelectedValuesList();
207        if (selected.isEmpty()) {
208            return;
209        }
210
211        filteredRecordings = filteredRecordings.stream()
212                .filter(entry -> matches(entry, list, selected))
213                .collect(Collectors.toList());
214        updateLists();
215    }
216
217    /**
218     * Resets a single filter by restoring {@code filteredRecordings}
219     * to its original state and updating UI lists.
220     *
221     */
222    private void resetSingleFilter() {
223        filteredRecordings = new ArrayList<>(originalRecordings);
224        updateLists();
225    }
226
227    /**
228     * Resets all filters and restores the full unfiltered dataset.
229     */
230    private void resetAllFilters() {
231        filteredRecordings = new ArrayList<>(originalRecordings);
232        updateLists();
233    }
234
235    /**
236     * Determines if a {@link RecordingEntry} matches the selected filter criteria.
237     *
238     * @param entry    the recording being tested
239     * @param list     the {@code JList} defining the filter dimension
240     * @param selected selected filter values
241     * @return {@code true} if the recording matches the criteria
242     */
243    private boolean matches(RecordingEntry entry, JList<String> list, List<String> selected) {
244        if (list == cellTypeList) {
245            return selected.contains(entry.getCellType());
246        } else if (list == probeNameList) {
247            return selected.contains(entry.getProbeName());
248        } else if (list == probeTypeList) {
249            return selected.contains(entry.getProbeType());
250        } else if (list == adjuvantList) {
251            return selected.contains(entry.getAdjuvant());
252        } else if (list == concentrationList) {
253            return selected.contains(String.valueOf(entry.getConcentration()));
254        }
255        return true;
256    }
257
258    /**
259     * Updates each filter list based on the attributes of the filtered recordings.
260     */
261    private void updateLists() {
262        updateList(cellTypeList,      filteredRecordings.stream().map(RecordingEntry::getCellType).collect(Collectors.toSet()));
263        updateList(probeNameList,     filteredRecordings.stream().map(RecordingEntry::getProbeName).collect(Collectors.toSet()));
264        updateList(probeTypeList,     filteredRecordings.stream().map(RecordingEntry::getProbeType).collect(Collectors.toSet()));
265        updateList(adjuvantList,      filteredRecordings.stream().map(RecordingEntry::getAdjuvant).collect(Collectors.toSet()));
266        updateList(concentrationList, filteredRecordings.stream().map(e -> String.valueOf(e.getConcentration())).collect(Collectors.toSet()));
267    }
268
269    /**
270     * Replaces the data model of a {@link JList} with a new set of values.
271     */
272    private void updateList(JList<String> list, Set<String> values) {
273        list.setListData(values.toArray(new String[0]));
274    }
275
276    /**
277     * Returns the currently filtered list of recordings.
278     */
279    public List<RecordingEntry> getFilteredRecordings() {
280        return filteredRecordings;
281    }
282
283    /**
284     * Returns {@code true} if the user canceled the dialog.
285     */
286    public boolean isCancelled() {
287        return cancelled;
288    }
289}
