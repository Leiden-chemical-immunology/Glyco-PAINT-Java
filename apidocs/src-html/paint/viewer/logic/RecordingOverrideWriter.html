001/******************************************************************************
002 *  Class:        RecordingOverrideWriter.java
003 *  Package:      paint.viewer.logic
004 *
005 *  PURPOSE:
006 *    Manages persistence of recording-level parameter overrides in the
007 *    PAINT viewer, enabling density ratio, variability, R² threshold,
008 *    and neighbour mode adjustments to be stored, updated, and restored
009 *    across sessions.
010 *
011 *  DESCRIPTION:
012 *    The {@code RecordingOverrideWriter} class writes and maintains a
013 *    structured CSV file named {@code Recording Override.csv} that
014 *    contains control parameter overrides for each recording. Each
015 *    entry represents a distinct configuration of user-applied square
016 *    control parameters, written with a timestamp and recording name.
017 *
018 *    The writer supports three levels of application scope:
019 *      • Recording — Applies overrides to the currently active recording.
020 *      • Experiment — Applies overrides to all recordings in an experiment.
021 *      • Project — Applies overrides globally to all recordings in a project.
022 *
023 *    The writer automatically creates the Viewer directory if missing,
024 *    validates the CSV header, replaces existing entries for the same
025 *    recording, and performs atomic write operations to ensure file safety.
026 *
027 *  KEY FEATURES:
028 *    • Persists per-recording parameter overrides in a consistent CSV format.
029 *    • Automatically creates and maintains headers and directory structure.
030 *    • Supports Recording, Experiment, and Project scopes.
031 *    • Performs atomic writes to prevent data corruption.
032 *

039 *  UPDATED:

046package paint.viewer.logic;
047
048import paint.shared.objects.Track;
049import paint.shared.utils.CalculateTau;
050import paint.shared.utils.PaintLogger;
051import paint.viewer.shared.SquareControlParams;
052import paint.viewer.utils.RecordingEntry;
053
054import java.io.IOException;
055import java.nio.file.Files;
056import java.nio.file.Path;
057import java.nio.file.StandardCopyOption;
058import java.time.LocalDateTime;
059import java.util.ArrayList;
060import java.util.List;
061
062import static paint.shared.utils.CalculateTau.calculateTau;
063import static paint.shared.utils.SharedSquareUtils.getTracksFromSelectedSquares;
064
065/**
066 * Handles the writing of per-recording override configurations for the
067 * PAINT viewer. The {@code RecordingOverrideWriter} manages the creation
068 * and updating of a CSV file that stores user-defined control parameters
069 * (density ratio, variability, R², and neighbour mode) for each recording.
070 *
071 * <p>Each record is timestamped and replaces any previous entry for the
072 * same recording. The file is stored in the {@code Viewer} directory
073 * under the project root.
074 *
075 * <p>This class mirrors {@link paint.viewer.logic.SquareOverrideWriter}
076 * but operates at the recording scope instead of per-square granularity.
077 */
078public class RecordingOverrideWriter {
079
080    private final Path csvFilePath;
081
082    /**
083     * Constructs a new {@code RecordingOverrideWriter} for the specified
084     * project. Ensures that the Viewer directory exists and initializes
085     * the target file path for the override CSV.
086     *
087     * @param projectPath the root project path where the Viewer directory resides
088     */
089    public RecordingOverrideWriter(Path projectPath) {
090        Path viewerPath = projectPath.resolve("Viewer");
091        if (Files.notExists(viewerPath)) {
092            try {
093                Files.createDirectories(viewerPath);          // creates all missing parents too
094            } catch (IOException e) {
095                PaintLogger.warnf("Failed to create Viewer directory: %s", e.getMessage());
096            }
097        }
098        this.csvFilePath = viewerPath.resolve("Recording Override.csv");
099    }
100
101    /**
102     * Applies the provided parameters and writes overrides based on the
103     * specified scope. The scope determines whether overrides are applied
104     * at the Recording, Experiment, or Project level.
105     *
106     * @param scope        the override scope ("Recording", "Experiment", or "Project")
107     * @param params       the parameter set containing density ratio, variability,
108     *                     R², and neighbour mode values
109     * @param recordings   the list of available recording entries in the current session
110     * @param currentIndex the index of the currently active recording
111     */
112    public void applyAndWrite(String scope,
113                              SquareControlParams params,
114                              List<RecordingEntry> recordings,
115                              int currentIndex) {
116        String timestamp = LocalDateTime.now().toString();
117
118        if ("Recording".equals(scope)) {
119            RecordingEntry recordingEntry = recordings.get(currentIndex);
120            writeOverrideRecord(recordingEntry.getRecordingName(), params, timestamp);
121            update(recordingEntry, params);
122
123        } else if ("Experiment".equals(scope)) {
124            RecordingEntry currentRecordingEntry = recordings.get(currentIndex);
125            for (RecordingEntry recordingEntry : recordings) {
126                if (recordingEntry.getExperimentName().equals(currentRecordingEntry.getExperimentName())) {
127                    writeOverrideRecord(recordingEntry.getRecordingName(), params, timestamp);
128                    update(recordingEntry, params);
129                }
130            }
131
132        } else if ("Project".equals(scope)) {
133            for (RecordingEntry recordingEntry : recordings) {
134                writeOverrideRecord(recordingEntry.getRecordingName(), params, timestamp);
135                update(recordingEntry, params);
136            }
137        }
138    }
139
140    /**
141     * Writes a single override record into the CSV file using the provided
142     * recording name, parameters, and timestamp. Existing entries for the
143     * same recording are replaced. Headers are automatically added if
144     * missing or invalid.
145     *
146     * @param recordingName the name of the recording to which the override applies
147     * @param params        the parameter values to be persisted
148     * @param timestamp     the ISO-formatted timestamp of the override entry
149     */
150    private void writeOverrideRecord(String recordingName,
151                                     SquareControlParams params,
152                                     String timestamp) {
153        PaintLogger.infof(
154                "Override for '%s': MinRequiredDensityRatio=%.0f, MaxAllowableVariability=%.0f, MinRequiredRSquared=%.2f, NeighbourMode=%s",
155                recordingName, params.minRequiredDensityRatio, params.maxAllowableVariability, params.minRequiredRSquared, params.neighbourMode
156        );
157
158        try {
159            List<String> lines = new ArrayList<>();
160            if (Files.exists(csvFilePath)) {
161                lines = Files.readAllLines(csvFilePath);
162            }
163
164            // Initialize header if missing or malformed
165            if (lines.isEmpty() || !lines.get(0).startsWith("recordingName,")) {
166                lines.clear();
167                lines.add("recordingName,timestamp,MinRequiredDensityRatio,MaxAllowableVariability,minRequiredRSquared,neighbourMode");
168            }
169
170            String prefix = recordingName + ",";
171            String newLine = recordingName + "," + timestamp + "," +
172                    params.minRequiredDensityRatio + "," + params.maxAllowableVariability + "," +
173                    params.minRequiredRSquared + "," + params.neighbourMode;
174
175            boolean replaced = false;
176            for (int i = 1; i < lines.size(); i++) {
177                if (lines.get(i).startsWith(prefix)) {
178                    lines.set(i, newLine);
179                    replaced = true;
180                    break;
181                }
182            }
183
184            if (!replaced) {
185                lines.add(newLine);
186            }
187
188            // Atomic write using a temporary file for reliability
189            Path tmpFilePath = csvFilePath.resolveSibling(csvFilePath.getFileName().toString() + ".tmp");
190            Files.write(tmpFilePath, lines);
191            Files.move(tmpFilePath, csvFilePath, StandardCopyOption.REPLACE_EXISTING);
192
193        } catch (IOException ex) {
194            PaintLogger.errorf("Error writing recording override: %s", ex.getMessage());
195            ex.printStackTrace();
196        }
197    }
198
199    /**
200     * Updates the in-memory recording entry based on the provided parameters
201     * and recalculates Tau and R² values from the selected squares.
202     *
203     * @param recordingEntry the recording entry to update
204     * @param params         the parameter set containing updated control values
205     */
206    private void update(RecordingEntry recordingEntry,
207                        SquareControlParams params) {
208
209        List<Track> tracksFromSelectedSquares = getTracksFromSelectedSquares(
210                recordingEntry.getRecording().getSquaresOfRecording());
211
212        CalculateTau.CalculateTauResult results = calculateTau(
213                tracksFromSelectedSquares, params.minRequiredRSquared);
214
215        recordingEntry.getRecording().setTau(results.getTau());
216        recordingEntry.getRecording().setRSquared(results.getRSquared());
217        recordingEntry.setMaxAllowableVariability(params.maxAllowableVariability);
218        recordingEntry.setMinRequiredDensityRatio(params.minRequiredDensityRatio);
219        recordingEntry.setNeighbourMode(params.neighbourMode);
220        recordingEntry.setMinRequiredRSquared(params.minRequiredRSquared);
221    }
222}
