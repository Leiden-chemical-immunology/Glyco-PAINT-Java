038package paint.viewer.logic;
039
040import paint.shared.objects.Square;
041import paint.viewer.panels.SquareGridPanel;
042
043import java.util.*;
044
045/**
046 * Handles cell assignment operations for a grid of {@link Square} objects.
047 * <p>
048 * Supports both assigning a cell ID to selected squares and undoing
049 * the last assignment operation by restoring the previous state.
050 */
051public class CellAssignmentManager {
052
053    private final Map<Integer, Integer>        squareAssignments = new HashMap<>();   // This holds the pairs cellId and SquareNumber
054    private final Deque<Map<Integer, Integer>> undoStack         = new ArrayDeque<>();
055
056    /**
057     * Assigns the specified cell ID to all currently selected squares in the grid.
058     * The current assignment state is saved before modification, allowing undo.
059     *
060     * @param cellId the identifier to assign to the selected squares
061     * @param grid   the grid panel containing the squares to be modified
062     */
063    public Map<Integer, Integer> assignUserSelectedSquares(int cellId, SquareGridPanel grid) {
064        Set<Integer> userSelectedSquaresNumbers = grid.getUserSelectedSquaresNumbers();
065
066        if (userSelectedSquaresNumbers.isEmpty()) {
067            return Collections.unmodifiableMap(squareAssignments);
068        }
069
070        // Save the current state for undo
071        undoStack.push(new HashMap<>(squareAssignments));
072
073        for (Square square : grid.getSquares()) {
074            if (userSelectedSquaresNumbers.contains(square.getSquareNumber())) {
075                square.setCellId(cellId);
076                squareAssignments.put(square.getSquareNumber(), cellId);    // Adds (or updates) the pair
077            }
078        }
079
080        grid.clearMouseSelection();
081        grid.repaint();
082        return Collections.unmodifiableMap(squareAssignments);
083    }
084
085    /**
086     * Undoes the most recent assignment operation by restoring the previous
087     * state from the undo stack. Updates the grid to reflect the restored
088     * assignments and triggers a repaint.
089     *
090     * @param grid the grid panel containing the squares to revert
091     */
092    public void undo(SquareGridPanel grid) {
093        if (undoStack.isEmpty()) {
094            return;
095        }
096
097        squareAssignments.clear();
098        squareAssignments.putAll(undoStack.pop());
099
100        for (Square sq : grid.getSquares()) {
101            int cellId = squareAssignments.containsKey(sq.getSquareNumber())
102                    ? squareAssignments.get(sq.getSquareNumber())
103                    : 0;
104            sq.setCellId(cellId);
105        }
106
107        grid.repaint();
108    }
109}
