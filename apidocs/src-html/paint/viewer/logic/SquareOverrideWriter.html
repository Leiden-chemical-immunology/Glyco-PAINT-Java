014 *    recording, square ID, assigned cell ID, and timestamp of the override.

016 *    The writer ensures the Viewer directory exists, validates and writes
017 *    headers if necessary, replaces entries for squares that have been
018 *    reassigned, and performs all writes atomically using a temporary file.
019 *
020 *  KEY FEATURES:
021 *    • Stores persistent square-to-cell mapping overrides.
022 *    • Automatically creates and maintains CSV structure and headers.
023 *    • Replaces existing entries for previously assigned squares.
024 *    • Performs atomic write operations using temporary files.

039package paint.viewer.logic;
040
041import paint.shared.utils.PaintLogger;
042import paint.viewer.utils.RecordingEntry;
043
044import java.io.IOException;
045import java.nio.file.Files;
046import java.nio.file.Path;
047import java.nio.file.StandardCopyOption;
048import java.time.LocalDateTime;
049import java.util.ArrayList;
050import java.util.List;
051import java.util.Map;
052
053/**
054 * Handles per-square cell assignment persistence by writing overrides to
055 * a CSV file named {@code Square Override.csv} located in the Viewer folder
056 * of the project. Each record links a specific square to a user-assigned
057 * cell ID, allowing persistent tracking of manual cell assignments.
058 * <p>
059 * Columns:
060 * <ul>
061 *   <li>experimentName</li>
062 *   <li>recordingName</li>
063 *   <li>squareId</li>
064 *   <li>cellId</li>
065 *   <li>timestamp</li>
066 * </ul>
067 * <p>
068 * The writer ensures the Viewer directory exists, headers are created if
069 * missing, and each update replaces any previous record for the same square.
070 * This mirrors {@link paint.viewer.logic.RecordingOverrideWriter} in structure
071 * and reliability, but applies to square-level (cell) granularity.
072 */
073public class SquareOverrideWriter {
074
075    private final Path csvFilePath;
076
077    /**
078     * Constructs a new {@code SquareOverrideWriter} for a given project.
079     * Ensures that the Viewer directory exists before assigning the file path.
080     *
081     * @param projectPath the root path of the project containing the Viewer directory
082     */
083    public SquareOverrideWriter(Path projectPath) {
084        Path viewerPath = projectPath.resolve("Viewer");
085        if (Files.notExists(viewerPath)) {
086            try {
087                Files.createDirectories(viewerPath);   // ensures Viewer folder exists
088            } catch (IOException e) {
089                PaintLogger.warnf("Failed to create Viewer directory: %s", e.getMessage());
090            }
091        }
092        this.csvFilePath = viewerPath.resolve("Square Override.csv");
093    }
094
095    /**
096     * Writes or updates cell assignment overrides for all selected squares
097     * within the specified recording entry. Each square’s assignment is
098     * recorded with an associated timestamp. If an entry for a square already
099     * exists, it is replaced with the new cell ID.
100     *
101     * @param recordingEntry    the recording containing the relevant squares
102     * @param squareAssignments the list of squares that have been assigned to a cell
103     */
104    public void writeSquareOverrides(RecordingEntry recordingEntry,
105                                     Map<Integer, Integer> squareAssignments) {
106        String timestamp      = LocalDateTime.now().toString();
107        String experimentName = recordingEntry.getExperimentName();
108        String recordingName  = recordingEntry.getRecordingName();
109
110        try {
111            List<String> lines = new ArrayList<>();
112            if (Files.exists(csvFilePath)) {
113                lines = Files.readAllLines(csvFilePath);
114            }
115
116            // Initialize CSV with headers if new or malformed
117            if (lines.isEmpty() || !lines.get(0).startsWith("experimentName,")) {
118                lines.clear();
119                lines.add("experimentName,recordingName,squareId,cellId,timestamp");
120            }
121
122            // Update or append records for each (squareNumber, cellId) pair
123            for (Map.Entry<Integer, Integer> entry : squareAssignments.entrySet()) {
124                int squareNumber = entry.getKey();
125                int cellId       = entry.getValue();
126
127                String prefix = experimentName + "," + recordingName + "," + squareNumber + ",";
128
129                boolean found = false;
130                for (int i = 1; i < lines.size(); i++) {
131                    if (lines.get(i).startsWith(prefix)) {
132                        found = true;
133                        if (cellId == 0) {
134                            // Remove entry if it exists and cellId == 0
135                            lines.remove(i);
136                        } else {
137                            // Replace existing record
138                            String newLine = experimentName + "," + recordingName + "," + squareNumber + "," + cellId + "," + timestamp;
139                            lines.set(i, newLine);
140                        }
141                        break;
142                    }
143                }
144
145                // Add new entry only if cellId > 0 and not found
146                if (cellId != 0 && !found) {
147                    String newLine = experimentName + "," + recordingName + "," + squareNumber + "," + cellId + "," + timestamp;
148                    lines.add(newLine);
149                }
150            }
151
152            // Atomic write via temporary file to prevent data loss
153            Path tmpFilePath = csvFilePath.resolveSibling(csvFilePath.getFileName().toString() + ".tmp");
154            Files.write(tmpFilePath, lines);
155            Files.move(tmpFilePath, csvFilePath, StandardCopyOption.REPLACE_EXISTING);
156
157        } catch (IOException ex) {
158            PaintLogger.errorf("Error writing square overrides: %s", ex.getMessage());
159            ex.printStackTrace();
160        }
161    }
162}
