036package paint.viewer.logic;
037
038import paint.viewer.panels.SquareGridPanel;
039import paint.viewer.shared.SquareControlParams;
040
041/**
042 * Handles configuration and control of a {@link SquareGridPanel}
043 * by applying user-defined visibility and threshold parameters.
044 */
045public class SquareControlHandler {
046
047    private SquareGridPanel grid;
048
049    /**
050     * Attaches the specified {@link SquareGridPanel} to this handler.
051     * All subsequent parameter applications will target this grid.
052     *
053     * @param panel the grid panel to be managed
054     */
055    public void attach(SquareGridPanel panel) {
056        this.grid = panel;
057    }
058
059    /**
060     * Applies the specified square control parameters to a grid panel.
061     * <p>
062     * If a panel is provided, it becomes the active target. If no panel
063     * is given, the method applies the parameters to the currently
064     * attached grid.
065     * </p>
066     *
067     * @param params the square control parameters including density ratio,
068     *               variability, R², and neighbour mode
069     * @param panel  optional {@link SquareGridPanel} target; if null, the
070     *               currently attached grid is used
071     */
072    public void apply(SquareControlParams params, SquareGridPanel panel) {
073        if (panel != null) {
074            this.grid = panel;
075        }
076        if (grid == null) {
077            return;
078        }
079
080        grid.setControlParameters(
081                params.minRequiredDensityRatio,
082                params.maxAllowableVariability,
083                params.minRequiredRSquared,
084                params.neighbourMode
085        );
086        grid.applyVisibilityFilter();
087    }
088}
