038package paint.viewer.shared;
039
040/**
041 * The {@code SquareControlParams} class encapsulates parameters used to control the
042 * behavior and filtering of square grid configurations within the PAINT viewer.
043 * <p>
044 * It defines numerical thresholds for density ratio, variability, and R², and
045 * specifies the neighborhood mode as a string parameter. Instances of this class
046 * are immutable.
047 * </p>
048 */
049public class SquareControlParams {
050    public final double minRequiredDensityRatio;
051    public final double maxAllowableVariability;
052    public final double minRequiredRSquared;
053    public final String neighbourMode;
054
055    /**
056     * Constructs a new immutable {@code SquareControlParams} instance with the
057     * specified control parameter values.
058     *
059     * @param minRequiredDensityRatio minimum required density ratio threshold
060     * @param maxAllowableVariability maximum allowable variability threshold
061     * @param minRequiredRSquared minimum R² value (0.0–1.0) representing statistical fit
062     * @param neighbourMode the neighborhood interaction mode
063     */
064    public SquareControlParams(double minRequiredDensityRatio,
065                               double maxAllowableVariability,
066                               double minRequiredRSquared,
067                               String neighbourMode) {
068        this.minRequiredDensityRatio = minRequiredDensityRatio;
069        this.maxAllowableVariability = maxAllowableVariability;
070        this.minRequiredRSquared     = minRequiredRSquared;
071        this.neighbourMode           = neighbourMode;
072    }
073
074    /**
075     * Returns a formatted string representation of this parameter set,
076     * including density ratio, variability, R², and neighbour mode values.
077     *
078     * @return string representation of this {@code SquareControlParams} instance
079     */
080    @Override
081    public String toString() {
082        return "SquareControlParams{" +
083                "densityRatio="     + minRequiredDensityRatio +
084                ", variability="    + maxAllowableVariability +
085                ", rSquared="       + minRequiredRSquared +
086                ", neighbourMode='" + neighbourMode + '\'' +
087                '}';
088    }
089}
