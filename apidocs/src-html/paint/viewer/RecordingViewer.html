041package paint.viewer;
042
043import paint.shared.config.PaintConfig;
044import paint.shared.dialogs.ProjectDialog;
045import paint.shared.dialogs.ProjectDialog.DialogMode;
046import paint.shared.utils.PaintLogger;
047import paint.shared.utils.PaintPrefs;
048import paint.viewer.utils.RecordingEntry;
049import paint.viewer.utils.RecordingLoader;
050
051import javax.swing.*;
052import java.nio.file.Path;
053import java.nio.file.Paths;
054import java.util.List;
055
056/**
057 * Entry point for the PAINT Viewer application.
058 * <p>
059 * Initializes preferences, logging, and configuration before launching the
060 * recording viewer interface. The viewer enables users to inspect and
061 * interact with experiment recordings loaded from a project.
062 * </p>
063 *
064 * <p><strong>Responsibilities:</strong></p>
065 * <ul>
066 *   <li>Load the last used project root directory via {@link PaintPrefs}.</li>
067 *   <li>Initialize configuration and logging via {@link PaintConfig} and {@link PaintLogger}.</li>
068 *   <li>Display a project specification dialog for user selection.</li>
069 *   <li>Load recordings from the selected project using {@link RecordingLoader}.</li>
070 *   <li>Launch the {@link RecordingViewerFrame} interface if valid recordings are found.</li>
071 * </ul>
072 */
073public class RecordingViewer {
074
075    /**
076     * Main entry point for the PAINT Viewer application.
077     * <p>
078     * This method initializes preferences, configuration, and logging, then
079     * opens the project dialog for user input. If valid recordings are loaded,
080     * it launches the viewer frame for visualization.
081     * </p>
082     *
083     * @param args command-line arguments (unused)
084     */
085    public static void main(String[] args) {
086        SwingUtilities.invokeLater(() -> {
087
088            // --- Step 1: Load the last used project root from preferences ---
089            String lastProject = PaintPrefs.getString("Path", "Project Root", System.getProperty("user.home"));
090            Path projectPath = Paths.get(lastProject);
091
092            // --- Step 2: Initialise logging/config ---
093            PaintConfig.initialise(projectPath);
094            PaintLogger.initialise(projectPath, "Viewer");
095
096            // --- Step 3: Open the Project Specification dialog directly ---
097            ProjectDialog specificationDialog =
098                    new ProjectDialog(null, projectPath, DialogMode.VIEWER);
099
100            // ✅ Callback for the OK button — launches the viewer
101            specificationDialog.setCalculationCallback(project -> {
102                try {
103                    // Load the data for experiments (images, squares, etc.)
104                    List<RecordingEntry> recordingEntries = RecordingLoader.loadFromProject(project);
105                    if (recordingEntries.isEmpty()) {
106                        JOptionPane.showMessageDialog(null,
107                                                      "No valid recordings found in selected experiments.",
108                                                      "No Recordings",
109                                                      JOptionPane.WARNING_MESSAGE);
110                        return false;
111                    }
112
113                    // With the recordingEntries info available, the viewer can be started
114                    RecordingViewerFrame viewer = new RecordingViewerFrame(project, recordingEntries);
115                    viewer.setVisible(true);
116                    return true;
117                } catch (Exception ex) {
118                    PaintLogger.errorf("Viewer launch failed: %s", ex.getMessage());
119                    JOptionPane.showMessageDialog(null,
120                                                  "Viewer launch failed:\n" + ex.getMessage(),
121                                                  "Error",
122                                                  JOptionPane.ERROR_MESSAGE);
123                    return false;
124                }
125            });
126
127            // --- Step 4: Show dialog and keep it open ---
128            specificationDialog.showDialog();
129        });
130    }
131}
