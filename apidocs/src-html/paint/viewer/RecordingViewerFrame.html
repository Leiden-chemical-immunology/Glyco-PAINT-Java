080/**
081 * The {@code RecordingViewerFrame} class defines the main window of the PAINT Viewer.
082 * It combines left and right image panels, navigation controls, and metadata panels into
083 * a cohesive interface for browsing, filtering, and analyzing experiment recordings.
084 * <p>
085 * Functionally, the class manages synchronization between user interactions and the
086 * underlying model objects — specifically {@link paint.viewer.utils.RecordingEntry}
087 * and {@link paint.shared.objects.Recording}. It enables users to:
088 * <ul>
089 *   <li>Navigate through multiple recordings within an experiment.</li>
090 *   <li>Filter visible squares using configurable thresholds.</li>
091 *   <li>Assign cell IDs interactively via selection dialogs.</li>
092 *   <li>Play TIFF/ND2 recordings directly from disk.</li>
093 *   <li>Preview live recalculations of Tau, R², and density metrics.</li>
094 * </ul>
095 * <p>
096 * This class is instantiated by {@link paint.viewer.RecordingViewer} after successful
097 * project initialization. All UI updates occur on the Swing event dispatch thread.
098 */
099public class RecordingViewerFrame extends JFrame
100        implements RecordingControlsPanel.Listener, NavigationPanel.Listener {

108    private final JLabel                   recordingLabel    = new JLabel("", SwingConstants.CENTER);
109
110    private       RecordingAttributesPanel attributesPanel;
111    private       NavigationPanel          navigationPanel;

113    private final CellAssignmentManager    assignmentManager = new CellAssignmentManager();
114    private final RecordingOverrideWriter  recordingOverrideWriter;

118    /**
119     * Constructs a {@code RecordingViewerFrame} that initializes and displays the complete
120     * recording viewer environment. The frame sets up grid visualization, navigation,
121     * control, and attribute panels while establishing event connections for user actions.
122     *
123     * @param project          the {@link Project} object providing experiment context and paths.
124     * @param recordingEntries list of {@link RecordingEntry} objects representing loaded recordings.
125     */
126    public RecordingViewerFrame(Project project, List<RecordingEntry> recordingEntries) {
127        super("Recording Viewer - " + project.getProjectRootPath().getFileName());
128        this.project                 = project;
129        this.recordingEntries        = recordingEntries;  // All the information is maintained here
130        this.recordingOverrideWriter = new RecordingOverrideWriter(project.getProjectRootPath());
131        this.squareOverrideWriter    = new SquareOverrideWriter(project.getProjectRootPath());
132
133
134        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
135        setLayout(new BorderLayout());
136        setResizable(false);
137
138        // Validate grid configuration
139        int     numberOfSquaresInRecording = PaintConfig.getInt("Generate Squares", "Number of Squares in Recording", -1);
140        int[]   validSquareLayouts         = {25, 100, 225, 400, 900};
141        boolean isValidSquareLayout        = false;
142
143        for (int valid : validSquareLayouts) {
144            if (numberOfSquaresInRecording == valid) {
145                isValidSquareLayout = true;
146                break;
147            }
148        }
149        if (!isValidSquareLayout) {
150            PaintLogger.errorf("Invalid square layout (d x d)");
151            return;
152        }
153        int numberOfSquareInOneDimension = (int) Math.sqrt(numberOfSquaresInRecording);
154
155        // --- Initialize panels and handlers ---
156        leftGridPanel = new SquareGridPanel(numberOfSquareInOneDimension, numberOfSquareInOneDimension);
157        controlHandler.attach(leftGridPanel);
158
159        attributesPanel = new RecordingAttributesPanel();
160        navigationPanel = new NavigationPanel(this);
161        RecordingControlsPanel controlsPanel = new RecordingControlsPanel(this);
162
163        // --- Build the main layout ---
164        JPanel imagesInner = new JPanel(new GridLayout(1, 2, 15, 0));
165        imagesInner.add(createSquareImagePanel(leftGridPanel));
166        imagesInner.add(createSquareImagePanel(rightImageLabel));
167
168        JPanel labelsPanel = new JPanel(new GridLayout(2, 1));
169        labelsPanel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
170        experimentLabel.setFont(new Font("SansSerif", Font.BOLD, 13));
171        recordingLabel.setFont(new Font("SansSerif", Font.PLAIN, 12));
172        labelsPanel.add(experimentLabel);
173        labelsPanel.add(recordingLabel);
174
175        JPanel imagesWithNav = new JPanel(new BorderLayout(15, 15));
176        imagesWithNav.setBorder(BorderFactory.createCompoundBorder(
177                BorderFactory.createLineBorder(Color.DARK_GRAY, 2),
178                BorderFactory.createEmptyBorder(15, 15, 15, 15)));
179        imagesWithNav.add(navigationPanel.getComponent(), BorderLayout.NORTH);
180        imagesWithNav.add(imagesInner, BorderLayout.CENTER);
181        imagesWithNav.add(labelsPanel, BorderLayout.SOUTH);
182
183        JPanel mainPanel = new JPanel(new BorderLayout());
184        mainPanel.add(attributesPanel.getComponent(), BorderLayout.WEST);
185        mainPanel.add(imagesWithNav, BorderLayout.CENTER);
186        mainPanel.add(controlsPanel.getComponent(), BorderLayout.EAST);
187        add(mainPanel, BorderLayout.CENTER);
188
189        // --- Close button setup ---
190        JButton closeButton = new JButton("Close Viewer");
191        closeButton.setFont(new Font("SansSerif", Font.PLAIN, 12));
192        closeButton.addActionListener(e -> dispose());
193
194        JPanel bottomPanel = new JPanel(new FlowLayout(FlowLayout.RIGHT, 10, 10));
195        bottomPanel.add(closeButton);
196        add(bottomPanel, BorderLayout.SOUTH);
197
198        setSize(1500, 700);
199        setLocationRelativeTo(null);
200
201        // Load first entry if available
202        if (!recordingEntries.isEmpty()) {
203            showRecordingEntry(0);
204        }
205    }

207    /**
208     * Creates a square container panel that holds the specified component (typically an image).
209     * Ensures consistent square proportions regardless of frame resizing.
210     *
211     * @param comp the child component to display inside the square panel.
212     * @return a configured {@link JPanel} maintaining a square aspect ratio.
213     */
214    private JPanel createSquareImagePanel(JComponent comp) {
215        JPanel panel = new JPanel(new BorderLayout()) {
216            public Dimension getPreferredSize() {
217                return new Dimension(NUMBER_PIXELS_WIDTH, NUMBER_PIXELS_HEIGHT);
218            }
219
220            public void setBounds(int x, int y, int w, int h) {
221                int size = Math.min(w, h);
222                super.setBounds(x, y, size, size);
223            }
224        };
225        panel.setBorder(BorderFactory.createLineBorder(Color.BLACK, 1));
226        panel.add(comp, BorderLayout.CENTER);
227        return panel;
228    }

230    /**
231     * Resizes an {@link ImageIcon} proportionally to the specified dimensions.
232     *
233     * @param icon image icon to scale.
234     * @param w    target width.
235     * @param h    target height.
236     * @return a new scaled {@link ImageIcon}, or {@code null} if the source is invalid.
237     */
238    private static ImageIcon scaleToFit(ImageIcon icon, int w, int h) {
239        if (icon == null || icon.getIconWidth() <= 0 || icon.getIconHeight() <= 0) {
240            return null;
241        }
242        Image scaled = icon.getImage().getScaledInstance(w, h, Image.SCALE_SMOOTH);
243        return new ImageIcon(scaled);
244    }

246    /**
247     * Updates the navigation buttons based on the current index position.
248     * Enables "previous" or "next" navigation only when appropriate.
249     */
250    private void updateNavButtons() {
251        navigationPanel.setEnabledState(currentIndex > 0, currentIndex < recordingEntries.size() - 1);
252    }
253
254    /**
255     * Displays a specific recording entry by index, updating both left and right
256     * image panels and all related attribute components.
257     *
258     * @param index the target recording index to display.
259     */
260    private void showRecordingEntry(int index) {
261        if (index < 0 || index >= recordingEntries.size()) {
262            return;
263        }
264        currentIndex = index;
265
266        RecordingEntry recordingEntry = recordingEntries.get(index);
267        leftGridPanel.setRecording(recordingEntry.getRecording());
268        leftGridPanel.setBackgroundImage(recordingEntry.getLeftImage());
269
270        int numberOfSquaresInRecording = PaintConfig.getInt("Generate Squares", "Number of Squares in Recording", -1);
271        leftGridPanel.setSquares(recordingEntry.getRecording().getSquaresOfRecording());
272
273        rightImageLabel.setIcon(scaleToFit(recordingEntry.getRightImage(), NUMBER_PIXELS_WIDTH, NUMBER_PIXELS_HEIGHT));
274        experimentLabel.setText("Experiment: " + recordingEntry.getExperimentName() + "   [Overall: " + (currentIndex + 1) + "/" + recordingEntries.size() + "]");
275        recordingLabel.setText("Recording: " + recordingEntry.getRecordingName());
276
277        attributesPanel.updateFromEntry(recordingEntry, numberOfSquaresInRecording);
278        updateNavButtons();
279        leftGridPanel.repaint();
280    }

322    /**
323     * Invoked when the user opens the filter dialog.
324     * The dialog allows narrowing the visible recording list by user-defined criteria.
325     * Once the user confirms, the filtered result is applied immediately to the viewer.
326     */
327    @Override
328    public void onFilterRequested() {
329        leftGridPanel.hideSquareInfoIfVisible();  // Close any popup first
330        FilterDialog dialog = new FilterDialog(this, recordingEntries);
331        dialog.setVisible(true);
332
333        if (!dialog.isCancelled()) {
334            List<RecordingEntry> filtered = dialog.getFilteredRecordings();
335            if (!filtered.isEmpty()) {
336                currentIndex = 0;
337                showRecordingEntry(0);
338            }
339        }
340    }

343     * Opens the Square Control dialog for adjusting visibility thresholds on the grid.
344     * The dialog enables configuration of minimum density ratio, maximum variability,
345     * minimum R², and neighbour mode. Changes can be previewed live or fully applied.
346     */
347    @Override
348    public void onSelectSquaresRequested() {
349        leftGridPanel.hideSquareInfoIfVisible();
350        RecordingEntry current = recordingEntries.get(currentIndex);

353                this,
354                leftGridPanel,
355                this,

357                        current.getMinRequiredDensityRatio(),
358                        current.getMaxAllowableVariability(),
359                        current.getMinRequiredRSquared(),
360                        "Free"
361                )
362        );
363        dialog.setVisible(true);
364    }

366    /**
367     * Opens a dialog for assigning the currently selected squares to a specific cell ID.
368     * Includes undo and cancel functions and disables selection when the dialog closes.
369     */
370    @Override
371    public void onAssignCellsRequested() {
372        leftGridPanel.hideSquareInfoIfVisible();
373        leftGridPanel.setSelectionEnabled(true);
374
375        final JFrame owner = this;
376        CellAssignmentDialog dialog = new CellAssignmentDialog(owner, new CellAssignmentDialog.Listener() {
377
378            public void onAssign(int cellId) {
379                Map<Integer, Integer> userSelectedSquares = assignmentManager.assignUserSelectedSquares(cellId, leftGridPanel);   // This has filled CellAssignmentManager.squareAssignments
380
381                // --- Persist cell assignments to Square Override.csv ---
382                RecordingEntry currentRecording = recordingEntries.get(currentIndex);
383
384                if (!userSelectedSquares.isEmpty()) {
385                    squareOverrideWriter.writeSquareOverrides(currentRecording, userSelectedSquares);
386                }
387            }
388
389            public void onUndo() {
390                assignmentManager.undo(leftGridPanel);
391            }
392
393            public void onCancelSelection() {
394                leftGridPanel.clearSelection();
395                leftGridPanel.applyVisibilityFilter();
396                leftGridPanel.repaint();
397            }
398        });
399
400        dialog.addWindowListener(new java.awt.event.WindowAdapter() {
401            public void windowClosed(java.awt.event.WindowEvent e) {
402                leftGridPanel.setSelectionEnabled(false);
403            }
404        });
405        dialog.setVisible(true);
406    }

449    /**
450     * Applies square control parameters from the dialog to the current grid or project scope.
451     * <p>
452     * When invoked in "Preview" mode, recalculates Tau, R², and density values dynamically
453     * without committing them to disk. For full application, thresholds are persisted via
454     * {@link RecordingOverrideWriter}.
455     *
456     * @param scope  the operational scope ("Preview" or "Apply").
457     * @param params parameter bundle defining the visibility thresholds and neighbour mode.
458     */
459    @Override
460    public void onApplySquareControl(String scope, SquareControlParams params) {
461        if ("Preview".equals(scope)) {
462            controlHandler.apply(params, leftGridPanel);
463            leftGridPanel.applyVisibilityFilter();
464
465            RecordingEntry currentRecordingEntry = recordingEntries.get(currentIndex);
466            int numSquares = PaintConfig.getInt("Generate Squares", "Number of Squares in Recording", -1);
467
468            // --- Compute Tau and R² for preview ---
469            List<Track> tracksFromSelectedSquares = getTracksFromSelectedSquares(currentRecordingEntry.getRecording().getSquaresOfRecording());
470            CalculateTau.CalculateTauResult results = calculateTau(tracksFromSelectedSquares, params.minRequiredRSquared);
471            if (results.getStatus() == TAU_SUCCESS) {
472                currentRecordingEntry.getRecording().setTau(results.getTau());
473                currentRecordingEntry.getRecording().setRSquared(results.getRSquared());
474            } else {
475                currentRecordingEntry.getRecording().setTau(NaN);
476                currentRecordingEntry.getRecording().setRSquared(NaN);
477                // No successful Tau calculation; retain existing state
478            }
479
480            // --- Compute density for current selection ---
481            double density = calculateDensity(
482                    tracksFromSelectedSquares.size(),
483                    calculateSquareArea(getNumberOfSelectedSquares(currentRecordingEntry.getRecording())),
484                    RECORDING_DURATION,
485                    currentRecordingEntry.getRecording().getConcentration()
486            );
487
488            // --- Reflect results in attribute preview panel ---
489            attributesPanel.updatePreview(
490                    currentRecordingEntry,
491                    numSquares,
492                    results.getTau(),
493                    density,
494                    params.minRequiredDensityRatio,
495                    params.maxAllowableVariability,
496                    params.minRequiredRSquared,
497                    params.neighbourMode
498            );
499
500            leftGridPanel.repaint();
501            return;
502        }
503
504        // --- Full application: persist thresholds and repaint ---
505        controlHandler.apply(params, leftGridPanel);
506        recordingOverrideWriter.applyAndWrite(scope, params, recordingEntries, currentIndex);
507        leftGridPanel.repaint();
508    }

514    /**
515     * Initiates playback of the ND2 or TIFF file corresponding to the current recording.
516     * Performs validation of user selection and file presence before launching playback.
517     * Errors and missing files are reported via dialog boxes.
518     */
519    @Override
520    public void onPlayRecordingRequested() {
521        leftGridPanel.hideSquareInfoIfVisible();
522
523        if (recordingEntries.isEmpty() || currentIndex < 0 || currentIndex >= recordingEntries.size()) {
524            PaintLogger.warnf("No recording selected to play.");
525            JOptionPane.showMessageDialog(this,
526                                          "No recording selected to play.",
527                                          "No Selection",
528                                          JOptionPane.WARNING_MESSAGE);
529            return;
530        }
531
532        RecordingEntry entry          = recordingEntries.get(currentIndex);
533        String         experimentName = entry.getExperimentName();
534        String         recordingName  = entry.getRecordingName();
535
536        // Determine the path for image playback
537        Path imagesRoot = project.getImagesRootPath();
538        if (imagesRoot == null) {
539            String stored = PaintPrefs.getString("Path", "Images Root", "");
540            if (stored != null && !stored.isEmpty()) {
541                imagesRoot = Paths.get(stored);
542            } else {
543                JOptionPane.showMessageDialog(this,
544                                              "No Images Root is defined.\nPlease set it in the Project Specification dialog.",
545                                              "Configuration Error",
546                                              JOptionPane.ERROR_MESSAGE);
547                return;
548            }
549        }
550
551        Path imagePath = imagesRoot.resolve(experimentName).resolve(recordingName + ".nd2");
552        if (!Files.exists(imagePath)) {
553            JOptionPane.showMessageDialog(this,
554                                          "Recording file not found:\n" + imagePath,
555                                          "File Missing",
556                                          JOptionPane.ERROR_MESSAGE);
557            return;
558        }
559
560        // Launch movie playback on a background thread
561        Thread movieThread = new Thread(() -> {
562            try {
563                TiffMoviePlayer player = new TiffMoviePlayer();
564                player.playMovie(imagePath.toString());
565            } catch (Exception ex) {
566                PaintLogger.errorf("Error playing recording: %s", ex.getMessage());
567                ex.printStackTrace();
568            }
569        }, "MoviePlayerThread");
570
571        movieThread.setDaemon(true);
572        movieThread.start();
573        PaintLogger.infof("Playing recording: %s / %s", experimentName, recordingName);
574    }
