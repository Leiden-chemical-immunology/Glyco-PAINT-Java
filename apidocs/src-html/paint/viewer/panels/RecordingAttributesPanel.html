011 *    recording entry, including probe characteristics, experiment parameters,
012 *    and derived statistics such as Tau, Density, and R² thresholds.

015 *    {@link javax.swing.table.DefaultTableModel}, allowing data to be
016 *    refreshed dynamically from a given {@link RecordingEntry}.
017 *    It supports both permanent updates and transient previews of
018 *    parameter changes.
019 *
020 *  KEY FEATURES:
021 *    • Displays probe, cell type, and recording statistics in a table view.
022 *    • Supports real-time preview updates without altering data objects.
023 *    • Formats numeric fields with configurable decimal precision.

038package paint.viewer.panels;
039
040import paint.viewer.utils.RecordingEntry;
041
042import javax.swing.*;
043import javax.swing.table.DefaultTableModel;
044import java.awt.*;
045
046/**
047 * A Swing panel that presents attributes of a {@link RecordingEntry} in tabular form.
048 * <p>
049 * The panel includes a scrollable, non-editable table showing probe, experiment,
050 * and computed parameters for the active recording. It supports both static
051 * updates (via {@link #updateFromEntry}) and temporary previews
052 * (via {@link #updatePreview}).
053 */
054public class RecordingAttributesPanel {
055
056    private final JPanel            root;
057    private final DefaultTableModel model;
058
059    /**
060     * Constructs a new {@code RecordingAttributesPanel}.
061     * <p>
062     * Initializes a styled, non-editable table for displaying recording attributes.
063     * The layout uses a {@link BorderLayout} with compound borders and a scroll pane
064     * for readability and integration within parent components.
065     */
066    public RecordingAttributesPanel() {
067        root = new JPanel(new BorderLayout());
068        root.setBorder(BorderFactory.createCompoundBorder(
069                BorderFactory.createLineBorder(Color.DARK_GRAY, 2),
070                BorderFactory.createEmptyBorder(10, 10, 10, 10)
071        ));
072        root.setPreferredSize(new Dimension(240, 0));
073
074        model = new DefaultTableModel(new Object[]{"Attr", "Val"}, 0) {
075            public boolean isCellEditable(int r, int c) {
076                return false;
077            }
078        };
079        JTable table = new JTable(model);
080        table.setRowHeight(22);
081        table.setFocusable(false);
082        table.setRowSelectionAllowed(false);
083        table.getTableHeader().setFont(new Font("SansSerif", Font.BOLD, 12));
084
085        JScrollPane scroll = new JScrollPane(table);
086        root.add(scroll, BorderLayout.CENTER);
087    }
088
089    /**
090     * Returns the root Swing component representing this panel.
091     *
092     * @return the root {@link JComponent}
093     */
094    public JComponent getComponent() {
095        return root;
096    }
097
098    /**
099     * Updates the displayed attributes using data from the specified recording entry.
100     *
101     * @param recordingEntry the {@link RecordingEntry} containing the data
102     * @param numSquares     the number of squares associated with this recording
103     */
104    public void updateFromEntry(RecordingEntry recordingEntry, int numSquares) {
105        model.setRowCount(0);
106        model.addRow(new Object[]{"Probe",             recordingEntry.getProbeName()});
107        model.addRow(new Object[]{"Probe Type",        recordingEntry.getProbeType()});
108        model.addRow(new Object[]{"Adjuvant",          recordingEntry.getAdjuvant()});
109        model.addRow(new Object[]{"Cell Type",         recordingEntry.getCellType()});
110        model.addRow(new Object[]{"Concentration",     recordingEntry.getConcentration()});
111        model.addRow(new Object[]{"Number of Spots",   recordingEntry.getNumberOfSpots()});
112        model.addRow(new Object[]{"Number of Tracks",  recordingEntry.getNumberOfTracks()});
113        model.addRow(new Object[]{"Number of Squares", numSquares});
114        model.addRow(new Object[]{"Threshold",         recordingEntry.getThreshold()});
115        model.addRow(new Object[]{"Tau",               format(recordingEntry.getTau(), 1)});
116        model.addRow(new Object[]{"Density",           recordingEntry.getDensity()});
117        model.addRow(new Object[]{"Min Density Ratio", recordingEntry.getMinRequiredDensityRatio()});
118        model.addRow(new Object[]{"Max Variability",   recordingEntry.getMaxAllowableVariability()});
119        model.addRow(new Object[]{"Min R²",            recordingEntry.getMinRequiredRSquared()});
120        model.addRow(new Object[]{"Neighbour Mode",    recordingEntry.getNeighbourMode()});
121    }
122
123    /**
124     * Temporarily updates the displayed attributes to preview parameter changes.
125     * <p>
126     * This does not modify the underlying {@link RecordingEntry}.
127     *
128     * @param recordingEntry  the current recording entry (for static attributes)
129     * @param numSquares      number of squares for the recording
130     * @param tau             Tau value to preview
131     * @param density         Density value to preview
132     * @param minDensityRatio minimum density ratio threshold
133     * @param maxVariability  maximum variability threshold
134     * @param minRSquared     minimum R² threshold
135     * @param neighbourMode   neighbour mode label (“Free”, “Relaxed”, “Strict”)
136     */
137    public void updatePreview(
138            RecordingEntry recordingEntry,
139            int            numSquares,
140            double         tau,
141            double         density,
142            double         minDensityRatio,
143            double         maxVariability,
144            double         minRSquared,
145            String         neighbourMode) {
146
147        model.setRowCount(0);
148        model.addRow(new Object[]{"Probe",               recordingEntry.getProbeName()});
149        model.addRow(new Object[]{"Probe Type",          recordingEntry.getProbeType()});
150        model.addRow(new Object[]{"Adjuvant",            recordingEntry.getAdjuvant()});
151        model.addRow(new Object[]{"Cell Type",           recordingEntry.getCellType()});
152        model.addRow(new Object[]{"Concentration",       recordingEntry.getConcentration()});
153        model.addRow(new Object[]{"Number of Spots",     recordingEntry.getNumberOfSpots()});
154        model.addRow(new Object[]{"Number of Tracks",    recordingEntry.getNumberOfTracks()});
155        model.addRow(new Object[]{"Number of Squares",   numSquares});
156        model.addRow(new Object[]{"Threshold",           recordingEntry.getThreshold()});
157        model.addRow(new Object[]{"Tau",                 format(tau, 1)});
158        model.addRow(new Object[]{"Density",             format(density, 2)});
159        model.addRow(new Object[]{"Min Density Ratio",   format(minDensityRatio, 1)});
160        model.addRow(new Object[]{"Max Variability",     format(maxVariability, 1)});
161        model.addRow(new Object[]{"Min R²",              format(minRSquared, 2)});
162        model.addRow(new Object[]{"Neighbour Mode",      neighbourMode});
163    }
164
165    /**
166     * Formats a numeric value to the specified precision, handling NaN and infinity cases.
167     *
168     * @param v the value to format
169     * @param p number of decimal places
170     * @return formatted string representation of the value
171     */
172    private static String format(double v, int p) {
173        if (Double.isNaN(v)) {
174            return "NaN";
175        }
176        if (Double.isInfinite(v)) {
177            return v > 0 ? "∞" : "-∞";
178        }
179        return String.format("%." + p + "f", v);
180    }
181}
