011 *      • “|<” — Jump to the first item

013 *      • “>”  — Move to the next item
014 *      • “>|” — Jump to the last item
015 *
016 *    A listener interface is used to connect navigation actions to custom
017 *    logic in the parent component. Button enabled states can be updated
018 *    dynamically to reflect navigation bounds.
019 *
020 *  KEY FEATURES:
021 *    • Simple reusable navigation panel.
022 *    • Listener-based event handling for flexibility.
023 *    • Dynamic enable/disable control for navigation limits.

038package paint.viewer.panels;
039
040import javax.swing.*;
041import java.awt.*;
042
043/**
044 * A simple reusable navigation component providing First, Previous, Next,
045 * and Last buttons for paging or record traversal within the viewer.
046 */
047public class NavigationPanel {
048
049    /**
050     * Listener interface for responding to navigation button events.
051     */
052    public interface Listener {
053        void onFirst();
054        void onPrev();
055        void onNext();
056        void onLast();
057    }
058
059    private final JPanel  root;
060    private final JButton firstBtn = new JButton("|<");
061    private final JButton prevBtn  = new JButton("<");
062    private final JButton nextBtn  = new JButton(">");
063    private final JButton lastBtn  = new JButton(">|");
064
065    /**
066     * Constructs a NavigationPanel with navigation buttons wired to a listener.
067     *
068     * @param listener a {@link Listener} implementation handling First, Previous,
069     *                 Next, and Last button events
070     */
071    public NavigationPanel(final Listener listener) {
072        root = new JPanel(new FlowLayout(FlowLayout.CENTER, 10, 10));
073        root.add(firstBtn);
074        root.add(prevBtn);
075        root.add(nextBtn);
076        root.add(lastBtn);
077
078        firstBtn.addActionListener(e -> listener.onFirst());
079        prevBtn.addActionListener( e -> listener.onPrev());
080        nextBtn.addActionListener( e -> listener.onNext());
081        lastBtn.addActionListener( e -> listener.onLast());
082    }
083
084    /**
085     * Returns the root Swing component representing this navigation panel.
086     *
087     * @return the root {@link JComponent}
088     */
089    public JComponent getComponent() {
090        return root;
091    }
092
093    /**
094     * Updates the enabled/disabled state of the navigation buttons.
095     *
096     * @param hasPrev {@code true} to enable the First and Previous buttons
097     * @param hasNext {@code true} to enable the Next and Last buttons
098     */
099    public void setEnabledState(boolean hasPrev, boolean hasNext) {
100        firstBtn.setEnabled(hasPrev);
101        prevBtn.setEnabled(hasPrev);
102        nextBtn.setEnabled(hasNext);
103        lastBtn.setEnabled(hasNext);
104    }
105}
