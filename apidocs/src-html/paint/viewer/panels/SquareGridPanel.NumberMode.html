005 *  PURPOSE:
006 *    Displays and manages an interactive grid of squares within the PAINT
007 *    viewer, supporting visual selection, annotation, and visibility control
008 *    features for experimental image data.
009 *
010 *  DESCRIPTION:
011 *    The {@code SquareGridPanel} provides a graphical interface that renders
012 *    a structured grid of {@link paint.shared.objects.Square} elements. Each
013 *    square can represent data such as density, variability, or R² statistics
014 *    and may be interactively selected, filtered, or annotated.
015 *
016 *    The panel supports dynamic updates, overlay shading, numeric display
017 *    modes, and contextual popups showing detailed square information. It can
018 *    operate independently or as part of a larger viewer managed by
019 *    {@link paint.viewer.logic.SquareControlHandler}.
020 *
021 *  KEY FEATURES:
022 *    • Renders an interactive, data-driven grid of squares.
023 *    • Supports shaded overlays, numeric modes, and border control.
024 *    • Allows user-driven selection and assignment of cell IDs.
025 *    • Integrates with visibility filtering based on configurable parameters.
026 *    • Provides contextual info popups for detailed square statistics.

060/**
061 * The {@code SquareGridPanel} is a graphical component that renders a grid of
062 * {@link paint.shared.objects.Square} objects, each representing an individual
063 * region of interest. It provides rich interactivity such as selection,
064 * annotation, and contextual information popups.
065 *
066 * <p>Users can interactively select squares, assign them to cells, and toggle
067 * visibility or shading modes. Control parameters for filtering and display
068 * can be applied through integration with {@code SquareControlHandler}.</p>
069 *
070 * <p>The panel automatically handles painting of overlays, borders, and
071 * numerical annotations depending on its configured state.</p>
072 */
073public class SquareGridPanel extends JPanel {
074
075    private       Recording    recording;
076    private final int          rows;
077    private final int          cols;
078
079    private       Image        backgroundImage;
080    private       List<Square> squares                 = new ArrayList<>();
081    private final Set<Integer> selectedSquaresNumbers  = new HashSet<>();
082
083    private       boolean      showBorders             = true;
084    private       boolean      showShading             = true;
085    private       Rectangle    selectionRect           = null;
086    private       Point        dragStart               = null;
087
088    private       double       minRequiredDensityRatio = 0.0;
089    private       double       maxAllowableVariability = Double.MAX_VALUE;
090    private       double       minRequiredRSquared     = 0.0;
091    private       String       neighbourMode           = "Free";
092    private       boolean      selectionEnabled        = false;
093    private final Set<Integer> dragSelectedSquares     = new HashSet<>();

095    /**
096     * Enumeration defining numeric display modes for squares.
097     * <ul>
098     *   <li>{@code NONE}: No numbers are displayed.</li>
099     *   <li>{@code LABEL}: Shows label numbers for selected squares.</li>
100     *   <li>{@code SQUARE}: Shows square indices for selected squares.</li>
101     * </ul>
102     */
103    public enum NumberMode {
104        NONE,    // No numbers are drawn on squares.
105        LABEL,   // Draws label numbers on selected squares.
106        SQUARE   // Draws square numbers on selected squares.
107    }
108
109    private NumberMode numberMode = NumberMode.NONE;

115            Color.RED,
116            Color.GREEN,
117            Color.BLUE,
118            Color.MAGENTA,
119            Color.ORANGE,
120            Color.CYAN
121    };

125    /**
126     * Constructs a {@code SquareGridPanel} with the specified grid size and sets
127     * up mouse listeners for interaction (selection and popups).
128     *
129     * @param rows number of grid rows
130     * @param cols number of grid columns
131     */
132    public SquareGridPanel(int rows, int cols) {
133        this.rows = rows;
134        this.cols = cols;
135
136        setPreferredSize(new Dimension(NUMBER_PIXELS_WIDTH, NUMBER_PIXELS_HEIGHT));
137
138        // --- Mouse listener for selection and info popups ---
139        addMouseListener(new MouseAdapter() {
140            @Override
141            public void mousePressed(MouseEvent e) {
142                if (!selectionEnabled) {
143                    return;
144                }
145                dragStart = e.getPoint();
146                selectionRect = new Rectangle(dragStart);
147            }
148
149            @Override
150            public void mouseReleased(MouseEvent e) {
151                if (!selectionEnabled) {
152                    return;
153                }
154                if (selectionRect != null) {
155                    selectSquaresInRect(selectionRect);
156                    selectionRect = null;
157                    SquareGridPanel.this.repaint();
158                }
159            }
160
161            @Override
162            public void mouseClicked(MouseEvent e) {
163                int squareW = getWidth() / cols;
164                int squareH = getHeight() / rows;
165                int col = e.getX() / squareW;
166                int row = e.getY() / squareH;
167
168                if (selectionEnabled) {
169                    for (Square sq : squares) {
170                        if (sq.getRowNumber() == row && sq.getColNumber() == col) {
171                            boolean newSel = !sq.isSelected();
172                            sq.setSelected(newSel);
173                            if (newSel) {
174                                selectedSquaresNumbers.add(sq.getSquareNumber());
175                            } else {
176                                selectedSquaresNumbers.remove(sq.getSquareNumber());
177                            }
178                            SquareGridPanel.this.repaint();
179                            break;
180                        }
181                    }
182                }
183
184                // Left-click shows info; right-click hides popup
185                if (SwingUtilities.isLeftMouseButton(e)) {
186                    showSquareInfo(e.getX(), e.getY());
187                } else if (SwingUtilities.isRightMouseButton(e)) {
188                    hideInfoPopup();
189                }
190            }
191        });
192
193        // --- Mouse motion listener for drag-selection ---
194        addMouseMotionListener(new MouseMotionAdapter() {
195            @Override
196            public void mouseDragged(MouseEvent e) {
197                if (!selectionEnabled) {
198                    return;
199                }
200                int x = Math.min(dragStart.x, e.getX());
201                int y = Math.min(dragStart.y, e.getY());
202                int w = Math.abs(dragStart.x - e.getX());
203                int h = Math.abs(dragStart.y - e.getY());
204                selectionRect = new Rectangle(x, y, w, h);
205                SquareGridPanel.this.repaint();
206            }
207        });
208    }

210    /**
211     * Displays an informational popup for the square under the given coordinates.
212     *
213     * @param mouseX x-coordinate of the mouse
214     * @param mouseY y-coordinate of the mouse
215     */
216    private void showSquareInfo(int mouseX, int mouseY) {
217        if (squares == null || squares.isEmpty()) {
218            return;
219        }
220
221        int squareW = getWidth() / cols;
222        int squareH = getHeight() / rows;
223        int row     = mouseY / squareH;
224        int col     = mouseX / squareW;
225        int index   = row * cols + col;
226
227        if (index < 0 || index >= squares.size()) {
228            return;
229        }
230
231        Square square = squares.get(index);
232        int trackCount = square.getNumberOfTracks();
233
234        String html = String.format(
235                "<html><body style='font-family:sans-serif;font-size:11px;'>"
236                        + "<b>Square %d</b>"
237                        + "<table style='margin-top:4px;'>"
238                        + "<tr><td style='padding-right:8px;'>Density:</td><td align='right'>%.4f</td></tr>"
239                        + "<tr><td style='padding-right:8px;'>Density Ratio:</td><td align='right'>%.1f</td></tr>"
240                        + "<tr><td>Variability:</td><td align='right'>%.1f</td></tr>"
241                        + "<tr><td>R²:</td><td align='right'>%.2f</td></tr>"
242                        + "<tr><td>Tracks:</td><td align='right'>%d</td></tr>"
243                        + "</table>"
244                        + "<div style='margin-top:4px; font-style:italic; color:#666;'>"
245                        + "Right-click anywhere to close"
246                        + "</div>"
247                        + "</body></html>",
248                square.getSquareNumber(),
249                square.getDensity(),
250                square.getDensityRatio(),
251                square.getVariability(),
252                square.getRSquared(),
253                trackCount
254        );
255
256        if (infoPopup == null) {
257            infoPopup = new JWindow(SwingUtilities.getWindowAncestor(this));
258            JLabel label = new JLabel(html);
259            label.setOpaque(true);
260            label.setBackground(new Color(255, 255, 255, 230));
261            label.setBorder(BorderFactory.createCompoundBorder(
262                    BorderFactory.createLineBorder(Color.DARK_GRAY),
263                    BorderFactory.createEmptyBorder(6, 8, 6, 8)
264            ));
265            infoPopup.add(label);
266            infoPopup.pack();
267            infoPopup.setAlwaysOnTop(true);
268        } else {
269            ((JLabel) infoPopup.getContentPane().getComponent(0)).setText(html);
270        }
271
272        Point panelScreen = getLocationOnScreen();
273        int popupX = panelScreen.x + col * squareW + squareW + 8;
274        int popupY = panelScreen.y + row * squareH + squareH / 4;
275
276        Rectangle screen = GraphicsEnvironment.getLocalGraphicsEnvironment().getMaximumWindowBounds();
277        if (popupX + infoPopup.getWidth() > screen.x + screen.width) {
278            popupX = panelScreen.x + col * squareW - infoPopup.getWidth() - 8;
279        }
280        if (popupY + infoPopup.getHeight() > screen.y + screen.height) {
281            popupY = screen.y + screen.height - infoPopup.getHeight() - 8;
282        }
283
284        infoPopup.setLocation(popupX, popupY);
285        infoPopup.setVisible(true);
286    }

288    /**
289     * Hides and disposes the info popup if currently visible.
290     */
291    private void hideInfoPopup() {
292        if (infoPopup != null) {
293            infoPopup.setVisible(false);
294            infoPopup.dispose();
295            infoPopup = null;
296        }

299    /**
300     * Selects all squares that intersect the specified rectangle.
301     *
302     * @param rect selection area in panel coordinates
303     */
304    private void selectSquaresInRect(Rectangle rect) {
305        int squareW = getWidth() / cols;
306        int squareH = getHeight() / rows;
307
308        dragSelectedSquares.clear();
309
310        for (Square square : squares) {
311            Rectangle r = new Rectangle(
312                    square.getColNumber() * squareW,
313                    square.getRowNumber() * squareH,
314                    squareW, squareH
315            );
316            if (rect.intersects(r)) {
317                square.setSelected(true);
318                selectedSquaresNumbers.add(square.getSquareNumber());
319                dragSelectedSquares.add(square.getSquareNumber());
320            }
321        }
322
323        this.repaint();
324    }

373    /**
374     * Paints all visual elements including the grid, overlays, and selections.
375     */
376    @Override
377    protected void paintComponent(Graphics g) {
378        super.paintComponent(g);
379
380        if (backgroundImage != null) {
381            g.drawImage(backgroundImage, 0, 0, getWidth(), getHeight(), this);
382        }
383        if (squares == null) {
384            return;
385        }
386
387        int        squareW = getWidth() / cols;
388        int        squareH = getHeight() / rows;
389        Graphics2D g2      = (Graphics2D) g;
390
391        // --- Draw cells, overlays, and borders ---
392        for (Square square : squares) {
393            int x = square.getColNumber() * squareW;
394            int y = square.getRowNumber() * squareH;
395
396            // 1) Assigned-cell shading (base layer)
397            if (square.getCellId() > 0 && showShading) {
398                Color baseColor = getColorForCell(square.getCellId());
399                int   rVal      = Math.min(255, baseColor.getRed() + 40);
400                int   gVal      = Math.min(255, baseColor.getGreen() + 40);
401                int   bVal      = Math.min(255, baseColor.getBlue() + 40);
402                g2.setColor(new Color(rVal, gVal, bVal, 100));
403                g2.fillRect(x, y, squareW, squareH);
404            }
405
406            // 2) User-selection highlight (always on top of any assigned shading)
407            if (showShading) {
408                if (selectedSquaresNumbers.contains(square.getSquareNumber())) {
409                    g2.setColor(new Color(255, 235, 0, 200));
410                    g2.fillRect(x, y, squareW, squareH);
411                } else if (square.isSelected()) {
412                    g2.setColor(new Color(255, 255, 255, 80));
413                    g2.fillRect(x, y, squareW, squareH);
414                }
415            }
416
417            // 3) Borders (only if enabled)
418            if (showBorders) {
419                if (square.getCellId() > 0 || square.isSelected() || selectedSquaresNumbers.contains(square.getSquareNumber())) {
420                    g2.setStroke(new BasicStroke(1.0f));
421                    g2.setColor(Color.WHITE);
422                    g2.drawRect(x, y, squareW, squareH);
423                }
424            }
425        }
426
427        // --- Draw numbers inside selected squares ---
428        for (Square sq : squares) {
429            if (sq.isSelected()) {
430                int x = sq.getColNumber() * squareW;
431                int y = sq.getRowNumber() * squareH;
432                if (numberMode == NumberMode.LABEL) {
433                    drawCenteredString(g2, String.valueOf(sq.getLabelNumber()), x, y, squareW, squareH);
434                } else if (numberMode == NumberMode.SQUARE) {
435                    drawCenteredString(g2, String.valueOf(sq.getSquareNumber()), x, y, squareW, squareH);
436                }
437            }
438        }
439
440        // --- Draw drag-selection rectangle ---
441        if (selectionRect != null && selectionEnabled) {
442            g2.setColor(new Color(255, 255, 180, 100));
443            g2.fill(selectionRect);
444            g2.setColor(Color.BLACK);
445            g2.setStroke(new BasicStroke(1.5f));
446            g2.draw(selectionRect);
447        }
448    }

450    /**
451     * Draws centered text inside a given square.
452     */
453    private void drawCenteredString(Graphics g, String text, int x, int y, int w, int h) {
454        Font original = g.getFont();
455        Font small = original.deriveFont(original.getSize2D() * 0.8f);
456        g.setFont(small);
457
458        FontMetrics fm = g.getFontMetrics();
459        int tx = x + (w - fm.stringWidth(text)) / 2;
460        int ty = y + (h + fm.getAscent() - fm.getDescent()) / 2;
461        g.setColor(Color.WHITE);
462        g.drawString(text, tx, ty);
463        g.setFont(original);
464    }

489    /**
490     * Returns the consistent color associated with a specific cell ID.
491     */
492    public static Color getColorForCell(int cellId) {
493        if (cellId <= 0) {
494            return Color.GRAY;
495        }
496        return CELL_COLORS[(cellId - 1) % CELL_COLORS.length];
497    }

525    /**
526     * Assigns all selected squares to a given cell ID.
527     */
528    public void assignSelectedToCell(int cellId) {
529        if (squares == null) {
530            return;
531        }
532        for (Square sq : squares) {
533            if (sq.isSelected()) {
534                sq.setCellId(cellId);
535            }
536        }
537        this.repaint();
538    }
