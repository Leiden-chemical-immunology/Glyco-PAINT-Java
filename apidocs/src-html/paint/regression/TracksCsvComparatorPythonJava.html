350    /**
351     * Strip threshold suffix ("-threshold-N") from recording name.
352     */
353    private static String normalizeRecordingName(String name) {
354        if (name == null) {
355            return "";
356        }
357        name = name.trim();
358        int idx = name.indexOf("-threshold");
359        if (idx > 0) {
360            name = name.substring(0, idx);
361        }
362        return name;
363    }
364
365    /**
366     * Find best matching recording name among candidates (ignoring threshold).
367     */
368    private static String findClosestRecording(String oldRecNorm, Set<String> candidates) {
369        for (String cand : candidates) {
370            String normCand = normalizeRecordingName(cand);
371            if (oldRecNorm.equalsIgnoreCase(normCand)
372                    || oldRecNorm.startsWith(normCand)
373                    || normCand.startsWith(oldRecNorm)) {
374                return cand;
375            }
376        }
377        return null;
378    }

380    /**
381     * Normalize column names, recording names, and track IDs.
382     * Then sort by Recording Name + Track Id.
383     */
384    private static List<Map<String, String>> normalizeAndSort(List<Map<String, String>> rows,
385                                                              String recordingColumn) {
386        if (rows.isEmpty()) {
387            return rows;
388        }
389
390        for (Map<String, String> row : rows) {
391            // unify recording name
392            String recName = row.getOrDefault(recordingColumn, "").trim().replace('\u00A0', ' ');
393            int idx = recName.toLowerCase(Locale.ROOT).indexOf("-threshold");
394            if (idx > 0) {
395                recName = recName.substring(0, idx);
396            }
397            row.put("Recording Name", recName);
398
399            // normalize alternate column names
400            for (Map.Entry<String, String> e : COLUMN_MAP.entrySet()) {
401                if (row.containsKey(e.getKey()) && !row.containsKey(e.getValue())) {
402                    row.put(e.getValue(), row.get(e.getKey()));
403                }
404            }
405
406            // ensure Track Id field exists
407            String id = row.getOrDefault("Track Id", "").trim();
408            row.put("Track Id", id);
409        }
410
411        // sort deterministically
412        rows.sort((a, b) -> {
413            String ra = a.getOrDefault("Recording Name", "").toLowerCase(Locale.ROOT);
414            String rb = b.getOrDefault("Recording Name", "").toLowerCase(Locale.ROOT);
415            int cmp = ra.compareTo(rb);
416            if (cmp != 0) {
417                return cmp;
418            }
419            double ia = parseDoubleSafe(a.getOrDefault("Track Id", ""));
420            double ib = parseDoubleSafe(b.getOrDefault("Track Id", ""));
421            if (Double.isFinite(ia) && Double.isFinite(ib)) {
422                return Double.compare(ia, ib);
423            }
424            return a.getOrDefault("Track Id", "").compareTo(b.getOrDefault("Track Id", ""));
425        });
426
427        System.out.println("‚úÖ Normalized + sorted by Recording Name + Track Id");
428        return rows;
429    }

470    // --- Matching + metrics ---
471    private static List<Map<String, String>> findMatches(MatchConfig cfg, Map<String, String> old,
472                                                         List<Map<String, String>> candidates, Set<String> used) {
473        List<Map<String, String>> matches = new ArrayList<>();
474
475        // Extract numerical fields from "old" track
476        int square     = parseIntSafe(old.get("Square Nr"));
477        int nSpots     = parseIntSafe(old.get("Nr Spots"));
478        int nGaps      = parseIntSafe(old.get("Nr Gaps"));
479        int longest    = parseIntSafe(old.get("Longest Gap"));
480        double durOld  = parseDoubleSafe(old.get("Track Duration"));
481        double dispOld = parseDoubleSafe(old.get("Track Displacement"));
482        double maxOld  = parseDoubleSafe(old.get("Track Max Speed"));
483        double medOld  = parseDoubleSafe(old.get("Track Median Speed"));
484        double distOld = parseDoubleSafe(old.get("Total Distance"));
485        double xOld    = parseDoubleSafe(old.get("Track X Location"));
486        double yOld    = parseDoubleSafe(old.get("Track Y Location"));
487        double confOld = parseDoubleSafe(old.get("Confinement Ratio"));
488
489        // Iterate over all potential candidates in the same recording
490        for (Map<String, String> cand : candidates) {
491            String newId = cand.getOrDefault("Track Id", "");
492            if (used.contains(newId)) {
493                continue;  // Skip if already assigned
494            }
495
496            // Compare integer-like columns exactly
497            int squareNew  = parseIntSafe(cand.get("Square Number"));
498            int nSpotsNew  = parseIntSafe(cand.get("Number of Spots"));
499            int nGapsNew   = parseIntSafe(cand.get("Number of Gaps"));
500            int longestNew = parseIntSafe(cand.get("Longest Gap"));
501            if (square != squareNew || nSpots != nSpotsNew || nGaps != nGapsNew || longest != longestNew) {
502                continue; // Not comparable structure
503            }
504
505            // Compare numeric columns within tolerance
506            double durNew  = parseDoubleSafe(cand.get("Track Duration"));
507            double dispNew = parseDoubleSafe(cand.get("Track Displacement"));
508            double maxNew  = parseDoubleSafe(cand.get("Track Max Speed"));
509            double medNew  = parseDoubleSafe(cand.get("Track Median Speed"));
510            double distNew = parseDoubleSafe(cand.get("Total Distance"));
511            double xNew    = parseDoubleSafe(cand.get("Track X Location"));
512            double yNew    = parseDoubleSafe(cand.get("Track Y Location"));
513            double confNew = parseDoubleSafe(cand.get("Confinement Ratio"));
514
515            boolean ok = true;
516            if (cfg.useDuration) {
517                ok &= Math.abs(durOld - durNew) <= DURATION_TOLERANCE;
518            }
519            if (cfg.useDisplacement) {
520                ok &= Math.abs(dispOld - dispNew) <= DISPLACEMENT_TOLERANCE;
521            }
522            if (cfg.useSpeed) {
523                ok &= Math.abs(maxOld - maxNew) <= SPEED_TOLERANCE &&
524                        Math.abs(medOld - medNew) <= SPEED_TOLERANCE;
525            }
526            if (cfg.useDistance) {
527                ok &= Math.abs(distOld - distNew) <= DIST_TOLERANCE;
528            }
529            if (cfg.useXY) {
530                ok &= Math.abs(xOld - xNew) <= XY_TOLERANCE &&
531                        Math.abs(yOld - yNew) <= XY_TOLERANCE;
532            }
533            if (cfg.useConfinement) {
534                ok &= Math.abs(confOld - confNew) <= CONFINEMENT_TOLERANCE;
535            }
536
537            if (ok) {
538                matches.add(cand);
539            }
540        }
541        return matches;
542    }

544    /**
545     * Compute the best possible match (lowest RMS score) between an unmatched
546     * "old" track and a set of "new" candidates, based on relative deviation
547     * weighted by field tolerances.
548     *
549     * @param old        reference track
550     * @param candidates candidate new tracks
551     * @return the map containing the best candidate ID, per-field deviations, and score
552     */
553    private static Map<String, Object> findBestCandidate(
554            Map<String, String> old, List<Map<String, String>> candidates) {
555
556        double bestScore = Double.POSITIVE_INFINITY;
557        Map<String, String> bestCand = null;
558        double durD  = 0;
559        double dispD = 0;
560        double maxD  = 0;
561        double medD  = 0;
562        double distD = 0;
563        double xD    = 0;
564        double yD    = 0;
565        double confD = 0;
566
567        for (Map<String, String> cand : candidates) {
568            // Normalized differences divided by their respective tolerance
569            double dur  = diff(old, cand, "Track Duration", DURATION_TOLERANCE);
570            double disp = diff(old, cand, "Track Displacement", DISPLACEMENT_TOLERANCE);
571            double max  = diff(old, cand, "Track Max Speed", SPEED_TOLERANCE);
572            double med  = diff(old, cand, "Track Median Speed", SPEED_TOLERANCE);
573            double dist = diff(old, cand, "Total Distance", DIST_TOLERANCE);
574            double x    = diff(old, cand, "Track X Location", XY_TOLERANCE);
575            double y    = diff(old, cand, "Track Y Location", XY_TOLERANCE);
576            double conf = diff(old, cand, "Confinement Ratio", CONFINEMENT_TOLERANCE);
577
578            // RMS score across all differences
579            double score = Math.sqrt((dur * dur + disp * disp + max * max + med * med +
580                    dist * dist + x * x + y * y + conf * conf) / 8.0);
581
582            if (Double.isFinite(score) && score < bestScore && score <= MAX_ACCEPTABLE_SCORE) {
583                bestScore = score;
584                bestCand  = cand;
585                durD      = dur;
586                dispD     = disp;
587                maxD      = max;
588                medD      = med;
589                distD     = dist;
590                xD        = x;
591                yD        = y;
592                confD     = conf;
593            }
594        }
595
596        // Prepare result summary
597        Map<String, Object> out = new LinkedHashMap<>();
598        if (bestCand != null) {
599            out.put("BestId", bestCand.get("Track Id"));
600            out.put("score", String.format(Locale.US, "%.3f", bestScore));
601        } else {
602            out.put("BestId", "-");
603            out.put("score", "No acceptable match");
604        }
605        out.put("durD", durD);
606        out.put("dispD", dispD);
607        out.put("maxD", maxD);
608        out.put("medD", medD);
609        out.put("distD", distD);
610        out.put("xD", xD);
611        out.put("yD", yD);
612        out.put("confD", confD);
613        return out;
614    }

690    /**
691     * Compute normalized difference divided by tolerance (used in RMS scoring).
692     */
693    private static double diff(Map<String, String> a, Map<String, String> b, String field, double tol) {
694        double va = parseDoubleSafe(a.get(field));
695        double vb = parseDoubleSafe(b.get(field));
696        if (Double.isNaN(va) || Double.isNaN(vb)) {
697            return Double.POSITIVE_INFINITY;
698        }
699        return Math.abs(va - vb) / tol;
700    }

705     * Safe numeric parsing ‚Äî returns NaN for empty or invalid input.
706     */
707    private static double parseDoubleSafe(String s) {
708        if (s == null || s.isEmpty()) {
709            return Double.NaN;
710        }
711        try {
712            return Double.parseDouble(s.trim());
713        } catch (Exception e) {
714            return Double.NaN;
715        }
716    }

720     */
721    private static int parseIntSafe(String s) {
722        if (s == null || s.isEmpty()) {
723            return -1;
724        }
725        try {
726            return (int) Math.round(Double.parseDouble(s.trim()));
727        } catch (Exception e) {
728            return -1;

732    /**
733     * Generic CSV reader returning list of maps (columnName ‚Üí value).
734     * No quoting is processed beyond trimming ‚Äî suitable for standard CSVs.
735     */
736    private static List<Map<String, String>> readCsv(Path path) throws IOException {
737        List<Map<String, String>> rows = new ArrayList<>();
738        try (BufferedReader br = Files.newBufferedReader(path)) {
739            String headerLine = br.readLine();
740            if (headerLine == null) {
741                return rows;
742            }
743            String[] headers = headerLine.split(",", -1);
744            String line;
745            while ((line = br.readLine()) != null) {
746                String[] parts = line.split(",", -1);
747                Map<String, String> map = new LinkedHashMap<>();
748                for (int i = 0; i < headers.length; i++) {
749                    map.put(headers[i].trim(), i < parts.length ? parts[i].trim() : "");
750                }
751                rows.add(map);
752            }
753        }
754        System.out.println("Loaded " + rows.size() + " rows from " + path.getFileName());
755        return rows;
756    }

761     * Escape CSV fields containing commas or quotes.
762     */
763    private static String escapeCsv(String s) {
764        if (s == null) {
765            return "";
766        }
767        if (s.contains(",") || s.contains("\"")) {
768            s = s.replace("\"", "\"\"");
769            return "\"" + s + "\"";
770        }
771        return s;
772    }

775     * Group list of maps by the value in a given column.
776     */
777    private static Map<String, List<Map<String, String>>> groupBy(List<Map<String, String>> rows, String col) {
778        Map<String, List<Map<String, String>>> grouped = new LinkedHashMap<>();
779        for (Map<String, String> r : rows) {
780            String key = r.getOrDefault(col, "").trim();
781            grouped.computeIfAbsent(key, k -> new ArrayList<>()).add(r);

799    /**
800     * Evaluate how much count changes when tightening per-field tolerance thresholds.
801     */
802    private static void optimizeTolerances(Path diagCsv, Path outCsv) throws IOException {
803        if (!Files.exists(diagCsv)) {
804            System.out.println("‚ö†Ô∏è No diagnostics available for optimization step.");
805            return;
806        }
807
808        Map<String, List<Double>> deviationsByField = new LinkedHashMap<>();
809
810        try (BufferedReader br = Files.newBufferedReader(diagCsv)) {
811            String line;
812            while ((line = br.readLine()) != null) {
813                String[] parts = line.split(",", -1);
814                if (parts.length < 12) {
815                    continue;
816                }
817                deviationsByField.computeIfAbsent("Duration Œî", k -> new ArrayList<>()).add(parseDoubleSafe(parts[3]));
818                deviationsByField.computeIfAbsent("Displacement Œî", k -> new ArrayList<>()).add(parseDoubleSafe(parts[4]));
819                deviationsByField.computeIfAbsent("Max Speed Œî", k -> new ArrayList<>()).add(parseDoubleSafe(parts[5]));
820                deviationsByField.computeIfAbsent("Median Speed Œî", k -> new ArrayList<>()).add(parseDoubleSafe(parts[6]));
821                deviationsByField.computeIfAbsent("Total Distance Œî", k -> new ArrayList<>()).add(parseDoubleSafe(parts[7]));
822                deviationsByField.computeIfAbsent("X Œî",              k -> new ArrayList<>()).add(parseDoubleSafe(parts[8]));
823                deviationsByField.computeIfAbsent("Y Œî",              k -> new ArrayList<>()).add(parseDoubleSafe(parts[9]));
824                deviationsByField.computeIfAbsent("Confinement Œî",    k -> new ArrayList<>()).add(parseDoubleSafe(parts[10]));
825            }
826        }
827
828        double[] testLevels = {5.0, 4.0, 3.0, 2.0, 1.0, 0.5};
829        double targetKeep = 98.0; // percent of matches to retain
830
831        try (PrintWriter pw = new PrintWriter(outCsv.toFile())) {
832            pw.println("Field,Total,Equal@5%,OptimalTolerance(%),Equal@Optimal(%)");
833
834            for (Map.Entry<String, List<Double>> e : deviationsByField.entrySet()) {
835                String field = e.getKey();
836                List<Double> devs = e.getValue();
837                devs.removeIf(d -> !Double.isFinite(d));
838                if (devs.isEmpty()) {
839                    continue;
840                }
841                int total = devs.size();
842
843                double equalAt5 = percentWithin(devs, 5.0);
844                double bestTol = 5.0;
845                double bestKeep = equalAt5;
846
847                for (double tol : testLevels) {
848                    double keep = percentWithin(devs, tol);
849                    if (keep < targetKeep) {
850                        break;
851                    }
852                    bestTol = tol;
853                    bestKeep = keep;
854                }
855
856                pw.printf(Locale.US, "%s,%d,%.2f,%.2f,%.2f%n",
857                          field, total, equalAt5, bestTol, bestKeep);
858            }
859        }
860    }

862    /**
863     * Helper: compute fraction of deviations within tolerance.
864     */
865    private static double percentWithin(List<Double> devs, double tol) {
866        long ok = devs.stream().filter(d -> Math.abs(d) <= tol).count();
867        return 100.0 * ok / devs.size();
868    }
