057/**

059 * all differences between them. The comparison is performed row-wise and
060 * field-wise, detecting added, missing, or differing values.

063 * Recordings) and adapts automatically based on which columns are present.
064 * It can also be executed from the command line or used programmatically.

081        final PrintStream fileOut = new PrintStream(Files.newOutputStream(logFile));

086        PrintStream dualOut = new PrintStream(new OutputStream() {
087            @Override
088            public void write(int b) {
089                originalOut.write(b);  // use original, not System.out
090                fileOut.write(b);

092
093            @Override
094            public void write(byte[] b, int off, int len) {
095                originalOut.write(b, off, len);
096                fileOut.write(b, off, len);

098
099            @Override
100            public void flush() {
101                originalOut.flush();
102                fileOut.flush();
103            }
104        }, true);
105
106        System.setOut(dualOut);
107        System.setErr(dualOut);
108        System.out.println("🧾 Logging to: " + logFile.toAbsolutePath());
109        System.out.println();

114    /**
115     * Compares two CSV files for differences based on their content, generates
116     * a detailed comparison report, and returns the total number of detected differences.
117     */
118    public static int compareFiles(Path oldCsv, Path newCsv) throws IOException {
119
120        // === Load both files ===
121        System.out.println("📥 Reading baseline file...");
122        List<Map<String, String>> oldRows = readCsv(oldCsv);
123        System.out.println("   → " + oldRows.size() + " rows loaded.");
124
125        System.out.println("📥 Reading test file...");
126        List<Map<String, String>> newRows = readCsv(newCsv);
127        System.out.println("   → " + newRows.size() + " rows loaded.");
128
129        if (oldRows.size() == newRows.size()) {
130            System.out.printf("✅ Same number of rows (%d). Ready for strict 1:1 comparison.%n", oldRows.size());
131        } else {
132            System.out.printf("⚠️  Different row counts: baseline=%d, test=%d%n", oldRows.size(), newRows.size());
133        }
134
135        // Build (Recording Name [+ Square Nr]) → list of rows
136        Map<String, List<Map<String, String>>> oldMulti = toMultiMap(oldRows);
137        Map<String, List<Map<String, String>>> newMulti = toMultiMap(newRows);
138
139        Set<String> allKeys = new TreeSet<>(oldMulti.keySet());
140        allKeys.addAll(newMulti.keySet());
141
142        System.out.println("\n🔎 Starting detailed comparison...");
143        System.out.println("   → Total unique (Recording Name [+ Square Nr]) keys: " + allKeys.size());
144        System.out.println();
145
146        List<String[]> diffs = new ArrayList<>();
147        int diffCount = 0;
148
149        int processed = 0;
150        int totalKeys = allKeys.size();
151
152        // === Core comparison loop ===
153        for (String key : allKeys) {
154            List<Map<String, String>> ol = oldMulti.getOrDefault(key, Collections.emptyList());
155            List<Map<String, String>> nl = newMulti.getOrDefault(key, Collections.emptyList());
156
157            int max = Math.max(ol.size(), nl.size());
158
159            for (int i = 0; i < max; i++) {
160                Map<String, String> o = (i < ol.size()) ? ol.get(i) : null;
161                Map<String, String> n = (i < nl.size()) ? nl.get(i) : null;
162
163                if (o == null && n != null) {
164                    String rec = safe(n.get("Recording Name"));
165                    String sq = n.containsKey("Square Number") ? safe(n.get("Square Number")) : "";
166                    diffs.add(new String[]{rec, sq, String.valueOf(i + 1), "", "", "", "Extra in NEW"});
167                    diffCount++;
168                    continue;
169                } else if (o != null && n == null) {
170                    String rec = safe(o.get("Recording Name"));
171                    String sq = o.containsKey("Square Number") ? safe(o.get("Square Number")) : "";
172                    diffs.add(new String[]{rec, sq, String.valueOf(i + 1), "", "", "", "Missing in NEW"});
173                    diffCount++;
174                    continue;
175                }
176
177                // Compare common fields
178                Set<String> fields = new LinkedHashSet<>();
179                fields.addAll(o.keySet());
180                fields.addAll(n.keySet());
181
182                String rec = safe(o.get("Recording Name"));
183                String sq = o.containsKey("Square Number") ? safe(o.get("Square Number")) : "";
184
185                for (String f : fields) {
186                    if (f == null || f.trim().isEmpty()) {
187                        continue;
188                    }
189                    if (IGNORE_COLUMNS.contains(f)) {
190                        continue;
191                    }
192
193                    String ov = clean(o.get(f));
194                    String nv = clean(n.get(f));
195
196                    if (valuesEqual(ov, nv)) {
197                        continue;
198                    }
199
200                    Double od = parseDouble(ov);
201                    Double nd = parseDouble(nv);
202                    String status = (od != null && nd != null)
203                            ? "NUMERIC DIFFERENCE" : "TEXT DIFFERENCE";
204
205                    diffs.add(new String[]{rec, sq, String.valueOf(i + 1), f, ov, nv, status});
206                    diffCount++;
207                }
208            }
209
210            processed++;
211            if (totalKeys <= 20) {
212                System.out.println("   ✓ Compared: " + key);
213            } else if (processed % 100 == 0) {
214                System.out.printf("   ...processed %d/%d keys%n", processed, totalKeys);
215            }
216        }
217
218        // === Grouped Difference Summary (aligned globally) ===
219        if (!diffs.isEmpty()) {
220            // Group by recording and square
221            Map<String, Map<String, List<String[]>>> grouped = new LinkedHashMap<>();
222            for (String[] row : diffs) {
223                String rec = row[0];
224                String sq = row[1].isEmpty() ? "—" : row[1];
225                grouped.computeIfAbsent(rec, r -> new LinkedHashMap<>())
226                        .computeIfAbsent(sq, s -> new ArrayList<>())
227                        .add(row);
228            }
229
230            // Compute global max field name width for perfect alignment
231            int globalFieldWidth = diffs.stream()
232                    .map(row -> row[3] == null ? 0 : row[3].length())
233                    .max(Integer::compareTo)
234                    .orElse(0) + 2;
235
236            System.out.println("\n🔎 Differences grouped by Square");
237            System.out.println("───────────────────────────────");
238
239            int total = 0;
240            Set<String> squaresWithDiffs = new TreeSet<>();
241
242            for (Map.Entry<String, Map<String, List<String[]>>> recEntry : grouped.entrySet()) {
243                System.out.println("Recording: " + recEntry.getKey());
244                for (Map.Entry<String, List<String[]>> sqEntry : recEntry.getValue().entrySet()) {
245                    String sq = sqEntry.getKey();
246                    if (!sq.equals("—")) {
247                        squaresWithDiffs.add(sq);
248                    }
249                    List<String[]> entries = sqEntry.getValue();
250
251                    System.out.println("  ▫ Square " + sq + ":");
252                    for (String[] row : entries) {
253                        String field = row[3];
254                        String oldV  = row[4];
255                        String newV  = row[5];
256                        String stat  = row[6];
257                        System.out.printf("     - %-" + globalFieldWidth + "s: '%s' vs '%s' (%s)%n",
258                                          field, oldV, newV, stat);
259                        total++;
260                    }
261                }
262                System.out.println();
263            }
264
265            System.out.printf("%n📊 Total differences listed: %d%n", total);
266            if (!squaresWithDiffs.isEmpty()) {
267                System.out.printf("🟧 Squares with at least one difference: %d (%s)%n",
268                                  squaresWithDiffs.size(),
269                                  String.join(", ", squaresWithDiffs));
270            }
271        } else {
272            System.out.println("\n✅ No differences detected.");
273        }
274        return diffCount;
275    }

278    private static String buildKey(Map<String, String> r) {
279        String rec = safe(r.get("Recording Name"));
280        String sq = r.containsKey("Square Number") ? safe(r.get("Square Number")) : "";
281        return sq.isEmpty() ? rec : rec + " - " + sq;
282    }
283
284    private static Map<String, List<Map<String, String>>> toMultiMap(List<Map<String, String>> rows) {
285        Map<String, List<Map<String, String>>> mm = new TreeMap<>();
286        for (Map<String, String> r : rows) {
287            String key = buildKey(r);
288            mm.computeIfAbsent(key, k -> new ArrayList<>()).add(r);
289        }
290        return mm;
291    }

293    private static List<Map<String, String>> readCsv(Path path) throws IOException {
294        List<Map<String, String>> rows = new ArrayList<>();
295        try (BufferedReader br = Files.newBufferedReader(path)) {
296            String headerLine = br.readLine();
297            if (headerLine == null) {
298                return rows;
299            }
300
301            String[] headers = headerLine.split(",", -1);
302            for (int i = 0; i < headers.length; i++) {
303                headers[i] = headers[i].trim();
304            }
305
306            String line;
307            while ((line = br.readLine()) != null) {
308                if (line.isEmpty()) {
309                    continue;
310                }
311                String[] vals = line.split(",", -1);
312                Map<String, String> row = new LinkedHashMap<>();
313                for (int i = 0; i < headers.length; i++) {
314                    String h = headers[i];
315                    String v = (i < vals.length ? vals[i] : "");
316                    row.put(h, v == null ? "" : v.trim());
317                }
318                rows.add(row);
319            }
320        }
321        return rows;
322    }

324    private static String clean(String s) {
325        if (s == null) {
326            return "";
327        }
328        String t = s.trim();
329        if (t.equalsIgnoreCase("nan") || t.equalsIgnoreCase("null")) {
330            return "";
331        }
332        return t;
333    }

335    private static boolean valuesEqual(String a, String b) {
336        if (Objects.equals(a, b)) {
337            return true;
338        }
339        Double da = parseDouble(a);
340        Double db = parseDouble(b);
341        if (da != null && db != null) {
342            double ra = Math.round(da * 1000.0) / 1000.0;
343            double rb = Math.round(db * 1000.0) / 1000.0;
344            return Double.compare(ra, rb) == 0;
345        }
346        return false;
347    }
348
349    private static Double parseDouble(String s) {
350        if (s == null || s.isEmpty()) {
351            return null;
352        }
353        try {
354            double v = Double.parseDouble(s);
355            return Double.isNaN(v) ? null : v;
356        } catch (Exception e) {
357            return null;
358        }
359    }
360
361    private static String safe(String s) {
362        return (s == null) ? "" : s;
363    }
