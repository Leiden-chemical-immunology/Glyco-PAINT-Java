061    // ---------------------------------------------------------------------
062    private static List<Map<String, String>> readCsv(Path p) throws IOException {
063        List<Map<String, String>> rows = new ArrayList<>();
064        try (BufferedReader br = Files.newBufferedReader(p)) {
065            String header = br.readLine();
066            if (header == null) return rows;
067            String[] h = header.split(",", -1);
068            String line;
069            while ((line = br.readLine()) != null) {
070                if (line.trim().isEmpty()) continue;
071                String[] parts = line.split(",", -1);
072                Map<String, String> m = new LinkedHashMap<>();
073                for (int i = 0; i < h.length; i++) {
074                    m.put(h[i].trim(), i < parts.length ? parts[i].trim() : "");
075                }
076                rows.add(m);
077            }
078        }
079        return rows;
080    }
081
082    // ---------------------------------------------------------------------
083    private static List<String[]> compare(List<Map<String, String>> oldRows, List<Map<String, String>> newRows) {
084        Map<String, Map<String, String>> newMap = new HashMap<>();
085        for (Map<String, String> n : newRows) newMap.put(key(n), n);
086
087        List<String[]> out = new ArrayList<>();
088        int missing = 0;
089        int showFirst = 10;
090
091        for (Map<String, String> o : oldRows) {
092            String k = key(o);
093            Map<String, String> n = newMap.get(k);
094            if (n == null) {
095                if (missing < showFirst) System.out.printf("‚ö†Ô∏è Missing key %s%n", k);
096                missing++;
097                continue;
098            }
099
100            List<String> row = new ArrayList<>();
101            row.add(o.getOrDefault("Recording Name", ""));
102            row.add(n.getOrDefault("Recording Name", ""));
103
104            for (String field : NUMERIC_FIELDS) {
105                String oldVal = o.getOrDefault(field, "");
106                String newVal = n.getOrDefault(field, "");
107
108                double oldNum = parseDouble(oldVal);
109                double newNum = parseDouble(newVal);
110                double diff = newNum - oldNum;
111                double diffPer = (oldNum == 0) ? 0 : (diff / oldNum) * 100.0;
112
113                row.add(oldVal);
114                row.add(newVal);
115                row.add(format(diff));
116                row.add(formatPercent(diffPer));  // üîπ use one-decimal formatting
117                row.add(""); // separator column
118            }
119
120            out.add(row.toArray(new String[0]));
121        }
122
123        if (missing > 0)
124            System.out.printf("üîé Keys not found in NEW: %,d (printed first %d above)%n", missing, Math.min(missing, 10));
125
126        System.out.printf("   ‚úÖ Matched %,d of %,d old rows%n", out.size(), oldRows.size());
127        return out;
128    }
129
130    // ---------------------------------------------------------------------
131    private static String key(Map<String, String> r) {
132        return r.getOrDefault("Recording Name", "") + "||" + r.getOrDefault("Square Nr", "");
133    }
134
135    private static double parseDouble(String s) {
136        if (s == null || s.isEmpty()) return 0.0;
137        try { return Double.parseDouble(s); }
138        catch (Exception e) { return 0.0; }
139    }
140
141    private static String format(double v) {
142        if (Double.isNaN(v)) return "";
143        return String.format(Locale.US, "%.4f", v);
144    }
145
146    private static String formatPercent(double v) {
147        if (Double.isNaN(v)) return "";
148        return String.format(Locale.US, "%.1f", v); // üîπ one decimal for percentages
149    }

151    private static List<String> header() {
152        List<String> h = new ArrayList<>();
153        h.add("Recording Name");
154        h.add("Recording Name Java");
155        for (String f : NUMERIC_FIELDS) {
156            h.add(f);
157            h.add(f + " Java");
158            h.add(f + " Diff");
159            h.add(f + " DiffPer");
160            h.add(""); // empty separator column
161        }
162        return h;
163    }
