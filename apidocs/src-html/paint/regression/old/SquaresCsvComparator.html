166    private static List<Map<String,String>> readCsv(Path p) throws IOException {
167        List<Map<String,String>> rows = new ArrayList<>();
168        try (BufferedReader br = Files.newBufferedReader(p)) {
169            String header = br.readLine();
170            if (header == null) return rows;
171            String[] h = header.split(",",-1);
172            String line;
173            while ((line = br.readLine()) != null) {
174                if (line.trim().isEmpty()) continue;
175                String[] parts = line.split(",",-1);
176                Map<String,String> m = new LinkedHashMap<>();
177                for (int i=0;i<h.length;i++) m.put(h[i].trim(), i<parts.length?parts[i].trim():"");
178                rows.add(m);
179            }
180        }
181        return rows;
182    }

184    /** Write normalized CSV using shared precision and zero→empty logic for old file. */
185    private static void writeCsv(List<Map<String, String>> rows, Path file) throws IOException {
186        if (rows.isEmpty()) return;
187        Files.createDirectories(file.getParent());
188
189        boolean isOldFile = file.getFileName().toString().toLowerCase(Locale.ROOT).contains("old");
190
191        try (PrintWriter pw = new PrintWriter(file.toFile())) {
192            List<String> header = new ArrayList<>();
193            header.add("Recording Name");
194            header.addAll(FIELD_MAP.keySet());
195            header.add("Selected");
196            pw.println(String.join(",", header));
197
198            for (Map<String, String> r : rows) {
199                List<String> vals = new ArrayList<>();
200                for (String col : header) {
201                    String val = r.getOrDefault(col, "");
202
203                    // ---- Handle numeric formatting ----
204                    if (NUMERIC_FIELDS.contains(col)) {
205                        Double num = parseDouble(val);
206                        if (num != null) {
207                            // 🔹 OLD FILES: apply "0 → empty" rule for specific fields
208                            if (isOldFile && num == 0.0 && Arrays.asList(
209                                    "R Squared",
210                                    "Median Short Track Duration",
211                                    "Median Long Track Duration",
212                                    "Total Displacement",
213                                    "Total Track Duration"
214                            ).contains(col)) {
215                                val = "";
216                            } else {
217                                int prec = EFFECTIVE_PRECISION_MAP.getOrDefault(
218                                        col, ROUNDING_MAP.getOrDefault(col, 3));
219                                double f = Math.pow(10, prec);
220                                double rounded = Math.round(num * f) / f;
221                                val = String.format(Locale.US, "%." + prec + "f", rounded);
222                            }
223                        }
224                    }
225
226                    vals.add(escapeCsv(val));
227                }
228                pw.println(String.join(",", vals));
229            }
230        }
231    }

233    private static String escapeCsv(String s) {
234        if (s==null) return "";
235        if (s.contains(",")||s.contains("\"")) return "\"" + s.replace("\"","\"\"") + "\"";
236        return s;
237    }

243    public static List<Map<String,String>> normalizeOld(List<Map<String,String>> oldRows){
244        List<Map<String,String>> out = new ArrayList<>();
245        for (Map<String,String> r: oldRows){
246            Map<String,String> n = new LinkedHashMap<>();
247            String rec = r.getOrDefault("Ext Recording Name","");
248            n.put("Recording Name", rec.replaceAll("-threshold-\\d+$","").trim());
249            for (String f: FIELD_MAP.keySet()) {
250                String v = r.getOrDefault(f,"").trim();
251                if (f.equals("Tau")) {
252                    try { if (!v.isEmpty() && Double.parseDouble(v)<0) v=""; } catch(Exception ignored){}
253                }
254                n.put(f,v);
255            }
256            adjustIndex(n,"Row Nr");
257            adjustIndex(n,"Col Nr");
258            n.put("Selected", String.valueOf(isSelected(n)));
259            out.add(n);
260        }
261        return out;
262    }

264    public static List<Map<String,String>> normalizeNew(List<Map<String,String>> newRows){
265        List<Map<String,String>> out = new ArrayList<>();
266        for (Map<String,String> r: newRows){
267            Map<String,String> n = new LinkedHashMap<>();
268            n.put("Recording Name", r.getOrDefault("Recording Name",""));
269            for (Map.Entry<String,String> e: FIELD_MAP.entrySet())
270                n.put(e.getKey(), r.getOrDefault(e.getValue(),""));
271            n.put("Selected", String.valueOf(isSelected(n)));
272            out.add(n);
273        }
274        return out;
275    }

277    private static void adjustIndex(Map<String,String> m, String k){
278        try {
279            String v=m.get(k);
280            if (v!=null&&!v.isEmpty()) {
281                int i=(int)Double.parseDouble(v);
282                m.put(k,String.valueOf(i-1));
283            }
284        } catch(Exception ignored){}
285    }

287    private static Map<String,Integer> computeEffectivePrecisions(
288            List<Map<String,String>> a,List<Map<String,String>> b,Set<String> numeric){
289        Map<String,Integer> res=new HashMap<>();
290        for (String f:numeric){
291            int pa=detectPrecision(a,f);
292            int pb=detectPrecision(b,f);
293            res.put(f,Math.min(pa,pb));
294        }
295        return res;
296    }
297
298    private static int detectPrecision(List<Map<String,String>> rows,String f){
299        int best=0;
300        for (Map<String,String> r: rows){
301            String s=r.get(f);
302            if (s!=null&&s.matches("^-?\\d+\\.\\d+$")){
303                int p=s.length()-s.indexOf('.')-1;
304                best=Math.max(best,p);
305            }
306        }
307        return best;
308    }

314    private static void compare(List<Map<String,String>> oldN,List<Map<String,String>> newN,Path outDir) throws IOException {
315        Path out=outDir.resolve("Squares Validation - Comparison.csv");
316        List<String[]> diffs=new ArrayList<>();
317        int total=0,diffCount=0;
318
319        Map<String,Map<String,String>> newMap=new HashMap<>();
320        for (Map<String,String> n:newN) newMap.put(key(n),n);
321
322        for (Map<String,String> o: oldN){
323            total++;
324            String k=key(o);
325            Map<String,String> n=newMap.get(k);
326            if (n==null) continue;
327
328            for (String f : FIELD_MAP.keySet()) {
329                String ov = o.getOrDefault(f, "");
330                String nv = n.getOrDefault(f, "");
331
332                if (NUMERIC_FIELDS.contains(f)) {
333                    Double da = parseDouble(ov);
334                    Double db = parseDouble(nv);
335                    double dev = relativeDeviation(ov, nv);
336                    double tol = TOLERANCE_MAP.getOrDefault(f, 5.0);
337                    int prec = EFFECTIVE_PRECISION_MAP.getOrDefault(f, ROUNDING_MAP.get(f));
338
339                    String status;
340                    if (da == null && db == null) {
341                        status = "BOTH EMPTY";
342                    } else if (Double.isNaN(dev)) {
343                        status = "NaN";
344                    } else if (dev <= tol) {
345                        status = "WITHIN " + tol + "%";
346                    } else {
347                        status = "DIFFERENT";
348                        diffCount++;
349                    }
350
351                    diffs.add(new String[]{
352                            o.get("Recording Name"),
353                            o.get("Square Nr"),
354                            f,
355                            ov,
356                            nv,
357                            String.valueOf(prec),
358                            String.format(Locale.US, "%.3f", dev),
359                            status
360                    });
361
362                } else if (!Objects.equals(ov, nv)) {
363                    diffs.add(new String[]{
364                            o.get("Recording Name"),
365                            o.get("Square Nr"),
366                            f,
367                            ov,
368                            nv,
369                            "",
370                            "",
371                            "TEXT DIFFERENCE"
372                    });
373                    diffCount++;
374                }
375            }
376            String sOld=o.get("Selected");
377            String sNew=n.get("Selected");
378            if (!Objects.equals(sOld,sNew)){
379                diffs.add(new String[]{o.get("Recording Name"),o.get("Square Nr"),"Selected",sOld,sNew,"","","DIFFERENT"});
380                diffCount++;
381            }
382
383            if (total%1000==0) System.out.printf("   ...processed %,d%n",total);
384        }
385
386        try(PrintWriter pw=new PrintWriter(out.toFile())){
387            pw.println("Recording Name,Square Nr,Field,Old Value,New Value,Precision Used,Relative Diff (%),Status");
388            for(String[] r:diffs) pw.println(String.join(",",r));
389            pw.println();
390            pw.printf("SUMMARY,,,,,,,%nDifferences,%d%n",diffCount);
391        }
392        System.out.printf("✅ Compared %,d squares — %d differences%n",total,diffCount);
393    }

399    private static boolean numericEqual(String f,String a,String b){
400        Double da=parseDouble(a), db=parseDouble(b);
401        if (da==null && db==null) return true;
402        if (da==null || db==null) return false;
403        int prec=EFFECTIVE_PRECISION_MAP.getOrDefault(f,ROUNDING_MAP.getOrDefault(f,3));
404        double fac=Math.pow(10,prec);
405        return Math.round(da*fac)==Math.round(db*fac);
406    }

408    private static double relativeDeviation(String a, String b) {
409        Double da = parseDouble(a);
410        Double db = parseDouble(b);
411
412        // Both missing or equivalent to zero → no deviation
413        if ((da == null || da == 0.0) && (db == null || db == 0.0)) return 0.0;
414
415        // One side empty → treat as absolute deviation
416        if (da == null || db == null) return 100.0; // arbitrary high deviation
417
418        double absA = Math.abs(da);
419        double absDiff = Math.abs(db - da);
420
421        // Avoid division by zero, use absolute diff if old value is near zero
422        if (absA < 1e-12) return absDiff * 100.0;
423
424        return absDiff / absA * 100.0;
425    }

432        if (s==null||s.isEmpty()||s.equalsIgnoreCase("NaN")) return null;
433        try { double v=Double.parseDouble(s); return Double.isNaN(v)?null:v; }
434        catch(Exception e){return null;}

437    private static boolean isSelected(Map<String,String> r){
438        Double dr=parseDouble(r.get("Density Ratio"));
439        Double var=parseDouble(r.get("Variability"));
440        Double r2=parseDouble(r.get("R Squared"));
441        return dr!=null&&var!=null&&r2!=null&&dr>=2.0&&var<10.0&&r2>0.1;
442    }

475    /**
476     * Evaluate how many values fall within various relative tolerance levels
477     * to identify optimal per-field thresholds.
478     */
479    private static void optimizeTolerances(Path comparisonCsv, Path outCsv) throws IOException {
480        if (!Files.exists(comparisonCsv)) {
481            System.out.println("⚠️ No comparison file found for tolerance optimization.");
482            return;
483        }
484
485        System.out.println("🔬 Analyzing tolerance levels across numeric fields...");
486
487        Map<String, List<Double>> diffsByField = new LinkedHashMap<>();
488
489        // --- Read comparison CSV and extract relative differences ---
490        try (BufferedReader br = Files.newBufferedReader(comparisonCsv, java.nio.charset.StandardCharsets.UTF_8)) {
491            String header = br.readLine(); // skip header
492            String line;
493            while ((line = br.readLine()) != null) {
494                if (line.trim().isEmpty() || line.startsWith("SUMMARY")) continue;
495                String[] parts = line.split(",", -1);
496                if (parts.length < 7) continue;
497                String field = parts[2].trim();
498                String rel = parts[6].trim();
499                if (rel.isEmpty() || rel.equalsIgnoreCase("NaN")) continue;
500                try {
501                    double d = Double.parseDouble(rel);
502                    diffsByField.computeIfAbsent(field, k -> new ArrayList<>()).add(d);
503                } catch (Exception ignored) {}
504            }
505        }
506
507        if (diffsByField.isEmpty()) {
508            System.out.println("ℹ️ No numeric deviations detected — skipping optimization.");
509            return;
510        }
511
512        double[] testLevels = {5.0, 4.0, 3.0, 2.0, 1.0, 0.5};
513        double targetKeep = 98.0; // target: retain at least this % of matches
514
515        // --- Write optimization summary in UTF-8 ---
516        try (PrintWriter pw = new PrintWriter(
517                Files.newBufferedWriter(outCsv, java.nio.charset.StandardCharsets.UTF_8))) {
518
519            pw.println("Field,Total,Equal@5%,OptimalTolerance(%),Equal@Optimal(%)");
520
521            for (Map.Entry<String, List<Double>> e : diffsByField.entrySet()) {
522                String field = e.getKey();
523                List<Double> diffs = e.getValue();
524                diffs.removeIf(d -> !Double.isFinite(d));
525                if (diffs.isEmpty()) continue;
526
527                int total = diffs.size();
528                double equalAt5 = percentWithin(diffs, 5.0);
529                double bestTol = 5.0;
530                double bestKeep = equalAt5;
531
532                for (double tol : testLevels) {
533                    double keep = percentWithin(diffs, tol);
534                    if (keep < targetKeep) break;
535                    bestTol = tol;
536                    bestKeep = keep;
537                }
538
539                pw.printf(Locale.US, "%s,%d,%.2f,%.2f,%.2f%n",
540                          field, total, equalAt5, bestTol, bestKeep);
541            }
542        }
543
544        System.out.println("📈 Tolerance optimization summary written: " + outCsv.toAbsolutePath());
545    }
546
547    /** Helper: compute fraction of deviations within given tolerance. */
548    private static double percentWithin(List<Double> vals, double tol) {
549        long ok = vals.stream().filter(v -> Math.abs(v) <= tol).count();
550        return 100.0 * ok / vals.size();
551    }
