109    private static List<Map<String, String>> readCsv(Path path) throws IOException {
110        List<Map<String, String>> rows = new ArrayList<>();
111        try (BufferedReader reader = new BufferedReader(new FileReader(path.toFile()))) {
112            String headerLine = reader.readLine();
113            if (headerLine == null) return rows;
114            String[] headers = headerLine.split(",", -1);
115            String line;
116            while ((line = reader.readLine()) != null) {
117                String[] values = line.split(",", -1);
118                Map<String, String> row = new LinkedHashMap<>();
119                for (int i = 0; i < Math.min(headers.length, values.length); i++) {
120                    row.put(headers[i].trim(), values[i].trim());
121                }
122                rows.add(row);
123            }
124        }
125        return rows;
126    }

129    private static List<Map<String, String>> normalizeOld(List<Map<String, String>> oldRows) {
130        List<Map<String, String>> norm = new ArrayList<>();
131        for (Map<String, String> row : oldRows) {
132            Map<String, String> n = new LinkedHashMap<>();
133            n.put("Recording Name", row.getOrDefault("Ext Recording Name", ""));
134
135            // Remove trailing "-threshold-<number>"
136            String recName = n.get("Recording Name");
137            if (recName != null) {
138                recName = recName.replaceAll("-threshold-\\d+$", "").trim();
139                n.put("Recording Name", recName);
140            }
141
142            for (String oldCol : FIELD_MAP.keySet()) {
143                String val = row.getOrDefault(oldCol, "").trim();
144
145                // Tau normalization
146                if (oldCol.equalsIgnoreCase("Tau")) {
147                    try {
148                        if (!val.isEmpty() && Double.parseDouble(val) < 0)
149                            val = "";
150                    } catch (NumberFormatException ignored) {}
151                }
152
153                // 0-to-empty normalization
154                if (oldCol.equalsIgnoreCase("R Squared")
155                        || oldCol.equalsIgnoreCase("Median Long Track Duration")
156                        || oldCol.equalsIgnoreCase("Median Short Track Duration")
157                        || oldCol.equalsIgnoreCase("Total Track Duration")
158                        || oldCol.equalsIgnoreCase("Density")
159                        || oldCol.equalsIgnoreCase("Density Ratio")
160                        || oldCol.equalsIgnoreCase("Variability")
161                        || oldCol.equalsIgnoreCase("Total Displacement")) {
162                    try {
163                        if (!val.isEmpty() && Double.parseDouble(val) == 0.0)
164                            val = "";
165                    } catch (NumberFormatException ignored) {}
166                }
167
168                n.put(oldCol, val);
169            }
170
171            // Normalize Row/Col to 0-based
172            try {
173                String rowVal = n.get("Row Nr");
174                if (rowVal != null && !rowVal.isEmpty()) {
175                    int rowNum = (int) Double.parseDouble(rowVal);
176                    n.put("Row Nr", String.valueOf(rowNum - 1));
177                }
178            } catch (NumberFormatException ignored) {}
179            try {
180                String colVal = n.get("Col Nr");
181                if (colVal != null && !colVal.isEmpty()) {
182                    int colNum = (int) Double.parseDouble(colVal);
183                    n.put("Col Nr", String.valueOf(colNum - 1));
184                }
185            } catch (NumberFormatException ignored) {}
186
187            // Compute Selected
188            n.put("Selected", String.valueOf(isSelected(n)));
189
190            norm.add(n);
191        }
192        return norm;
193    }

195    private static List<Map<String, String>> normalizeNew(List<Map<String, String>> newRows) {
196        List<Map<String, String>> norm = new ArrayList<>();
197        for (Map<String, String> row : newRows) {
198            Map<String, String> n = new LinkedHashMap<>();
199            n.put("Recording Name", row.getOrDefault("Recording Name", ""));
200            for (Map.Entry<String, String> entry : FIELD_MAP.entrySet()) {
201                String oldCol = entry.getKey();
202                String newCol = entry.getValue();
203                n.put(oldCol, row.getOrDefault(newCol, ""));
204            }
205            // NEW is already 0-based; nothing to adjust for row/col
206            // Compute Selected
207            n.put("Selected", String.valueOf(isSelected(n)));
208
209            norm.add(n);
210        }
211        return norm;
212    }

215    private static Map<String, Integer> computeEffectivePrecisions(
216            List<Map<String, String>> oldRows,
217            List<Map<String, String>> newRows,
218            Set<String> numericFields) {
219
220        Map<String, Integer> precisionMap = new HashMap<>();
221        for (String field : numericFields) {
222            int oldPrec = detectPrecision(oldRows, field);
223            int newPrec = detectPrecision(newRows, field);
224            precisionMap.put(field, Math.min(oldPrec, newPrec));
225        }
226        return precisionMap;
227    }
228
229    private static int detectPrecision(List<Map<String, String>> rows, String field) {
230        int best = 0;
231        for (Map<String, String> row : rows) {
232            String s = row.get(field);
233            if (s != null) {
234                s = s.trim();
235                if (s.matches("^-?\\d+\\.\\d+$")) {
236                    int p = s.length() - s.indexOf('.') - 1;
237                    if (p > best) best = p;
238                }
239            }
240        }
241        return best;
242    }

250    private static void writeCsv(List<Map<String, String>> rows, Path file) throws IOException {
251        if (rows.isEmpty()) return;
252        Files.createDirectories(file.getParent());
253        try (PrintWriter pw = new PrintWriter(file.toFile())) {
254            // Ensure "Selected" exists and appears last in header (after FIELD_MAP)
255            List<String> header = new ArrayList<>();
256            header.add("Recording Name");
257            header.addAll(FIELD_MAP.keySet());
258            if (!header.contains("Selected")) header.add("Selected");
259            pw.println(String.join(",", header));
260
261            for (Map<String, String> r : rows) {
262                List<String> vals = new ArrayList<>();
263                for (String c : header) {
264                    String val = r.get(c);
265                    if (val == null) val = "";
266                    String out = val;
267
268                    if (DOUBLE_FIELDS.contains(c)) {
269                        Double num = parseDouble(val);
270                        if (num != null) {
271                            int digits = EFFECTIVE_PRECISION_MAP.getOrDefault(
272                                    c, ROUNDING_MAP.getOrDefault(c, 3));
273                            double factor = Math.pow(10, digits);
274                            double rounded = Math.round(num * factor) / factor;
275                            out = String.format(Locale.US, "%." + digits + "f", rounded);
276                        }
277                    }
278                    vals.add(escapeCsv(out));
279                }
280                pw.println(String.join(",", vals));
281            }
282        }
283    }

286    private static boolean compareDoubles(String field, String a, String b) {
287        String ta = (a == null) ? "" : a.trim();
288        String tb = (b == null) ? "" : b.trim();
289
290        boolean aSpecial = isZeroNaNOrEmpty(ta);
291        boolean bSpecial = isZeroNaNOrEmpty(tb);
292        if (aSpecial && bSpecial) return true;
293
294        if (field.equalsIgnoreCase("Tau")) {
295            try {
296                if (!ta.isEmpty() && Double.parseDouble(ta) < 0 && isZeroNaNOrEmpty(tb))
297                    return true;
298            } catch (NumberFormatException ignored) {}
299        }
300
301        Double dx = parseDouble(ta);
302        Double dy = parseDouble(tb);
303        if (dx != null && dy != null) {
304            int d = EFFECTIVE_PRECISION_MAP.getOrDefault(field, ROUNDING_MAP.getOrDefault(field, 3));
305            double f = Math.pow(10, d);
306            double rx = Math.round(dx * f) / f;
307            double ry = Math.round(dy * f) / f;
308            return Double.compare(rx, ry) == 0;
309        }
310        return false;
311    }
312
313    private static boolean isZeroNaNOrEmpty(String s) {
314        if (s == null) return true;
315        String t = s.trim();
316        if (t.isEmpty()) return true;
317        if (t.equalsIgnoreCase("NaN")) return true;
318        try {
319            double v = Double.parseDouble(t);
320            return Math.abs(v) < 1e-12 || Double.isNaN(v);
321        } catch (Exception e) {
322            return false;
323        }
324    }
325
326    private static Double parseDouble(String s) {
327        if (s == null || s.isEmpty() || s.equalsIgnoreCase("NaN")) return null;
328        try {
329            double v = Double.parseDouble(s);
330            return Double.isNaN(v) ? null : v;
331        } catch (Exception e) {
332            return null;
333        }
334    }
335
336    /** Selected := (Density Ratio >= 2) AND (Variability < 10) AND (R Squared > 0.1) */
337    private static boolean isSelected(Map<String, String> row) {
338        Double densityRatio = parseDouble(row.get("Density Ratio"));
339        Double variability  = parseDouble(row.get("Variability"));
340        Double rSquared     = parseDouble(row.get("R Squared"));
341        if (densityRatio == null || variability == null || rSquared == null) return false;
342        return densityRatio >= 2.0 && variability < 10.0 && rSquared > 0.1;
343    }

345    private static String quote(String s) {
346        if (s == null) return "";
347        String t = s.replace("\"", "\"\"");
348        if (t.contains(",") || t.contains("\"") || t.contains(" ")) return "\"" + t + "\"";
349        return t;
350    }
351
352    private static String escapeCsv(String s) {
353        if (s == null) return "";
354        if (s.contains(",") || s.contains("\"")) return "\"" + s.replace("\"", "\"\"") + "\"";
355        return s;
356    }

360    private static void compare(List<Map<String, String>> oldNorm, List<Map<String, String>> newNorm) throws IOException {
361        Path reportFile = Paths.get("/Users/Hans/Desktop/Squares Validation - Comparison.csv");
362
363        // Sort inputs
364        Comparator<Map<String, String>> sorter = Comparator
365                .comparing((Map<String, String> r) -> r.get("Recording Name"))
366                .thenComparing(r -> {
367                    String s = r.get("Square Nr");
368                    try { return Integer.parseInt(s); } catch (Exception e) { return 0; }
369                });
370        oldNorm.sort(sorter);
371        newNorm.sort(sorter);
372
373        Map<String, Map<String, String>> newMap = new HashMap<>();
374        for (Map<String, String> n : newNorm) newMap.put(compositeKey(n), n);
375
376        List<String[]> reportRows = new ArrayList<>();
377
378        int matched = 0, mismatched = 0, missing = 0, extra = 0, selectedMismatches = 0;
379
380        for (Map<String, String> oldRow : oldNorm) {
381            String key = compositeKey(oldRow);
382            Map<String, String> newRow = newMap.get(key);
383
384            if (newRow == null) {
385                reportRows.add(new String[]{ oldRow.get("Recording Name"), oldRow.get("Square Nr"),
386                        "", "", "", "", "", "Missing in NEW" });
387                missing++;
388                continue;
389            }
390
391            boolean diff = false;
392
393            // Compare numeric & mapped fields
394            for (String field : FIELD_MAP.keySet()) {
395                String oldVal = oldRow.getOrDefault(field, "");
396                String newVal = newRow.getOrDefault(field, "");
397
398                boolean equal = DOUBLE_FIELDS.contains(field)
399                        ? compareDoubles(field, oldVal, newVal)
400                        : Objects.equals(oldVal, newVal);
401
402                if (!equal) {
403                    diff = true;
404                    int prec = EFFECTIVE_PRECISION_MAP.getOrDefault(field, ROUNDING_MAP.getOrDefault(field, -1));
405
406                    // Relative difference
407                    String relDiffStr = "";
408                    Double dx = parseDouble(oldVal);
409                    Double dy = parseDouble(newVal);
410                    if (dx != null && dy != null && Math.abs(dx) > 1e-12) {
411                        double relDiff = Math.abs((dy - dx) / dx) * 100.0;
412                        relDiffStr = String.format(Locale.US, "%.2f", relDiff);
413                    }
414
415                    reportRows.add(new String[]{
416                            oldRow.get("Recording Name"),
417                            oldRow.get("Square Nr"),
418                            field,
419                            oldVal,
420                            newVal,
421                            String.valueOf(prec),
422                            relDiffStr,
423                            "DIFFERENT"
424                    });
425                }
426            }
427
428            // Compare Selected flag
429            String selOld = String.valueOf(isSelected(oldRow));
430            String selNew = String.valueOf(isSelected(newRow));
431            if (!selOld.equalsIgnoreCase(selNew)) {
432                diff = true;
433                selectedMismatches++;
434                reportRows.add(new String[]{
435                        oldRow.get("Recording Name"),
436                        oldRow.get("Square Nr"),
437                        "Selected",
438                        selOld,
439                        selNew,
440                        "",         // precision N/A
441                        "",         // relative diff N/A
442                        "DIFFERENT"
443                });
444            }
445
446            if (diff) mismatched++; else matched++;
447        }
448
449        // Extra rows in NEW
450        Set<String> oldKeys = new HashSet<>();
451        for (Map<String, String> o : oldNorm) oldKeys.add(compositeKey(o));
452        for (Map<String, String> n : newNorm) {
453            if (!oldKeys.contains(compositeKey(n))) {
454                reportRows.add(new String[]{
455                        n.get("Recording Name"), n.get("Square Nr"),
456                        "", "", "", "", "", "Extra in NEW"
457                });
458                extra++;
459            }
460        }
461
462        // Sort report rows by key
463        reportRows.sort(Comparator
464                                .comparing((String[] r) -> r[0] == null ? "" : r[0])
465                                .thenComparing(r -> {
466                                    try { return Integer.parseInt(r[1]); } catch (Exception e) { return 0; }
467                                }));
468
469        // Write report
470        try (PrintWriter report = new PrintWriter(reportFile.toFile())) {
471            report.println("Recording Name,Square Nr,Field,Old Value,New Value,Precision Used,Relative Difference (%),Status");
472            for (String[] row : reportRows) {
473                List<String> escaped = new ArrayList<>();
474                for (String v : row) escaped.add(quote(v));
475                report.println(String.join(",", escaped));
476            }
477            report.println();
478            report.println("SUMMARY,,,,,,,");
479            report.printf("Matched,%d,,,,,,%n", matched);
480            report.printf("Mismatched,%d,,,,,,%n", mismatched);
481            report.printf("Missing,%d,,,,,,%n", missing);
482            report.printf("Extra,%d,,,,,,%n", extra);
483            report.printf("Selected mismatches,%d,,,,,,%n", selectedMismatches);
484            report.flush();
485        }
486
487        // Console summary
488        System.out.printf("✅ Comparison complete. Report: %s%n", reportFile);
489        System.out.printf("Matched: %d, Mismatched: %d, Missing: %d, Extra: %d, Selected mismatches: %d%n",
490                          matched, mismatched, missing, extra, selectedMismatches);
491    }

496    /**
497     * Write an overview CSV listing for each square:
498     * Selected (old/new), Tau, Density Ratio, and Variability,
499     * plus combined selection state (Both/OnlyOld/OnlyNew).
500     */
501    private static void writeSelectedOverview(List<Map<String, String>> oldNorm,
502                                              List<Map<String, String>> newNorm) throws IOException {
503        Path overviewFile = Paths.get("/Users/Hans/Desktop/Squares Validation - Selected Overview.csv");
504
505        Comparator<Map<String, String>> sorter = Comparator
506                .comparing((Map<String, String> r) -> r.get("Recording Name"))
507                .thenComparing(r -> {
508                    String s = r.get("Square Nr");
509                    try { return Integer.parseInt(s); } catch (Exception e) { return 0; }
510                });
511        oldNorm.sort(sorter);
512        newNorm.sort(sorter);
513
514        Map<String, Map<String, String>> newMap = new HashMap<>();
515        for (Map<String, String> n : newNorm) newMap.put(compositeKey(n), n);
516
517        Set<String> allKeys = new TreeSet<>();
518        for (Map<String, String> o : oldNorm) allKeys.add(compositeKey(o));
519        for (Map<String, String> n : newNorm) allKeys.add(compositeKey(n));
520
521        try (PrintWriter pw = new PrintWriter(overviewFile.toFile())) {
522            pw.println("Recording Name,Square Nr,"
523                               + "Selected (Old),Selected (New),Both Selected,Only Old Selected,Only New Selected,"
524                               + "Tau (Old),Tau (New),Density Ratio (Old),Density Ratio (New),Variability (Old),Variability (New)");
525
526            for (String key : allKeys) {
527                String rec = key.split(" - ")[0];
528                String sq  = key.split(" - ").length > 1 ? key.split(" - ")[1] : "";
529
530                Map<String, String> oldRow = oldNorm.stream()
531                        .filter(r -> compositeKey(r).equals(key)).findFirst().orElse(null);
532                Map<String, String> newRow = newMap.get(key);
533
534                boolean selOldBool = oldRow != null && isSelected(oldRow);
535                boolean selNewBool = newRow != null && isSelected(newRow);
536
537                String selOld = oldRow != null ? String.valueOf(selOldBool) : "";
538                String selNew = newRow != null ? String.valueOf(selNewBool) : "";
539
540                // Combined flags
541                String bothSelected = (selOldBool && selNewBool) ? "true" : "";
542                String onlyOldSelected = (selOldBool && !selNewBool) ? "true" : "";
543                String onlyNewSelected = (!selOldBool && selNewBool) ? "true" : "";
544
545                String tauOld = oldRow != null ? oldRow.getOrDefault("Tau", "") : "";
546                String tauNew = newRow != null ? newRow.getOrDefault("Tau", "") : "";
547
548                String drOld = oldRow != null ? oldRow.getOrDefault("Density Ratio", "") : "";
549                String drNew = newRow != null ? newRow.getOrDefault("Density Ratio", "") : "";
550
551                String varOld = oldRow != null ? oldRow.getOrDefault("Variability", "") : "";
552                String varNew = newRow != null ? newRow.getOrDefault("Variability", "") : "";
553
554                pw.printf("%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s%n",
555                          quote(rec), quote(sq),
556                          quote(selOld), quote(selNew),
557                          quote(bothSelected), quote(onlyOldSelected), quote(onlyNewSelected),
558                          quote(tauOld), quote(tauNew),
559                          quote(drOld), quote(drNew),
560                          quote(varOld), quote(varNew));
561            }
562        }
563
564        System.out.printf("📊 Selected overview written: %s%n", overviewFile);
565    }
