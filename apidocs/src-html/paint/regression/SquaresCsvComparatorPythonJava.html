240    /**
241     * Reads a CSV file from the given path and parses its content into a list of maps.
242     * Each map represents a row, where the keys are the column headers
243     * and the values are the corresponding cell values.
244     *
245     * @param p the path to the CSV file to be read
246     * @return a list of maps where each map corresponds to a row in the CSV file with key-value pairs of column headers to cell values
247     * @throws IOException if there is an error opening or reading the file
248     */
249    private static List<Map<String, String>> readCsv(Path p) throws IOException {
250        List<Map<String, String>> rows = new ArrayList<>();
251        try (BufferedReader br = Files.newBufferedReader(p)) {
252            String header = br.readLine();
253            if (header == null) {
254                return rows;
255            }
256            String[] h = header.split(",", -1);
257            String line;
258            while ((line = br.readLine()) != null) {
259                if (line.trim().isEmpty()) {
260                    continue;
261                }
262                String[] parts = line.split(",", -1);
263                Map<String, String> m = new LinkedHashMap<>();
264                for (int i = 0; i < h.length; i++) {
265                    m.put(h[i].trim(), i < parts.length ? parts[i].trim() : "");
266                }
267                rows.add(m);
268            }
269        }
270        return rows;
271    }

273    /**
274     * Writes a list of rows represented as key-value pairs into a CSV file at the specified location.
275     * The method creates the necessary directories if they do not exist and processes special formatting
276     * requirements for numeric fields and headers.
277     *
278     * @param rows a list of maps where each map represents a row of the CSV. Keys are column names, and values are cell data.
279     * @param file the path to the CSV file to be written. Directories in the path will be created if they do not exist.
280     * @throws IOException if an I/O error occurs during file creation or writing.
281     */
282    private static void writeCsv(List<Map<String, String>> rows, Path file) throws IOException {
283        if (rows.isEmpty()) {
284            return;
285        }
286        Files.createDirectories(file.getParent());
287
288        // Determine whether this is the "old" file (Python source)
289        boolean isOldFile = file.getFileName().toString().toLowerCase(Locale.ROOT).contains("old");
290
291        try (PrintWriter pw = new PrintWriter(file.toFile())) {
292            // Write header
293            List<String> header = new ArrayList<>();
294            header.add("Recording Name");
295            header.addAll(FIELD_MAP.keySet());
296            header.add("Selected");
297            pw.println(String.join(",", header));
298
299            // Write data rows
300            for (Map<String, String> r : rows) {
301                List<String> vals = new ArrayList<>();
302                for (String col : header) {
303                    String val = r.getOrDefault(col, "");
304
305                    // Handle numeric columns: round or blank-out invalid zero values
306                    if (NUMERIC_FIELDS.contains(col)) {
307                        Double num = parseDouble(val);
308                        if (num != null) {
309                            if (isOldFile && num == 0.0 && Arrays.asList(
310                                    "R Squared",
311                                    "Median Short Track Duration",
312                                    "Median Long Track Duration",
313                                    "Total Displacement",
314                                    "Total Track Duration",
315                                    "Variability",
316                                    "Density",
317                                    "Density Ratio"
318                            ).contains(col)) {
319                                // Blank these zeros in the old file
320                                val = "";
321                            } else {
322                                int prec = EFFECTIVE_PRECISION_MAP.getOrDefault(
323                                        col, ROUNDING_MAP.getOrDefault(col, 3));
324                                double f = Math.pow(10, prec);
325                                double rounded = Math.round(num * f) / f;
326                                val = String.format(Locale.US, "%." + prec + "f", rounded);
327                            }
328                        }
329                    }
330                    vals.add(escapeCsv(val));
331                }
332                pw.println(String.join(",", vals));
333            }
334        }
335    }

337    /**
338     * Escapes a string for safe usage in a CSV file. If the string contains
339     * a comma or a double quote, it will be wrapped in double quotes,
340     * and any double quotes within the string will be escaped by doubling them.
341     * If the input is null, an empty string is returned.
342     *
343     * @param s the string to be escaped for CSV formatting; may be null
344     * @return a CSV-safe version of the input string, or an empty string if the input is null
345     */
346    private static String escapeCsv(String s) {
347        if (s == null) {
348            return "";
349        }
350        if (s.contains(",") || s.contains("\"")) {
351            return "\"" + s.replace("\"", "\"\"") + "\"";
352        }
353        return s;
354    }

358    /**
359     * Normalizes and transforms a list of rows (represented as maps)
360     * by cleaning and remapping fields, adjusting indices, and
361     * computing selection flags.
362     *
363     * @param oldRows a list of maps representing rows of data to be normalized.
364     *                Each map contains key-value pairs of the original data.
365     * @return a list of maps where each map represents a normalized row of data
366     *         with cleaned and remapped fields, adjusted indices, and computed
367     *         selection flags.
368     */
369    public static List<Map<String, String>> normalizeOld(List<Map<String, String>> oldRows) {
370        List<Map<String, String>> out = new ArrayList<>();
371        for (Map<String, String> r : oldRows) {
372            Map<String, String> n = new LinkedHashMap<>();
373
374            // Normalize recording name (strip "-threshold-N")
375            String rec = r.getOrDefault("Ext Recording Name", "");
376            n.put("Recording Name", rec.replaceAll("-threshold-\\d+$", "").trim());
377
378            // Map and clean all fields
379            for (String f : FIELD_MAP.keySet()) {
380                String v = r.getOrDefault(f, "").trim();
381                if (f.equals("Tau")) {
382                    try {
383                        if (!v.isEmpty() && Double.parseDouble(v) < 0) {
384                            v = "";
385                        }
386                    } catch (Exception ignored) {
387                    }
388                }
389                n.put(f, v);
390            }
391
392            // Adjust 1-based row/column indices to 0-based
393            adjustIndex(n, "Row Nr");
394            adjustIndex(n, "Col Nr");
395
396            // Compute selection flag based on field criteria
397            n.put("Selected", String.valueOf(isSelected(n)));
398            out.add(n);
399        }
400        return out;
401    }

403    /**
404     * Normalizes a list of maps by restructuring and updating each map's key-value pairs
405     * based on predefined mappings and logic.
406     *
407     * @param newRows a list of maps representing rows of data to be normalized
408     * @return a list of normalized maps with updated key-value pairs
409     */
410    public static List<Map<String, String>> normalizeNew(List<Map<String, String>> newRows) {
411        List<Map<String, String>> out = new ArrayList<>();
412        for (Map<String, String> r : newRows) {
413            Map<String, String> n = new LinkedHashMap<>();
414            n.put("Recording Name", r.getOrDefault("Recording Name", ""));
415            for (Map.Entry<String, String> e : FIELD_MAP.entrySet()) {
416                n.put(e.getKey(), r.getOrDefault(e.getValue(), ""));
417            }
418            n.put("Selected", String.valueOf(isSelected(n)));
419            out.add(n);
420        }
421        return out;
422    }

424    /**
425     * Converts a 1-based integer field to 0-based if it parses as a number.
426     *
427     * @param m row map to mutate
428     * @param k key whose value should be decremented (Row Nr / Col Nr)
429     */
430    private static void adjustIndex(Map<String, String> m, String k) {
431        try {
432            String v = m.get(k);
433            if (v != null && !v.isEmpty()) {
434                int i = (int) Double.parseDouble(v);
435                m.put(k, String.valueOf(i - 1));
436            }
437        } catch (Exception ignored) {
438        }
439    }

441    /**
442     * Detects the shared numeric precision (decimal places) for all
443     * numeric fields across two datasets by taking the minimum precision
444     * observed per field.
445     *
446     * @param a       first dataset (normalized rows)
447     * @param b       second dataset (normalized rows)
448     * @param numeric set of field names to consider as numeric
449     * @return map of field → effective precision
450     */
451    private static Map<String, Integer> computeEffectivePrecisions(
452            List<Map<String, String>> a, List<Map<String, String>> b, Set<String> numeric) {
453        Map<String, Integer> res = new HashMap<>();
454        for (String f : numeric) {
455            int pa = detectPrecision(a, f);
456            int pb = detectPrecision(b, f);
457            res.put(f, Math.min(pa, pb));
458        }
459        return res;
460    }
461
462    /**
463     * Determines the maximum number of fractional digits present for a field
464     * within a list of row maps.
465     *
466     * @param rows rows to scan
467     * @param f    field name
468     * @return the highest number of decimals seen; 0 if none
469     */
470    private static int detectPrecision(List<Map<String, String>> rows, String f) {
471        int best = 0;
472        for (Map<String, String> r : rows) {
473            String s = r.get(f);
474            if (s != null && s.matches("^-?\\d+\\.\\d+$")) {
475                int p = s.length() - s.indexOf('.') - 1;
476                best = Math.max(best, p);
477            }
478        }
479        return best;
480    }

484    /**
485     * Compares normalized old/new CSVs field-by-field and writes the
486     * per-field comparison report, including differences, missing values,
487     * and field-level summaries.
488     *
489     * @param oldN   normalized old data
490     * @param newN   normalized new data
491     * @param outDir directory where the comparison CSV is written
492     * @throws IOException if writing the comparison file fails
493     */
494    private static void compareStatus(List<Map<String, String>> oldN, List<Map<String, String>> newN, Path outDir) throws IOException {
495        Path out = outDir.resolve("Squares Validation - Comparison.csv");
496
497        List<String[]> diffs = new ArrayList<>();
498        int total = 0, diffCount = 0;
499        Set<String> squaresWithDiffs = new HashSet<>();
500
501        // Track field-level differences
502        Map<String, Integer> diffByField = new HashMap<>();
503        Map<String, Integer> missingByField = new HashMap<>();
504
505        // Build lookup map for new dataset
506        Map<String, Map<String, String>> newMap = new HashMap<>();
507        for (Map<String, String> n : newN) {
508            newMap.put(key(n), n);
509        }
510
511        // Iterate all old records and compare with matching new record
512        for (Map<String, String> o : oldN) {
513            total++;
514            String k = key(o);
515            Map<String, String> n = newMap.get(k);
516            if (n == null) {
517                continue;
518            }
519
520            for (String f : FIELD_MAP.keySet()) {
521                String ov = o.getOrDefault(f, "");
522                String nv = n.getOrDefault(f, "");
523
524                // Handle optional 0/empty fields gracefully
525                if (isOptionalZeroField(f) && (isZeroOrEmpty(ov) || isZeroOrEmpty(nv))) {
526                    boolean ovEmpty = isZeroOrEmpty(ov);
527                    boolean nvEmpty = isZeroOrEmpty(nv);
528                    if (ovEmpty && nvEmpty) {
529                        continue; // both empty → skip
530                    }
531
532                    // One side missing → record as "MISSING"
533                    if (ovEmpty) {
534                        ov = "";
535                    }
536                    if (nvEmpty) {
537                        nv = "";
538                    }
539                    diffs.add(new String[]{o.get("Recording Name"), o.get("Square Nr"), f, ov, nv, "", "", "MISSING"});
540                    missingByField.merge(f, 1, Integer::sum);
541                    squaresWithDiffs.add(k);
542                    continue;
543                }
544
545                // Numeric comparison logic
546                if (NUMERIC_FIELDS.contains(f)) {
547                    Double da = parseDouble(ov);
548                    Double db = parseDouble(nv);
549
550                    if ((da == null || isZeroOrEmpty(ov)) && (db == null || isZeroOrEmpty(nv))) {
551                        continue;
552                    }
553
554                    if ((da == null || isZeroOrEmpty(ov)) || (db == null || isZeroOrEmpty(nv))) {
555                        if (isZeroOrEmpty(ov)) {
556                            ov = "";
557                        }
558                        if (isZeroOrEmpty(nv)) {
559                            nv = "";
560                        }
561                        diffs.add(new String[]{o.get("Recording Name"), o.get("Square Nr"), f, ov, nv, "", "", "MISSING"});
562                        missingByField.merge(f, 1, Integer::sum);
563                        squaresWithDiffs.add(k);
564                        continue;
565                    }
566
567                    // Compute relative deviation and status
568                    double dev = relativeDeviation(ov, nv);
569                    double tol = TOLERANCE_MAP.getOrDefault(f, 5.0);
570                    int prec = EFFECTIVE_PRECISION_MAP.getOrDefault(f, ROUNDING_MAP.getOrDefault(f, 3));
571
572                    if (Double.isNaN(dev)) {
573                        continue;
574                    }
575
576                    String status;
577                    if (Math.abs(da - db) < 1e-12) {
578                        status = "EQUAL";
579                    } else if (dev <= tol) {
580                        status = "WITHIN " + tol + "%";
581                    } else {
582                        status = "DIFFERENT";
583                        diffCount++;
584                        diffByField.merge(f, 1, Integer::sum);
585                        squaresWithDiffs.add(k);
586                    }
587
588                    diffs.add(new String[]{o.get("Recording Name"), o.get("Square Nr"), f,
589                            ov, nv, String.valueOf(prec),
590                            String.format(Locale.US, "%.3f", dev), status});
591                }
592            }
593
594            // Compare the "Selected" field between old/new
595            String sOld = o.get("Selected");
596            String sNew = n.get("Selected");
597            if (!Objects.equals(sOld, sNew)) {
598                diffs.add(new String[]{o.get("Recording Name"), o.get("Square Nr"),
599                        "Selected", sOld, sNew, "", "", "DIFFERENT"});
600                diffCount++;
601                diffByField.merge("Selected", 1, Integer::sum);
602                squaresWithDiffs.add(k);
603            }
604
605            if (total % 1000 == 0) {
606                System.out.printf("   ...processed %,d%n", total);
607            }
608        }
609
610        // Write the comparison file
611        try (PrintWriter pw = new PrintWriter(out.toFile())) {
612            pw.println("Recording Name,Square Nr,Field,Old Value,New Value,Precision Used,Relative Diff (%),Status");
613            for (String[] r : diffs) {
614                pw.println(String.join(",", r));
615            }
616            pw.println();
617            pw.printf("SUMMARY,,,,,,,%nDifferences,%d%n", diffCount);
618
619            // Overview of differences and missing values
620            if (!diffByField.isEmpty()) {
621                pw.println();
622                pw.println("Field Difference Overview (Status=DIFFERENT):");
623                pw.println("Field,Count");
624                diffByField.entrySet().stream()
625                        .sorted(Map.Entry.<String, Integer>comparingByValue().reversed())
626                        .forEach(e -> pw.printf("%s,%d%n", e.getKey(), e.getValue()));
627            }
628            if (!missingByField.isEmpty()) {
629                pw.println();
630                pw.println("Field Missing Overview (one side empty):");
631                pw.println("Field,Count");
632                missingByField.entrySet().stream()
633                        .sorted(Map.Entry.<String, Integer>comparingByValue().reversed())
634                        .forEach(e -> pw.printf("%s,%d%n", e.getKey(), e.getValue()));
635            }
636        }
637
638        // Console summary
639        System.out.printf("✅ Compared %,d squares — %d differences in %d squares%n",
640                          total, diffCount, squaresWithDiffs.size());
641    }

645    /**
646     * Writes a detailed numeric comparison for all numeric fields,
647     * including absolute and percentage differences per field.
648     *
649     * @param oldN    normalized old data
650     * @param newN    normalized new data
651     * @param outFile output CSV file to create
652     * @throws IOException if writing fails
653     */
654    private static void writeDetailed(List<Map<String, String>> oldN, List<Map<String, String>> newN, Path outFile) throws IOException {
655        Map<String, Map<String, String>> newMap = new HashMap<>();
656        for (Map<String, String> n : newN) {
657            newMap.put(key(n), n);
658        }
659
660        List<List<String>> rows = new ArrayList<>();
661        int matched = 0, total = oldN.size();
662
663        for (Map<String, String> o : oldN) {
664            Map<String, String> n = newMap.get(key(o));
665            if (n == null) {
666                continue;
667            }
668            matched++;
669
670            List<String> line = new ArrayList<>();
671            line.add(o.getOrDefault("Recording Name", ""));
672            line.add(n.getOrDefault("Recording Name", ""));
673
674            // For each numeric field, compute old/new/diff/diff%
675            for (String field : NUMERIC_FIELDS_LIST) {
676                String ov = o.getOrDefault(field, "");
677                String nv = n.getOrDefault(field, "");
678
679                double oldNum = toDouble(ov);
680                double newNum = toDouble(nv);
681                double diff = newNum - oldNum;
682                double diffPer = (oldNum == 0.0) ? 0.0 : (diff / oldNum) * 100.0;
683
684                line.add(ov);
685                line.add(nv);
686                line.add(format4(diff));
687                line.add(format1(diffPer)); // 1 decimal place for %
688                line.add(""); // separator column
689            }
690            rows.add(line);
691        }
692
693        // Write output CSV with grouped columns per field
694        try (PrintWriter pw = new PrintWriter(Files.newBufferedWriter(outFile))) {
695            List<String> h = new ArrayList<>();
696            h.add("Recording Name");
697            h.add("Recording Name Java");
698            for (String f : NUMERIC_FIELDS_LIST) {
699                h.add(f);
700                h.add(f + " Java");
701                h.add(f + " Diff");
702                h.add(f + " DiffPer");
703                h.add(""); // separator
704            }
705            pw.println(String.join(",", h));
706            for (List<String> r : rows) {
707                pw.println(String.join(",", r));
708            }
709        }
710
711        System.out.printf("Detailed numeric rows written: %d (matched from %d)%n", matched, total);
712        System.out.println("Written: " + outFile.toAbsolutePath());
713    }

771    /**
772     * Analyze the comparison report to estimate a tighter per-field tolerance
773     * that still keeps at least 98% of cases within range.
774     *
775     * @param comparisonCsv input comparison CSV produced by {@link #compareStatus(List, List, Path)}
776     * @param outCsv        output CSV summarizing suggested tolerances
777     * @throws IOException if reading or writing fails
778     */
779    private static void optimizeTolerances(Path comparisonCsv, Path outCsv) throws IOException {
780        if (!Files.exists(comparisonCsv)) {
781            System.out.println("No comparison file found for tolerance optimization.");
782            return;
783        }
784
785        Map<String, List<Double>> diffsByField = new LinkedHashMap<>();
786
787        // Read all numeric differences from the comparison file
788        try (BufferedReader br = Files.newBufferedReader(comparisonCsv, java.nio.charset.StandardCharsets.UTF_8)) {
789            String line;
790            while ((line = br.readLine()) != null) {
791                if (line.trim().isEmpty() || line.startsWith("SUMMARY")) {
792                    continue;
793                }
794                String[] parts = line.split(",", -1);
795                if (parts.length < 7) {
796                    continue;
797                }
798                String field = parts[2].trim();
799                String rel = parts[6].trim();
800                if (rel.isEmpty() || rel.equalsIgnoreCase("NaN")) {
801                    continue;
802                }
803                try {
804                    double d = Double.parseDouble(rel);
805                    diffsByField.computeIfAbsent(field, k -> new ArrayList<>()).add(d);
806                } catch (Exception ignored) {
807                }
808            }
809        }
810
811        if (diffsByField.isEmpty()) {
812            System.out.println("No numeric deviations detected — skipping optimization.");
813            return;
814        }
815
816        // Test tolerance levels from 5% down to 0.5%
817        double[] testLevels = {5.0, 4.0, 3.0, 2.0, 1.0, 0.5};
818        double targetKeep = 98.0;
819
820        try (PrintWriter pw = new PrintWriter(Files.newBufferedWriter(outCsv, java.nio.charset.StandardCharsets.UTF_8))) {
821            pw.println("Field,Total,Equal@5%,OptimalTolerance(%),Equal@Optimal(%)");
822
823            for (Map.Entry<String, List<Double>> e : diffsByField.entrySet()) {
824                String field = e.getKey();
825                List<Double> diffs = e.getValue();
826                diffs.removeIf(d -> !Double.isFinite(d));
827                if (diffs.isEmpty()) {
828                    continue;
829                }
830
831                int total = diffs.size();
832                double within5 = percentWithin(diffs, 5.0);
833                double bestTol = 5.0, bestKeep = within5;
834
835                // Find the smallest tolerance that still keeps ≥98% within range
836                for (double tol : testLevels) {
837                    double keep = percentWithin(diffs, tol);
838                    if (keep < targetKeep) {
839                        break;
840                    }
841                    bestTol = tol;
842                    bestKeep = keep;
843                }
844
845                pw.printf(Locale.US, "%s,%d,%.2f,%.2f,%.2f%n", field, total, within5, bestTol, bestKeep);
846            }
847        }
848
849        System.out.println("Tolerance optimization summary written: " + outCsv.toAbsolutePath());
850    }

852    /**
853     * Percentage of values whose absolute magnitude is within the given tolerance.
854     *
855     * @param vals list of numeric deviations (percent)
856     * @param tol  tolerance threshold (percent)
857     * @return percent of entries with |value| ≤ tol
858     */
859    private static double percentWithin(List<Double> vals, double tol) {
860        long ok = vals.stream().filter(v -> Math.abs(v) <= tol).count();
861        return 100.0 * ok / vals.size();
862    }

876     *

923    /**
924     * Computes the relative deviation (percent) between two numeric values given as strings.
925     * Returns NaN if either value is missing/invalid. If old is zero and new is non-zero,
926     * returns {@link Double#POSITIVE_INFINITY}.
927     *
928     * @param oldVal old value (string)
929     * @param newVal new value (string)
930     * @return relative deviation in percent, NaN if not applicable
931     */
932    private static double relativeDeviation(String oldVal, String newVal) {
933        Double oldNum = parseDouble(oldVal);
934        Double newNum = parseDouble(newVal);
935        if (oldNum == null || newNum == null) {
936            return Double.NaN;
937        }
938        if (oldNum == 0.0 && newNum == 0.0) {
939            return 0.0;
940        }
941        if (oldNum == 0.0) {
942            return Double.POSITIVE_INFINITY;
943        }
944        return Math.abs((newNum - oldNum) / oldNum) * 100.0;
945    }

949     * <ul>
950     *     <li>Density Ratio ≥ 2.0</li>
