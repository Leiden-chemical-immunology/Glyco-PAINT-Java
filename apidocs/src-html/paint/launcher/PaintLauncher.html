013/**
014 * The PaintLauncher class provides functionality for launching and managing
015 * a graphical user interface (GUI) designed to assist users in navigating through
016 * workflow-related applications in the Glyco-PAINT software suite.
017 * <p>
018 * The GUI includes components like buttons, labels, and panels arranged in a specific layout
019 * to represent a workflow. The class utilizes mappings between user-friendly labels
020 * and associated application identifiers, enabling the conditional display of actionable
021 * buttons or visual-only workflow steps based on the map contents.
022 * <p>
023 * Features:
024 * - Customized GUI appearance with consistent sizing, fonts, and colors.
025 * - Creation of customizable buttons and section labels with hover effects and styling.
026 * - Dynamic rendering of workflow paths using arrows and defined visual elements.
027 * - Integration to external application launching using associated identifiers.
028 * - Non-resizable, centered application window with a footer.
029 * <p>
030 * The GUI is intended to provide a user-friendly gateway for interacting
031 * with Glyco-PAINT tools or associated processes.
032 */
033public class PaintLauncher {
034
035    private static final Map<String, String> APP_MAP = new LinkedHashMap<>();
036    private static final String JAR_DIR = "jars";
037
038    static {
039        APP_MAP.put("Get Omero", "paint-get-omero");
040        APP_MAP.put("Create Experiment", "paint-create-experiment");
041        APP_MAP.put("Run TrackMate", null); // visual only
042        APP_MAP.put("Generate Squares", "paint-generate-squares");
043        APP_MAP.put("Viewer", "paint-viewer");
044        APP_MAP.put("Analyse Results", null); // visual only
045    }

048    /**
049     * Initializes and displays the main graphical user interface (GUI) for the Glyco-PAINT Launcher.
050     *
051     * This method sets up the main window of the application with a fixed size and layout, featuring
052     * a title, a workflow panel with buttons and visual elements, and a footer. Buttons in the workflow
053     * panel are dynamically created based on the {@code APP_MAP} field, each associated with specific
054     * functionalities or applications. The GUI elements use customized styles, colors, and layouts
055     * optimized for a consistent appearance.
056     *
057     * The window is designed to be non-resizable, centered on the screen, and includes functionality
058     * for launching external applications defined by app mappings.
059     */
060    private static void createAndShowGUI() {
061        JFrame frame = new JFrame("Glyco-PAINT Launcher");
062        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
063        frame.setSize(420, 600);
064        frame.setResizable(false);
065        frame.setLocationRelativeTo(null);
066
067        Color bg           = new Color(35, 39, 42);
068        Color fg           = new Color(240, 240, 240);
069        Color accent       = new Color(0x4E9AF1);
070        Color runColor     = new Color(0xBFD7F8);      // "Run TrackMate"
071        Color analyseColor = new Color(0xD8E7FB);  // "Analyse Results"
072        Color arrowMain    = new Color(210, 210, 210);
073        Color arrowShadow  = new Color(80, 80, 80);
074
075        JPanel root = new JPanel(new BorderLayout());
076        root.setBackground(bg);
077        root.setBorder(new EmptyBorder(20, 20, 20, 20));
078
079        JLabel title = new JLabel("Glyco-PAINT Launcher", SwingConstants.CENTER);
080        title.setForeground(fg);
081        title.setFont(new Font("SansSerif", Font.BOLD, 20));
082        title.setBorder(new EmptyBorder(0, 0, 20, 0));
083        root.add(title, BorderLayout.NORTH);
084
085        // Main workflow panel
086        JPanel flowPanel = new JPanel();
087        flowPanel.setLayout(new BoxLayout(flowPanel, BoxLayout.Y_AXIS));
088        flowPanel.setBackground(bg);
089
090        int index = 0;
091        int total = APP_MAP.size();
092        Dimension standardSize = new Dimension(280, 45);
093
094        for (Map.Entry<String, String> entry : APP_MAP.entrySet()) {
095            String label = entry.getKey();
096            String target = entry.getValue();
097
098            JComponent comp;
099            if (target == null) {
100                Color boxColor = "Run TrackMate".equals(label) ? runColor : analyseColor;
101                comp = createSectionLabel(label, boxColor, Color.DARK_GRAY, standardSize);
102            } else {
103                JButton btn = createStyledButton(label, accent, fg, bg, standardSize);
104                btn.addActionListener(e -> launchApp(target, frame));
105                comp = btn;
106            }
107
108            comp.setAlignmentX(Component.CENTER_ALIGNMENT);
109            flowPanel.add(comp);
110
111            if (++index < total) {
112                JLabel arrow = createBoldArrow(arrowMain, arrowShadow);
113                flowPanel.add(arrow);
114            }
115        }
116
117        // Add vertical glue to keep the bottom visible
118        flowPanel.add(Box.createVerticalGlue());
119        root.add(flowPanel, BorderLayout.CENTER);
120
121        JLabel footer = new JLabel("Hans Bakker • © 2025", SwingConstants.CENTER);
122        footer.setForeground(new Color(160, 160, 160));
123        footer.setFont(new Font("SansSerif", Font.PLAIN, 12));
124        footer.setBorder(new EmptyBorder(20, 0, 0, 0));
125        root.add(footer, BorderLayout.SOUTH);
126
127        frame.setContentPane(root);
128        frame.setVisible(true);
129    }

131    /**
132     * Creates a styled JButton with custom appearance and behavior.
133     * <p>
134     * This method sets properties such as font, colors, size, cursor, and border
135     * to create a visually consistent button. It also adds mouse listeners to
136     * provide a hover effect by adjusting the background color.
137     *
138     * @param text   The text to be displayed on the button.
139     * @param accent The primary accent color for the button (used as the background color).
140     * @param fg     The color for the button's text.
141     * @param bg     The color for the background (though not directly set here, it may be relevant for overridden methods).
142     * @param size   The dimensions of the button.
143     * @return A JButton instance with the applied styles and behavior.
144     */
145    private static JButton createStyledButton(String text, Color accent, Color fg, Color bg, Dimension size) {
146        JButton btn = new JButton(text);
147        btn.setPreferredSize(size);
148        btn.setMaximumSize(size);
149        btn.setMinimumSize(size);
150        btn.setFocusPainted(false);
151        btn.setFont(new Font("SansSerif", Font.BOLD, 14));
152        btn.setForeground(fg);
153        btn.setBackground(accent);
154        btn.setBorder(BorderFactory.createEmptyBorder(10, 20, 10, 20));
155        btn.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
156        btn.setOpaque(true);
157
158        btn.addMouseListener(new MouseAdapter() {
159            @Override
160            public void mouseEntered(MouseEvent e) {
161                btn.setBackground(accent.brighter());
162            }
163
164            @Override
165            public void mouseExited(MouseEvent e) {
166                btn.setBackground(accent);
167            }
168        });
169
170        return btn;
171    }

173    /**
174     * Creates a styled JLabel with custom appearance and size. The label is centered
175     * with the specified text, background color, foreground color, and dimensions.
176     *
177     * @param text    The text to be displayed on the label.
178     * @param bgColor The background color of the label.
179     * @param fg      The foreground color (text color) of the label.
180     * @param size    The dimensions of the label (preferred, maximum, and minimum sizes).
181     * @return A JLabel instance with the applied styles and specified properties.
182     */
183    private static JLabel createSectionLabel(String text, Color bgColor, Color fg, Dimension size) {
184        JLabel label = new JLabel(text, SwingConstants.CENTER);
185        label.setOpaque(true);
186        label.setBackground(bgColor);
187        label.setForeground(fg);
188        label.setFont(new Font("SansSerif", Font.BOLD, 14));
189        label.setBorder(BorderFactory.createEmptyBorder(10, 20, 10, 20));
190        label.setPreferredSize(size);
191        label.setMaximumSize(size);
192        label.setMinimumSize(size);
193        return label;
194    }

196    /**
197     * Creates a JLabel representing a bold arrow with customized appearance.
198     * <p>
199     * The arrow is displayed as a large, bold, and centered downward arrow ("↓")
200     * with a shadow effect, using the specified main and shadow colors. This method
201     * customizes the label's font, alignment, border, and preferred size.
202     *
203     * @param main   The primary color of the arrow.
204     * @param shadow The color used to create the shadow effect for the arrow.
205     * @return A JLabel instance with a styled bold arrow and shadow effect.
206     */
207    private static JLabel createBoldArrow(Color main, Color shadow) {
208        JLabel arrow = new JLabel("↓", SwingConstants.CENTER) {
209            @Override
210            protected void paintComponent(Graphics g) {
211                Graphics2D g2 = (Graphics2D) g.create();
212                g2.setFont(getFont());
213                FontMetrics fm = g2.getFontMetrics();
214                String s = getText();
215                int x = (getWidth() - fm.stringWidth(s)) / 2;
216                int y = (getHeight() + fm.getAscent()) / 2 - 4;
217
218                // shadow
219                g2.setColor(shadow);
220                g2.drawString(s, x + 1, y + 1);
221                // main arrow
222                g2.setColor(main);
223                g2.drawString(s, x, y);
224                g2.dispose();
225            }
226        };
227        arrow.setFont(new Font("SansSerif", Font.BOLD, 32)); // larger, thicker arrow
228        arrow.setBorder(new EmptyBorder(10, 0, 10, 0));
229        arrow.setAlignmentX(Component.CENTER_ALIGNMENT);
230        arrow.setPreferredSize(new Dimension(300, 35));
231        return arrow;
232    }

234    /**
235     * Launches an external application by locating a JAR file with a specified prefix
236     * and executing it as a new process. The method searches for the JAR files in a
237     * predefined directory and displays error messages if no matching JAR is found or
238     * if the process fails to start.
239     *
240     * @param jarPrefix The prefix of the JAR file name to be located and executed.
241     * @param parent    The parent component for displaying error messages in dialog boxes.
242     */
243    private static void launchApp(String jarPrefix, Component parent) {
244        File jarDir = new File(System.getProperty("user.dir"), JAR_DIR);
245        if (!jarDir.exists() || !jarDir.isDirectory()) {
246            JOptionPane.showMessageDialog(parent,
247                                          "JAR directory not found:\n" + jarDir.getAbsolutePath(),
248                                          "Error", JOptionPane.ERROR_MESSAGE);
249            return;
250        }
251
252        File[] matches = jarDir.listFiles(f -> f.getName().startsWith(jarPrefix) && f.getName().endsWith(".jar"));
253
254        if (matches == null || matches.length == 0) {
255            JOptionPane.showMessageDialog(parent,
256                                          "No matching JAR found for prefix:\n" + jarPrefix,
257                                          "Error", JOptionPane.ERROR_MESSAGE);
258            return;
259        }
260
261        File jarFile = matches[0];
262        try {
263            new ProcessBuilder("java", "-jar", jarFile.getAbsolutePath())
264                    .inheritIO()
265                    .start();
266        } catch (IOException e) {
267            JOptionPane.showMessageDialog(parent,
268                                          "Failed to launch:\n" + jarFile.getName() + "\n\n" + e.getMessage(),
269                                          "Error", JOptionPane.ERROR_MESSAGE);
270        }
271    }

273    public static void main(String[] args) {
274        SwingUtilities.invokeLater(PaintLauncher::createAndShowGUI);
275    }
