014 *    deletes empty subdirectories, and displays a summary of remaining files.

017 *    • Directory selection via JFileChooser.
018 *    • Processing of "Fileset" directories (via ProcessOmeroFiles).
019 *    • Automatic relocation of valid files to root folder.

021 *    • Scrollable summary dialog showing remaining files.

038package paint.getomero;
039
040import javax.swing.*;
041import java.awt.*;
042import java.awt.event.ActionEvent;
043import java.io.File;
044import java.io.IOException;
045
046/**
047 * A graphical user interface for processing Omero-exported "Fileset" directories.
048 * <p>
049 * This Swing-based tool enables users to:
050 * <ul>
051 *   <li>Select a root directory containing "Fileset" subdirectories.</li>
052 *   <li>Automatically move valid files to the root directory.</li>
053 *   <li>Delete empty "Fileset" folders after processing.</li>
054 *   <li>Display a scrollable list of remaining files.</li>
055 * </ul>
056 * The interface provides buttons for browsing, processing, and closing, with clear
057 * error handling for invalid directories or I/O failures.
058 */
059public class GetOmeroUI extends JFrame {
060
061    private final JTextField directoryField;
062
063    /**
064     * Constructs a new instance of the {@code GetOmeroUI} class and initializes
065     * the application window with controls for directory selection, processing,
066     * and viewing results.
067     */
068    public GetOmeroUI() {
069        super("Get Omero");
070        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
071
072        // Base font (plain, small)
073        Font baseFont = new JLabel().getFont().deriveFont(Font.PLAIN, 12f);
074
075        JPanel content = new JPanel(new BorderLayout(8, 8));
076        content.setBorder(BorderFactory.createEmptyBorder(12, 12, 12, 12));
077
078        // --- Directory selection panel ---
079        JPanel dirPanel = new JPanel(new GridBagLayout());
080        GridBagConstraints gbc = new GridBagConstraints();
081        gbc.insets = new Insets(2, 2, 2, 2);
082        gbc.gridy = 0;
083
084        JLabel dirLabel = new JLabel("Directory:");
085        dirLabel.setFont(baseFont);
086
087        directoryField = new JTextField(System.getProperty("user.home"));
088        directoryField.setEditable(false);
089        directoryField.setFont(baseFont);
090        directoryField.setPreferredSize(new Dimension(320, 24));
091
092        JButton browseButton = new JButton("Browse…");
093        browseButton.setFont(baseFont);
094        browseButton.setMargin(new Insets(2, 8, 2, 8));
095        browseButton.addActionListener(this::onBrowse);
096
097        gbc.gridx  = 0;
098        gbc.anchor = GridBagConstraints.WEST;
099        dirPanel.add(dirLabel, gbc);
100
101        gbc.gridx   = 1;
102        gbc.fill    = GridBagConstraints.HORIZONTAL;
103        gbc.weightx = 1.0;
104        dirPanel.add(directoryField, gbc);
105
106        gbc.gridx   = 2;
107        gbc.fill    = GridBagConstraints.NONE;
108        gbc.weightx = 0.0;
109        dirPanel.add(browseButton, gbc);
110
111        // --- Action buttons ---
112        JPanel buttonPanel = new JPanel(new FlowLayout(FlowLayout.RIGHT, 8, 4));
113        JButton processButton = new JButton("Process");
114        JButton closeButton = new JButton("Close");
115        processButton.setFont(baseFont);
116        closeButton.setFont(baseFont);
117        buttonPanel.add(processButton);
118        buttonPanel.add(closeButton);
119
120        processButton.addActionListener(this::onProcess);
121        closeButton.addActionListener(e -> dispose());
122
123        // Layout
124        content.add(dirPanel, BorderLayout.CENTER);
125        content.add(buttonPanel, BorderLayout.SOUTH);
126
127        add(content, BorderLayout.CENTER);
128        pack();
129
130        // Fix width = 500, keep auto height
131        Dimension size = getSize();
132        setSize(new Dimension(500, size.height));
133        setLocationRelativeTo(null);
134        setVisible(true);
135    }
136
137    /**
138     * Handles the action event triggered when the user clicks the "Browse" button.
139     * Opens a directory chooser dialog and updates the directory path display.
140     *
141     * @param e the triggered {@link ActionEvent}
142     */
143    private void onBrowse(ActionEvent e) {
144        JFileChooser chooser = new JFileChooser(directoryField.getText());
145        chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
146        if (chooser.showOpenDialog(this) == JFileChooser.APPROVE_OPTION) {
147            File selected = chooser.getSelectedFile();
148            directoryField.setText(selected.getAbsolutePath());
149        }
150    }
151
152    /**
153     * Executes when the "Process" button is clicked. Processes all "Fileset" subdirectories
154     * of the selected folder by invoking {@code ProcessOmeroFiles.process()}, then displays
155     * a summary of remaining files in a scrollable dialog.
156     *
157     * @param e the triggered {@link ActionEvent}
158     */
159    private void onProcess(ActionEvent e) {
160        String path = directoryField.getText();
161        if (path == null || path.isEmpty()) {
162            JOptionPane.showMessageDialog(this, "Please select a directory first.",
163                                          "Error", JOptionPane.ERROR_MESSAGE);
164            return;
165        }
166        File rootDir = new File(path);
167        if (!rootDir.isDirectory()) {
168            JOptionPane.showMessageDialog(this, "Invalid directory.",
169                                          "Error", JOptionPane.ERROR_MESSAGE);
170            return;
171        }
172
173        try {
174            ProcessOmeroFiles.process(rootDir);
175
176            // Collect visible files now in the directory (skip dotfiles)
177            File[] remaining = rootDir.listFiles(f -> f.isFile() && !f.getName().startsWith("."));
178            StringBuilder sb = new StringBuilder();
179            if (remaining != null && remaining.length > 0) {
180                for (File f : remaining) {
181                    sb.append(f.getName()).append("\n");
182                }
183            } else {
184                sb.append("(No files found)");
185            }
186
187            // Display them in a scrollable dialog
188            JTextArea area = new JTextArea(sb.toString(), 15, 50);
189            area.setEditable(false);
190            area.setFont(new JLabel().getFont().deriveFont(Font.PLAIN, 12f));
191            JScrollPane scroll = new JScrollPane(area);
192
193            JOptionPane.showMessageDialog(
194                    this,
195                    scroll,
196                    "Processing completed. Files now in directory:",
197                    JOptionPane.INFORMATION_MESSAGE
198            );
199
200        } catch (IOException ex) {
201            JOptionPane.showMessageDialog(this,
202                                          "Error during processing:\n" + ex.getMessage(),
203                                          "Error",
204                                          JOptionPane.ERROR_MESSAGE);
205        }
206    }
207
208    /**
209     * Launches the Get Omero GUI.
210     */
211    public static void main(String[] args) {
212        SwingUtilities.invokeLater(GetOmeroUI::new);
213    }
214}
