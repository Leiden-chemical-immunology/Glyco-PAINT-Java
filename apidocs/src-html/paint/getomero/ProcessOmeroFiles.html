011 *    (typically produced by Omero exports). It moves all valid files from

013 *    "Fileset" folders. It is used internally by {@link GetOmeroUI} and can
014 *    also be invoked headlessly for batch cleanup operations.
015 *
016 *  KEY FEATURES:
017 *    • Detects and processes subdirectories beginning with "Fileset".
018 *    • Moves contained files directly into the parent directory.
019 *    • Deletes empty "Fileset" folders after transfer.
020 *    • Supports safe overwrite using StandardCopyOption.REPLACE_EXISTING.
021 *    • Designed for batch automation and GUI invocation.

037package paint.getomero;
038
039import java.io.File;
040import java.io.IOException;
041import java.nio.file.Files;
042import java.nio.file.StandardCopyOption;
043
044/**
045 * Provides functionality to handle and reorganize files stored in directories
046 * starting with the prefix {@code "Fileset"}. Extracts contained files and moves
047 * them into the root directory, then deletes the empty subdirectories.
048 */
049public class ProcessOmeroFiles {
050
051    /**
052     * Processes the specified root directory by identifying and handling subdirectories
053     * whose names start with {@code "Fileset"}. Extracts files from these subdirectories,
054     * moves them into the root directory, and deletes the now-empty folders.
055     *
056     * @param rootDir the root directory to process; must be non-null and a valid directory
057     * @throws IOException if the directory is invalid, no "Fileset" folders exist,
058     *                     or if file operations fail during processing
059     */
060    public static void process(File rootDir) throws IOException {
061        if (rootDir == null || !rootDir.isDirectory()) {
062            throw new IOException("Invalid root directory: " + rootDir);
063        }
064
065        // Collect Fileset* directories
066        File[] filesetDirs = rootDir.listFiles(f -> f.isDirectory() && f.getName().startsWith("Fileset"));
067        if (filesetDirs == null || filesetDirs.length == 0) {
068            throw new IOException("No Fileset directories found in: " + rootDir.getAbsolutePath());
069        }
070
071        // Process each Fileset directory
072        for (File fsDir : filesetDirs) {
073            File[] files = fsDir.listFiles(f -> f.isFile() && !f.getName().startsWith("."));
074            if (files != null) {
075                for (File f : files) {
076                    Files.move(
077                            f.toPath(),
078                            new File(rootDir, f.getName()).toPath(),
079                            StandardCopyOption.REPLACE_EXISTING
080                    );
081                }
082            }
083            // Delete the now-empty Fileset directory
084            Files.delete(fsDir.toPath());
085        }
086    }
087}
