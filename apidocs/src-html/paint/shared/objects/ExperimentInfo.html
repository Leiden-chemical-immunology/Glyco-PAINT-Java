001/******************************************************************************
002 *  Class:        ExperimentInfo.java
003 *  Package:      paint.shared.objects
004 *
005 *  PURPOSE:
006 *    Represents metadata and associated data structures describing a single
007 *    experiment recording within the PAINT analysis framework.
008 *
009 *  DESCRIPTION:
010 *    The {@code ExperimentInfo} class encapsulates all metadata fields that
011 *    describe an experiment, such as condition number, probe information,
012 *    concentration, and threshold values. In addition, it maintains references
013 *    to related entities, including {@link Square} and {@link Track} objects,
014 *    and optionally a {@link tech.tablesaw.api.Table} containing track data.
015 *
016 *    This class provides constructors to create instances from explicit
017 *    parameters or from key-value maps (as when reading experiment info
018 *    from CSV or JSON sources). All fields are mutable through standard
019 *    accessor and mutator methods.
020 *
021 *  KEY FEATURES:
022 *    • Encapsulates all experimental metadata.
023 *    • Links to associated {@link Square} and {@link Track} entities.
024 *    • Provides map-based initialization for tabular imports.
025 *    • Supports formatted string export for diagnostics and logging.
026 *
027 *  AUTHOR:
028 *    Hans Bakker
029 *
030 *  MODULE:
031 *    paint-shared-utils
032 *
033 *  UPDATED:
034 *    2025-10-28
035 *

101    /**

103     *
104     * @param experimentName  the experiment name
105     * @param recordingName   the recording name
106     * @param conditionNumber the condition number
107     * @param replicateNumber the replicate number
108     * @param probeName       the probe name
109     * @param probeType       the probe type
110     * @param cellType        the cell type
111     * @param adjuvant        the adjuvant used
112     * @param concentration   the concentration value
113     * @param processFlag     whether this recording should be processed
114     * @param threshold       the threshold for processing
115     */
116    public ExperimentInfo(String  experimentName,
117                          String  recordingName,
118                          int     conditionNumber,
119                          int     replicateNumber,
120                          String  probeName,
121                          String  probeType,
122                          String  cellType,
123                          String  adjuvant,
124                          double  concentration,
125                          boolean processFlag,
126                          double threshold) {
127        this.experimentName  = experimentName;
128        this.recordingName   = recordingName;
129        this.conditionNumber = conditionNumber;
130        this.replicateNumber = replicateNumber;
131        this.probeName       = probeName;
132        this.probeType       = probeType;
133        this.cellType        = cellType;
134        this.adjuvant        = adjuvant;
135        this.concentration   = concentration;
136        this.processFlag     = processFlag;
137        this.threshold       = threshold;
138    }
139
140    /**
141     * Constructs an {@code ExperimentInfo} instance from a map of string key-value pairs.
142     * <p>
143     * Expected keys include:
144     * <ul>
145     *   <li>Experiment Name</li>
146     *   <li>Recording Name</li>
147     *   <li>Condition Number</li>
148     *   <li>Replicate Number</li>
149     *   <li>Probe Name</li>
150     *   <li>Probe Type</li>
151     *   <li>Cell Type</li>
152     *   <li>Adjuvant</li>
153     *   <li>Concentration</li>
154     *   <li>Process Flag</li>
155     *   <li>Threshold</li>
156     * </ul>
157     *
158     * @param row the map of column names to values
159     */
160    public ExperimentInfo(Map<String, String> row) {
161        try {
162            this.experimentName  = row.get("Experiment Name");
163            this.recordingName   = row.get("Recording Name");
164            this.conditionNumber = parseInt(row.get("Condition Number"));
165            this.replicateNumber = parseInt(row.get("Replicate Number"));
166            this.probeName       = row.get("Probe Name");
167            this.probeType       = row.get("Probe Type");
168            this.cellType        = row.get("Cell Type");
169            this.adjuvant        = row.get("Adjuvant");
170            this.concentration   = parseDouble(row.get("Concentration"));
171            this.processFlag     = Miscellaneous.isBooleanTrue(row.get("Process Flag"));
172            this.threshold       = parseDouble(row.get("Threshold"));
173        } catch (Exception e) {
174            PaintLogger.errorf("Problem parsing Experiment Info");
175            PaintLogger.errorf(row.toString());
176            StringWriter sw = new StringWriter();
177            e.printStackTrace(new PrintWriter(sw));
178            PaintLogger.errorf("An exception occurred:\n" + sw);
179        }
180    }

333    /**
334     * Returns a formatted string representation of this {@code ExperimentInfo}.
335     *
336     * @return formatted experiment information as string
337     */
338    @Override
339    public String toString() {
340        StringBuilder sb = new StringBuilder();
341        sb.append("\tExperiment Info");
342        sb.append(String.format("\t              Experiment Name               : %s\n", experimentName));
343        sb.append(String.format("\t              Recording Name                : %s%n", recordingName));
344        sb.append(String.format("\t              Condition Nr                  : %d%n", conditionNumber));
345        sb.append(String.format("\t              Replicate Nr                  : %d%n", replicateNumber));
346        sb.append(String.format("\t              Probe Name                    : %s%n", probeName));
347        sb.append(String.format("\t              Probe Type                    : %s%n", probeType));
348        sb.append(String.format("\t              Cell Type                     : %s%n", cellType));
349        sb.append(String.format("\t              Adjuvant                      : %s%n", adjuvant));
350        sb.append(String.format("\t              Concentration                 : %.2f%n", concentration));
351        sb.append(String.format("\t              Threshold                     : %.2f%n", threshold));
352        if (tracks != null) {
353            sb.append(String.format("\t              Number of tracks              : %d%n", tracks.size()));
354        }
355        if (squares != null) {
356            sb.append(String.format("\t              Number of squares             : %d%n", squares.size()));
357        }
358
359        int numberOfSquaresWithTracks = 0;
360        for (Square square : squares) {
361            if (square.getTracks() != null && !square.getTracks().isEmpty()) {
362                numberOfSquaresWithTracks++;
363            }
364        }
365
366        sb.append(String.format("\tNumber of squares with tracks : %d%n", numberOfSquaresWithTracks));
367        return sb.toString();
368    }
