001/******************************************************************************
002 *  Class:        Square.java
003 *  Package:      paint.shared.objects
004 *
005 *  PURPOSE:
006 *    Represents a rectangular spatial region ("square") within a recording or
007 *    experiment in the PAINT analysis framework.
008 *
009 *  DESCRIPTION:
010 *    The {@code Square} class models a subregion of an experimental image grid.
011 *    Each square stores its spatial coordinates, computed statistics, flags,
012 *    and references to related {@link Track} objects. Squares can be generated
013 *    automatically from the total number of regions in a recording or defined
014 *    explicitly from coordinates.
015 *
016 *    The class also provides utilities for calculating theoretical square areas
017 *    and introspective initialization of numeric fields.
018 *
019 *  KEY FEATURES:
020 *    • Encapsulates position, dimensions, and analysis results of a square.
021 *    • Supports linking of {@link Track} objects and {@link tech.tablesaw.api.Table}.
022 *    • Provides automatic coordinate calculation based on grid size.
023 *    • Offers formatted diagnostic output and NaN initialization for doubles.
024 *
025 *  AUTHOR:

107    /**
108     * Creates a {@code Square} with explicit coordinates and identifiers.
109     *
110     * @param uniqueKey      unique identifier
111     * @param experimentName experiment name
112     * @param recordingName  recording name
113     * @param squareNumber   sequential number
114     * @param rowNumber      row index in grid
115     * @param colNumber      column index in grid
116     * @param x0             left coordinate in pixels
117     * @param y0             top coordinate in pixels
118     * @param x1             right coordinate in pixels
119     * @param y1             bottom coordinate in pixels
120     */
121    public Square(String uniqueKey,
122                  String experimentName,
123                  String recordingName,
124                  int squareNumber,
125                  int rowNumber,
126                  int colNumber,
127                  double x0,
128                  double y0,
129                  double x1,
130                  double y1) {
131
132        initialiseDoublesToNaN();
133        this.uniqueKey       = uniqueKey;
134        this.experimentName  = experimentName;
135        this.recordingName  = recordingName;
136        this.squareNumber    = squareNumber;
137        this.rowNumber       = rowNumber;
138        this.colNumber       = colNumber;
139        this.x0              = round(x0, 2);
140        this.y0              = round(y0, 2);
141        this.x1              = round(x1, 2);
142        this.y1              = round(y1, 2);
143    }

145    /**
146     * Creates a {@code Square} automatically based on its sequential number and
147     * the total number of squares in the recording.
148     *
149     * @param squareNumber               sequential number of the square
150     * @param numberOfSquaresInRecording total number of squares in the recording
151     */
152    public Square(int squareNumber, int numberOfSquaresInRecording) {
153        int numberSquaresInRow = (int) Math.sqrt(numberOfSquaresInRecording);
154        double width           = IMAGE_WIDTH / numberSquaresInRow;
155        double height          = IMAGE_HEIGHT / numberSquaresInRow;
156
157        initialiseDoublesToNaN();
158
159        colNumber = squareNumber % numberSquaresInRow;
160        rowNumber = squareNumber / numberSquaresInRow;
161
162        x0 = round(colNumber * width, 2);
163        x1 = round((colNumber + 1) * width, 2);
164        y0 = round(rowNumber * height, 2);
165        y1 = round((rowNumber + 1) * width, 2);
166    }

287    /**
288     * Initializes all double fields in this object to {@code NaN}.
289     * Used to ensure undefined numeric values are recognizable.
290     */
291    private void initialiseDoublesToNaN() {
292        for (Field f : getClass().getDeclaredFields()) {
293            if (f.getType() == double.class) {
294                try {
295                    f.setDouble(this, Double.NaN);
296                } catch (IllegalAccessException e) {
297                    throw new RuntimeException(e);
298                }
299            }
300        }
301    }

303    /**
304     * Calculates the theoretical square area for a given recording grid size.
305     *
306     * @param nrSquares total number of squares in the recording
307     * @return area of one square (in image units)
308     */
309    public static double calculateSquareArea(int nrSquares) {
310        return IMAGE_WIDTH * IMAGE_HEIGHT / nrSquares;
311    }

317    /**
318     * Returns a formatted textual representation of this square and its metrics.
319     *
320     * @return formatted string
321     */
322    @Override
323    public String toString() {
324        StringBuilder sb = new StringBuilder();
325
326        sb.append("\n----------------------------------------------------------------------\n")
327                .append("Square ").append(squareNumber)
328                .append(" (Experiment: ").append(experimentName != null ? experimentName : "N/A").append(")\n")
329                .append(" (Recording: ").append(recordingName != null ? recordingName : "N/A").append(")\n")
330                .append("----------------------------------------------------------------------\n");
331
332        sb.append(String.format("Row,Col Number                 : %d,%d%n", rowNumber, colNumber));
333        sb.append(String.format("Coordinates [x0,y0]-[x1,y1]    : [%.2f, %.2f] - [%.2f, %.2f]%n", x0, y0, x1, y1));
334        sb.append(String.format("Unique Key                     : %s%n", uniqueKey != null ? uniqueKey : "N/A"));
335        sb.append(String.format("Label Number                   : %d%n", labelNumber));
336        sb.append(String.format("Cell ID                        : %d%n", cellId));
337
338        sb.append(String.format("Selected                       : %b%n", selected));
339        sb.append(String.format("Square Manually Excluded       : %b%n", squareManuallyExcluded));
340        sb.append(String.format("Image Excluded                 : %b%n", imageExcluded));
341
342        sb.append(String.format("Number of Tracks               : %d%n", numberOfTracks));
343        sb.append(String.format("Variability                    : %.4f%n", variability));
344        sb.append(String.format("Density                        : %.4f%n", density));
345        sb.append(String.format("Density Ratio                  : %.4f%n", densityRatio));
346        sb.append(String.format("Tau                            : %.4f%n", tau));
347        sb.append(String.format("R²                             : %.4f%n", rSquared));
348
349        sb.append(String.format("Median Diffusion Coefficient   : %.4f%n", medianDiffusionCoefficient));
350        sb.append(String.format("Median Diffusion CoefficientExt: %.4f%n", medianDiffusionCoefficientExt));
351        sb.append(String.format("Median Displacement            : %.4f%n", medianDisplacement));
352        sb.append(String.format("Max Displacement               : %.4f%n", maxDisplacement));
353        sb.append(String.format("Total Displacement             : %.4f%n", totalDisplacement));
354
355        sb.append(String.format("Median Max Speed               : %.4f%n", medianMaxSpeed));
356        sb.append(String.format("Max Max Speed                  : %.4f%n", maxMaxSpeed));
357        sb.append(String.format("Median Mean Speed              : %.4f%n", medianMedianSpeed));
358        sb.append(String.format("Max Mean Speed                 : %.4f%n", maxMedianSpeed));
359
360        sb.append(String.format("Max Track Duration             : %.4f%n", maxTrackDuration));
361        sb.append(String.format("Total Track Duration           : %.4f%n", totalTrackDuration));
362        sb.append(String.format("Median Track Duration          : %.4f%n", medianTrackDuration));
363
364        if (tracks != null) {
365            sb.append(String.format("Tracks attached                : %d%n", tracks.size()));
366        }
367        if (tracksTable != null) {
368            sb.append("Tracks table available%n");
369        }
370
371        return sb.toString();
372    }

378    /** Example driver for debugging square area calculations. */
379    public static void main(String[] args) {
380        List<Square> squares = new ArrayList<>();
381        for (int i = 0; i < 21; i++) {
382            squares.add(new Square(i, 100));
383        }
384        System.out.println(squares);
385
386        double areaOriginal         = calcSquareAreaOriginal(20);
387        double areaNew              = calculateSquareArea(400);
388        double difference           = areaNew - areaOriginal;
389        double percentualDifference = (areaNew - areaOriginal) / areaOriginal * 100;
390        System.out.println("Area original: " + areaOriginal);
391        System.out.println("Area new: " + areaNew);
392        System.out.printf("Difference: %.6f%n", difference);
393        System.out.printf("Percentual difference: %.4f%%%n", percentualDifference);
394    }
395
396    private static double calcSquareAreaOriginal(int nrSquaresInRow) {
397        double micrometerPerPixel = 0.1602804;
398        int pixelsPerImage = 512;
399        double micrometerPerImageAxis  = micrometerPerPixel * pixelsPerImage;
400        double micrometerPerSquareAxis = micrometerPerImageAxis / nrSquaresInRow;
401        return micrometerPerSquareAxis * micrometerPerSquareAxis;
402    }
