056/**
057 * Provides all CSV input/output operations for {@link Square} entities.
058 *
059 * <p>Handles CSV reading, schema validation, entity conversion, and table
060 * appending in a consistent manner across PAINT’s square-level datasets.</p>
061 */
062public class SquareTableIO extends BaseTableIO {
063
064    // ───────────────────────────────────────────────────────────────────────────────
065    // TABLE CREATION
066    // ───────────────────────────────────────────────────────────────────────────────
067
068    /**
069     * Creates an empty {@link Table} for square data with the defined schema.
070     *
071     * @return a new empty table named “Squares”
072     */
073    public Table emptyTable() {
074        return newEmptyTable("Squares", SQUARES_COLS, SQUARES_TYPES);
075    }
076
077    // ───────────────────────────────────────────────────────────────────────────────
078    // ENTITY ⇄ TABLE CONVERSION
079    // ───────────────────────────────────────────────────────────────────────────────
080
081    /**
082     * Converts a list of {@link Square} entities into a {@link Table}
083     * matching the {@code squares.csv} schema.
084     *
085     * @param squares list of {@link Square} objects to convert
086     * @return a schema-compliant {@link Table} populated with square data
087     */
088    public Table toTable(List<Square> squares) {
089        Table table = emptyTable();
090
091        for (Square square : squares) {
092            Row tablesawRow = table.appendRow();
093            tablesawRow.setString( "Unique Key",                       square.getUniqueKey());
094            tablesawRow.setString( "Experiment Name",                  square.getExperimentName());
095            tablesawRow.setString( "Recording Name",                   square.getRecordingName());
096            tablesawRow.setInt(    "Square Number",                    square.getSquareNumber());
097            tablesawRow.setInt(    "Row Number",                       square.getRowNumber());
098            tablesawRow.setInt(    "Column Number",                    square.getColNumber());
099            tablesawRow.setInt(    "Label Number",                     square.getLabelNumber());
100            tablesawRow.setInt(    "Cell ID",                          square.getCellId());
101            tablesawRow.setBoolean("Selected",                         square.isSelected());
102            tablesawRow.setBoolean("Square Manually Excluded",         square.isSquareManuallyExcluded());
103            tablesawRow.setBoolean("Image Excluded",                   square.isImageExcluded());
104            tablesawRow.setDouble( "X0",                               square.getX0());
105            tablesawRow.setDouble( "Y0",                               square.getY0());
106            tablesawRow.setDouble( "X1",                               square.getX1());
107            tablesawRow.setDouble( "Y1",                               square.getY1());
108            tablesawRow.setInt(    "Number of Tracks",                 square.getNumberOfTracks());
109            tablesawRow.setDouble( "Variability",                      square.getVariability());
110            tablesawRow.setDouble( "Density",                          square.getDensity());
111            tablesawRow.setDouble( "Density Ratio",                    square.getDensityRatio());
112            tablesawRow.setDouble( "Density Ratio Ori",                square.getDensityRatioOri());
113            tablesawRow.setDouble( "Tau",                              square.getTau());
114            tablesawRow.setDouble( "R Squared",                        square.getRSquared());
115            tablesawRow.setDouble( "Median Diffusion Coefficient",     square.getMedianDiffusionCoefficient());
116            tablesawRow.setDouble( "Median Diffusion Coefficient Ext", square.getMedianDiffusionCoefficientExt());
117            tablesawRow.setDouble( "Median Displacement",              square.getMedianDisplacement());
118            tablesawRow.setDouble( "Max Displacement",                 square.getMaxDisplacement());
119            tablesawRow.setDouble( "Total Displacement",               square.getTotalDisplacement());
120            tablesawRow.setDouble( "Median Max Speed",                 square.getMedianMaxSpeed());
121            tablesawRow.setDouble( "Max Max Speed",                    square.getMaxMaxSpeed());
122            tablesawRow.setDouble( "Median Mean Speed",                square.getMedianMedianSpeed());
123            tablesawRow.setDouble( "Max Mean Speed",                   square.getMaxMedianSpeed());
124            tablesawRow.setDouble( "Max Track Duration",               square.getMaxTrackDuration());
125            tablesawRow.setDouble( "Total Track Duration",             square.getTotalTrackDuration());
126            tablesawRow.setDouble( "Median Track Duration",            square.getMedianTrackDuration());
127        }
128        return table;
129    }

131    /**
132     * Converts a {@link Table} into a list of {@link Square} entities.
133     *
134     * <p>The table must conform to the {@code squares.csv} schema.
135     * Each row is mapped directly to a {@code Square} instance.</p>
136     *
137     * @param table the validated table to convert
138     * @return a list of {@link Square} entities
139     */
140    public List<Square> toEntities(Table table) {
141        List<Square> squares = new ArrayList<>();
142
143        for (Row tablesawRow : table) {
144            Square square = new Square();
145            square.setUniqueKey(                     tablesawRow.getString(  "Unique Key"));
146            square.setExperimentName(                tablesawRow.getString(  "Experiment Name"));
147            square.setRecordingName(                 tablesawRow.getString(  "Recording Name"));
148            square.setSquareNumber(                  tablesawRow.getInt(     "Square Number"));
149            square.setRowNumber(                     tablesawRow.getInt(     "Row Number"));
150            square.setColNumber(                     tablesawRow.getInt(     "Column Number"));
151            square.setLabelNumber(                   tablesawRow.getInt(     "Label Number"));
152            square.setCellId(                        tablesawRow.getInt(     "Cell ID"));
153            square.setSelected(                      tablesawRow.getBoolean( "Selected"));
154            square.setSquareManuallyExcluded(        tablesawRow.getBoolean( "Square Manually Excluded"));
155            square.setImageExcluded(                 tablesawRow.getBoolean( "Image Excluded"));
156            square.setX0(                            tablesawRow.getDouble(  "X0"));
157            square.setY0(                            tablesawRow.getDouble(  "Y0"));
158            square.setX1(                            tablesawRow.getDouble(  "X1"));
159            square.setY1(                            tablesawRow.getDouble(  "Y1"));
160            square.setNumberOfTracks(                tablesawRow.getInt(     "Number of Tracks"));
161            square.setVariability(                   tablesawRow.getDouble(  "Variability"));
162            square.setDensity(                       tablesawRow.getDouble(  "Density"));
163            square.setDensityRatio(                  tablesawRow.getDouble(  "Density Ratio"));
164            square.setDensityRatioOri(               tablesawRow.getDouble(  "Density Ratio Ori"));
165            square.setTau(                           tablesawRow.getDouble(  "Tau"));
166            square.setRSquared(                      tablesawRow.getDouble(  "R Squared"));
167            square.setMedianDiffusionCoefficient(    tablesawRow.getDouble(  "Median Diffusion Coefficient"));
168            square.setMedianDiffusionCoefficientExt( tablesawRow.getDouble(  "Median Diffusion Coefficient Ext"));
169            square.setMedianDisplacement(            tablesawRow.getDouble(  "Median Displacement"));
170            square.setMaxDisplacement(               tablesawRow.getDouble(  "Max Displacement"));
171            square.setTotalDisplacement(             tablesawRow.getDouble(  "Total Displacement"));
172            square.setMedianMaxSpeed(                tablesawRow.getDouble(  "Median Max Speed"));
173            square.setMaxMaxSpeed(                   tablesawRow.getDouble(  "Max Max Speed"));
174            square.setMedianMedianSpeed(             tablesawRow.getDouble(  "Median Mean Speed"));
175            square.setMaxMedianSpeed(                tablesawRow.getDouble(  "Max Mean Speed"));
176            square.setMaxTrackDuration(              tablesawRow.getDouble(  "Max Track Duration"));
177            square.setTotalTrackDuration(            tablesawRow.getDouble(  "Total Track Duration"));
178            square.setMedianTrackDuration(           tablesawRow.getDouble(  "Median Track Duration"));
179            squares.add(square);
180        }
181        return squares;
182    }

188    /**
189     * Reads a {@code squares.csv} file and validates its schema.
190     *
191     * @param csvPath the path to the CSV file
192     * @return a schema-compliant {@link Table} of square data
193     * @throws IOException if the file cannot be read or validated
194     */
195    public Table readCsv(Path csvPath) throws IOException {
196        return readCsvWithSchema(csvPath, SQUARES, SQUARES_COLS, SQUARES_TYPES, false);
197    }
198
199    /**
200     * Appends all rows from a source {@link Table} into a target {@link Table},
201     * enforcing the {@code squares.csv} schema.
202     *
203     * <p>Performs basic type handling (STRING, INTEGER, DOUBLE, BOOLEAN) and
204     * preserves missing values.</p>
205     *
206     * @param target the destination table
207     * @param source the source table to append
208     */
209    public void appendInPlace(Table target, Table source) {
210        for (Row row : source) {
211            Row newRow = target.appendRow();
212            for (String col : SQUARES_COLS) {
213                Column<?> targetCol = target.column(col);
214                if (targetCol.type() == ColumnType.STRING) {
215                    newRow.setString(col, row.getString(col));
216                } else if (targetCol.type() == ColumnType.INTEGER) {
217                    newRow.setInt(col, row.getInt(col));
218                } else if (targetCol.type() == ColumnType.DOUBLE) {
219                    newRow.setDouble(col, row.getDouble(col));
220                } else if (targetCol.type() == ColumnType.BOOLEAN) {
221                    newRow.setBoolean(col, row.getBoolean(col));
222                }
223            }
224        }
225    }
