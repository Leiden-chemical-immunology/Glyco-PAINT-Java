053/**
054 * Provides CSV I/O and schema enforcement for {@link Track} entities.
055 *
056 * <p>This class encapsulates reading, writing, and conversion logic for
057 * {@code tracks.csv} and guarantees alignment with the schema definitions
058 * provided in {@link PaintConstants}.</p>
059 */
060public class TrackTableIO extends BaseTableIO {
061
062    // ───────────────────────────────────────────────────────────────────────────────
063    // TABLE CREATION
064    // ───────────────────────────────────────────────────────────────────────────────
065
066    /**
067     * Creates an empty {@link Table} for tracks with the correct schema.
068     *
069     * @return a new empty {@link Table} with all track columns defined
070     */
071    public Table emptyTable() {
072        return newEmptyTable("Tracks", TRACKS_COLS, TRACKS_TYPES);
073    }
074
075    // ───────────────────────────────────────────────────────────────────────────────
076    // ENTITY ⇄ TABLE CONVERSION
077    // ───────────────────────────────────────────────────────────────────────────────
078
079    /**
080     * Converts a list of {@link Track} objects into a {@link Table}.
081     *
082     * @param tracks list of {@link Track} entities to convert
083     * @return a {@link Table} containing one row per track, schema validated
084     */
085    public Table toTable(List<Track> tracks) {
086        Table table = emptyTable();
087        for (Track track : tracks) {
088            Row tablesawRow = table.appendRow();
089            tablesawRow.setString( "Unique Key",                track.getUniqueKey());
090            tablesawRow.setString( "Experiment Name",           track.getExperimentName());
091            tablesawRow.setString( "Recording Name",            track.getRecordingName());
092            tablesawRow.setInt(    "Track Id",                  track.getTrackId());
093            tablesawRow.setInt(    "Number of Spots",           track.getNumberOfSpots());
094            tablesawRow.setInt(    "Number of Gaps",            track.getNumberOfGaps());
095            tablesawRow.setInt(    "Longest Gap",               track.getLongestGap());
096            tablesawRow.setDouble( "Track Duration",            track.getTrackDuration());
097            tablesawRow.setDouble( "Track X Location",          track.getTrackXLocation());
098            tablesawRow.setDouble( "Track Y Location",          track.getTrackYLocation());
099            tablesawRow.setDouble( "Track Displacement",        track.getTrackDisplacement());
100            tablesawRow.setDouble( "Track Max Speed",           track.getTrackMaxSpeed());
101            tablesawRow.setDouble( "Track Median Speed",        track.getTrackMedianSpeed());
102            tablesawRow.setDouble( "Diffusion Coefficient",     track.getDiffusionCoefficient());
103            tablesawRow.setDouble( "Diffusion Coefficient Ext", track.getDiffusionCoefficientExt());
104            tablesawRow.setDouble( "Total Distance",            track.getTotalDistance());
105            tablesawRow.setDouble( "Confinement Ratio",         track.getConfinementRatio());
106            tablesawRow.setInt(    "Square Number",             track.getSquareNumber());
107            tablesawRow.setInt(    "Label Number",              track.getLabelNumber());
108        }
109        return table;
110    }
111
112    /**
113     * Converts a {@link Table} into a list of {@link Track} entities.
114     *
115     * @param table the validated {@link Table} to convert
116     * @return a list of {@link Track} entities populated from the table
117     */
118    public List<Track> toEntities(Table table) {
119        List<Track> tracks = new ArrayList<>();
120        for (Row row : table) {
121            Track track = new Track();
122            track.setUniqueKey(              row.getString( "Unique Key"));
123            track.setExperimentName(         row.getString( "Experiment Name"));
124            track.setRecordingName(          row.getString( "Recording Name"));
125            track.setTrackId(                row.getInt(    "Track Id"));
126            track.setNumberOfSpots(          row.getInt(    "Number of Spots"));
127            track.setNumberOfGaps(           row.getInt(    "Number of Gaps"));
128            track.setLongestGap(             row.getInt(    "Longest Gap"));
129            track.setTrackDuration(          row.getDouble( "Track Duration"));
130            track.setTrackXLocation(         row.getDouble( "Track X Location"));
131            track.setTrackYLocation(         row.getDouble( "Track Y Location"));
132            track.setTrackDisplacement(      row.getDouble( "Track Displacement"));
133            track.setTrackMaxSpeed(          row.getDouble( "Track Max Speed"));
134            track.setTrackMedianSpeed(       row.getDouble( "Track Median Speed"));
135            track.setDiffusionCoefficient(   row.getDouble( "Diffusion Coefficient"));
136            track.setDiffusionCoefficientExt(row.getDouble( "Diffusion Coefficient Ext"));
137            track.setTotalDistance(          row.getDouble( "Total Distance"));
138            track.setConfinementRatio(       row.getDouble( "Confinement Ratio"));
139            track.setSquareNumber(           row.getInt(    "Square Number"));
140            track.setLabelNumber(            row.getInt(    "Label Number"));
141            tracks.add(track);
142        }
143        return tracks;
144    }
145
146    // ───────────────────────────────────────────────────────────────────────────────
147    // CSV READ / APPEND
148    // ───────────────────────────────────────────────────────────────────────────────
149
150    /**
151     * Reads a CSV file into a {@link Table} using the expected track schema.
152     *
153     * @param csvPath path to the {@code tracks.csv} file
154     * @return the parsed and validated {@link Table}
155     * @throws IOException if the file cannot be read or validated
156     */
157    public Table readCsv(Path csvPath) throws IOException {
158        return readCsvWithSchema(csvPath, TRACKS, TRACKS_COLS, TRACKS_TYPES, false);
159    }
160
161    /**
162     * Appends all rows from the source {@link Table} into the target {@link Table}.
163     *
164     * <p>This method performs manual type matching and enforces the schema
165     * for all columns defined in {@code TRACKS_COLS}.</p>
166     *
167     * @param target the destination {@link Table}
168     * @param source the source {@link Table}
169     */
170    public void appendInPlace(Table target, Table source) {
171        for (Row row : source) {
172            Row newRow = target.appendRow();
173            for (String col : TRACKS_COLS) {
174                Column<?> targetCol = target.column(col);
175
176                if (targetCol.type() == ColumnType.STRING) {
177                    newRow.setString(col, row.getString(col));
178                } else if (targetCol.type() == ColumnType.INTEGER) {
179                    newRow.setInt(col, row.getInt(col));
180                } else if (targetCol.type() == ColumnType.DOUBLE) {
181                    newRow.setDouble(col, row.getDouble(col));
182                } else if (targetCol.type() == ColumnType.BOOLEAN) {
183                    newRow.setBoolean(col, row.getBoolean(col));
184                }
185            }
186        }
187    }
188

197     * @return a {@link Track} populated with values from the row
198     */

200        Track track = new Track();
201        track.setUniqueKey(                row.getString(  "Unique Key"));
202        track.setExperimentName(           row.getString(  "Experiment Name"));
203        track.setRecordingName(            row.getString(  "Recording Name"));
204        track.setTrackId(                  row.getInt(     "Track Id"));
205        track.setNumberOfSpots(            row.getInt(     "Number of Spots"));
206        track.setNumberOfGaps(             row.getInt(     "Number of Gaps"));
207        track.setLongestGap(               row.getInt(     "Longest Gap"));
208        track.setTrackDuration(            row.getDouble(  "Track Duration"));
209        track.setTrackXLocation(           row.getDouble(  "Track X Location"));
210        track.setTrackYLocation(           row.getDouble(  "Track Y Location"));
211        track.setTrackDisplacement(        row.getDouble(  "Track Displacement"));
212        track.setTrackMaxSpeed(            row.getDouble(  "Track Max Speed"));
213        track.setTrackMedianSpeed(         row.getDouble(  "Track Median Speed"));
214        track.setDiffusionCoefficient(     row.getDouble(  "Diffusion Coefficient"));
215        track.setDiffusionCoefficientExt(  row.getDouble(  "Diffusion Coefficient Ext"));
216        track.setTotalDistance(            row.getDouble(  "Total Distance"));
217        track.setConfinementRatio(         row.getDouble(  "Confinement Ratio"));
218        track.setSquareNumber(             row.getInt(     "Square Number"));
219        track.setLabelNumber(              row.getInt(     "Label Number"));
220        return track;
221    }
