038package paint.shared.io;
039
040import paint.shared.objects.Experiment;
041import paint.shared.objects.Recording;
042import paint.shared.objects.Square;
043import paint.shared.utils.PaintLogger;
044import tech.tablesaw.api.Table;
045
046import java.nio.file.Path;
047import java.util.List;
048
049import static paint.shared.constants.PaintConstants.*;
050import static paint.shared.utils.SharedSquareUtils.filterTracksInSquare;
051
052/**
053 * Provides centralized functionality for loading all experiment-related data:
054 * recordings, tracks, and (for mature projects) squares.
055 *
056 * <p>The class orchestrates reading and schema validation of CSV files into
057 * structured {@link Experiment} objects. It leverages the specialized
058 * {@link RecordingTableIO}, {@link TrackTableIO}, and {@link SquareTableIO}
059 * classes to ensure consistent I/O and validation logic.</p>
060 *
061 * <p>This class is a static-only loader with a single entry point
062 * {@link #loadExperiment(Path, String, boolean)}.</p>
063 */
064public final class ExperimentDataLoader {
065
066    /**
067     * Private constructor to prevent instantiation.
068     */
069    private ExperimentDataLoader() {
070    }
071
072    // ───────────────────────────────────────────────────────────────────────────────
073    // EXPERIMENT LOADER
074    // ───────────────────────────────────────────────────────────────────────────────
075
076    /**
077     * Loads an entire experiment from a given project folder.
078     *
079     * <p>This method sequentially performs the following steps:</p>
080     * <ol>
081     *   <li>Loads and validates {@code recordings.csv}, converting rows into
082     *       {@link Recording} entities.</li>
083     *   <li>Loads and validates {@code tracks.csv}, associating tracks with each
084     *       recording based on the “Recording Name” column.</li>
085     *   <li>If {@code matureProject} is {@code true}, loads and validates
086     *       {@code squares.csv}, mapping tracks into squares using the spatial
087     *       boundaries defined for each recording.</li>
088     * </ol>
089     *
090     * <p>All CSVs are validated for column names, order, and data types
091     * according to schema definitions in {@link paint.shared.constants.PaintConstants}.</p>
092     *
093     * @param projectPath   the root project directory
094     * @param experimentName the name of the experiment folder (relative to the project path)
095     * @param matureProject  {@code true} to include square-level loading and mapping
096     * @return a fully populated {@link Experiment} object, or {@code null} if any
097     *         data file fails to load or validate
098     */
099    public static Experiment loadExperiment(Path projectPath,
100                                            String experimentName,
101                                            boolean matureProject) {
102
103        Path experimentPath   = projectPath.resolve(experimentName);
104        Experiment experiment = new Experiment(experimentName);
105
106        // ─── Recordings ───────────────────────────────────────────────────────
107        RecordingTableIO recIO = new RecordingTableIO();
108        List<Recording> recordings;
109        try {
110            Table recTable = recIO.readCsvWithSchema(
111                    experimentPath.resolve(RECORDINGS_CSV),
112                    RECORDINGS,
113                    RECORDINGS_COLS,
114                    RECORDINGS_TYPES,
115                    false
116            );
117            recordings = recIO.toEntities(recTable);
118            recordings.forEach(experiment::addRecording);
119        } catch (Exception e) {
120            PaintLogger.errorf("Failed to read %s in %s", RECORDINGS_CSV, experimentName);
121            return null;
122        }
123
124        // ─── Tracks ───────────────────────────────────────────────────────────
125        TrackTableIO trackIO = new TrackTableIO();
126        Table tracksTable;
127        try {
128            tracksTable = trackIO.readCsvWithSchema(
129                    experimentPath.resolve(TRACKS_CSV),
130                    TRACKS,
131                    TRACKS_COLS,
132                    TRACKS_TYPES,
133                    false
134            );
135        } catch (Exception e) {
136            PaintLogger.errorf("Failed to read %s in %s", TRACKS_CSV, experimentName);
137            return null;
138        }
139
140        // ─── Attach tracks to recordings ──────────────────────────────────────
141        PaintLogger.debugf("Found %d tracks", tracksTable.rowCount());
142        for (Recording recording : recordings) {
143            if (!recording.isProcessFlag()) {
144                continue; // skip non-processed recordings
145            }
146
147            Table recTracks = tracksTable.where(
148                    tracksTable.stringColumn("Recording Name")
149                            .isEqualTo(recording.getRecordingName()));
150
151            PaintLogger.debugf("Found %d tracks for recording '%s'",
152                               recTracks.rowCount(), recording.getRecordingName());
153
154            recording.setTracks(trackIO.toEntities(recTracks));
155            recording.setTracksTable(recTracks);
156        }
157
158        // ─── Squares (only for mature projects) ───────────────────────────────
159        if (matureProject) {
160            SquareTableIO squareIO = new SquareTableIO();
161            Table squaresTable;
162            try {
163                squaresTable = squareIO.readCsvWithSchema(
164                        experimentPath.resolve(SQUARES_CSV),
165                        SQUARES,
166                        SQUARES_COLS,
167                        SQUARES_TYPES,
168                        false
169                );
170            } catch (Exception e) {
171                PaintLogger.errorf("Failed to read %s in %s", SQUARES_CSV, experimentName);
172                return null;
173            }
174
175            int numberOfRecordings          = recordings.size();
176            int numberOfSquares             = squaresTable.rowCount();
177            int numberOfSquaresPerRecording = (numberOfRecordings == 0) ? 0 : numberOfSquares / numberOfRecordings;
178
179            int numberOfRows;
180            if (numberOfSquaresPerRecording > 0) {
181                numberOfRows = (int) Math.round(Math.sqrt(numberOfSquaresPerRecording));
182            } else {
183                numberOfRows = 0;
184            }
185
186            // Validate square layout geometry
187            if (numberOfRows > 0 && numberOfRows * numberOfRows != numberOfSquaresPerRecording) {
188                PaintLogger.errorf("Invalid squares layout in experiment '%s'", experimentName);
189                System.exit(-1);
190            }
191
192            // ─── Assign squares and map tracks into them ──────────────────────
193            for (Recording rec : recordings) {
194                Table recSquares = squaresTable.where(
195                        squaresTable.stringColumn("Recording Name")
196                                .matchesRegex("^" + rec.getRecordingName() + "(?:-threshold-\\d{1,3})?$"));
197
198                rec.addSquares(squareIO.toEntities(recSquares));
199
200                if (numberOfRows > 0) {
201                    int lastRowCol = numberOfRows - 1;
202                    Table recTracks = rec.getTracksTable();
203
204                    for (Square square : rec.getSquaresOfRecording()) {
205                        Table squareTracks = filterTracksInSquare(recTracks, square, lastRowCol);
206                        square.setTracks(trackIO.toEntities(squareTracks));
207                    }
208                }
209            }
210        }
211
212        return experiment;
213    }
214}
