039import tech.tablesaw.api.Row;

041import tech.tablesaw.columns.Column;
042
043import java.io.IOException;
044import java.nio.file.Path;
045import java.util.ArrayList;
046import java.util.List;
047
048import static paint.shared.constants.PaintConstants.EXPERIMENT_INFO_COLS;
049import static paint.shared.constants.PaintConstants.EXPERIMENT_INFO_TYPES;

051/**
052 * Provides table input/output utilities for {@code ExperimentInfo}
053 * records (per-recording metadata).
054 *
055 * <p>This class enforces a fixed schema defined by
056 * {@code EXPERIMENT_INFO_COLS} and {@code EXPERIMENT_INFO_TYPES}.
057 * It supports:</p>
058 * <ul>
059 *   <li>Creating an empty table with the correct schema
060 *       via {@link #emptyTable()}.</li>
061 *   <li>Converting between lists of {@link ExperimentInfo} objects
062 *       and a {@link tech.tablesaw.api.Table}
063 *       via {@link #toTable(java.util.List)} and
064 *       {@link #toEntities(tech.tablesaw.api.Table)}.</li>
065 *   <li>Reading CSV files into validated tables
066 *       via {@link #readCsv(java.nio.file.Path)}.</li>
067 *   <li>Appending rows from one table to another with type-safe coercion
068 *       via {@link #appendInPlace(tech.tablesaw.api.Table, tech.tablesaw.api.Table)}.</li>
069 * </ul>
070 *
071 * <p>Validation ensures that column order, names, and types match the expected
072 * schema, while allowing some flexibility (e.g. {@code INTEGER -> DOUBLE} upcasts).</p>
073 */
074public class ExperimentInfoTableIO extends BaseTableIO {
075
076    /**
077     * Creates an empty {@link Table} with the {@code Experiment Info} schema.
078     *
079     * <p>The table has all expected columns defined by
080     * {@code EXPERIMENT_INFO_COLS} and {@code EXPERIMENT_INFO_TYPES}, but
081     * contains zero rows.</p>
082     *
083     * @return a new empty {@code Table} ready to receive rows with the
084     * Experiment Info schema
085     */
086    public Table emptyTable() {
087        return newEmptyTable("Experiment Info", EXPERIMENT_INFO_COLS, EXPERIMENT_INFO_TYPES);
088    }
089
090    /**
091     * Converts a list of {@link ExperimentInfo} objects into a {@link Table}
092     * with a fixed schema.
093     *
094     * <p>Each {@code ExperimentInfo} is mapped to a single row in the table with
095     * the following columns:</p>
096     * <ul>
097     *   <li>{@code Recording Name} (String)</li>
098     *   <li>{@code Condition Number} (int)</li>
099     *   <li>{@code Replicate Number} (int)</li>
100     *   <li>{@code Probe Name} (String)</li>
101     *   <li>{@code Probe Type} (String)</li>
102     *   <li>{@code Cell Type} (String)</li>
103     *   <li>{@code Adjuvant} (String)</li>
104     *   <li>{@code Concentration} (double)</li>
105     *   <li>{@code Process Flag} (boolean)</li>
106     *   <li>{@code Threshold} (double)</li>
107     * </ul>
108     *
109     * <p>The schema is enforced by starting from an {@link #emptyTable()} with all
110     * expected columns pre-defined.</p>
111     *
112     * @param infos the list of {@code ExperimentInfo} objects to convert
113     * @return a {@code Table} containing one row per experiment
114     */
115    public Table toTable(List<ExperimentInfo> infos) {
116        Table table = emptyTable();
117        for (ExperimentInfo info : infos) {
118            Row row = table.appendRow();
119            row.setString("Experiment Name",  info.getExperimentName());
120            row.setString("Recording Name",   info.getRecordingName());
121            row.setInt(   "Condition Number", info.getConditionNumber());
122            row.setInt(   "Replicate Number", info.getReplicateNumber());
123            row.setString("Probe Name",       info.getProbeName());
124            row.setString("Probe Type",       info.getProbeType());
125            row.setString("Cell Type",        info.getCellType());
126            row.setString("Adjuvant",         info.getAdjuvant());
127            row.setDouble("Concentration",    info.getConcentration());
128            row.setBoolean("Process Flag",    info.isProcessFlag());
129            row.setDouble("Threshold",        info.getThreshold());
130        }
131        return table;
132    }
133
134    /**
135     * Converts a {@link Table} into a list of {@link ExperimentInfo} objects.
136     *
137     * <p>The table is expected to have already been validated against the
138     * experiment schema. Each row is mapped to a new {@code ExperimentInfo}
139     * instance, with values extracted from the following columns:</p>
140     * <ul>
141     *   <li>{@code Recording Name} → {@code ExperimentInfo.setRecordingName(String)}</li>
142     *   <li>{@code Condition Number} → {@code ExperimentInfo.setConditionNumber(int)}</li>
143     *   <li>{@code Replicate Number} → {@code ExperimentInfo.setReplicateNumber(int)}</li>
144     *   <li>{@code Probe Name} → {@code ExperimentInfo.setProbeName(String)}</li>
145     *   <li>{@code Probe Type} → {@code ExperimentInfo.setProbeType(String)}</li>
146     *   <li>{@code Cell Type} → {@code ExperimentInfo.setCellType(String)}</li>
147     *   <li>{@code Adjuvant} → {@code ExperimentInfo.setAdjuvant(String)}</li>
148     *   <li>{@code Concentration} → {@code ExperimentInfo.setConcentration(double)}</li>
149     *   <li>{@code Process Flag} → {@code ExperimentInfo.setProcessFlag(boolean)}</li>
150     *   <li>{@code Threshold} → {@code ExperimentInfo.setThreshold(double)}</li>
151     * </ul>
152     *
153     * @param table the validated table to convert
154     * @return a list of {@code ExperimentInfo} objects populated from the table rows
155     */
156    public List<ExperimentInfo> toEntities(Table table) {
157        List<ExperimentInfo> infos = new ArrayList<>();
158        for (Row tablesawRow : table) {
159            ExperimentInfo experimentInfo = new ExperimentInfo();
160            experimentInfo.setExperimentName(  tablesawRow.getString( "Experiment Name"));
161            experimentInfo.setRecordingName(   tablesawRow.getString( "Recording Name"));
162            experimentInfo.setConditionNumber( tablesawRow.getInt(    "Condition Number"));
163            experimentInfo.setReplicateNumber( tablesawRow.getInt(    "Replicate Number"));
164            experimentInfo.setProbeName(       tablesawRow.getString( "Probe Name"));
165            experimentInfo.setProbeType(       tablesawRow.getString( "Probe Type"));
166            experimentInfo.setCellType(        tablesawRow.getString( "Cell Type"));
167            experimentInfo.setAdjuvant(        tablesawRow.getString( "Adjuvant"));
168            experimentInfo.setConcentration(   tablesawRow.getDouble( "Concentration"));
169            experimentInfo.setProcessFlag(     tablesawRow.getBoolean("Process Flag"));
170            experimentInfo.setThreshold(       tablesawRow.getDouble( "Threshold"));
171            infos.add(experimentInfo);
172        }
173        return infos;
174    }
175
176    /**
177     * Reads a CSV file as a {@link Table}, enforcing the
178     * {@code Experiment Info} schema.
179     *
180     * <p>The schema is defined by {@code EXPERIMENT_INFO_COLS} (column names) and
181     * {@code EXPERIMENT_INFO_TYPES} (column types). Header order and types are
182     * validated before returning the table.</p>
183     *
184     * @param csvPath the path to the CSV file
185     * @return a {@code Table} containing the experiment info data
186     * @throws IOException if the file cannot be read or parsed
187     */
188    public Table readCsv(Path csvPath) throws IOException {
189        return readCsvWithSchema(csvPath, "Experiment Info",
190                                 EXPERIMENT_INFO_COLS, EXPERIMENT_INFO_TYPES, false);
191    }
192
193    /**
194     * Appends all rows from a source {@link Table} into a target {@link Table},
195     * enforcing the {@code Experiment Info} schema.
196     *
197     * <p>Behavior:</p>
198     * <ul>
199     *   <li>Rows are appended one by one to the {@code target}.</li>
200     *   <li>Columns are matched against {@code EXPERIMENT_INFO_COLS} with types from
201     *       {@code EXPERIMENT_INFO_TYPES}.</li>
202     *   <li>If the source table is missing a column, that column is skipped.</li>
203     *   <li>Missing cell values in the source remain missing in the destination.</li>
204     *   <li>{@code INTEGER -> DOUBLE} upcasts are allowed when the schema expects a double.</li>
205     *   <li>Other type mismatches throw an {@link IllegalArgumentException}.</li>
206     * </ul>
207     *
208     * @param target the destination table to which rows will be appended
209     * @param source the source table providing rows and column values
210     * @throws IllegalArgumentException if the source contains an unsupported column type
211     */
212    public void appendInPlace(Table target, Table source) {
213        if (source.isEmpty()) {
214            return; // nothing to do
215        }
216
217        for (Row srcRow : source) {
218            Row dst = target.appendRow();
219            int r   = srcRow.getRowNumber();
220
221            for (int i = 0; i < EXPERIMENT_INFO_COLS.length; i++) {
222                String col = EXPERIMENT_INFO_COLS[i];
223                ColumnType expected = EXPERIMENT_INFO_TYPES[i];
224
225                if (!source.columnNames().contains(col)) {
226                    continue; // Source missing the column — skip
227                }
228
229                Column<?> sCol = source.column(col);
230                if (sCol.isMissing(r)) {
231                    continue; // Leave as missing in the destination
232                }
233
234                if (expected.equals(ColumnType.STRING)) {
235                    dst.setString(col, source.stringColumn(col).get(r));
236                } else if (expected.equals(ColumnType.INTEGER)) {
237                    dst.setInt(col, source.intColumn(col).getInt(r));
238                } else if (expected.equals(ColumnType.DOUBLE)) {
239                    if (sCol.type().equals(ColumnType.INTEGER)) {
240                        dst.setDouble(col, source.intColumn(col).getInt(r));
241                    } else {
242                        dst.setDouble(col, source.doubleColumn(col).getDouble(r));
243                    }
244                } else if (expected.equals(ColumnType.BOOLEAN)) {
245                    dst.setBoolean(col, source.booleanColumn(col).get(r));
246                } else {
247                    throw new IllegalArgumentException("Unsupported type: " + expected);
248                }
249            }
250        }
251    }
