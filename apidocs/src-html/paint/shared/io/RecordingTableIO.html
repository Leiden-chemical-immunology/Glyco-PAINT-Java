057/**
058 * Provides table I/O utilities for {@link Recording} entities.
059 *
060 * <p>This class manages reading, writing, and schema enforcement for
061 * {@code recordings.csv} files. Each method ensures full consistency
062 * with the column definitions in PaintConstants.</p>
063 */
064public class RecordingTableIO extends BaseTableIO {
065
066    // ───────────────────────────────────────────────────────────────────────────────
067    // TABLE CREATION
068    // ───────────────────────────────────────────────────────────────────────────────
069
070    /**
071     * Creates an empty {@link Table} for recordings with the full schema applied.
072     *
073     * @return a new empty {@code Table} with the “Recordings” schema
074     */
075    public Table emptyTable() {
076        return newEmptyTable("Recordings", RECORDINGS_COLS, RECORDINGS_TYPES);
077    }
078
079    // ───────────────────────────────────────────────────────────────────────────────
080    // ENTITY ⇄ TABLE CONVERSION
081    // ───────────────────────────────────────────────────────────────────────────────
082
083    /**
084     * Converts a list of {@link Recording} entities into a fully typed
085     * {@link Table} matching the {@code recordings.csv} schema.
086     *
087     * @param recordings the list of {@code Recording} objects to convert
088     * @return a {@code Table} populated with recording data
089     */
090    public Table toTable(List<Recording> recordings) {
091        Table table = emptyTable();
092        for (Recording recording : recordings) {
093            Row row = table.appendRow();
094            row.setString(  "Experiment Name",                   recording.getExperimentName());
095            row.setString(  "Recording Name",                    recording.getRecordingName());
096            row.setInt(     "Condition Number",                  recording.getConditionNumber());
097            row.setInt(     "Replicate Number",                  recording.getReplicateNumber());
098            row.setString(  "Probe Name",                        recording.getProbeName());
099            row.setString(  "Probe Type",                        recording.getProbeType());
100            row.setString(  "Cell Type",                         recording.getCellType());
101            row.setString(  "Adjuvant",                          recording.getAdjuvant());
102            row.setDouble(  "Concentration",                     recording.getConcentration());
103            row.setBoolean( "Process Flag",                      recording.isProcessFlag());
104            row.setDouble(  "Threshold",                         recording.getThreshold());
105            row.setInt(     "Number of Spots",                   recording.getNumberOfSpots());
106            row.setInt(     "Number of Tracks",                  recording.getNumberOfTracks());
107            row.setInt(     "Number of Squares in Background",   recording.getNumberOfSquaresInBackground());
108            row.setInt(     "Number of Tracks in Background",    recording.getNumberOfTracksInBackground());
109            row.setDouble(  "Average Tracks in Background",      recording.getAverageTracksInBackGround());
110            row.setInt(     "Number of Spots in All Tracks",     recording.getNumberOfSpotsInAllTracks());
111            row.setInt(     "Number of Frames",                  recording.getNumberOfFrames());
112            row.setDouble(  "Run Time",                          recording.getRunTime());
113            row.setDateTime("Time Stamp",                        recording.getTimeStamp());
114            row.setBoolean( "Exclude",                           recording.isExclude());
115            row.setDouble(  "Tau",                               recording.getTau());
116            row.setDouble(  "R Squared",                         recording.getRSquared());
117            row.setDouble(  "Density",                           recording.getDensity());
118        }
119        return table;
120    }
121
122    /**
123     * Converts a {@link Table} into a list of {@link Recording} entities.
124     *
125     * <p>The table must conform to the schema defined by
126     * {@code RECORDINGS_COLS} and {@code RECORDINGS_TYPES}.
127     * Each row is mapped one-to-one to a {@code Recording} object.</p>
128     *
129     * @param table the validated {@link Table} to convert
130     * @return a list of {@code Recording} entities populated from the table
131     */
132    public List<Recording> toEntities(Table table) {
133        List<Recording> recordings = new ArrayList<>();
134        for (Row tablesawRow : table) {
135            Recording recording = new Recording();
136            recording.setExperimentName(              tablesawRow.getString(   "Experiment Name"));
137            recording.setRecordingName(               tablesawRow.getString(   "Recording Name"));
138            recording.setConditionNumber(             tablesawRow.getInt(      "Condition Number"));
139            recording.setReplicateNumber(             tablesawRow.getInt(      "Replicate Number"));
140            recording.setProbeName(                   tablesawRow.getString(   "Probe Name"));
141            recording.setProbeType(                   tablesawRow.getString(   "Probe Type"));
142            recording.setCellType(                    tablesawRow.getString(   "Cell Type"));
143            recording.setAdjuvant(                    tablesawRow.getString(   "Adjuvant"));
144            recording.setConcentration(               tablesawRow.getDouble(   "Concentration"));
145            recording.setProcessFlag(                 tablesawRow.getBoolean(  "Process Flag"));
146            recording.setThreshold(                   tablesawRow.getDouble(   "Threshold"));
147            recording.setNumberOfSpots(               tablesawRow.getInt(      "Number of Spots"));
148            recording.setNumberOfTracks(              tablesawRow.getInt(      "Number of Tracks"));
149            recording.setNumberOfSquaresInBackground( tablesawRow.getInt(      "Number of Squares in Background"));
150            recording.setNumberOfTracksInBackground(  tablesawRow.getInt(      "Number of Tracks in Background"));
151            recording.setAverageTracksInBackGround(   tablesawRow.getDouble(   "Average Tracks in Background"));
152            recording.setNumberOfSpotsInAllTracks(    tablesawRow.getInt(      "Number of Spots in All Tracks"));
153            recording.setNumberOfFrames(              tablesawRow.getInt(      "Number of Frames"));
154            recording.setRunTime(                     tablesawRow.getDouble(   "Run Time"));
155            recording.setTimeStamp(                   tablesawRow.getDateTime( "Time Stamp"));
156            recording.setExclude(                     tablesawRow.getBoolean(  "Exclude"));
157            recording.setTau(                         tablesawRow.getDouble(   "Tau"));
158            recording.setRSquared(                    tablesawRow.getDouble(   "R Squared"));
159            recording.setDensity(                     tablesawRow.getDouble(   "Density"));
160            recordings.add(recording);
161        }
162        return recordings;
163    }

165    // ───────────────────────────────────────────────────────────────────────────────
166    // CSV READ / APPEND

169    /**
170     * Reads a CSV file containing recording data into a validated {@link Table}.
171     *
172     * @param csvPath path to the {@code recordings.csv} file
173     * @return a {@link Table} conforming to the recordings schema
174     * @throws IOException if the file cannot be read or parsed
175     */
176    public Table readCsv(Path csvPath) throws IOException {
177        return readCsvWithSchema(csvPath, RECORDINGS, RECORDINGS_COLS, RECORDINGS_TYPES, false);
178    }
179
180    /**
181     * Appends all rows from a source {@link Table} into a target {@link Table},
182     * enforcing the recordings schema.
183     *
184     * <p>Behavior:</p>
185     * <ul>
186     *   <li>All columns are appended in schema order.</li>
187     *   <li>Supports basic type coercion (INTEGER → DOUBLE).</li>
188     *   <li>Missing values are preserved as missing.</li>
189     *   <li>Both tables must share the same schema.</li>
190     * </ul>
191     *
192     * @param target the destination table
193     * @param source the source table to append from
194     */
195    public void appendInPlace(Table target, Table source) {
196        for (Row row : source) {
197            Row newRow = target.appendRow();
198
199            for (String col : RECORDINGS_COLS) {
200                Column<?> targetCol = target.column(col);
201
202                if (targetCol.type() == ColumnType.STRING) {
203                    newRow.setString(col, row.getString(col));
204                } else if (targetCol.type() == ColumnType.INTEGER) {
205                    newRow.setInt(col, row.getInt(col));
206                } else if (targetCol.type() == ColumnType.DOUBLE) {
207                    newRow.setDouble(col, row.getDouble(col));
208                } else if (targetCol.type() == ColumnType.BOOLEAN) {
209                    newRow.setBoolean(col, row.getBoolean(col));
210                }
211            }
212        }
213    }
