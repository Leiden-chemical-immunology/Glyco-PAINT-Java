005 *  PURPOSE:
006 *    Validates the `recordings.csv` file within a PAINT experiment project.
007 *    Ensures correct header order, data types, and logical consistency across
008 *    condition groups shared with `experiment_info.csv`.
009 *
010 *  DESCRIPTION:
011 *    • Verifies that the header matches {@link PaintConstants#RECORDINGS_COLS}.
012 *    • Checks that each column’s data type conforms to
013 *      {@link PaintConstants#RECORDINGS_TYPES}.
014 *    • Performs a consistency check ensuring that all rows with the same
015 *      “Condition Number” share identical Probe, Cell Type, Adjuvant, and
016 *      Concentration values, consistent with `experiment_info.csv`.
017 *
018 *  RESPONSIBILITIES:
019 *    • Detect header mismatches or missing columns.
020 *    • Validate cell-level data types against the schema definition.
021 *    • Detect logical inconsistencies across repeated condition groups.
022 *
023 *  USAGE EXAMPLE:
024 *    File csv = new File("recordings.csv");
025 *    RecordingsValidator validator = new RecordingsValidator();
026 *    ValidationResult result = validator.validateWithConsistency(csv, "Experiment A");
027 *    if (!result.isValid()) { result.printSummary(); }
028 *

047package paint.shared.validate;
048
049import paint.shared.constants.PaintConstants;
050import tech.tablesaw.api.ColumnType;
051
052import java.io.File;
053import java.util.Arrays;
054import java.util.List;
055
056/**
057 * Validates the structural and logical integrity of {@code recordings.csv}.
058 * <p>
059 * Performs schema validation (header + types) and an additional
060 * condition-based consistency check to ensure uniformity of experimental metadata.
061 * </p>
062 */
063public final class RecordingsValidator extends AbstractFileValidator {
064
065    // ───────────────────────────────────────────────────────────────────────────────
066    // HEADER VALIDATION
067    // ───────────────────────────────────────────────────────────────────────────────
068
069    /**
070     * Validates that the header matches {@link PaintConstants#RECORDINGS_COLS}.
071     *
072     * @param actualHeader the CSV header read from the file
073     * @param result       validation result collector
074     */
075    @Override
076    protected void validateHeader(List<String> actualHeader, ValidationResult result) {
077        List<String> expectedHeader = Arrays.asList(PaintConstants.RECORDINGS_COLS);
078        headersMatch(expectedHeader, actualHeader, result);
079    }
080
081    // ───────────────────────────────────────────────────────────────────────────────
082    // TYPE VALIDATION
083    // ───────────────────────────────────────────────────────────────────────────────
084
085    /**
086     * Returns the expected column data types as defined in
087     * {@link PaintConstants#RECORDINGS_TYPES}.
088     *
089     * @return array of expected {@link ColumnType}s
090     */
091    @Override
092    protected ColumnType[] getExpectedTypes() {
093        return PaintConstants.RECORDINGS_TYPES;
094    }
095
096    // ───────────────────────────────────────────────────────────────────────────────
097    // CONSISTENCY CHECK
098    // ───────────────────────────────────────────────────────────────────────────────
099
100    /**
101     * Performs full validation — including header, type, and condition-based consistency checks.
102     *
103     * @param file           the {@code recordings.csv} file to validate
104     * @return {@link ValidationResult} summarizing detected issues or confirming validity
105     */
106    public ValidationResult validateWithConsistency(File file) {
107        ValidationResult result = validate(file);
108        if (result.isValid()) {
109            result.merge(ConditionConsistencyChecker.check(file));
110        }
111        return result;
112    }
113}
