001/******************************************************************************
002 *  Class:        ValidationHandler.java
003 *  Package:      paint.shared.validate
004 *
005 *  PURPOSE:
006 *    Performs batch validation of core PAINT experiment CSV files across one
007 *    or more experiments within a project directory. Each experiment’s
008 *    files are validated for schema, type, and consistency integrity.
009 *
010 *  DESCRIPTION:
011 *    • Iterates over a list of experiment names and validates selected CSV files.
012 *    • Routes validation to specific validator classes depending on file type
013 *      (Experiment Info, Recordings, Squares, or Tracks).
014 *    • Aggregates all validation messages into a unified report.
015 *    • Outputs concise problem lines in the format:
016 *         [ExperimentName] - [FileName] - [Problem summary]
017 *    • Provides verbose logging output when {@link PaintRuntime#isVerbose()} is enabled.
018 *
019 *  RESPONSIBILITIES:
020 *    • Coordinate multi-file validation across experiments.
021 *    • Integrate individual validators into a unified workflow.
022 *    • Provide human-readable reporting and console logging.
023 *
024 *  USAGE EXAMPLE:
025 *    List<String> experiments = Arrays.asList("221012", "221108");
026 *    List<String> files = Arrays.asList(EXPERIMENT_INFO_CSV, RECORDINGS_CSV);
027 *    ValidationHandler.validate(projectPath, experiments, files);
028 *
029 *  DEPENDENCIES:
030 *    – paint.shared.validate.{ExperimentInfoValidator, RecordingsValidator,
031 *      SquaresValidator, TracksValidator, ValidationResult}

065/**
066 * Central coordinator for validating multiple experiment CSV files within
067 * a PAINT project. Executes specific validators per file type and aggregates
068 * results into a combined {@link ValidationResult}.
069 */
070public final class ValidationHandler {
071
072    // ───────────────────────────────────────────────────────────────────────────────
073    // MAIN VALIDATION ENTRY POINTS
074    // ───────────────────────────────────────────────────────────────────────────────
075
076    /**
077     * Validates a single file type across multiple experiments.
078     *
079     * @param projectPath     the base path of the PAINT project
080     * @param experimentNames list of experiment directories to validate
081     * @param fileName        the file name to validate (e.g., "Experiment Info.csv")
082     * @return accumulated {@link ValidationResult} across all experiments
083     */
084    public static ValidationResult validateExperiments(Path projectPath,
085                                                       List<String> experimentNames,
086                                                       String fileName) {
087        return validateExperiments(projectPath, experimentNames, Collections.singletonList(fileName));
088    }
089
090    /**
091     * Validates one or more specified files across multiple experiments.
092     *
093     * @param projectPath     the base path of the PAINT project
094     * @param experimentNames list of experiment directories to validate
095     * @param fileNames       list of CSV file names to validate
096     * @return {@link ValidationResult} with accumulated validation issues
097     */
098    public static ValidationResult validateExperiments(Path projectPath,
099                                                       List<String> experimentNames,
100                                                       List<String> fileNames) {
101
102        List<String>     report    = new ArrayList<>();
103        ValidationResult overall   = new ValidationResult();
104        boolean          isVerbose = PaintRuntime.isVerbose();
105
106        for (String expName : experimentNames) {
107            Path expDir = projectPath.resolve(expName);
108
109            if (!Files.isDirectory(expDir)) {
110                String msg = "[" + expName + "] - Directory - Missing experiment directory: " + expDir;
111                overall.addError(msg);
112                report.add(msg);
113                continue;
114            }
115
116            if (isVerbose) {
117                PaintLogger.infof("   Validating experiment: %s", expName);
118            }
119
120            for (String fileName : fileNames) {
121                Path filePath = expDir.resolve(fileName);
122
123                if (!Files.exists(filePath)) {
124                    String msg = "[" + expName + "] - " + fileName + " - Missing file";
125                    overall.addError(msg);
126                    report.add(msg);
127                    continue;
128                }
129
130                ValidationResult res = runValidator(fileName, filePath.toFile());
131
132                if (!res.isValid()) {
133                    for (String err : res.getErrors()) {
134                        String msg = formatMessage(expName, fileName, err);
135                        overall.addError(msg);
136                        report.add(msg);
137                    }
138                }
139            }
140        }
141
142        if (isVerbose) {
143            PaintLogger.blankline();
144        }
145
146        if (!report.isEmpty()) {
147            overall.setReport(String.join("\n", report));
148        }
149
150        return overall;
151    }
152
153    // ───────────────────────────────────────────────────────────────────────────────
154    // VALIDATOR ROUTING LOGIC
155    // ───────────────────────────────────────────────────────────────────────────────
156
157    /**
158     * Routes validation based on a file name pattern to the appropriate validator.
159     *
160     * @param fileName        the CSV file name
161     * @param file            the {@link File} object to validate
162     * @return a {@link ValidationResult} containing file-specific validation output
163     */
164    private static ValidationResult runValidator(String fileName, File file) {
165        String lower = fileName.toLowerCase();
166        if (lower.contains("experiment")) {
167            return new ExperimentInfoValidator().validateWithConsistency(file);
168        } else if (lower.contains("recording")) {
169            return new RecordingsValidator().validateWithConsistency(file);
170        } else if (lower.contains("track")) {
171            return new TracksValidator().validate(file);
172        } else if (lower.contains("square")) {
173            return new SquaresValidator().validate(file);
174        } else {
175            ValidationResult res = new ValidationResult();
176            res.addError("Unknown file: " + fileName);
177            return res;
178        }
179    }
180
181    /**
182     * Formats a validation message consistently for inclusion in the summary report.
183     */
184    private static String formatMessage(String expName, String fileName, String err) {
185        String flattened = err.replace("\n", " ").replaceAll("\\s+", " ").trim();
186
187        if (flattened.startsWith("[" + expName + "]")) {
188            return "[" + expName + "] - " + fileName + " - "
189                    + flattened.substring(flattened.indexOf("]") + 1).trim();
190        }
191        return "[" + expName + "] - " + fileName + " - " + flattened;
192    }
193
194    // ───────────────────────────────────────────────────────────────────────────────
195    // CONSOLE WRAPPER
196    // ───────────────────────────────────────────────────────────────────────────────
197
198    /**
199     * Performs validation and logs results directly to the console.
200     *
201     * @param projectPath     project root directory
202     * @param experimentNames experiment directories to validate
203     * @param fileNames       CSV file names to validate
204     */
205    public static void validate(Path projectPath, List<String> experimentNames, List<String> fileNames) {
206
207        PaintLogger.infof("Validating experiments: %s", experimentNames);
208        PaintLogger.blankline();
209        PaintLogger.infof("Validating files: %s", fileNames);
210        PaintLogger.blankline();
211
212        ValidationResult validateResult = validateExperiments(projectPath, experimentNames, fileNames);
213
214        if (!validateResult.isValid()) {
215            for (String line : validateResult.getReport().split("\n")) {
216                PaintLogger.errorf(line);
217            }
218        } else {
219            PaintLogger.infof("All validations passed");
220        }
221    }
