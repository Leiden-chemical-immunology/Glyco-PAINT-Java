043package paint.shared.validate;
044
045import paint.shared.constants.PaintConstants;
046import tech.tablesaw.api.ColumnType;
047
048import java.util.Arrays;
049import java.util.List;
050
051/**
052 * Validates the schema of {@code tracks.csv} by checking header correctness
053 * and column data types according to {@link PaintConstants#TRACKS_COLS} and
054 * {@link PaintConstants#TRACKS_TYPES}.
055 */
056public final class TracksValidator extends AbstractFileValidator {
057
058    // ───────────────────────────────────────────────────────────────────────────────
059    // HEADER VALIDATION
060    // ───────────────────────────────────────────────────────────────────────────────
061
062    /**
063     * Validates that the header matches {@link PaintConstants#TRACKS_COLS}.
064     *
065     * @param actualHeader the CSV header read from the file
066     * @param result       validation result collector
067     */
068    @Override
069    protected void validateHeader(List<String> actualHeader, ValidationResult result) {
070        List<String> expectedHeader = Arrays.asList(PaintConstants.TRACKS_COLS);
071        headersMatch(expectedHeader, actualHeader, result);
072    }
073
074    // ───────────────────────────────────────────────────────────────────────────────
075    // TYPE VALIDATION
076    // ───────────────────────────────────────────────────────────────────────────────
077
078    /**
079     * Returns the expected column data types as defined in
080     * {@link PaintConstants#TRACKS_TYPES}.
081     *
082     * @return array of expected {@link ColumnType}s
083     */
084    @Override
085    protected ColumnType[] getExpectedTypes() {
086        return PaintConstants.TRACKS_TYPES;
087    }
088}
