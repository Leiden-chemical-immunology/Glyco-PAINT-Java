003 *  Package:      paint.shared.validate
004 *
005 *  PURPOSE:
006 *    Validates the existence and integrity of experiment image directories and
007 *    corresponding recording files referenced by Experiment Info CSV files.
008 *
009 *  DESCRIPTION:
010 *    • Ensures each experiment listed in the project has a matching image
011 *      directory under the global image root.
012 *    • Checks that each experiment folder contains its required
013 *      “Experiment Info.csv” metadata file.
014 *    • Parses each CSV file and confirms that all recordings marked for
015 *      processing (Process Flag = true) have corresponding `.nd2` image files
016 *      present in the expected image directory.
017 *
018 *  RESPONSIBILITIES:
019 *    • Detect missing experiment directories or image files.
020 *    • Cross-validate metadata and file system structure.
021 *    • Provide detailed reporting via {@link ValidationResult}.
022 *
023 *  USAGE EXAMPLE:
024 *    List<String> experiments = Arrays.asList("221108", "221122");
025 *    ValidationResult result = ImageRootValidator.validateImageRoot(
026 *        Paths.get("/Users/hans/Paint Test Project"),
027 *        Paths.get("/Volumes/Extreme Pro/Omero"),
028 *        experiments
029 *    );
030 *    System.out.println(result.getReport());
031 *
032 *  DEPENDENCIES:

051package paint.shared.validate;
052
053import org.apache.commons.csv.CSVFormat;
054import org.apache.commons.csv.CSVParser;
055import org.apache.commons.csv.CSVRecord;
056
057import java.io.IOException;
058import java.io.Reader;
059import java.nio.file.Files;
060import java.nio.file.Path;
061import java.nio.file.Paths;
062import java.util.Arrays;
063import java.util.List;
064
065/**
066 * Validates that all required experiment and image directories exist, and that
067 * each recording flagged for processing in an Experiment Info file has a
068 * corresponding `.nd2` image file in the expected location.
069 */
070public final class ImageRootValidator {
071
072    /**
073     * Standard file name for experiment metadata.
074     */
075    private static final String EXPERIMENT_INFO_CSV = "Experiment Info.csv";
076
077    // ───────────────────────────────────────────────────────────────────────────────
078    // MAIN ENTRY POINT (TEST HARNESS)
079    // ───────────────────────────────────────────────────────────────────────────────
080
081    /**
082     * Standalone test harness for local validation of project/image directory pairs.
083     *
084     * @param args command-line arguments (unused)
085     * @throws IOException if any I/O error occurs while reading CSV files
086     */
087    public static void main(String[] args) {
088        List<String> experiments = Arrays.asList("221108", "221122");
089
090        ValidationResult result = ImageRootValidator.validateImageRoot(
091                Paths.get("/Users/hans/Paint Test Project"),
092                Paths.get("/Volumes/Extreme Pro/Omero"),
093                experiments
094        );
095
096        System.out.println(result.getReport());
097    }
098
099    // ───────────────────────────────────────────────────────────────────────────────
100    // CORE VALIDATION LOGIC
101    // ───────────────────────────────────────────────────────────────────────────────
102
103    /**
104     * Validates that all required recording image files exist under the specified
105     * image root for each experiment in the provided list.
106     *
107     * <p>For each experiment:
108     * <ul>
109     *   <li>Confirms the corresponding image directory exists under {@code imagesRoot}.</li>
110     *   <li>Confirms the experiment folder under {@code projectRoot} contains
111     *       the {@code Experiment Info.csv} file.</li>
112     *   <li>Parses the CSV to locate all recordings marked with
113     *       {@code Process Flag = true}, and verifies each has a matching
114     *       {@code .nd2} file in the image directory.</li>
115     * </ul>
116     *
117     * @param projectRoot     path to the local PAINT project root
118     * @param imagesRoot      path to the global image repository root
119     * @param experimentNames list of experiment identifiers to check
120     * @return a {@link ValidationResult} containing all missing files/directories
121     */
122    public static ValidationResult validateImageRoot(Path projectRoot,
123                                                     Path imagesRoot,
124                                                     List<String> experimentNames) {
125        ValidationResult result = new ValidationResult();
126
127        for (String experiment : experimentNames) {
128            Path experimentDir = projectRoot.resolve(experiment);
129            Path imageDir      = imagesRoot.resolve(experiment);
130
131            // ── 1. Check image directory existence
132            if (!Files.isDirectory(imageDir)) {
133                result.addError("[" + experiment + "] Missing Image Root: " + imageDir);
134                continue;
135            }
136
137            // ── 2. Check Experiment Info CSV presence
138            Path expInfoFile = experimentDir.resolve(EXPERIMENT_INFO_CSV);
139            if (!Files.exists(expInfoFile)) {
140                result.addError("[" + experiment + "] Missing " + EXPERIMENT_INFO_CSV + " in " + experimentDir);
141                continue;
142            }
143
144            // ── 3. Parse and verify each recording entry - this is a try with resources construct
145            try (Reader reader = Files.newBufferedReader(expInfoFile);
146                 CSVParser parser = CSVFormat.Builder.create(CSVFormat.DEFAULT)
147                         .setHeader()
148                         .setSkipHeaderRecord(true)
149                         .build()
150                         .parse(reader)) {
151
152                for (CSVRecord record : parser) {
153                    String recordingName = record.get("Recording Name");
154                    String processFlag = record.get("Process Flag").trim().toLowerCase();
155
156                    if ("true".equals(processFlag)) {
157                        Path recordingFile = imageDir.resolve(recordingName + ".nd2");
158                        if (!Files.exists(recordingFile)) {
159                            result.addError("[" + experiment + "] Missing recording file: " + recordingFile);
160                        }
161                    }
162                }
163
164            } catch (IOException e) {
165                result.addError("[" + experiment + "] Error reading " + expInfoFile + ": " + e.getMessage());
166            }
167        }
168
169        // ── 4. Generate the summary report
170        if (!result.hasErrors()) {
171            result.setReport("All required image directories and files exist.");
172        } else {
173            result.setReport(String.join("\n", result.getErrors()));
174        }
175
176        return result;
177    }
178
179    /**
180     * Private constructor to prevent instantiation.
181     */
182    private ImageRootValidator() {
183        // Deliberately empty
184    }
185}
