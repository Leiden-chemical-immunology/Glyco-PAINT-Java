048package paint.shared.validate;
049
050import paint.shared.constants.PaintConstants;
051import tech.tablesaw.api.ColumnType;
052
053import java.io.File;
054import java.util.Arrays;
055import java.util.List;
056
057/**
058 * Validates the structure and data of {@code experiment_info.csv}.
059 * <p>
060 * In addition to header and type validation, this class runs an additional
061 * consistency check to ensure uniform attribute values across all rows sharing
062 * the same condition number.
063 * </p>
064 */
065public final class ExperimentInfoValidator extends AbstractFileValidator {
066
067    // ───────────────────────────────────────────────────────────────────────────────
068    // HEADER VALIDATION
069    // ───────────────────────────────────────────────────────────────────────────────
070
071    /**
072     * Validates that the CSV header matches {@link PaintConstants#EXPERIMENT_INFO_COLS}.
073     *
074     * @param actualHeader actual CSV header read from the file
075     * @param result       validation result collector
076     */
077    @Override
078    protected void validateHeader(List<String> actualHeader, ValidationResult result) {
079        List<String> expectedHeader = Arrays.asList(PaintConstants.EXPERIMENT_INFO_COLS);
080        headersMatch(expectedHeader, actualHeader, result);
081    }
082
083    // ───────────────────────────────────────────────────────────────────────────────
084    // TYPE VALIDATION
085    // ───────────────────────────────────────────────────────────────────────────────
086
087    /**
088     * Returns the expected column data types as defined in
089     * {@link PaintConstants#EXPERIMENT_INFO_TYPES}.
090     *
091     * @return array of expected {@link ColumnType}s
092     */
093    @Override
094    protected ColumnType[] getExpectedTypes() {
095        return PaintConstants.EXPERIMENT_INFO_TYPES;
096    }
097
098    // ───────────────────────────────────────────────────────────────────────────────
099    // CONSISTENCY CHECK
100    // ───────────────────────────────────────────────────────────────────────────────
101
102    /**
103     * Runs header and type validation, followed by a consistency check ensuring
104     * all rows with the same "Condition Number" have identical associated attributes.
105     *
106     * @param file           the {@code experiment_info.csv} file to validate
107     * @return a {@link ValidationResult} summarizing all detected issues
108     */
109    public ValidationResult validateWithConsistency(File file) {
110        ValidationResult result = validate(file);
111        if (result.isValid()) {
112            result.merge(ConditionConsistencyChecker.check(file));
113        }
114        return result;
115    }
116}
