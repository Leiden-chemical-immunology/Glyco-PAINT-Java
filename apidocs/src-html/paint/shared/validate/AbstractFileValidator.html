045package paint.shared.validate;
046
047import org.apache.commons.csv.CSVFormat;
048import org.apache.commons.csv.CSVParser;
049import org.apache.commons.csv.CSVRecord;
050import tech.tablesaw.api.ColumnType;
051
052import java.io.*;
053import java.nio.charset.StandardCharsets;
054import java.time.LocalDateTime;
055import java.time.format.DateTimeFormatter;
056import java.time.format.DateTimeFormatterBuilder;
057import java.time.temporal.ChronoField;
058import java.util.HashSet;
059import java.util.List;
060import java.util.Set;
061import java.util.stream.Collectors;
062
063import static paint.shared.utils.Miscellaneous.checkBooleanValue;
064
065/**
066 * Abstract superclass for all CSV file validators in PAINT.
067 * <p>
068 * Provides header, type, and optional value-validation mechanisms that are
069 * common across multiple experiment file formats.
070 * Concrete subclasses specify expected headers and types.
071 * </p>
072 */
073public abstract class AbstractFileValidator {
074
075    // ───────────────────────────────────────────────────────────────────────────────
076    // INTERNAL STATE
077    // ───────────────────────────────────────────────────────────────────────────────
078
079    /**
080     * Tracks which column type errors have already been reported to avoid duplicates.
081     */
082    private final Set<String> reportedTypeErrors = new HashSet<>();
083
084    /**
085     * Flexible timestamp parser supporting multiple formats with optional fractions.
086     */
087    private static final DateTimeFormatter FLEXIBLE_DATE_TIME = new DateTimeFormatterBuilder()
088            .appendPattern("[yyyy-MM-dd'T'HH:mm:ss][dd/MM/yyyy'T'HH:mm:ss]")
089            .optionalStart()
090            .appendFraction(ChronoField.NANO_OF_SECOND, 1, 9, true)
091            .optionalEnd()
092            .toFormatter();
093
094    // ───────────────────────────────────────────────────────────────────────────────
095    // VALIDATION ENTRY POINTS
096    // ───────────────────────────────────────────────────────────────────────────────
097
098    /**
099     * Validates both headers and row values of the specified CSV file.
100     *
101     * @param file        CSV file to validate
102     * @param checkValues whether to validate row values against expected types
103     * @return a {@link ValidationResult} containing all detected issues
104     */
105    public ValidationResult validate(File file, boolean checkValues) {
106        ValidationResult result = new ValidationResult();
107
108        CSVFormat format = CSVFormat.DEFAULT.builder()
109                .setDelimiter(',')
110                .setHeader()
111                .setSkipHeaderRecord(false)
112                .setIgnoreSurroundingSpaces(true)
113                .build();
114
115        try (Reader reader = new InputStreamReader(new FileInputStream(file), StandardCharsets.UTF_8);
116             CSVParser parser = format.parse(reader)) {
117
118            List<String> header = parser.getHeaderNames();
119            List<CSVRecord> records = parser.getRecords();
120
121            validateHeader(header, result);
122
123            if (!checkValues || result.hasErrors()) {
124                return result; // Stop early if only header check or header invalid
125            }
126
127            ColumnType[] types = getExpectedTypes();
128            boolean reportedMismatch = false;
129
130            for (int i = 0; i < records.size(); i++) {
131                CSVRecord record = records.get(i);
132                String[] row = new String[header.size()];
133
134                for (int j = 0; j < header.size(); j++) {
135                    row[j] = record.get(j);
136                }
137
138                if (row.length != types.length) {
139                    if (!reportedMismatch) {
140                        result.addError("Row " + (i + 2) + " has " + row.length
141                                                + " values; expected " + types.length
142                                                + ". Possible formatting issue (e.g., extra/missing commas).");
143                        reportedMismatch = true;
144                    }
145                    continue;
146                }
147                rowMatchesTypes(row, types, header, i + 2, result);
148            }
149        } catch (IOException e) {
150            result.addError("Error reading file: " + e.getMessage());
151        }
152
153        return result;
154    }
155
156    /**
157     * Validates headers only (no row data).
158     *
159     * @param file CSV file to validate
160     * @return {@link ValidationResult} with header validation results
161     */
162    public ValidationResult validateHeadersOnly(File file) {
163        return validate(file, false);
164    }
165
166    /**
167     * Validates both headers and values.
168     *
169     * @param file CSV file to validate
170     * @return {@link ValidationResult} describing validation results
171     */
172    public ValidationResult validate(File file) {
173        return validate(file, true);
174    }
175
176    // ───────────────────────────────────────────────────────────────────────────────
177    // ABSTRACT DEFINITIONS
178    // ───────────────────────────────────────────────────────────────────────────────
179
180    /**
181     * Subclasses implement header validation logic for a specific CSV schema.
182     *
183     * @param actualHeader header names read from the CSV
184     * @param result       result accumulator for recording mismatches
185     */
186    protected abstract void validateHeader(List<String> actualHeader, ValidationResult result);
187
188    /**
189     * Returns the expected column types for the CSV file.
190     *
191     * @return array of {@link ColumnType} values
192     */
193    protected abstract ColumnType[] getExpectedTypes();
194
195    // ───────────────────────────────────────────────────────────────────────────────
196    // HELPER METHODS
197    // ───────────────────────────────────────────────────────────────────────────────
198
199    /**
200     * Compares expected and actual header sequences, recording mismatches.
201     *
202     * @param expected expected header names
203     * @param actual   actual header names
204     * @param result   validation result for error reporting
205     */
206    protected void headersMatch(List<String> expected, List<String> actual, ValidationResult result) {
207        if (expected.equals(actual)) {
208            return;
209        }
210
211        List<String> missing = expected.stream()
212                .filter(h -> !actual.contains(h))
213                .collect(Collectors.toList());
214
215        List<String> unexpected = actual.stream()
216                .filter(h -> !expected.contains(h))
217                .collect(Collectors.toList());
218
219        StringBuilder error = new StringBuilder("Header mismatch:");
220
221        if (!missing.isEmpty()) {
222            error.append("\n- Missing headers: ").append(missing);
223        }
224
225        if (!unexpected.isEmpty()) {
226            error.append("\n- Unexpected headers: ").append(unexpected);
227        }
228
229        result.addError(error.toString());
230    }
231
232    /**
233     * Checks whether all row values conform to expected column types.
234     *
235     * @param row      string array of CSV values
236     * @param types    expected column types
237     * @param headers  header names
238     * @param rowIndex row index (1-based)
239     * @param result   validation accumulator
240      */
241    protected void rowMatchesTypes(String[]         row,
242                                      ColumnType[]     types,
243                                      List<String>     headers,
244                                      int              rowIndex,
245                                      ValidationResult result) {
246        for (int i = 0; i < types.length; i++) {
247            String value = row[i];
248            ColumnType type = types[i];
249            if (!canParse(value, type)) {
250                String colName = headers.size() > i ? headers.get(i) : "Column " + (i + 1);
251                String key = colName + " type " + type.name();
252                if (!reportedTypeErrors.contains(key)) {
253                    result.addError("Some values in column '" + colName +
254                                            "' are invalid for type " + type.name() + ".");
255                    reportedTypeErrors.add(key);
256                }
257            }
258        }
259    }
260
261    /**
262     * Determines if a value can be parsed as the given column type.
263     *
264     * @param value raw string value
265     * @param type  expected type
266     * @return {@code true} if value is parseable or empty; otherwise {@code false}
267     */
268    private boolean canParse(String value, ColumnType type) {
269        if (value == null || value.trim().isEmpty()) {
270            return true;
271        }
272
273        try {
274            switch (type.name()) {
275                case "INTEGER":
276                    Integer.parseInt(value);
277                    break;
278                case "DOUBLE":
279                    Double.parseDouble(value);
280                    break;
281                case "BOOLEAN":
282                    String v = value.trim().toLowerCase();
283                    if (!checkBooleanValue(v)) {
284                        throw new IllegalArgumentException("Invalid boolean: " + value);
285                    }
286                    break;
287                case "LOCAL_DATE_TIME":
288                    LocalDateTime.parse(value, FLEXIBLE_DATE_TIME);
289                    break;
290                case "STRING":
291                    break;
292                default:
293                    return false;
294            }
295            return true;
296        } catch (Exception e) {
297            return false;
298        }
299    }
