077    /**
078     * Filters track coordinates to include only those within the specified square.
079     * <p>
080     * This method handles inclusive or exclusive boundaries depending on whether
081     * the square is in the last column or row. It operates on a Tablesaw table
082     * containing at least "Track X Location" and "Track Y Location" columns.
083     * </p>
084     *
085     * @param tracks     table of track data with X/Y coordinate columns
086     * @param square     the square region defining coordinate boundaries
087     * @param lastRowCol index of the last row/column in the grid (for boundary handling)
088     * @return new {@link Table} containing only the tracks located within the specified square
089     */
090    public static Table filterTracksInSquare(Table tracks, Square square, int lastRowCol) {
091        double x0 = square.getX0();
092        double y0 = square.getY0();
093        double x1 = square.getX1();
094        double y1 = square.getY1();
095
096        boolean isLastCol = square.getColNumber() == lastRowCol;
097        boolean isLastRow = square.getRowNumber() == lastRowCol;
098
099        double left   = Math.min(x0, x1);
100        double right  = Math.max(x0, x1);
101        double top    = Math.min(y0, y1);
102        double bottom = Math.max(y0, y1);
103
104        DoubleColumn x = tracks.doubleColumn("Track X Location");
105        DoubleColumn y = tracks.doubleColumn("Track Y Location");
106
107        Selection selX = isLastCol
108                ? x.isBetweenInclusive(left, right)
109                : x.isGreaterThanOrEqualTo(left).and(x.isLessThan(right));
110
111        Selection selY = isLastRow
112                ? y.isBetweenInclusive(top, bottom)
113                : y.isGreaterThanOrEqualTo(top).and(y.isLessThan(bottom));
114
115        return tracks.where(selX.and(selY));
116    }

122    /**
123     * Applies a visibility filter to all squares in a recording based on numeric
124     * thresholds and optional neighbour-based retention logic.
125     * <p>
126     * <b>Pass 1:</b> Marks each square selected if it meets:
127     * <ul>
128     *   <li>Density ratio ≥ minDensityRatio</li>
129     *   <li>Variability ≤ maxVariability</li>
130     *   <li>R² ≥ minRSquared and not NaN</li>
131     * </ul>
132     * <b>Pass 2:</b> If {@code neighbourMode != "Free"}, keeps only squares having
133     * at least one selected neighbour:
134     * <ul>
135     *   <li>“Relaxed” → corner or edge adjacency allowed (dr ≤ 1 & dc ≤ 1)</li>
136     *   <li>“Strict” → edge adjacency only (dr = 1 & dc = 0 or vice versa)</li>
137     * </ul>
138     *
139     * @param squares         the list of squares
140     * @param minDensityRatio minimum density ratio for selection
141     * @param maxVariability  maximum allowed variability
142     * @param minRSquared     minimum R² value for selection
143     * @param neighbourMode   neighbour logic: "Free", "Relaxed", or "Strict"
144     */
145    public static void applyVisibilityFilter(List<Square> squares,
146                                             double minDensityRatio,
147                                             double maxVariability,
148                                             double minRSquared,
149                                             String neighbourMode) {
150
151        if (squares == null || squares.isEmpty()) {
152            return;
153        }
154
155        int visibleBasic = 0;
156
157        // Pass 1 — Numeric filter
158        for (Square square : squares) {
159            boolean passes = square.getDensityRatio() >= minDensityRatio
160                    && square.getVariability() <= maxVariability
161                    && square.getRSquared() >= minRSquared
162                    && !Double.isNaN(square.getRSquared());
163
164            square.setSelected(passes);
165            if (passes) {
166                visibleBasic++;
167            }
168        }
169
170        // Pass 2 — Neighbour-based refinement
171        if ("Free".equalsIgnoreCase(neighbourMode)) {
172            return; // No neighbour constraints
173        }
174
175        Set<Square> keep     = new HashSet<>();
176        int keptCount        = 0;
177
178        for (Square square : squares) {
179            if (!square.isSelected()) {
180                continue;
181            }
182
183            boolean hasNeighbour = false;
184            int r = square.getRowNumber();
185            int c = square.getColNumber();
186
187            for (Square other : squares) {
188                if (other == square || !other.isSelected()) {
189                    continue;
190                }
191                int dr = Math.abs(other.getRowNumber() - r);
192                int dc = Math.abs(other.getColNumber() - c);
193
194                if ("Relaxed".equalsIgnoreCase(neighbourMode)) {
195                    // Corner or edge adjacency allowed
196                    if (dr <= 1 && dc <= 1 && (dr + dc) > 0) {
197                        hasNeighbour = true;
198                        break;
199                    }
200                } else if ("Strict".equalsIgnoreCase(neighbourMode)) {
201                    // Only direct-edge adjacency allowed
202                    if ((dr == 1 && dc == 0) || (dr == 0 && dc == 1)) {
203                        hasNeighbour = true;
204                        break;
205                    }
206                }
207            }
208
209            if (hasNeighbour) {
210                keep.add(square);
211                keptCount++;
212            }
213        }
214
215        // --- Apply neighbour filtering result ---
216        for (Square sq : squares) {
217            if (!keep.contains(sq)) {
218                sq.setSelected(false);
219            }
220        }
221
222        PaintLogger.debugf("NeighbourMode [%s] neighbour-filtered: %d / %d retained",
223                           neighbourMode, keptCount, visibleBasic);
224    }

226    /**
227     * Retrieves a list of tracks from all selected squares in the provided recording.
228     * If a square is selected and contains tracks, those tracks are added to the result.
229     *
230     * @param squares The squares to be checked.
231     * @return A list of tracks from all selected squares in the recording.
232     */
233    public static List<Track> getTracksFromSelectedSquares(List<Square> squares) {
234        List<Track> selectedTracks = new ArrayList<>();
235        for (Square square : squares) {
236            if (square.isSelected() && square.getTracks() != null) {
237                selectedTracks.addAll(square.getTracks());
238            }
239        }
240        return selectedTracks;
241    }
242
243    /**
244     * Counts the number of squares marked as selected in the provided recording.
245     *
246     * @param recording The recording object containing a collection of squares.
247     * @return The count of squares that are marked as selected.
248     */
249    public static int getNumberOfSelectedSquares(Recording recording) {
250        int count = 0;
251        for (Square square : recording.getSquaresOfRecording()) {
252            if (square.isSelected()) {
253                count++;
254            }
255        }
256        return count;
257    }

259    /**
260     * Calculates the density of tracks in a specified area over time and considering a concentration factor.
261     *
262     * The density is computed as the number of tracks divided by the area,
263     * then divided by the time, and finally divided by the concentration.
264     * If any of the area, time, or concentration values are non-positive, an IllegalArgumentException is thrown.
265     *
266     * @param nrTracks      The total number of tracks to calculate density for.
267     * @param area          The area over which the density is calculated. Must be positive.
268     * @param time          The time period over which the density is calculated. Must be positive.
269     * @param concentration The concentration factor to apply to the density calculation. Must be positive.
270     * @return The calculated density as a double.
271     * @throws IllegalArgumentException if area, time, or concentration is less than or equal to zero.
272     */
273    public static double calculateDensity(int nrTracks, double area, double time, double concentration) {
274        if (area <= 0 || time <= 0 || concentration <= 0) {
275            throw new IllegalArgumentException("Area, time, and concentration must be positive");
276        }
277
278        double density  = nrTracks / area;
279        density        /= time;
280        density        /= concentration;
281
282        return density;
283    }
