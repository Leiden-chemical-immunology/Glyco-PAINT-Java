079    /**
080     * Formats a duration given in seconds into a human-readable string such as
081     * "1 hour 2 minutes 5 seconds".
082     *
083     * @param totalSeconds duration in seconds
084     * @return formatted human-readable duration string
085     */
086    public static String formatDuration(int totalSeconds) {
087        int hours   = totalSeconds / 3600;
088        int minutes = (totalSeconds % 3600) / 60;
089        int secs    = totalSeconds % 60;
090
091        StringBuilder sb = new StringBuilder();
092        if (hours > 0) {
093            sb.append(hours).append(" hour").append(hours == 1 ? "" : "s").append(" ");
094        }
095        if (minutes > 0) {
096            sb.append(minutes).append(" minute").append(minutes == 1 ? "" : "s").append(" ");
097        }
098        if (secs > 0 || sb.length() == 0) {
099            sb.append(secs).append(" second").append(secs == 1 ? "" : "s");
100        }
101
102        return sb.toString().trim();
103    }
104
105    /**
106     * Converts a {@link Duration} object into a human-readable string.
107     *
108     * @param duration duration instance to format
109     * @return formatted string representation of the duration
110     */
111    public static String formatDuration(Duration duration) {
112        int totalSeconds = (int) duration.getSeconds();
113        return formatDuration(totalSeconds);
114    }

116    // ───────────────────────────────────────────────────────────────────────────────
117    // ERROR HANDLING UTILITIES
118    // ───────────────────────────────────────────────────────────────────────────────
119
120    /**
121     * Extracts a concise user-friendly message from a {@link Throwable}.
122     * If no colon is present in the exception message, returns the raw text.
123     *
124     * @param t exception or error object
125     * @return trimmed message string or an empty string if {@code t} is null
126     */
127    public static String friendlyMessage(Throwable t) {
128        if (t == null) {
129            return "";
130        }
131        String m     = t.toString();
132        int    colon = m.lastIndexOf(':');
133        return (colon != -1) ? m.substring(colon + 1).trim() : m;
134    }

140    /**
141     * Rounds a floating-point number to a specified number of decimal places.
142     *
143     * @param value    numeric value to round
144     * @param decimals number of decimal places
145     * @return rounded numeric value
146     */
147    public static double round(double value, int decimals) {
148        if (decimals < 0) {
149            return value;
150        } else {
151            double scale = Math.pow(10, decimals);
152            return Math.round(value * scale) / scale;
153        }
154    }

157    // BOOLEAN VALUE UTILITIES

160    /**
161     * Evaluates a string as a boolean. Supports variations such as
162     * {@code "true", "yes", "y", "1"} for {@code true} and
163     * {@code "false", "no", "n", "0"} for {@code false}.
164     * Invalid values log an error via {@link PaintLogger}.
165     *
166     * @param value input string
167     * @return boolean interpretation of the input
168     */
169    public static boolean isBooleanValue(String value) {
170        if (value.equals("true") || value.equals("1") ||
171                value.equals("yes") || value.equals("y") || value.equals("t")) {
172            return true;
173        } else if (value.equals("false") || value.equals("0") ||
174                value.equals("no") || value.equals("n") || value.equals("f")) {
175            return false;
176        } else {
177            PaintLogger.errorf("Invalid boolean value: %s", value);
178            return false;
179        }
180    }
181
182    /**
183     * Checks whether the provided string represents a valid boolean expression.
184     *
185     * @param value input string to check
186     * @return {@code true} if recognizable as boolean, {@code false} otherwise
187     */
188    public static Boolean checkBooleanValue(String value) {
189        value                 = value.trim().toLowerCase();
190        Set<String> yesValues = new HashSet<>(Arrays.asList("y", "ye", "yes", "ok", "true", "t", "1"));
191        Set<String> noValues  = new HashSet<>(Arrays.asList("n", "no", "false", "f", "0"));
192        return yesValues.contains(value) || noValues.contains(value);
193    }
194
195    /**
196     * Parses a flexible string-based boolean expression into a {@code Boolean}.
197     * <p>
198     * Logs an error if the input cannot be parsed and defaults to {@code false}.
199     *
200     * @param value input string
201     * @return parsed boolean value, defaults to {@code false} for invalid input
202     */
203    public static Boolean getBooleanValue(String value) {
204        value                 = value.trim().toLowerCase();
205        Set<String> yesValues = new HashSet<>(Arrays.asList("y", "ye", "yes", "ok", "true", "t", "1"));
206        Set<String> noValues  = new HashSet<>(Arrays.asList("n", "no", "false", "f", "0"));
207
208        if (yesValues.contains(value)) {
209            return true;
210        } else if (noValues.contains(value)) {
211            return false;
212        } else {
213            PaintLogger.errorf("Invalid boolean value: %s, default to 'false'", value);
214            return false;
215        }
216    }
217
218    /**
219     * Returns {@code true} if the input string is a recognized affirmative boolean value.
220     *
221     * @param value input string
222     * @return {@code true} if value is affirmative (e.g., "y", "yes", "true", "1")
223     */
224    public static Boolean isBooleanTrue(String value) {
225        value = value.trim().toLowerCase();
226        Set<String> yesValues = new HashSet<>(Arrays.asList("y", "ye", "yes", "ok", "true", "t", "1"));
227        return yesValues.contains(value);
228    }
