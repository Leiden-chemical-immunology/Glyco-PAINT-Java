005 *  PURPOSE:
006 *    Manages application-level preferences for the Glyco-PAINT application,
007 *    stored in a macOS-style Property List (plist) file. Provides thread-safe
008 *    access to get and set preference values (strings, booleans, integers),
009 *    automatically saving changes and allowing reload from disk.
010 *
011 *  DESCRIPTION:
012 *    • On class load, attempts to parse an existing plist file at
013 *      '~/Library/Preferences/Glyco-PAINT.plist', falling back to an empty
014 *      dictionary if missing or unreadable.
015 *    • Provides synchronized getters (getString, getBoolean, getInt) and setters
016 *      (putString, putBoolean, putInt) with default handling.
017 *    • Persists all modifications immediately via XML save.
018 *    • Supports manual reload from disk to pick up external edits.
019 *
020 *  RESPONSIBILITIES:
021 *    • Centralised preference storage for runtime and configuration settings.
022 *    • Ensures atomic and thread-safe preference access.
023 *    • Abstracts away plist file handling and persistence details.
024 *
025 *  USAGE EXAMPLE:
026 *    boolean verbose = PaintPrefs.getBoolean("Runtime", "Verbose", false);
027 *    PaintPrefs.putString("Project", "RootPath", "/Users/hans/project");
028 *    PaintPrefs.reload();
029 *

048package paint.shared.utils;
049
050import com.dd.plist.NSDictionary;
051import com.dd.plist.NSObject;
052import com.dd.plist.PropertyListParser;
053
054import java.io.File;
055
056/**
057 * Provides static, thread-safe access to application preferences for Glyco-PAINT.
058 * <p>
059 * Preferences are stored in {@code ~/Library/Preferences/Glyco-PAINT.plist} using
060 * Apple’s Property List (plist) format. The class handles reading, writing, and
061 * automatic persistence of key-value pairs for various data types.
062 * </p>
063 * <ul>
064 *   <li>Thread-safe access (synchronized methods)</li>
065 *   <li>Automatic saving of preferences after updates</li>
066 *   <li>Reload capability for external modifications</li>
067 *   <li>Fallback to default values if keys are missing</li>
068 * </ul>
069 */
070public final class PaintPrefs {
071
072    // ───────────────────────────────────────────────────────────────────────────────
073    // CONSTANTS AND STATE
074    // ───────────────────────────────────────────────────────────────────────────────
075
076    private static final File PREF_FILE = new File(
077            System.getProperty("user.home"),
078            "Library/Preferences/Glyco-PAINT.plist"
079    );
080
081    private static NSDictionary plist = new NSDictionary();
082
083    static {
084        try {
085            if (PREF_FILE.exists()) {
086                plist = (NSDictionary) PropertyListParser.parse(PREF_FILE);
087            }
088        } catch (Exception e) {
089            PaintLogger.errorf("Could not load plist: " + e.getMessage());
090            plist = new NSDictionary();
091        }
092    }
093
094    /**
095     * Private constructor to prevent instantiation.
096     */
097    private PaintPrefs() {
098        // Deliberately empty
099    }
100
101    // ───────────────────────────────────────────────────────────────────────────────
102    // GET METHODS
103    // ───────────────────────────────────────────────────────────────────────────────
104
105    /**
106     * Retrieves a boolean preference value or writes the given default if not present.
107     *
108     * @param section      section name (namespace)
109     * @param key          preference key
110     * @param defaultValue default to return and store if missing
111     * @return stored boolean or default
112     */
113    public static synchronized boolean getBoolean(String section,
114                                                  String key,
115                                                  boolean defaultValue) {
116        NSDictionary sectionDict = (NSDictionary) plist.objectForKey(section);
117        if (sectionDict == null) {
118            sectionDict = new NSDictionary();
119            plist.put(section, sectionDict);
120        }
121        NSObject obj = sectionDict.objectForKey(key);
122        if (obj == null) {
123            sectionDict.put(key, defaultValue);
124            save();
125            return defaultValue;
126        }
127        return Boolean.parseBoolean(obj.toString());
128    }
129
130    /**
131     * Retrieves a string preference value or writes the given default if missing.
132     *
133     * @param section      section name (namespace)
134     * @param key          preference key
135     * @param defaultValue default to return and store if missing
136     * @return stored string or default
137     */
138    public static synchronized String getString(String section,
139                                                String key,
140                                                String defaultValue) {
141        NSDictionary sectionDict = (NSDictionary) plist.objectForKey(section);
142        if (sectionDict == null) {
143            sectionDict = new NSDictionary();
144            plist.put(section, sectionDict);
145        }
146        NSObject obj = sectionDict.objectForKey(key);
147        if (obj == null) {
148            sectionDict.put(key, defaultValue);
149            save();
150            return defaultValue;
151        }
152        return obj.toString();
153    }
154
155    // ───────────────────────────────────────────────────────────────────────────────
156    // PUT METHODS (SECTIONED)
157    // ───────────────────────────────────────────────────────────────────────────────
158
159    /**
160     * Stores a boolean value within a named section and persists the change.
161     */
162    public static synchronized void putBoolean(String section,
163                                               String key,
164                                               boolean value) {
165        NSDictionary sectionDict = (NSDictionary) plist.objectForKey(section);
166        if (sectionDict == null) {
167            sectionDict = new NSDictionary();
168            plist.put(section, sectionDict);
169        }
170        sectionDict.put(key, value);
171        save();
172    }
173
174    /**
175     * Stores a string value within a named section and persists the change.
176     */
177    public static synchronized void putString(String section,
178                                              String key,
179                                              String value) {
180        NSDictionary sectionDict = (NSDictionary) plist.objectForKey(section);
181        if (sectionDict == null) {
182            sectionDict = new NSDictionary();
183            plist.put(section, sectionDict);
184        }
185        sectionDict.put(key, value);
186        save();
187    }
188
189    // ───────────────────────────────────────────────────────────────────────────────
190    // PUT METHODS (TOP-LEVEL)
191    // ───────────────────────────────────────────────────────────────────────────────
192
193    /**
194     * Stores a string at the top-level (no section) and persists it.
195     */
196    public static synchronized void putString(String key, String value) {
197        plist.put(key, value);
198        save();
199    }
200
201    /**
202     * Stores a boolean at the top-level (no section) and persists it.
203     */
204    public static synchronized void putBoolean(String key, boolean value) {
205        plist.put(key, value);
206        save();
207    }
208
209    /**
210     * Stores an integer at the top-level (no section) and persists it.
211     */
212    public static synchronized void putInt(String key, int value) {
213        plist.put(key, value);
214        save();
215    }
216
217    // ───────────────────────────────────────────────────────────────────────────────
218    // FILE MANAGEMENT
219    // ───────────────────────────────────────────────────────────────────────────────
220
221    /**
222     * Saves the current preferences to the plist file.
223     * Logs an error via {@link PaintLogger} if saving fails.
224     */
225    private static synchronized void save() {
226        try {
227            PropertyListParser.saveAsXML(plist, PREF_FILE);
228        } catch (Exception e) {
229            PaintLogger.errorf("Could not save plist: %s", e.getMessage());
230        }
231    }
232
233    /**
234     * Reloads preferences from disk, replacing the in-memory dictionary.
235     * If loading fails, resets to a new empty dictionary.
236     */
237    public static synchronized void reload() {
238        try {
239            if (PREF_FILE.exists()) {
240                plist = (NSDictionary) PropertyListParser.parse(PREF_FILE);
241            } else {
242                plist = new NSDictionary();
243            }
244        } catch (Exception e) {
245            PaintLogger.errorf("Could not reload plist: %s", e.getMessage());
246            plist = new NSDictionary();
247        }
248    }
249}
