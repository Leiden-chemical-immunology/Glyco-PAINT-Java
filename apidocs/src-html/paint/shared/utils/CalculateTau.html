003 *  Package:      generatesquares.calc
004 *
005 *  PURPOSE:
006 *    Computes the exponential decay time constant (Tau) and R² value from
007 *    track duration data using non-linear least squares fitting.
008 *
009 *  DESCRIPTION:
010 *    This class provides static methods for fitting mono-exponential decay
011 *    models to track lifetime distributions. It evaluates the quality of
012 *    the fit through the R² metric and classifies outcomes by fit status.
013 *    The class is non-instantiable and intended as a computational utility
014 *    used by higher-level analysis classes such as CalculateAttributes.
015 *
016 *  RESPONSIBILITIES:
017 *    • Build frequency distributions of track durations
018 *    • Perform exponential decay fitting using Levenberg–Marquardt optimization
019 *    • Estimate Tau and R² metrics
020 *    • Return structured results with fit status indicators
021 *
022 *  USAGE EXAMPLE:
023 *    CalculateTauResult result = CalculateTau.calcTau(tracks, 50, 0.95);
024 *    if (result.getStatus() == CalculateTauResult.Status.TAU_SUCCESS) {
025 *        double tau = result.getTau();
026 *        double r2  = result.getRSquared();
027 *    }
028 *
029 *  DEPENDENCIES:

066    /**
067     * Calculates the tau (time constant) and goodness-of-fit (R-squared) from a given set of track durations using an exponential decay model.
068     * The method performs a series of checks and validations, fits the data to a model,
069     * and evaluates the quality of the fit to decide the result status.
070     *
071     * @param tracks              the list of Track objects from which durations are extracted. Each Track represents a single time measurement.
072     *                            points status.
073     * @param minRequiredRSquared the minimum allowable value for the R-squared of the fit. If the computed R-squared is less than this threshold, the method will return a too-low
074     *                            R-squared status.
075     * @return a {@code CalculateTauResult} object containing the calculated tau, R-squared value, and the status of the calculation. Possible statuses include success, insufficient
076     * points, no fit, or R-squared too low.
077     */
078    public static CalculateTauResult calculateTau(List<Track> tracks,
079                                                  double minRequiredRSquared) {
080
081        // 1) Extract durations
082        final int n = tracks.size();
083        double[] durations = new double[n];
084        for (int i = 0; i < n; i++) {
085            durations[i] = tracks.get(i).getTrackDuration();
086        }
087
088        // 2) Build frequency distribution (sorted by duration)
089        Map<Double, Integer> freq = createFrequencyDistribution(durations);
090
091        // Need at least 2 distinct x-values to fit
092        if (freq.size() < 2) {
093            return new CalculateTauResult(0.0, 0.0, CalculateTauResult.Status.TAU_NO_FIT);
094        }
095
096        // 3) Convert to arrays for fitting
097        double[] x = new double[freq.size()];
098        double[] y = new double[freq.size()];
099        int k = 0;
100        for (Map.Entry<Double, Integer> e : freq.entrySet()) {
101            x[k] = e.getKey();
102            y[k] = e.getValue();
103            k++;
104        }
105
106        // 4) Fit and evaluate quality
107        CalculateTauExpDecayFitter.FitResult fitResult = CalculateTauExpDecayFitter.fit(x, y);
108
109        // 5) Reject non-finite results
110        if (!Double.isFinite(fitResult.rSquared) || !Double.isFinite(fitResult.tauMs)) {
111            return new CalculateTauResult(fitResult.tauMs, fitResult.rSquared, CalculateTauResult.Status.TAU_NO_FIT);
112        }
113
114        // 6) Apply threshold
115        if (fitResult.rSquared < minRequiredRSquared) {
116            return new CalculateTauResult(fitResult.tauMs, fitResult.rSquared, CalculateTauResult.Status.TAU_RSQUARED_TOO_LOW);
117        }
118
119        return new CalculateTauResult(fitResult.tauMs, fitResult.rSquared, CalculateTauResult.Status.TAU_SUCCESS);
120    }

122    /**
123     * Represents the result of the tau calculation process, containing the
124     * tau value, R-squared value, and the status of the calculation.
125     * <p>
126     * This result is produced after performing an exponential decay model
127     * fitting on given track durations, evaluating the quality of the fit,
128     * and determining if the computation meets the required conditions.
129     */
130    public static class CalculateTauResult {
131
132        private final double tau;
133        private final double rsquared;
134        private final Status status;
135
136        public CalculateTauResult(double tau, double rsquared, Status status) {
137            this.tau      = tau;
138            this.rsquared = rsquared;
139            this.status   = status;
140        }
141
142        public enum Status {
143            TAU_SUCCESS,
144            TAU_INSUFFICIENT_POINTS,
145            TAU_RSQUARED_TOO_LOW,
146            TAU_NO_FIT
147        }
148
149        public double getTau() {
150            return tau;
151        }
152
153        public double getRSquared() {
154            return rsquared;
155        }
156
157        public Status getStatus() {
158            return status;
159        }

162    /**
163     * Creates a frequency distribution from an array of track durations.
164     * The method calculates the number of occurrences for each unique duration
165     * and stores the result in a map where the keys are unique durations and
166     * the values are their respective frequencies.
167     *
168     * @param trackDurations an array of double values representing track durations.
169     *                       If the array is null, an empty map is returned.
170     * @return a map containing unique track durations as keys and their
171     * corresponding frequencies as values.
172     */
173    private static Map<Double, Integer> createFrequencyDistribution(double[] trackDurations) {
174        Map<Double, Integer> frequencyDistribution = new TreeMap<>();
175        if (trackDurations == null) {
176            return frequencyDistribution;
177        }
178
179        for (double duration : trackDurations) {
180            Integer prev = frequencyDistribution.get(duration);
181            frequencyDistribution.put(duration, (prev == null ? 1 : prev + 1));
182        }
183        return frequencyDistribution;
184    }

186    /**
187     * A utility class performing exponential decay fitting. This class enables
188     * the computation of an exponential decay's time constant (tau) and
189     * the goodness-of-fit (R-squared) through least-squares optimization.
190     */
191    private static class CalculateTauExpDecayFitter {
192
193        private CalculateTauExpDecayFitter() {
194        }
195
196        /**
197         * Immutable fit result.
198         */
199        private static final class FitResult {
200            final double tauMs;
201            final double rSquared;
202
203            FitResult(double tauMs, double rSquared) {
204                this.tauMs    = tauMs;
205                this.rSquared = rSquared;
206            }
207        }
208
209        /**
210         * Fits a model to the given data points using a non-linear least squares optimization
211         * and returns the results as a {@code FitResult}.
212         * The model assumes an exponential decay function of the form:
213         * y = m * exp(-t * x) + b.
214         *
215         * @param x the independent variable data points (input values)
216         * @param y the dependent variable data points (output values)
217         *          Both arrays must have the same length and contain at least two elements.
218         * @return a {@code FitResult} containing the calculated time constant in milliseconds (tauMs)
219         *         and the coefficient of determination (rSquared).
220         *         Returns a {@code FitResult} with NaN values if the input is invalid
221         *         or the fitting process fails.
222         */
223        private static FitResult fit(double[] x, double[] y) {
224            if (x == null || y == null || x.length != y.length || x.length < 2) {
225                return new FitResult(NaN, NaN);
226            }
227
228            final MultivariateJacobianFunction model = p -> {
229                double m = p.getEntry(0);
230                double t = p.getEntry(1);
231                double b = p.getEntry(2);
232
233                double[] values = new double[x.length];
234                double[][] jac = new double[x.length][3];
235
236                for (int i = 0; i < x.length; i++) {
237                    double e = FastMath.exp(-t * x[i]);
238                    values[i] = m * e + b;
239                    jac[i][0] = e;              // d/dm
240                    jac[i][1] = -m * x[i] * e;  // d/dt
241                    jac[i][2] = 1.0;            // d/db
242                }
243                return new org.apache.commons.math3.util.Pair<>(
244                        new ArrayRealVector(values, false),
245                        new Array2DRowRealMatrix(jac, false)
246                );
247            };
248
249            double[] p0 = initialGuess(x, y);
250            double[] w = new double[y.length];
251            Arrays.fill(w, 1.0);
252
253            LeastSquaresProblem problem = new LeastSquaresBuilder()
254                    .start(p0)
255                    .model(model)
256                    .target(y)
257                    .weight(new DiagonalMatrix(w))
258                    .maxEvaluations(10_000)
259                    .maxIterations(1_000)
260                    .build();
261
262            try {
263                LeastSquaresOptimizer.Optimum opt =
264                        new LevenbergMarquardtOptimizer().optimize(problem);
265
266                double[] p = opt.getPoint().toArray();
267                double m = p[0], t = p[1], b = p[2];
268                double tauMs = (t > 0.0) ? (1000.0 / t) : NaN;
269                double r2 = computeRSquared(x, y, m, t, b);
270
271                return new FitResult(tauMs, r2);
272            } catch (Throwable ignored) {
273                return new FitResult(NaN, NaN);
274            }
275        }

277        /**
278         * Generates an initial guess for the parameters of an exponential decay model
279         * based on the provided input data.
280         * This method calculates preliminary values for the model parameters
281         * which can be used as a starting point for further optimization or fitting.
282         *
283         * @param x the independent variable data points (input values).
284         *          The array must have at least two elements.
285         * @param y the dependent variable data points (output values).
286         *          The array must have the same length as the {@code x} array.
287         * @return a double array containing the initial guesses for the model parameters:
288         *         - {@code m}: the scaling factor
289         *         - {@code t}: the time constant
290         *         - {@code b}: the baseline offset
291         */
292        private static double[] initialGuess(double[] x, double[] y) {
293            double minY = Arrays.stream(y).min().orElse(0);
294            double maxY = Arrays.stream(y).max().orElse(1);
295            double maxX = Arrays.stream(x).max().orElse(1);
296
297            double b = Math.max(0.0, minY);
298            double m = Math.max(1e-6, maxY - b);
299
300            double eps  = Math.max(1e-6, 0.01 * m);
301            double sumX = 0, sumXX = 0, sumY = 0, sumXY = 0;
302            int    n    = 0;
303
304            for (int i = 0; i < x.length; i++) {
305                double yiAdj = y[i] - b;
306                if (yiAdj > eps) {
307                    double lx = x[i];
308                    double ly = FastMath.log(yiAdj);
309                    sumX += lx;
310                    sumXX += lx * lx;
311                    sumY += ly;
312                    sumXY += lx * ly;
313                    n++;
314                }
315            }
316
317            double t;
318            if (n >= 2) {
319                double denominator = (n * sumXX - sumX * sumX);
320                double slope = (denominator == 0.0)
321                        ? -1.0
322                        : (n * sumXY - sumX * sumY) / denominator;
323                t = Math.max(1e-9, -slope);
324            } else {
325                t = 1.0 / Math.max(1e-3, maxX);
326            }
327
328            // Clamp to reasonable ranges
329            m = clamp(m, 1e-9, 1e9);
330            t = clamp(t, 1e-9, 1e3);
331            b = clamp(b, 0.0, Math.max(1.0, maxY));
332
333            return new double[]{m, t, b};
334        }

336        /**
337         * Computes the coefficient of determination (R-squared) for the given data points
338         * and model parameters. The R-squared value indicates how well the model fits
339         * the data, with a value ranging from 0 (no fit) to 1 (perfect fit).
340         *
341         * @param x the independent variable data points (input values).
342         *          Must be an array of doubles with length matching the {@code y} array.
343         * @param y the dependent variable data points (output values).
344         *          Must be an array of doubles with length matching the {@code x} array.
345         * @param m the scaling factor (model parameter).
346         * @param t the time constant (model parameter).
347         * @param b the baseline offset (model parameter).
348         * @return the R-squared value, representing the goodness of fit of the model to the data.
349         *         Returns NaN if the total sum of squares is zero.
350         */
351        private static double computeRSquared(double[] x, double[] y, double m, double t, double b) {
352            double meanY  = Arrays.stream(y).average().orElse(0);
353            double ssRes  = 0.0;
354            double  ssTot = 0.0;
355            for (int i = 0; i < y.length; i++) {
356                double predicted = m * FastMath.exp(-t * x[i]) + b;
357                ssRes += Math.pow(y[i] - predicted, 2);
358                ssTot += Math.pow(y[i] - meanY, 2);
359            }
360            return (ssTot == 0.0) ? NaN : 1.0 - (ssRes / ssTot);
361        }
362
363        private static double clamp(double v, double lo, double hi) {
364            return Math.max(lo, Math.min(hi, v));
365        }
366    }

372     * @param x the independent variable data points
373     * @param y the dependent variable data points
374     * @return an array of double values, where the first element is the time constant (tau) and the second

379        return new double[]{result.tauMs, result.rSquared};
