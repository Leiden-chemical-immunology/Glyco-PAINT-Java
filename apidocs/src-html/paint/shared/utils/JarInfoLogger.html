041import java.net.URL;
042import java.security.CodeSource;
043import java.time.*;
044import java.time.format.DateTimeFormatter;
045import java.time.format.DateTimeParseException;

052/**
053 * Static utility for reading manifest metadata from JAR archives.
054 * <p>
055 * The {@code JarInfoLogger} reads manifest files, extracts implementation
056 * details (title, version, vendor, build date), and formats time data into
057 * Amsterdam time for uniform reporting. It can operate on any class
058 * contained in a JAR file.
059 */
060public final class JarInfoLogger {
061
062    // ───────────────────────────────────────────────────────────────────────────────
063    // CONSTANTS
064    // ───────────────────────────────────────────────────────────────────────────────
065
066    private static final ZoneId AMS = ZoneId.of("Europe/Amsterdam");
067    private static final DateTimeFormatter OUT_FMT =
068            DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss z").withZone(AMS);
069
070    /**
071     * Prevents instantiation of this static utility class.
072     */
073    private JarInfoLogger() {
074        // Prevent instantiation
075    }
076
077    // ───────────────────────────────────────────────────────────────────────────────
078    // PUBLIC API
079    // ───────────────────────────────────────────────────────────────────────────────
080
081    /**
082     * Reads JAR manifest information for the given class.
083     * <p>
084     * The method first attempts to locate the manifest in the JAR containing
085     * the provided class. If not found, it scans the classpath for the first
086     * available {@code META-INF/MANIFEST.MF}.
087     *
088     * @param clazz the class from which to derive the JAR source
089     * @return a {@link JarInfo} object containing manifest metadata, or {@code null} if unavailable
090     */
091    public static JarInfo getJarInfo(Class<?> clazz) {
092        try {
093            // Try to load from the JAR that defined this class
094            Manifest manifest = manifestFromCodeSource(clazz);
095            if (manifest == null) {
096                // Fallback: first MANIFEST.MF on classpath
097                manifest = manifestFromClasspath(clazz);
098            }
099            if (manifest == null) {
100                return null;
101            }
102
103            Attributes attribute   = manifest.getMainAttributes();
104            String     ts          = attribute.getValue("Build-Timestamp"); // optional
105            String     formattedTs = formatAmsterdam(ts);
106
107            return new JarInfo(
108                    attribute.getValue("Implementation-Title"),
109                    attribute.getValue("Implementation-Version"),
110                    attribute.getValue("Implementation-Vendor"),
111                    formattedTs,
112                    attribute.getValue("Specification-Title"),
113                    attribute.getValue("Specification-Version"),
114                    attribute.getValue("Specification-Vendor")
115            );
116        } catch (Exception e) {
117            PaintLogger.errorf("JarInfoLogger.getJarInfo(): %s", e);
118            return null;
119        }
120    }

126    /**
127     * Attempts to read a manifest directly from the JAR file that defined
128     * the specified class.
129     *
130     * @param clazz class whose code source will be used to locate the JAR
131     * @return the parsed {@link Manifest}, or {@code null} if unavailable
132     */
133    private static Manifest manifestFromCodeSource(Class<?> clazz) {
134        try {
135            CodeSource cs = clazz.getProtectionDomain().getCodeSource();
136            if (cs == null) {
137                return null;
138            }
139
140            URL loc     = cs.getLocation();
141            String path = loc.toString();
142
143            if (path.endsWith(".jar") || path.contains(".jar!")) {
144                try (JarFile jf = new JarFile(loc.getPath().replace("file:", ""))) {
145                    return jf.getManifest();
146                } catch (Exception ignored) {
147                    String jarUrl = (path.endsWith(".jar")
148                            ? "jar:" + path + "!/META-INF/MANIFEST.MF"
149                            : path.substring(0, path.indexOf("!")) + "!/META-INF/MANIFEST.MF");
150                    try (InputStream is = new URL(jarUrl).openStream()) {
151                        return new Manifest(is);
152                    }
153                }
154            }
155            return null;
156        } catch (Exception e) {
157            return null;
158        }
159    }
160
161    /**
162     * Reads the first available {@code META-INF/MANIFEST.MF} from the classpath.
163     *
164     * @param clazz class whose {@link ClassLoader} will be used for resource lookup
165     * @return the first {@link Manifest} found, or {@code null} if none exist
166     */
167    private static Manifest manifestFromClasspath(Class<?> clazz) {
168        try {
169            ClassLoader cl = clazz.getClassLoader();
170            Enumeration<URL> en = cl.getResources("META-INF/MANIFEST.MF");
171            while (en.hasMoreElements()) {
172                URL u = en.nextElement();
173                try (InputStream is = u.openStream()) {
174                    return new Manifest(is);
175                }
176            }
177        } catch (Exception ignored) {
178        }
179        return null;
180    }

186    /**
187     * Converts a UTC timestamp string to Amsterdam local time, formatted as
188     * {@code yyyy-MM-dd HH:mm:ss z}.
189     * <p>
190     * Supported input formats:
191     * <ul>
192     *   <li>ISO 8601 (e.g. {@code 2025-10-14T06:39:16Z})</li>
193     *   <li>{@code yyyy-MM-dd HH:mm:ss} (assumed UTC)</li>
194     *   <li>{@code yyyy-MM-dd'T'HH:mm:ssXXX}</li>
195     * </ul>
196     *
197     * @param utcString raw timestamp string from the manifest
198     * @return formatted string (e.g., {@code "2025-10-14 08:39:16 CET"}), or {@code null} if invalid
199     */
200    public static String formatAmsterdam(String utcString) {
201        if (utcString == null || utcString.trim().isEmpty()) {
202            return null;
203        }
204
205        // Try ISO 8601 (e.g. 2025-10-14T06:39:16Z)
206        try {
207            Instant ins = Instant.parse(utcString.trim());
208            return OUT_FMT.format(ins);
209        } catch (DateTimeParseException ignored) {
210        }
211
212        // Try "yyyy-MM-dd HH:mm:ss" (assume UTC)
213        try {
214            LocalDateTime ldt = LocalDateTime.parse(utcString.trim(),
215                                                    DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss"));
216            return OUT_FMT.format(ldt.toInstant(ZoneOffset.UTC));
217        } catch (DateTimeParseException ignored) {
218        }
219
220        // Try "yyyy-MM-dd'T'HH:mm:ssXXX"
221        try {
222            OffsetDateTime odt = OffsetDateTime.parse(utcString.trim(),
223                                                      DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ssXXX"));
224            return OUT_FMT.format(odt.toInstant());
225        } catch (DateTimeParseException ignored) {
226        }
227
228        return null;
229    }

235    /**
236     * Prints manifest information for this utility’s own class.
237     * <p>
238     * Intended for manual or command-line testing.
239     *
240     * @param args command-line arguments (unused)
241     */
242    public static void main(String[] args) {
243        JarInfo info = getJarInfo(JarInfoLogger.class);
244        System.out.println(Objects.toString(info, "No manifest information found."));
245    }

251    /**
252     * Immutable container class for JAR manifest metadata.
253     */
254    public static final class JarInfo {
255
256        public final String implementationTitle;
257        public final String implementationVersion;
258        public final String implementationVendor;
259        public final String implementationDate;
260
261        public final String specificationTitle;
262        public final String specificationVersion;
263        public final String specificationVendor;
264
265        /**
266         * Constructs a new {@code JarInfo} container with all metadata values.
267         *
268         * @param implTitle    implementation title
269         * @param implVersion  implementation version
270         * @param implVendor   implementation vendor
271         * @param implDate     implementation build date/time
272         * @param specTitle    specification title
273         * @param specVersion  specification version
274         * @param specVendor   specification vendor
275         */
276        public JarInfo(String implTitle,
277                       String implVersion,
278                       String implVendor,
279                       String implDate,
280                       String specTitle,
281                       String specVersion,
282                       String specVendor) {
283            this.implementationTitle   = implTitle;
284            this.implementationVersion = implVersion;
285            this.implementationVendor  = implVendor;
286            this.implementationDate    = implDate;
287
288            this.specificationTitle    = specTitle;
289            this.specificationVersion  = specVersion;
290            this.specificationVendor   = specVendor;
291        }
292
293        /**
294         * @return a concise human-readable summary of implementation metadata.
295         */
296        @Override
297        public String toString() {
298            return String.format(
299                    "Implementation Title  : %s%n" +
300                            "Implementation Version: %s%n" +
301                            "Implementation Vendor : %s%n" +
302                            "Implementation Date   : %s%n",
303                    implementationTitle, implementationVersion,
304                    implementationVendor, implementationDate
305            );
306        }
307
308        /**
309         * @return a detailed report including specification metadata.
310         */
311        public String toStringLONG() {
312            return String.format(
313                    "Implementation Title  : %s%n" +
314                            "Implementation Version: %s%n" +
315                            "Implementation Vendor : %s%n" +
316                            "Implementation Date   : %s%n" +
317                            "Specification Title   : %s%n" +
318                            "Specification Version : %s%n" +
319                            "Specification Vendor  : %s",
320                    implementationTitle, implementationVersion, implementationVendor,
321                    implementationDate, specificationTitle, specificationVersion, specificationVendor
322            );
323        }
324    }
