051package paint.shared.utils;
052
053import javax.swing.*;
054import java.awt.*;
055import java.nio.file.Files;
056import java.nio.file.Path;
057import java.nio.file.Paths;
058
059import static paint.shared.constants.PaintConstants.PAINT_CONFIGURATION_JSON;
060
061/**
062 * Validates and retrieves a usable project root folder path for the PAINT application.
063 * <p>
064 * If the stored path is invalid or missing the configuration file, the user is prompted
065 * with a warning dialog and a native macOS folder chooser to select a valid directory.
066 * The chosen path is stored persistently via {@link PaintPrefs}.
067 * </p>
068 */
069public final class ValidProjectPath {
070
071    /**
072     * Private constructor to prevent instantiation.
073     */
074    private ValidProjectPath() {
075    }
076
077    // ───────────────────────────────────────────────────────────────────────────────
078    // MAIN VALIDATION LOGIC
079    // ───────────────────────────────────────────────────────────────────────────────
080
081    /**
082     * Retrieves and validates the configured project root directory.
083     * <p>
084     * Steps performed:
085     * <ol>
086     *   <li>Reads the stored project path from {@link PaintPrefs}.</li>
087     *   <li>Checks that the directory exists and contains the required
088     *       {@code PAINT_CONFIGURATION_JSON} file.</li>
089     *   <li>If invalid, prompts the user with warnings and a native folder chooser.</li>
090     *   <li>Updates preferences with the newly selected valid folder.</li>
091     * </ol>
092     * <p>
093     * Returns {@code null} if the user cancels folder selection.
094     * </p>
095     *
096     * @return a valid {@link Path} to the project root directory, or {@code null} if the user cancelled
097     */
098    public static Path getValidProjectPath() {
099
100        boolean needToAsk = false;
101        Path projectPath = Paths.get(PaintPrefs.getString("Path", "Project Root",
102                                                          System.getProperty("user.home")));
103
104        // Validate existence of the configured path
105        if (!Files.isDirectory(projectPath)) {
106            JOptionPane.showMessageDialog(null,
107                                          "The configured project path is invalid:\n" + projectPath +
108                                                  "\n\nPlease select a valid project folder.",
109                                          "Invalid Project Path",
110                                          JOptionPane.WARNING_MESSAGE);
111            projectPath = Paths.get(System.getProperty("user.home"));
112            needToAsk   = true;
113        }
114
115        // Validate configuration file presence
116        Path confPath = projectPath.resolve(PAINT_CONFIGURATION_JSON);
117        if (!Files.isRegularFile(confPath)) {
118            JOptionPane.showMessageDialog(null,
119                                          "The project folder does not contain:\n" + confPath +
120                                                  "\n\nPlease select a valid project folder.",
121                                          "Invalid Project Path",
122                                          JOptionPane.WARNING_MESSAGE);
123            projectPath = Paths.get(System.getProperty("user.home"));
124            needToAsk   = true;
125        }
126
127        // Ask user to select a valid folder if required
128        if (needToAsk) {
129            FileDialog chooser = new FileDialog((Frame) null, "Select Project Folder", FileDialog.LOAD);
130            System.setProperty("apple.awt.fileDialogForDirectories", "true");
131            chooser.setVisible(true);
132            System.clearProperty("apple.awt.fileDialogForDirectories");
133
134            String dir  = chooser.getDirectory();
135            String file = chooser.getFile();
136
137            if (dir != null && file != null) {
138                projectPath = Paths.get(dir, file);
139                PaintPrefs.putString("Path", "Project Root", projectPath.toString());
140            } else {
141                JOptionPane.showMessageDialog(null,
142                                              "No project folder selected.\nExiting.",
143                                              "Operation Cancelled",
144                                              JOptionPane.ERROR_MESSAGE);
145                return null; // Abort startup
146            }
147        }
148        return projectPath;
149    }
150}
