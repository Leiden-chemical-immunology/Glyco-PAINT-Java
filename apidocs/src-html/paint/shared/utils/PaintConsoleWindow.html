019 *    • Thread-safe message logging and dynamic color highlighting.
020 *    • Scroll lock and caret control for controlled auto-scrolling.
021 *    • Interactive buttons for saving, closing, and navigating problem messages.
022 *    • Pattern-based highlighting of "Error", "Warning", and "Exception".

125    /**
126     * Closes and disposes of the console window, clearing all references and data.
127     */
128    public static synchronized void close() {
129        if (frame != null) {
130            SwingUtilities.invokeLater(() -> {
131                frame.dispose();
132                frame = null;
133                textPane = null;
134                doc = null;
135                scrollLock = null;
136                problemPositions.clear();
137                currentProblemIndex = -1;
138            });
139        }
140    }
141
142    /**
143     * Attaches automatic console closure when a given dialog is disposed.
144     */
145    public static void closeOnDialogDispose(JDialog dialog) {
146        dialog.addWindowListener(new java.awt.event.WindowAdapter() {
147            @Override
148            public void windowClosed(java.awt.event.WindowEvent e) {
149                close();
150            }
151        });
152    }

155     * Closes the console window if currently visible.
156     */
157    public static void closeIfVisible() {

160                frame.setVisible(false);
161                frame.dispose();
162                frame = null;

200    private static void ensureConsoleCreated() {
201        if (frame == null) {
202            createConsole();
203        }
204    }

206    private static void createConsole() {
207        createConsole("Paint Console");
208    }

210    /**
211     * Creates and displays the console window with default components and layout.
212     */
213    private static void createConsole(String title) {
214        frame = new JFrame(title != null ? title : "Paint Console");
215        frame.setSize(1200, 400);
216        frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
217        frame.setLayout(new BorderLayout());
218
219        textPane = new JTextPane();
220        textPane.setEditable(false);
221        textPane.setFont(new Font(Font.MONOSPACED, Font.PLAIN, 12));
222        doc = textPane.getStyledDocument();
223
224        DefaultCaret caret = (DefaultCaret) textPane.getCaret();
225        caret.setUpdatePolicy(DefaultCaret.NEVER_UPDATE);
226
227        JScrollPane scrollPane = new JScrollPane(textPane);
228
229        JPanel controlPanel = new JPanel(new BorderLayout());
230        scrollLock = new JCheckBox("Scroll Lock");
231        scrollLock.addActionListener(e -> updateTitleForScrollLock(scrollLock.isSelected()));
232        controlPanel.add(scrollLock, BorderLayout.WEST);
233
234        JPanel buttonPanel      = new JPanel(new FlowLayout(FlowLayout.RIGHT));
235        JButton highlightButton = new JButton("Highlight Problems");
236        highlightButton.addActionListener(e -> highlightProblemsOrNext());
237        JButton saveButton  = new JButton("Save");
238        JButton closeButton = new JButton("Close");
239        saveButton.addActionListener(PaintConsoleWindow::saveConsoleContent);
240        closeButton.addActionListener(e -> close());
241
242        buttonPanel.add(highlightButton);
243        buttonPanel.add(saveButton);
244        buttonPanel.add(closeButton);
245        controlPanel.add(buttonPanel, BorderLayout.EAST);
246
247        frame.add(scrollPane, BorderLayout.CENTER);
248        frame.add(controlPanel, BorderLayout.SOUTH);
249        frame.setLocationRelativeTo(null);
250        frame.setVisible(true);
251    }

257    /**
258     * Appends colored text to the console.
259     */
260    private static void appendText(String text, Color color) {
261        try {
262            Style style = textPane.addStyle("Style", null);
263            StyleConstants.setForeground(style, color);
264            doc.insertString(doc.getLength(), text, style);
265
266            if (!scrollLock.isSelected()) {
267                textPane.setCaretPosition(doc.getLength());
268            }
269        } catch (BadLocationException e) {
270            e.printStackTrace();
271        }
272    }
273
274    /**
275     * Saves console output to a user-selected file.
276     */
277    private static void saveConsoleContent(ActionEvent e) {
278        JFileChooser chooser = new JFileChooser();
279        chooser.setDialogTitle("Save Console Output");
280        int choice = chooser.showSaveDialog(frame);
281        if (choice == JFileChooser.APPROVE_OPTION) {
282            File file = chooser.getSelectedFile();
283            try (BufferedWriter writer = new BufferedWriter(new FileWriter(file))) {
284                writer.write(textPane.getText());
285                JOptionPane.showMessageDialog(frame, "Console saved to " + file.getAbsolutePath());
286            } catch (IOException ex) {
287                JOptionPane.showMessageDialog(frame,
288                                              "Failed to save file:\n" + ex.getMessage(),
289                                              "Error", JOptionPane.ERROR_MESSAGE);
290            }
291        }
292    }

298    /**
299     * Highlights detected problems in the console output or navigates to the next highlighted problem.
300     * <p>
301     * This method performs two primary actions:
302     * 1. If no problems have been highlighted yet, it identifies problems in the text output
303     * (e.g., errors, warnings, exceptions) and highlights them.
304     * 2. If problems are already highlighted, it navigates to the next problem in the list,
305     * looping back to the first problem when the end of the list is reached.
306     * <p>
307     * Behavior:
308     * - If the scroll lock button is present and not already selected, it is activated to
309     * ensure that the highlighting and navigation operate properly.
310     * - Problems are identified using a predefined pattern (e.g., "Error", "Warning", "Exception").
311     * - If no problems are found during the initial search, a dialog is displayed informing
312     * the user that no issues were detected.
313     * - Navigation across detected problems is cyclical, wrapping around to the first
314     * problem after reaching the last one.
315     * <p>
316     * Thread Safety:
317     * This method is not explicitly synchronized and may require synchronization if called
318     * in a multithreaded environment to ensure that shared state (such as problemPositions
319     * and currentProblemIndex) is safely updated.
320     * <p>
321     * Effects:
322     * - Updates the scroll lock state and the console title if necessary.
323     * - Highlights newly identified problem areas in the console output.
324     * - Modifies the `currentProblemIndex` to track the currently selected problem.
325     */
326    private static void highlightProblemsOrNext() {
327        if (scrollLock != null && !scrollLock.isSelected()) {
328            scrollLock.setSelected(true);
329            updateTitleForScrollLock(true);
330        }
331
332        if (problemPositions.isEmpty()) {
333            // First click: find and highlight
334            highlightProblems();
335            if (!problemPositions.isEmpty()) {
336                currentProblemIndex = 0;
337                selectProblem(problemPositions.get(currentProblemIndex));
338            } else {
339                JOptionPane.showMessageDialog(frame, "No problems found (error, warning, exception).");
340            }
341        } else {
342            // Subsequent clicks: move to next match
343            currentProblemIndex++;
344            if (currentProblemIndex >= problemPositions.size()) {
345                currentProblemIndex = 0; // wrap around
346            }
347            selectProblem(problemPositions.get(currentProblemIndex));
348        }
349    }

351    /**
352     * Highlights problems in the console output based on predefined patterns.
353     * <p>
354     * This method scans the console's text content and highlights specific terms
355     * that indicate potential problems, such as "error", "warning", and "exception".
356     * The method uses regular expressions to identify these terms, including
357     * variations in capitalization and tolerance for ANSI color codes.
358     * <p>
359     * The detected terms are highlighted with different colors:
360     * - "Error" and "Exception" are highlighted in pink.
361     * - "Warning" and "Warn" are highlighted in orange.
362     * <p>
363     * Detected problem positions are stored in a list for further navigation or processing.
364     * <p>
365     * Functionality:
366     * - Removes all existing highlights before applying new ones.
367     * - Clears the previously recorded problem positions and resets the current
368     * problem index to indicate no problem is currently selected.
369     * - Performs a case-insensitive search, tolerating leading/trailing spaces
370     * and ANSI formatting codes.
371     * - Marks detected terms with appropriate highlight colors.
372     * - Adds the start position of each detected problem to the list for future reference.
373     * <p>
374     * Design notes:
375     * - The method is private and applies highlights within the text pane associated
376     * with the console.
377     * - If an exception occurs while applying highlights (due to invalid positions),
378     * it is logged but does not interrupt execution.
379     * <p>
380     * Thread Safety:
381     * This method is not thread-safe and should be externally synchronized if
382     * accessed from multiple threads, as shared structures like problemPositions
383     * and currentProblemIndex are modified.
384     */
385    private static void highlightProblems() {
386        Highlighter highlighter = textPane.getHighlighter();
387        highlighter.removeAllHighlights();
388        problemPositions.clear();
389        currentProblemIndex = -1;
390
391        // Get full text as-is
392        String fullText = textPane.getText();
393
394        // Unified regex to catch most variants: ERROR, Error, WARN, Warning, etc.
395        // Also tolerates ANSI color codes (\u001B[...]m) and leading/trailing spaces.
396        Pattern pattern = Pattern.compile(
397                "(?i)(?:\\u001B\\[[;\\d]*m)?\\b(?:error|warn|warning|exception)\\b(?:\\u001B\\[[;\\d]*m)?"
398        );
399
400        Matcher matcher = pattern.matcher(fullText);
401        while (matcher.find()) {
402            try {
403                String match = matcher.group().toLowerCase();
404                Color color = match.contains("error") || match.contains("exception")
405                        ? Color.PINK : Color.ORANGE;
406
407                highlighter.addHighlight(
408                        matcher.start(),
409                        matcher.end(),
410                        new DefaultHighlighter.DefaultHighlightPainter(color)
411                );
412                problemPositions.add(matcher.start());
413            } catch (BadLocationException e) {
414                e.printStackTrace();
415            }
416        }
417    }

419    /**
420     * Highlights all occurrences of a given regex pattern in the provided text with the specified color.
421     *
422     * @param fullText the text in which to search for the pattern
423     * @param regex the regular expression pattern to search for
424     * @param color the color used for highlighting the matched pattern
425     */
426    private static void highlightPattern(String fullText, String regex, Color color) {
427        java.util.regex.Pattern pattern = java.util.regex.Pattern.compile(regex);
428        java.util.regex.Matcher matcher = pattern.matcher(fullText);
429
430        while (matcher.find()) {
431            int start = matcher.start();
432            int end = matcher.end();
433            try {
434                textPane.getHighlighter().addHighlight(
435                        start, end,
436                        new DefaultHighlighter.DefaultHighlightPainter(color)
437                );
438                problemPositions.add(start);
439            } catch (BadLocationException e) {
440                e.printStackTrace();
441                break;
442            }
443        }
444    }

446    /**
447     * Highlights and selects a specific text region in a text pane based on the given position.
448     * This method applies a yellow highlight to the specified region and sets the text caret within the region,
449     * making the region both focused and visible. Previous focus highlights are cleared before applying the new one.
450     *
451     * @param pos the starting position of the text region to be highlighted and selected
452     */
453    private static void selectProblem(int pos) {
454        try {
455            Highlighter highlighter = textPane.getHighlighter();
456            Highlighter.HighlightPainter focusPainter = new DefaultHighlighter.DefaultHighlightPainter(Color.YELLOW);
457
458            for (Highlighter.Highlight h : highlighter.getHighlights()) {
459                if (h.getStartOffset() == pos) {
460                    textPane.requestFocus();
461                    textPane.setCaretPosition(h.getEndOffset());
462                    textPane.select(h.getStartOffset(), h.getEndOffset());
463                    highlighter.addHighlight(h.getStartOffset(), h.getEndOffset(), focusPainter);
464                    textPane.scrollRectToVisible(textPane.modelToView(h.getStartOffset()));
465                    return;
466                }
467            }
468        } catch (BadLocationException e) {
469            e.printStackTrace();
470        }
471    }

477    /**
478     * Updates window title to reflect scroll lock state.
479     */
480    private static void updateTitleForScrollLock(boolean locked) {
481        if (frame == null) {
482            return;
483        }
484        String baseTitle = frame.getTitle().replace(" [Scroll Locked]", "");
485        frame.setTitle(locked ? baseTitle + " [Scroll Locked]" : baseTitle);
486    }
