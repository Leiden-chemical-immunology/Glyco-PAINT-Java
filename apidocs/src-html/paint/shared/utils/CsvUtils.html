078    /**
079     * Counts the number of records in a CSV file where the column
080     * {@code "Process Flag"} has a value considered as {@code true}.
081     * <p>
082     * Accepted truthy values are: {@code "true"}, {@code "yes"}, {@code "y"},
083     * and {@code "1"} (case-insensitive).
084     *
085     * @param filePath path to the CSV file
086     * @return number of records marked as processed;
087     *         returns {@code 0} if no valid "Process Flag" column is found
088     */
089    public static int countProcessed(Path filePath) {
090        int count = 0;
091
092        try {
093            CSVFormat format = CSVFormat.DEFAULT.builder()
094                    .setHeader()
095                    .setSkipHeaderRecord(true)
096                    .build();
097
098            try (CSVParser parser = CSVParser.parse(filePath.toFile(),
099                                                    StandardCharsets.UTF_8,
100                                                    format)) {
101
102                String processFlagKey = null;
103                for (String header : parser.getHeaderMap().keySet()) {
104                    if (header.trim().equalsIgnoreCase("Process Flag")) {
105                        processFlagKey = header;
106                        break;
107                    }
108                }
109
110                if (processFlagKey == null) {
111                    return 0;
112                }
113
114                for (CSVRecord record : parser) {
115                    String val = record.get(processFlagKey);
116                    if (Miscellaneous.isBooleanTrue(val)) {
117                        count++;
118                    }
119                }
120            }
121        } catch (IOException e) {
122            PaintLogger.errorf("CsvUtils.countProcessed() failed for %s: %s",
123                               filePath, e.getMessage());
124        }
125
126        return count;
127    }

133    /**
134     * Concatenates multiple CSV files into a single CSV file, preserving headers.
135     * <p>
136     * The output file includes one header row (from the first valid input file)
137     * and all data rows from subsequent files. Optionally, the input files can be
138     * deleted after successful concatenation.
139     *
140     * @param inputFiles   list of input CSV file paths
141     * @param outputFile   target output file path
142     * @param deleteInputs if {@code true}, delete input files after processing
143     * @throws IOException if any I/O error occurs during reading or writing
144     */
145    public static void concatenateCsvFiles(List<Path> inputFiles,
146                                           Path outputFile,
147                                           boolean deleteInputs) throws IOException {
148
149        boolean headerWritten = false;
150        CSVPrinter printer = null;
151        List<Path> processed = new ArrayList<>();
152
153        try (BufferedWriter writer = Files.newBufferedWriter(outputFile)) {
154            for (Path inputFile : inputFiles) {
155                if (!Files.exists(inputFile)) {
156                    PaintLogger.warnf("Skipping missing file: %s", inputFile);
157                    continue;
158                }
159
160                try (Reader reader = Files.newBufferedReader(inputFile);
161                     CSVParser parser = CSVFormat.DEFAULT.builder()
162                             .setHeader()
163                             .setSkipHeaderRecord(true)
164                             .build()
165                             .parse(reader)) {
166
167                    if (!headerWritten) {
168                        printer = new CSVPrinter(writer, CSVFormat.DEFAULT.builder()
169                                .setHeader(parser.getHeaderMap().keySet().toArray(new String[0]))
170                                .build());
171                        headerWritten = true;
172                    }
173
174                    for (CSVRecord record : parser) {
175                        printer.printRecord(record);
176                    }
177
178                    processed.add(inputFile);
179                }
180            }
181
182            if (printer != null) {
183                printer.flush();
184            }
185        }
186
187        if (deleteInputs) {
188            for (Path f : processed) {
189                try {
190                    Files.deleteIfExists(f);
191                } catch (IOException e) {
192                    PaintLogger.warnf("Could not delete %s: %s", f, e.getMessage());
193                }
194            }
195        }
196    }

198    /**
199     * Concatenates CSV files located in subdirectories of a given project path.
200     * <p>
201     * Each subdirectory corresponds to an experiment and contains a CSV file
202     * with the specified {@code fileName}. The resulting concatenated CSV file
203     * is written to the root project directory.
204     *
205     * @param projectPath     base directory containing experiment subdirectories
206     * @param fileName        CSV filename to concatenate from each subdirectory
207     * @param experimentNames list of experiment directory names
208     * @throws IOException if an I/O error occurs during processing
209     */
210    public static void concatenateNamedCsvFiles(Path projectPath,
211                                                String fileName,
212                                                List<String> experimentNames) throws IOException {
213        List<Path> inputs = new ArrayList<>();
214        for (String exp : experimentNames) {
215            inputs.add(projectPath.resolve(exp).resolve(fileName));
216        }
217
218        Path output = projectPath.resolve(fileName);
219        concatenateCsvFiles(inputs, output, false);
220    }

226    /**
227     * Adds or updates a {@code "Case"} column in CSV files within the specified
228     * experiment subdirectories.
229     * <p>
230     * If the column exists, its values are overwritten. If not, it is appended
231     * to the end of the record. Each processed file is atomically replaced
232     * after writing the updated version.
233     *
234     * @param root            base path containing experiment subdirectories
235     * @param fileName        target CSV filename within each subdirectory
236     * @param experimentNames list of experiment subdirectory names
237     * @param caseName        value to assign to the "Case" column
238     * @throws IOException if an I/O error occurs during processing
239     */
240    public static void addCase(Path root,
241                               String fileName,
242                               List<String> experimentNames,
243                               String caseName) throws IOException {
244
245        for (String exp : experimentNames) {
246            Path csvPath = root.resolve(exp).resolve(fileName);
247            if (!Files.exists(csvPath)) {
248                PaintLogger.warnf("CsvUtils.addCase(): file not found %s", csvPath);
249                continue;
250            }
251
252            Path tempPath = csvPath.resolveSibling(fileName + ".tmp");
253
254            CSVFormat readFormat = CSVFormat.DEFAULT.builder()
255                    .setHeader()
256                    .setSkipHeaderRecord(true)
257                    .build();
258
259            try (Reader reader = Files.newBufferedReader(csvPath);
260                 CSVParser parser = new CSVParser(reader, readFormat);
261                 BufferedWriter writer = Files.newBufferedWriter(tempPath)) {
262
263                Map<String, Integer> headerMap = parser.getHeaderMap();
264                if (headerMap == null || headerMap.isEmpty()) {
265                    PaintLogger.warnf("CsvUtils.addCase(): missing or invalid header in %s", csvPath);
266                    continue;
267                }
268
269                boolean      hasCaseColumn = headerMap.containsKey("Case");
270                List<String> headers       = new ArrayList<>(headerMap.keySet());
271                if (!hasCaseColumn) {
272                    headers.add("Case");
273                }
274
275                // Define write format only once, now with explicit header
276                CSVFormat writeFormat = CSVFormat.DEFAULT.builder()
277                        .setHeader(headers.toArray(new String[0]))
278                        .build();
279
280                try (CSVPrinter printer = new CSVPrinter(writer, writeFormat)) {
281                    for (CSVRecord record : parser) {
282                        List<String> row = new ArrayList<>();
283                        for (String h : headerMap.keySet()) {
284                            if (h.equals("Case")) {
285                                row.add(caseName); // overwrite if exists
286                            } else {
287                                row.add(record.get(h));
288                            }
289                        }
290                        if (!hasCaseColumn) {
291                            row.add(caseName);
292                        }
293                        printer.printRecord(row);
294                    }
295                    printer.flush();
296                }
297
298                Files.move(tempPath, csvPath, StandardCopyOption.REPLACE_EXISTING);
299                PaintLogger.debugf("Updated 'Case' column in %s", csvPath);
300            }
301        }
302    }
