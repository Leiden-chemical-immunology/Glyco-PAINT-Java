030package paint.shared.config;
031
032import com.google.gson.JsonElement;
033import com.google.gson.JsonObject;
034import com.google.gson.JsonParser;
035
036import java.io.FileReader;
037import java.io.IOException;
038import java.util.ArrayList;
039import java.util.LinkedHashMap;
040import java.util.List;
041import java.util.Map;
042
043// +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
044// NOTE: All `.get(...)`, `.has(...)`, and `.getAsXxx(...)` methods refer to Gson's JsonObject / JsonElement API.
045// +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
046
047/**
048 * The {@code SweepConfig} class provides functionality for loading, parsing,
049 * and retrieving sweep-related configuration from a JSON file.
050 * The configuration is stored internally as a {@link JsonObject}, and methods
051 * are provided to access specific types of data.
052 *
053 * <p>Typical usage:
054 * <pre>
055 *   SweepConfig cfg = new SweepConfig("path/to/file.json");
056 *   Map&lt;String, List&lt;Number&gt;&gt; values = cfg.getActiveSweepValues("CategoryName");
057 * </pre>
058 *
059 * <p>Key behaviours:
060 * <ul>
061 *   <li>Loads JSON from a file path in constructor.</li>
062 *   <li>Processes a category section (if present) to find attributes flagged `true`.</li>
063 *   <li>For each enabled attribute, collects numeric values from its corresponding JSON object.</li>
064 *   <li>Returns an ordered map (in insertion order) of attribute → list of numbers.</li>
065 * </ul>
066 */
067public class SweepConfig {
068
069    private final JsonObject root;
070
071    /**
072     * Constructs a SweepConfig object by parsing a JSON configuration file.
073     * This constructor reads the specified file, parses its content as JSON,
074     * and initializes the configuration data.
075     *
076     * @param filePath the path to the JSON file containing the sweep configuration.
077     * @throws IOException if an I/O error occurs while reading the file.
078     */
079    public SweepConfig(String filePath) throws IOException {
080        try (FileReader reader = new FileReader(filePath)) {
081            this.root = JsonParser.parseReader(reader).getAsJsonObject();
082        }
083    }
084
085    /**
086     * Retrieves the active sweep values for a specified category.
087     * <p>
088     * This method processes the JSON configuration to extract and return
089     * a map containing the category attributes and their corresponding
090     * active numeric values. Only enabled attributes (true) in the category
091     * section are included.
092     *
093     * @param category the name of the category whose sweep values are to be fetched.
094     *                 It corresponds to a key in the root JSON object.
095     * @return a map of attributes to lists of numeric values. The map is empty
096     * if the category does not exist or no active values are found.
097     */
098    public Map<String, List<Number>> getActiveSweepValues(String category) {
099        Map<String, List<Number>> activeValues = new LinkedHashMap<>();
100
101        JsonObject sweep = (root.has(category) && root.get(category).isJsonObject())
102                ? root.getAsJsonObject(category)
103                : null;
104        if (sweep == null) {
105            System.out.println("Sweep Section '" + category + "' not found");
106            return activeValues;
107        }
108
109        for (Map.Entry<String, JsonElement> entry : sweep.entrySet()) {
110            String attribute = entry.getKey();
111            boolean enabled = entry.getValue().getAsBoolean();
112
113            if (enabled && root.has(attribute) && root.get(attribute).isJsonObject()) {
114                JsonObject section = root.getAsJsonObject(attribute);
115                List<Number> values = new ArrayList<>();
116
117                for (Map.Entry<String, JsonElement> valEntry : section.entrySet()) {
118                    JsonElement val = valEntry.getValue();
119                    if (val.isJsonPrimitive() && val.getAsJsonPrimitive().isNumber()) {
120                        Number num = val.getAsNumber();
121                        // Keep as Integer if it has no decimal part
122                        if (num.doubleValue() == num.intValue()) {
123                            values.add(num.intValue());
124                        } else {
125                            values.add(num.doubleValue());
126                        }
127                    }
128                }
129                activeValues.put(attribute, values);
130            }
131        }
132        return activeValues;
133    }
134}
