050/**
051 * The {@code PaintConfig} class provides a configuration management system
052 * for handling configuration data, organized into named sections and key-value pairs.
053 * The configuration is stored as a JSON file and includes methods for reading,
054 * updating, and persisting values. It supports automatic creation of default
055 * configurations and offers both static and instance-level access.
056 *
057 * <p>This class follows a singleton pattern for managing a global configuration
058 * instance, accessed via {@link #instance()}. It also supports initializing and
059 * switching between different configuration files.
060 *
061 * <p>Key functional areas:
062 * <ul>
063 *   <li>Loading existing configuration or creating defaults if missing</li>
064 *   <li>Saving configuration back to disk in pretty-printed JSON using Gson</li>
065 *   <li>Case-insensitive lookup of section names and keys</li>
066 *   <li>Typed getter and setter methods for String, int, double and boolean</li>
067 *   <li>Static convenience methods for global use</li>
068 * </ul>
069 */
070public class PaintConfig {

104    /**
105     * Initializes the global configuration instance with the specified project path.
106     * If an instance already exists, a warning is logged if the path differs.
107     *
108     * @param projectPath the base project directory used to resolve the config file
109     */
110    public static void initialise(Path projectPath) {
111        Path configPath = projectPath.resolve(PAINT_CONFIGURATION_JSON);
112        if (INSTANCE == null) {
113            synchronized (PaintConfig.class) {
114                if (INSTANCE == null) {
115                    INSTANCE = new PaintConfig(configPath);
116                }
117            }
118        } else {
119            Path newConfigPath = projectPath.resolve(PAINT_CONFIGURATION_JSON);
120            if (!INSTANCE.path.equals(newConfigPath)) {
121                PaintLogger.warnf(
122                        "PaintConfig already initialised at %s (attempted reinit with %s)\n",
123                        INSTANCE.path, newConfigPath
124                );
125            }
126        }
127    }
128
129    /**
130     * Reinitialises the global configuration instance with a new project path.
131     *
132     * @param projectPath the base project directory used to resolve the config file
133     */
134    public static void reinitialise(Path projectPath) {
135        synchronized (PaintConfig.class) {
136            INSTANCE = null;
137            initialise(projectPath);
138            PaintLogger.debugf("PaintConfig reinitialised at %s",
139                               projectPath.resolve(PAINT_CONFIGURATION_JSON));
140        }
141    }

143    /**
144     * Returns the singleton instance of {@code PaintConfig}, initializing it
145     * if necessary with the default user-home based path.
146     *
147     * @return the singleton {@code PaintConfig} instance
148     */
149    public static PaintConfig instance() {
150        if (INSTANCE == null) {
151            Path defaultPath = Paths.get(System.getProperty("user.home"), PAINT_CONFIGURATION_JSON);
152            initialise(defaultPath);
153        }
154        return INSTANCE;
155    }

161    /**
162     * Ensures that the configuration data has been loaded into memory.
163     * If not already loaded, reads the JSON file or creates defaults.
164     */
165    private synchronized void ensureLoaded() {
166        if (this.configData != null) {
167            return;
168        }
169
170        if (Files.exists(path)) {
171            try (Reader reader = Files.newBufferedReader(path)) {
172                this.configData = GSON.fromJson(reader, JsonObject.class);
173                if (this.configData == null) {
174                    this.configData = new JsonObject();
175                }
176            } catch (IOException | JsonParseException e) {
177                PaintLogger.errorf("Failed to load config file: %s", e.getMessage());
178                this.configData = new JsonObject();
179            }
180        } else {
181            PaintLogger.warnf("No config file exists at %s, default created.\n", path);
182            this.configData = new JsonObject();
183            loadDefaults();
184            save();
185        }
186    }

236    /**
237     * Saves the current configuration data to disk in pretty-printed JSON format.
238     */
239    public void save() {
240        ensureLoaded();
241        try {
242            Files.createDirectories(path.getParent());
243            try (Writer writer = Files.newBufferedWriter(path)) {
244                GSON.toJson(configData, writer);
245            }
246        } catch (IOException e) {
247            PaintLogger.errorf("Failed to save config file: %s", e.getMessage());
248        }
249    }

465    /**
466     * Retrieves a String value from the specified section and key.
467     * If the key is missing or invalid, logs a warning, writes the default value,
468     * and returns the default.
469     *
470     * @param section      the section name in the configuration
471     * @param key          the key to retrieve (case-insensitive)
472     * @param defaultValue default String value if missing
473     * @return the String value from configuration, or defaultValue if not present/valid
474     */
475    private String getStringInternal(String section, String key, String defaultValue) {
476        ensureLoaded();
477        JsonObject sec = getSection(section);
478        if (sec != null) {
479            String realKey = findKeyIgnoreCase(sec, key);
480            if (realKey != null && sec.get(realKey).isJsonPrimitive()) {
481                return sec.getAsJsonPrimitive(realKey).getAsString();
482            }
483        }
484        PaintLogger.warnf("No value for '%s' found, default '%s' applied", key, defaultValue);
485        setStringValue(section, key, defaultValue, true);
486        return defaultValue;
487    }
488
489    /**
490     * Retrieves an int value from the specified section and key.
491     * If missing or invalid, logs, writes, and returns the default.
492     *
493     * @param section      the section name
494     * @param key          the key name
495     * @param defaultValue default integer value
496     * @return the integer from config or default if missing/invalid
497     */
498    private int getIntInternal(String section, String key, int defaultValue) {
499        ensureLoaded();
500        JsonObject sec = getSection(section);
501        if (sec != null) {
502            String realKey = findKeyIgnoreCase(sec, key);
503            if (realKey != null && sec.get(realKey).isJsonPrimitive()) {
504                try {
505                    return sec.getAsJsonPrimitive(realKey).getAsInt();
506                } catch (NumberFormatException e) {
507                    PaintLogger.warnf("Invalid '%s', default %d applied", realKey, defaultValue);
508                    setIntValue(section, key, defaultValue, true);
509                }
510            }
511        }
512        PaintLogger.warnf("No value for '%s', default %d applied", key, defaultValue);
513        setIntValue(section, key, defaultValue, true);
514        return defaultValue;
515    }

571    /**
572     * Searches for a key in the given JsonObject, ignoring case. Returns the actual
573     * key if found, or null if not found.
574     *
575     * @param obj the JsonObject to search
576     * @param key the key to find (case-insensitive)
577     * @return the real key in obj if found; null otherwise
578     */
579    private String findKeyIgnoreCase(JsonObject obj, String key) {
580        if (obj == null || key == null) {
581            return null;
582        }
583        for (String k : obj.keySet()) {
584            if (k.equalsIgnoreCase(key)) {
585                return k;
586            }
587        }
588        return null;
589    }

591    /**
592     * Retrieves the specified section from the loaded configuration, ignoring case differences
593     * in section names.
594     *
595     * @param section the section name (case-insensitive)
596     * @return the JsonObject for the section if found; null if no such section exists
597     */
598    public JsonObject getSection(String section) {
599        ensureLoaded();
600        return findSectionIgnoreCase(section);
601    }

603    /**
604     * Retrieves or creates a section in the configuration. If the section does not exist,
605     * it is added to the configuration store.
606     *
607     * @param section the section name to retrieve or create
608     * @return the existing or newly created JsonObject section
609     */
610    private JsonObject getOrCreateSection(String section) {
611        ensureLoaded();
612        JsonObject sec = findSectionIgnoreCase(section);
613        if (sec != null) {
614            return sec;
615        }
616        JsonObject newSec = new JsonObject();
617        configData.add(section, newSec);
618        return newSec;
619    }

621    /**
622     * Searches for a section in the configuration data by name, ignoring case.
623     *
624     * @param section the name of the section (case-insensitive)
625     * @return the JsonObject corresponding to the section if found; null otherwise
626     */
627    private JsonObject findSectionIgnoreCase(String section) {
628        ensureLoaded();
629        for (String s : configData.keySet()) {
630            if (s.equalsIgnoreCase(section)) {
631                return configData.getAsJsonObject(s);
632            }
633        }
634        return null;
635    }
