042/**
043 * The {@code TrackMateConfig} class represents configuration parameters used in a
044 * tracking system. It encapsulates various settings related to frame gap,
045 * localization techniques, distance thresholds, and track management
046 * functionalities like merging and splitting.
047 *
048 * <p>This class is immutable as all properties are final and set through
049 * constructors. This allows for thread-safe usage and ensures the
050 * configuration data remains consistent throughout the application.</p>
051 *
052 * <p>Key capabilities:
053 * <ul>
054 *   <li>Specification of tracking parameters such as maximum frame gap, linking
055 *       costs, maximum allowed distances for linking, gap closing, splitting,
056 *       and merging.</li>
057 *   <li>Enabling or disabling specific track behaviors such as subpixel
058 *       localization, track merging, splitting, and gap closing.</li>
059 *   <li>Setting constraints on track construction, including minimum spots per
060 *       track, maximum spots in an image, and other related parameters.</li>
061 *   <li>Configuration of optional operations such as median filtering.</li>
062 *   <li>Provides a utility method to write the configuration to a file in a
063 *       structured string form.</li>
064 * </ul>
065 * </p>
066 */
067public class TrackMateConfig {
068
069    private final int     maxFrameGap;
070    private final double  alternativeLinkingCostFactor;
071    private final boolean doSubpixelLocalization;
072    private final int     minNumberOfSpotsInTrack;
073    private final double  linkingMaxDistance;
074    private final int     maxNumberOfSpotsInImage;
075    private final int     maxNumberOfSecondsPerImage;
076    private final double  gapClosingMaxDistance;
077    private final int     targetChannel;
078    private final double  splittingMaxDistance;
079    private final String  trackColouring;
080    private final double  radius;
081    private final boolean allowGapClosing;
082    private final boolean medianFiltering;
083    private final boolean allowTrackSplitting;
084    private final boolean allowTrackMerging;
085    private final double  mergingMaxDistance;
086
087    /**
088     * Constructs a configuration object for TrackMate with various parameters used for
089     * defining track linking, merging, splitting, and other track-based operations.
090     *
091     * @param maxFrameGap                  Maximum allowable frame gap for linking spots in a track.
092     * @param alternativeLinkingCostFactor Factor affecting the costs during alternative linking of spots.
093     * @param doSubpixelLocalization       Flag indicating whether subpixel spot localization should be performed.
094     * @param minNrSpotsInTrack            Minimum number of spots required for a track to be considered valid.
095     * @param linkingMaxDistance           Maximum allowable distance for linking neighboring spots.
096     * @param maxNumberOfSpotsInImage      Maximum number of spots allowed in a single image/frame.
097     * @param maxNumberOfSecondsPerImage   Maximum number of seconds allowed for a single image/frame.
098     * @param gapClosingMaxDistance        Maximum distance permitted for closing gaps between track segments.
099     * @param targetChannel                The target channel to be used for track analysis.
100     * @param splittingMaxDistance         Maximum allowable distance for splitting tracks.
101     * @param trackColouring               Colouring scheme to use for visualizing tracks.
102     * @param radius                       The radius used for spatial filtering of the spots.
103     * @param allowGapClosing              Flag indicating whether gap closing in tracks is allowed.
104     * @param doMedianFiltering            Flag indicating whether median filtering should be applied to the tracks.
105     * @param allowTrackSplitting          Flag indicating if track splitting at branching points is allowed.
106     * @param allowTrackMerging            Flag indicating if merging of separate tracks is allowed.
107     * @param mergingMaxDistance           Maximum allowable distance for merging tracks.
108     */
109    private TrackMateConfig(
110
111            int     maxFrameGap,
112            double  alternativeLinkingCostFactor,
113            boolean doSubpixelLocalization,
114            int     minNrSpotsInTrack,
115            double  linkingMaxDistance,
116            int     maxNumberOfSpotsInImage,
117            int     maxNumberOfSecondsPerImage,
118            double  gapClosingMaxDistance,
119            int     targetChannel,
120            double  splittingMaxDistance,
121            String  trackColouring,
122            double  radius,
123            boolean allowGapClosing,
124            boolean doMedianFiltering,
125            boolean allowTrackSplitting,
126            boolean allowTrackMerging,
127            double  mergingMaxDistance) {
128
129        this.maxFrameGap                  = maxFrameGap;
130        this.alternativeLinkingCostFactor = alternativeLinkingCostFactor;
131        this.doSubpixelLocalization       = doSubpixelLocalization;
132        this.minNumberOfSpotsInTrack      = minNrSpotsInTrack;
133        this.linkingMaxDistance           = linkingMaxDistance;
134        this.maxNumberOfSpotsInImage      = maxNumberOfSpotsInImage;
135        this.maxNumberOfSecondsPerImage   = maxNumberOfSecondsPerImage;
136        this.gapClosingMaxDistance        = gapClosingMaxDistance;
137        this.targetChannel                = targetChannel;
138        this.splittingMaxDistance         = splittingMaxDistance;
139        this.trackColouring               = trackColouring;
140        this.radius                       = radius;
141        this.allowGapClosing              = allowGapClosing;
142        this.medianFiltering              = doMedianFiltering;
143        this.allowTrackSplitting          = allowTrackSplitting;
144        this.allowTrackMerging            = allowTrackMerging;
145        this.mergingMaxDistance           = mergingMaxDistance;
146        
147    }
148
149    /**
150     * Constructs a configuration object for TrackMate using the parameters provided
151     * in the given PaintConfig instance. The configuration parameters set here are
152     * used for defining track-related operations such as linking, merging, splitting,
153     * and visualization.
154     */
155    public TrackMateConfig() {
156        
157        this.maxFrameGap                  = PaintConfig.getInt(    SECTION_TRACKMATE, MAX_FRAME_GAP, 3);
158        this.alternativeLinkingCostFactor = PaintConfig.getDouble( SECTION_TRACKMATE, ALTERNATIVE_LINKING_COST_FACTOR,2.0);
159        this.doSubpixelLocalization       = PaintConfig.getBoolean(SECTION_TRACKMATE, DO_SUBPIXEL_LOCALIZATION,       true);
160        this.minNumberOfSpotsInTrack      = PaintConfig.getInt(    SECTION_TRACKMATE, MIN_NR_SPOTS_IN_TRACK,          3);
161        this.linkingMaxDistance           = PaintConfig.getDouble( SECTION_TRACKMATE, LINKING_MAX_DISTANCE,           0.6);
162        this.maxNumberOfSpotsInImage      = PaintConfig.getInt(    SECTION_TRACKMATE, MAX_NR_SPOTS_IN_IMAGE,          2000000);
163        this.maxNumberOfSecondsPerImage   = PaintConfig.getInt(    SECTION_TRACKMATE, MAX_NR_SECONDS_PER_IMAGE,       2000);
164        this.gapClosingMaxDistance        = PaintConfig.getDouble( SECTION_TRACKMATE, GAP_CLOSING_MAX_DISTANCE,       1.2);
165        this.targetChannel                = PaintConfig.getInt(    SECTION_TRACKMATE, TARGET_CHANNEL,                 1);
166        this.splittingMaxDistance         = PaintConfig.getDouble( SECTION_TRACKMATE, SPLITTING_MAX_DISTANCE,         1.0);
167        this.trackColouring               = PaintConfig.getString( SECTION_TRACKMATE, TRACK_COLOURING,                "TRACK_DURATION");
168        this.radius                       = PaintConfig.getDouble( SECTION_TRACKMATE, RADIUS,                         1.0);
169        this.allowGapClosing              = PaintConfig.getBoolean(SECTION_TRACKMATE, ALLOW_GAP_CLOSING,              true);
170        this.medianFiltering              = PaintConfig.getBoolean(SECTION_TRACKMATE, DO_MEDIAN_FILTERING,            false);
171        this.allowTrackSplitting          = PaintConfig.getBoolean(SECTION_TRACKMATE, ALLOW_TRACK_SPLITTING,          false);
172        this.allowTrackMerging            = PaintConfig.getBoolean(SECTION_TRACKMATE, ALLOW_TRACK_MERGING,            false);
173        this.mergingMaxDistance           = PaintConfig.getDouble( SECTION_TRACKMATE, MERGING_MAX_DISTANCE,           1.0);
174        
175    }

271    /**
272     * Writes the given TrackMateConfig to the specified file path as a structured string.
273     *
274     * @param trackMateConfig the configuration to write
275     * @param filePath        the path of the file to write the configuration to
276     */
277    public static void trackMateConfigToFile(TrackMateConfig trackMateConfig, Path filePath) {
278        String formattedString = trackMateConfig.toString();
279
280        try {
281            Files.write(filePath, formattedString.getBytes(StandardCharsets.UTF_8));
282        } catch (IOException e) {
283            // Intentionally left blank; log an error if desired
284        }
285    }
